           // store widget form, question heading and questions in global variables
    var audienceForm = document.getElementById("audiences");
    var optionsHeading = document.getElementById("options-heading");
    var contactOptions = document.querySelectorAll("#contact-options > li");

    // call showOptions to show matching questions on change in form
    audienceForm.addEventListener("change", function () {
        showOptions();
    });

    // call showOptions when loading the page to show the currently selected options
    window.addEventListener("load", function(){
        showOptions();
    });

    function showOptions() {
        var selectedOption = audienceForm.options[audienceForm.selectedIndex].value;

        // show the questions heading when a specific audience is selected
        if (selectedOption != '') {
            optionsHeading.classList.remove('hidden');
        }
        else {
            optionsHeading.classList.add('hidden');
        }

        // show items that match with the selected audience 
        // loop over each item
        for (i = 0; i < contactOptions.length; i++) {
            var item = contactOptions[i];

            // reset: hide item
            item.classList.add('hidden');

            // get item values
            var itemValues = item.getAttribute('data-audiences').split(" ");

            // if one of the item values matches with the selected option, show the item
            for (y = 0; y < itemValues.length; y++) {
                var itemValue = itemValues[y];
                if (itemValue == selectedOption) {
                    item.classList.remove('hidden');
                }
            }
        }
    }
