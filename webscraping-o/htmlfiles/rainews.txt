function initTouchMenu() {
  var menu, menuBtn;
  menu = $('.menuTematiche');
  menuBtn = menu.find('span.menuIcon');
  menu.unbind('touchstart touchend click');
  menuBtn.unbind('touchstart touchend click');
  menu.bind('touchend', function() {
    // just need to bind a click handler to make it receive simulated hover events
    var prev, par;
    menu.toggleClass('expanded');
    if (!menu.is('.expanded')) {
      par = menuBtn.parent();
      prev = menuBtn.prev();
      menuBtn.remove();
      window.setTimeout(function() {
        if (prev.length > 0) {
          menuBtn.insertBefore(prev);
        } else {
          par.prepend(menuBtn);
        }
      }, 0);
    }
  });
};
