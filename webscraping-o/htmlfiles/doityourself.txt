    function getVisibility(elem, setAttr = true) {
        elem = $(elem);
        var visible = 0;
        var elemTop = elem.offset().top;
        var elemBottom = elemTop + elem.outerHeight();
        var screenTop = $(window).scrollTop();
        var screenBottom = screenTop + $(window).outerHeight();

        if (elemBottom <= screenTop || elemTop >= screenBottom) {
            // If the element is completely outside of viewport
            visible = 0;
        } else if ((elemTop >= screenTop && elemBottom <= screenBottom) || (elemTop <= screenTop && elemBottom >= screenBottom)) {
            // If the element is completely within of viewport
            //         or if the top of element is above viewport and bottom of element is below viewport
            visible = 100;
        } else if ((elemBottom - elemTop) > (screenBottom - screenTop)) {
            // if the element is larger than viewport
            if (elemTop > screenTop) {
                // if top is within viewport
                visible = ((screenBottom - elemTop) / (screenBottom - screenTop)) * 100;
            } else {
                // assuming bottom is within viewport
                visible = ((elemBottom - screenTop) / (screenBottom - screenTop)) * 100;
            }
        } else if (elemTop < screenTop) {
            // if the top of the element is above viewport, but assuming bottom is within viewport
            visible = ((elemBottom - screenTop) / (elemBottom - elemTop)) * 100.0
        } else if (elemBottom > screenBottom) {
            // if the bottom of the element is below viewport, but assuming top is within viewport
            visible = ((screenBottom - elemTop) / (elemBottom - elemTop)) * 100.0
        }

        if (setAttr)
            elem.data(DATA_PERCENT_VIEW, visible);

        return visible;
    }


    function refreshAds(adsSlotsToRefresh) {
        /*
        * Function accepts array of ads slots ID [Can be obtained by obj.getSlotId()]
        *   Ad slot can't be refreshed within 30 seconds time frame
        */
        let adsToRefresh = []; // Holds the ads slots that will be refreshed
        const minimumViewTime = 30; // Time in seconds
        let currentTime = new Date().getTime()

        for (const i in adsSlotsToRefresh) {
            if (adSlotsMap.get(adsSlotsToRefresh[i]) === undefined) continue;
            let adViewTime = (currentTime - adSlotsMap.get(adsSlotsToRefresh[i])[1]) / 1000

            //Checks if ad never been refreshed or it have been displayed longer then the minimumViewTime
            let canRefreshAd = adSlotsMap.get(adsSlotsToRefresh[i])[1] === -1 || adViewTime >= minimumViewTime;
            if (canRefreshAd && adSlotsMap.get(adsSlotsToRefresh[i])[0] !== "undefined" && typeof adSlotsMap.get(adsSlotsToRefresh[i])[0].getSlotId !== "undefined") {
                adsToRefresh.push(adSlotsMap.get(adsSlotsToRefresh[i])[0])
                adSlotsMap.get(adsSlotsToRefresh[i])[1] = currentTime;
            }
        }

        if (adsToRefresh.length > 0) {
            googletag.pubads().refresh(adsToRefresh);
        }
    }

    function processApstag(arrSlots) {

                    arrGptAdRefrash.push(oSlot);

    function splitContent(text, box, maxRow, offset, url = "") {
        var re = /[^\x00-\xff]/g;
        var style = getComputedStyle(box, null);
        var w = parseInt(style.width);
        var mSize = parseInt(style.fontSize);

        var count = Math.floor(w / mSize);
        var hasDouble = text.match(re);
        var len = hasDouble ? (hasDouble.length + text.length) / 2 : text.length / 2;
        var maxSize = count * maxRow;
        if (len > maxSize) {
            for (var i = maxSize; i < text.length; i++) {
                var mText = text.substr(0, i);
                var has = mText.match(re);
                var mLen = has ? (has.length + mText.length) / 2 : mText.length / 2;
                if (mLen >= maxSize - offset) {
                    text = mText + '...' + '<a href="' + url + '">Read More</a>';
                    break;
                }
            }
        }
        box.innerHTML = text;
    }
