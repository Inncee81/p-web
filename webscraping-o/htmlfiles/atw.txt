   var iSplash, timer_cube;
$(document).ready(function() {
    setTimeout('$(".cube-sides .extra").addClass("shadow")', 550);
    setTimeout('$(".cp-sides").removeClass("cp-sides")', 400);
    setTimeout('$(".extra").removeClass("cp-extra")', 400);
    setTimeout('$(".vps").removeClass("cp-vps")', 600);
    setTimeout('$(".webhosting").removeClass("cp-webhosting")', 800);
    setTimeout('$(".server").removeClass("cp-server")', 1000);
    setTimeout('$(".etc").removeClass("cp-etc")', 1200);
    setTimeout('$(".domain").removeClass("cp-domain")', 1400);
    setTimeout('iSplash.init()', 1600);



    iSplash = (function() {
        var frame_time_slider = 4,
            frame_time_user = 8;

        getBottom = function(current) {
            switch(current) {
                case "domain":
                case "webhosting":
                case "vps":
                case "server":
                    return "extra";
                case "etc":
                    return "domain";
                case "extra":
                    return "webhosting";

            }
        },

        reset = function() {
            console.log( $("div[class^='frame-']").not(".active").removeClass("animated bounceInLeft bounceInRight bounceOutRight bounceOutLeft").addClass("hidden") );
        }



        setFrame = function(name) {
            var pos = {};

              $(".cube-sides .cube-side").removeClass("shadow active");
              $(".cube-sides ." + getBottom(name)).addClass("shadow");
              $(".cube-sides ." + name).addClass("active");
              $(".cube-sides").removeClass("cp-dir-domain cp-dir-webhosting cp-dir-vps cp-dir-extra cp-dir-server cp-dir-etc").addClass("cp-dir-" + name);

              if($("ul.splash-control > li.active").prevAll('[data-attr="' + name + '"]').length) {
                  pos = {
                      out: {
                          start: 0,
                          end: 1000,
                          easing: "easeInQuint"
                      },
                      in: {
                          start: -1000,
                          end: 0,
                          easing: "easeOutQuint"
                      }
                  }
              } else {
                  pos = {
                      out: {
                          start: 0,
                          end: -1000,
                          easing: "easeInQuint"
                      },
                      in: {
                          start: 1000,
                          end: 0,
                          easing: "easeOutQuint"
                      }
                  }
              }

            var obj_out = $("div[class^='frame-'].active").fadeOut();
            $({countNum: pos.out.start}).animate({countNum: pos.out.end}, {
                duration: 800,
                easing: pos.in.easing,
                start: function() {
                    obj_out.removeClass("active");
                },
                step: function () {
                    obj_out.css('-moz-transform',"translateX(" + this.countNum + "px)");
                    obj_out.css('transform',"translateX(" + this.countNum + "px)");
                },
                complete: function () {
                    obj_out.addClass("hidden");
                }
            });

            var obj_in = $("div[class~='frame-"+ name + "']").fadeIn();
            $({countNum: pos.in.start}).delay(200).animate({countNum: pos.in.end}, {
                duration: 800,
                easing: pos.in.easing,
                start: function() {
                    obj_in.removeClass("hidden").addClass("active");
                },
                step: function () {
                    obj_in.css('-moz-transform',"translateX(" + this.countNum + "px)");
                    obj_in.css('transform',"translateX(" + this.countNum + "px)");
                },
                complete: function () {
                }
            });


            // indicator

              $("ul.splash-control > li").removeClass("active");
              $("li[data-attr='" + name + "']").addClass("active");
              $(".indicator").attr("data-current", name);
        }


        startIndicator = function(sec) {
              var i = $(".indicator").css("width", "0%");
              i.removeClass("splash-indicator-anim");
              i[0].offsetWidth = i[0].offsetWidth;
              i.css("-webkit-animation-duration", sec + "s");
              i.css("-moz-animation-duration", sec + "s");
              i.css("-o-animation-duration", sec + "s");
              i.css("-ms-animation-duration", sec + "s");
              i.css("animation-duration", sec + "s");
              i.addClass("splash-indicator-anim");
        }

        startAnim = function() {
            setFrame("domain");
            startTimer();
              startIndicator(frame_time_slider);
        },

        startTimer = function() {
            if(timer_cube)
                clearInterval(timer_cube);

            timer_cube = setInterval("iSplash.showNext()", frame_time_slider * 1000);
        },

        showNext = function() {
              var next = $("ul.splash-control > li.active").next().first().attr("data-attr");

              if(typeof next == "undefined")
                  next = "domain";

              setFrame(next);
              startIndicator(frame_time_slider);
        },

        init = function() {
            $("ul.splash-control > li").click(function() {
                if($(this).hasClass("active"))
                    return;

                if($("div[class^='frame-']").hasClass("animated"))
                    return;



                clearInterval(timer_cube);
                setTimeout("iSplash.startTimer()", (frame_time_user - frame_time_slider) * 1000);
                startIndicator(frame_time_user);
                setFrame($(this).attr("data-attr"));
            });
            startAnim();
        }


        return {
            reset: reset,
            setFrame: setFrame,
            showNext: showNext,
            startTimer: startTimer,
            init: init
        };
    })();


});
  </script>
