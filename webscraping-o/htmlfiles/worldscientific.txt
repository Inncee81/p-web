              var boxwidth = 300;
var backandfor = 0;

var controllerwidth = 0;
var numberofslides = 0;
var currentslide = 0;
var ctime = 500;//complete time for flipping over
var interwidth;
var countchild;
var numberofboxes;

function restartall()
{
  interwidth = $(".col-sm-12").width();// get the width of the content div
  //console.log(interwidth);
  countchild = document.getElementById("controller").childElementCount;// count number of book
  numberofboxes = interwidth / boxwidth;// currently the box width is 300px
  if(countchild == Math.floor(numberofboxes) || countchild < Math.floor(numberofboxes))
  {
    sliderwidth = interwidth;
    controllerwidth = interwidth;
    $(".catalogues").width(interwidth/countchild);// make sure the boxes are on the center of the content div
    numberofslides = 1;
    document.getElementById("arrowback").style.display = "none";
    document.getElementById("arrowfor").style.display = "none";
    currentslide = 1;
  }
  else
  {
    numbooksperrow = Math.floor(interwidth/boxwidth);//find out one page can hold how many books per frame
    expectedwidth = Math.ceil(interwidth/numbooksperrow);// find out the catalogues expected width
    $(".catalogues").width(expectedwidth);
    sliderwidth = interwidth;
    controllerwidth = expectedwidth*countchild;
    backandfor = controllerwidth - sliderwidth;
    if(backandfor > sliderwidth){backandfor = sliderwidth;}
    numberofslides = Math.round(controllerwidth / sliderwidth);
    //console.log(numberofslides);
    currentslide = 1;
    document.getElementById("arrowback").style.display = "none";
    document.getElementById("arrowfor").style.display = "block";
  }
  conheight = $(".catalogues").height();
  //con = $("#controller").offset();
  //console.log(controllerwidth);
  document.getElementById("arrowback").style.top = (conheight/2)+"px";
  document.getElementById("arrowfor").style.top = (conheight/2)+"px";
  $("#slideronstaticpage").width(sliderwidth);
  $("#controller").width(controllerwidth);
}
function next(move)
{
  currentslide++;
  if(move != "no"){$("#controller").animate({left: "-="+backandfor+"px"}, ctime);}
  if(currentslide == numberofslides){document.getElementById("arrowfor").style.display = "none";}
  if(currentslide <= numberofslides && numberofslides > 1){document.getElementById("arrowback").style.display = "block";}
}
function back(move)
{
