	$(document).ready(function(){

		$('.access-page-form-btn').click($.fn.jovebox.close);

		$('.access-page-submit-btn').click(function(){
        	if($('#access-page-email').length && (!$('#access-page-email').val().match(/.*@.*\..*/) || $('#access-page-email').val().match(/\.{2,}/))) {
				$('.access-page-status').html('Please make sure this is a valid email address.').addClass('show');
			} else {
				var access_page_email = $('#access-page-email').val(); 
				$.post( "/access/ajax?action=send_access_page_email", {access_page_email: access_page_email}).done(function(data) {
					//alert(data);
					if(data == '1') { // SUCCESS
						
						// update modal message with success
						$('.access-page-form-btn').html('Close');
						$('.access-page-header h4').html("Please check your inbox to review your institution's access to JoVE");
						$('.access-page-content').hide();

					} else if (data == '-1') {
						// invalid email address
						$('.access-page-status').html('Please make sure this is a valid email address.').addClass('show');
					} else if (data == '-2') {
						// invalid institution
						$('.access-page-status').html('Your institution was not found.').addClass('show');
					} else if (data == '-3') {
						// institution does not have access
						$('.access-page-status').html('Your institution does not subscribe to joVE.').addClass('show');
					} else if (data == '-4') {
						// email address is unsubscribed
						$('.access-page-status').html('Your email address is on the unsubscribed list.').addClass('show');
					} else {
						// unknown error
						$('.access-page-status').html('An unknown error has occurred.').addClass('show');
					}
				});
			}
		});

	});
