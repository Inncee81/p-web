   function umu_initSearchBlock() {
            var suggestionEnabled = "True";
            var suggestionUserEnabled = "False";

            if (suggestionEnabled === "True" || suggestionUserEnabled === "True") {
                enableAutoComplete();
            }

            $(window).on('changed.zf.mediaquery',
                function(event, newSize, oldSize) {
                    if (oldSize === "large" && newSize === "medium" && suggestionEnabled === "False") {
                        enableScrollToSearchField();
                    } else if (newSize === "large" && oldSize === "medium" && suggestionEnabled === "False") {
                        disableScrollToSearchField();
                    }
                });

            function enableAutoComplete() {
                $("#searchField").autocomplete({
                    source: function (request, response) {
                        getFindSuggestions(request, response);
                    },
                    minLength: 2,
                    open: function (event, ui) {
                        var acData = $(this).data("uiAutocomplete");
                        acData.menu.element.find("div").each(function () {
                            var me = $(this);
                            var keywords = acData.term.split(" ").join("|");
                            me.html(me.text().replace(new RegExp("(" + keywords + ")", "gi"), "<b>$1</b>")); //Fetmarkera text i sökförslag matchande sökfråga
                        });
                    },
                    messages: {
                        noResults: function () {
                            if ($("html").attr("lang") === "en") {
                                return "No search results";
                            } else {
                                return "Inga sökförslag";
                            }
                        },  
                        results: function (amount) {
                            if ($("html").attr("lang") === "en") {
                                return amount + " search result" + (amount > 1 ? "s are available" : " is available") + ", use the up and down arrow keys to navigate";
                            } else {
                                return amount + " sökförslag är" + (amount > 1 ? " tillgängliga" : " tillgängligt") + ", använd upp och ned på piltangenterna för att navigera";
                            }
                        }
                    },
                    position: {
                        my: "left-43, top+0" //Positionering dropdown suggestions
                    },
                    select: function (event, ui) {
                        var value = ui.item.value;
                        $("#searchField").val(value); //Uppdatera sökfält med aktuell sökfråga innan klick
                        $("#searchButton").click();
                    }
                });

                $("#searchField").data("uiAutocomplete")._resizeMenu = function () {
                    var width = $(".sokBlockMainField").outerWidth(false);
                    this.menu.element.outerWidth(width);
                };
            }

            function getFindSuggestions(request, response) {
                $.ajax({
                    url: "https://www.umu.se/utbildning/api/utbsuggestion",
                    dataType: "json",
                    type: "POST",
                    data: { query: request.term, lang: document.documentElement.lang },
                    success: function (data) {
                        var result = $.ui.autocomplete.filter(data, request.term);
                        response(result.slice(0, 10));
                    },
                    error: function (data) {
                        console.log("Search error");
                    }
                });
            }

            function enableScrollToSearchField() {
                $("#searchField").on("focus",
                    function() {
                        $('html, body').animate({
                            scrollTop: $(this).offset().top - 10
                        });
                    });
            }

            function disableScrollToSearchField() {
                $("#searchField").off("focus");
            }

            if (!Foundation.MediaQuery.atLeast('large')) {
                enableScrollToSearchField();
            } else {
                disableScrollToSearchField();
            }
        }
