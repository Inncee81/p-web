           (function () {
                            const form = document.querySelector('#sfmc_form');
                            const submit = document.querySelector('#sfmc_form_submit');
                            const params = [
                              'utm_campaign',
                              'utm_source',
                              'utm_medium',
                              'utm_term',
                              'lead_source',
                              'domain_source',
                              'source_channel'
                            ]
                            const getQueryParameters = function (str) {
                              return (str || document.location.search).replace(/(^\?)/, '').split("&").map(function (n) {
                                return n = n.split("="), this[n[0]] = n[1], this
                              }.bind({}))[0];
                            }
                            const urlParams = getQueryParameters();
                            const getHiddenQueryParams = function () {
                              const source = window.location.origin + window.location.pathname;
                              let inputs = [];
                              params.forEach(function (item) {
                                if (urlParams.hasOwnProperty(item) && urlParams[item] !== '') {
                                  inputs.push({
                                    title: item,
                                    val: urlParams[item]
                                  });
                                }
                              })
                              inputs.push({
                                title: 'webpage_url',
                                val: source || ''
                              });
                              return inputs;
                            }
                            const hiddenInputs = getHiddenQueryParams(params);
                            const renderHiddenInputs = function (inputs) {
                              inputs.forEach(function (item, index) {
                                let input = document.createElement('input');
                                input.setAttribute('type', 'hidden');
                                input.setAttribute('id', item.title);
                                input.setAttribute('name', item.title);
                                input.setAttribute('placeholder', item.title);
                                input.setAttribute('value', item.val);
                                form.appendChild(input);
                              })
                            }
                            renderHiddenInputs(hiddenInputs);
                            const serialize = function (form) {
                              let serialized = [];
                              for (let i = 0; i < form.elements.length; i++) {
                                let field = form.elements[i];
                                if (!field.name || field.disabled || field.type === 'file' || field.type === 'reset' || field.type === 'submit' || field.type === 'button') continue;
                                if (field.type === 'select-multiple') {
                                  for (let n = 0; n < field.options.length; n++) {
                                    if (!field.options[n].selected) continue;
                                    serialized.push(encodeURIComponent(field.name) + "=" + encodeURIComponent(field.options[n].value));
                                  }
                                }
                                else if ((field.type !== 'checkbox' && field.type !== 'radio') || field.checked) {
                                  serialized.push(encodeURIComponent(field.name) + "=" + encodeURIComponent(field.value));
                                }
                              }
                              return serialized.join('&');
                            };

                            const customEventTrigger = function(eventName, element, data){
                              if (window.CustomEvent && typeof window.CustomEvent === 'function') {
                                var event = new CustomEvent(eventName, {detail: data});
                              } else {
                                var event = document.createEvent('CustomEvent');
                                event.initCustomEvent(eventName, true, true, data);
                              }
                              element.dispatchEvent(event);
                            }

                            const showMessage = function(){
                              form.classList.add('sfmc_submitted');
                            };

                            function submitForm(event) {
                              event.preventDefault();
                              const formIsComplete = form.checkValidity();
                              if (formIsComplete){

                                const formData = serialize(form);
                                const request = new XMLHttpRequest();
                                request.open(form.getAttribute('method'), form.getAttribute('action'), true);
                                request.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
                                request.onload = function() {
                                  if (this.status >= 200 && this.status < 400) {
                                    console.log(this.status + ' - Success!');
                                  } else {
                                    console.log(this.status + ' - We reached our target server, but it returned an error: ');
                                  }
                                  showMessage();
                                  customEventTrigger('form_has_posted', form);
                                };
                                request.onerror = function() {
                                  console.log(this.status + 'There was a connection error of some sort');
                                  showMessage();
                                  customEventTrigger('form_has_posted', form);
                                };
                                request.send(formData);

                              } else {
                                form.reportValidity();
                              }
                            }
                            form.addEventListener('form_has_posted', function(){
                              console.log('form_has_posted');
                            })
                            submit.addEventListener('click', submitForm);
                          }());
