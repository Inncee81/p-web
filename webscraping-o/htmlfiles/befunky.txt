        (function() {
            var container = document.getElementById('transitioning_images_467963');
            var images = container.querySelectorAll('img');
            var index = 0;
            var interval = 5000;

            if (!false) {
                initImageTransitions();
            } else {
                window.registerLazyComponent(container, initImageTransitions);
            }

            function initImageTransitions() {
                // Load the first image
                loadImage(images[0], function() {

                    // Abort if user prefers reduced motion
                    if (window.matchMedia('(prefers-reduced-motion)').matches) return;

                    // Start loading next image as soon as the page has loaded
                    var startTime = Date.now();
                    BeFunky.onLoad(function() { loadNextImage(startTime); });
                });
            }

            function loadNextImage(startTime) {
                // If user enters app, stop transitioning
                if (window.isAppOpen) return;

                index++;
                if (index >= images.length) index = 0;

                startTime = startTime || Date.now();
                loadImage(images[index], function() {
                    var endTime = Date.now();
                    var loadTime = endTime - startTime;
                    if (loadTime > interval) {
                        displayNextImage();
                    } else {
                        setTimeout(displayNextImage, interval - loadTime);
                    }
                });
            }

            function loadImage(image, callback) {
                loadLazyImage(image);
                if (!image.complete) {
                    image.onload = callback;
                } else {
                    callback();
                }
            }

            function displayNextImage() {
                var activeClass = 'transitioning-images__image--active';
                container.querySelector('.' + activeClass).classList.remove(activeClass);
                images[index].classList.add(activeClass);
                loadNextImage();
            }
        })();
