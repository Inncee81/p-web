    contentCanceledHintComponent = this;

        contentCanceledHintComponent.initVisibleTexts();
        contentCanceledHintComponent.initEventhandlers();
    }
    //contentCanceledHintComponent.initAutoHideTimeout();
};
contentCanceledHintFormController.prototype.initEventhandlers = function () {
    // handles close button click
    if (contentCanceledHintComponent.closeBtn !== undefined && contentCanceledHintComponent.closeBtn !== null) {
        contentCanceledHintComponent.closeBtn.addEventListener('click', contentCanceledHintComponent.onContentCanceledHintFormClose);
    }
    if (contentCanceledHintComponent.contentCanceledHintLink !== undefined && contentCanceledHintComponent.contentCanceledHintLink !== null) {
        contentCanceledHintComponent.contentCanceledHintLink.addEventListener('click', contentCanceledHintComponent.onContentCanceledHintFormClose);
    }
    if (contentCanceledHintComponent.mainElement !== undefined && contentCanceledHintComponent.mainElement !== null) {
        // disable autohide timeout after mouse enters newsletter form
        contentCanceledHintComponent.mainElement.addEventListener('mouseenter', contentCanceledHintComponent.deleteAutoHideTimeout);
        // activate autohide timeout after mouse out newsletter form
        contentCanceledHintComponent.mainElement.addEventListener('mouseout', contentCanceledHintComponent.initAutoHideTimeout);
    }
};
contentCanceledHintFormController.prototype.initVisibleTexts = function () {

contentCanceledHintFormController.prototype.createCookie = function (name, value, days) {
    let expires = '';
    if (days) {
        let date = new Date();
        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
        expires = '; expires=' + date.toUTCString();
    }
    document.cookie = name + '=' + value + expires + '; path=/';
};
contentCanceledHintFormController.prototype.readCookie = function (name) {
    let nameEQ = name + '=';
    let ca = document.cookie.split(';');
    for (let i = 0; i < ca.length; i++) {
        let c = ca[i];
        while (c.charAt(0) === ' ') c = c.substring(1, c.length);
        if (c.indexOf(nameEQ) === 0) return c.substring(nameEQ.length, c.length);
    }
    return null;
};
contentCanceledHintFormController.prototype.eraseCookie = function (name) {
    contentCanceledHintComponent.createCookie(name, '', -1);
};
