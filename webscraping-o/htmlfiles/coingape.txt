    function getCoinGeckoData(){
        var map = {
            bitcoin:{ticker:'btc'},
            ripple:{ticker:'xrp'},
            ethereum:{ticker:'eth'},
            mimblewimblecoin:{ticker:'mwc'},
        }

        var coinGeckoUrl = "https://api.coingecko.com/api/v3/simple/price?ids=bitcoin%2Cmimblewimblecoin%2Cripple%2Cethereum&vs_currencies=usd%2Cgbp%2Ceur%2Cinr";

        var xHttpGecko = new XMLHttpRequest();
        var count = 0;
        xHttpGecko.onreadystatechange = function(){
            if(this.readyState==4 && this.status==200){
                var resp = JSON.parse(xHttpGecko.responseText);
                for(var j in resp){
                    (function(i){
                        var url = "https://api.coingecko.com/api/v3/coins/"+i+"/market_chart?vs_currency=usd&days=1";
                        var req = new XMLHttpRequest();
                        req.onreadystatechange = function(){
                            if(this.readyState==4 && this.status==200){
                                var respChart = JSON.parse(req.responseText).prices;
    
                                var change = (respChart[respChart.length-1][1]-respChart[0][1])/respChart[0][1];
                                map[i]['change'] = (change*100).toFixed(2);
                                map[i]['quotes'] = resp[i];
                                map[i]['live_price'] = respChart[respChart.length-1][1].toFixed(2);
                                count++;
                                if(count==4){
                                    setTickers(map);
                                }
                            }
                        }
                        req.open("GET", url, true);
                        req.send();
                    }(j))
                }
            }
        }
        xHttpGecko.open("GET", coinGeckoUrl, true);
        xHttpGecko.send();
    }
