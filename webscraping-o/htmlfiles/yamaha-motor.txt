function onYouTubeIframeAPIReady() {

  var youtubeVideos = document.querySelectorAll("iframe.youtube-video");
  [].forEach.call(youtubeVideos, function (ytVid) {
    if (!ytVid.id) return;
    
    //first frame needs autoplay and fail detection + fallback GIF
    if (ytVid.className.indexOf("first") != -1) options = {
      events: {
        'onReady': onPlayerReady,
        'onStateChange': onPlayerStateChange,
        playerVars: {
          'autoplay': 1,
          'controls': 0,

    }};
    
    ytVid.player = new YT.Player(ytVid.id, options);
  });
}

function onPlayerReady(event) {
    event.target.playVideo();
}

//detect autoplay fail + serve GIF
var videoStatus = "";
function onPlayerStateChange(event) {

    if (videoStatus === "" && event.data == -1) videoStatus = "unstarted";
    else if (videoStatus == "unstarted" && event.data == 3) videoStatus = "buffering";
    else if (videoStatus == "buffering" && event.data == -1) {

      //FAIL DETECTED
      videoStatus = "failedstart";
      if ($(window).width() < 600) fallbackGif = themePath  + "/public/assets/movies/movie-mobile.gif";
      else fallbackGif = themePath  + "/public/assets/movies/movie-tablet.gif";

      $(event.target.a).hide().parents(".mp-slider-img").first().css("background-image", "url('" + fallbackGif + "')");
      loadDeferredImages();
    }
    //success detection
    else if (videoStatus != "success" && event.data == 1 && videoStatus != "looped") {
      videoStatus = "success";
      loadDeferredImages();
    }

    //loop detection - next slide
    if (videoStatus == "success" && event.data == 3 && event.target.getCurrentTime() === 0) {
      videoStatus = "looped";
      $(".mp-slider").slick("slickNext").slick("slickPlay");
    }
}
