function AdnetLoadScript(url) {
    if (AdnetDebug) console.log("[adnet]: AdnetLoadScript", url);

    if (url === "") return;

    for (let i = 0; i < AdnetLoadedScripts.length; i++) {

function AdnetRunBanner(target, banners) {
    if (AdnetDebug) console.log("[adnet]: AdnetRunBanner", target, banners);
    if (banners === null) {
        return;
    }
    
    if (banners.length < 1) {
        return;
    }
    /*setup*/
    var hasTouchScreen = false;
    var UA = navigator.userAgent;

    var isComputer = true;
    var isMobile = false;
    var isTablet = false;

    if ("maxTouchPoints" in navigator) {
        hasTouchScreen = navigator.maxTouchPoints > 0;
    } else if ("msMaxTouchPoints" in navigator) {
        hasTouchScreen = navigator.msMaxTouchPoints > 0;
    } else {
        var mQ = window.matchMedia && matchMedia("(pointer:coarse)");
        if (mQ && mQ.media === "(pointer:coarse)") {
            hasTouchScreen = !!mQ.matches;
        } else if ('orientation' in window) {
            hasTouchScreen = true; // deprecated, but good fallback
        } else {
            // Only as a last resort, fall back to user agent sniffing       
            hasTouchScreen = (
                /\b(BlackBerry|webOS|iPhone|IEMobile)\b/i.test(UA) ||
                /\b(Android|Windows Phone|iPad|iPod)\b/i.test(UA)
            );
        }
    }

    if (hasTouchScreen) {
        isTablet = /(ipad|tablet|(android(?!.*mobile))|(windows(?!.*phone)(.*touch))|kindle|playbook|silk|(puffin(?!.*(IP|AP|WP))))/.test(UA.toLowerCase());
        if (isTablet) {
            isComputer = false;
            isMobile = false;
        } else {
            isComputer = false;
            isMobile = true;
        }
    }

    if (AdnetDebug) {
        console.log("[adnet]: flags hasTouchScreen", hasTouchScreen, "isTablet", isTablet, "isComputer", isComputer, "isMobile", isMobile)
    }

    /*process banners*/
    for (let i = 0; i < banners.length; i++) {
        banners[i].b = Date.parse(banners[i].begin);
        banners[i].e = Date.parse(banners[i].end);
    }

    let currentDate = new Date();

    // console.log(banners, banners.length, currentDate);

    /*one banner - show aa is*/
    if (banners.length === 1) {
        console.log("[adnet]: only one banner avaible")
        if ((banners[0].b < currentDate) && (currentDate < banners[0].e) || (banners[0].Banner.stub)) {
            AdnetDisplayBanner(target, banners[0]);
        } else {
            if (AdnetDebug) {
                console.log("[adnet]: AdnetRunBanner: Single banner has expired"); 
            }
        }
        return;
    }

    var f = [];
    var stub = {};

    for (let i = 0; i < banners.length; i++) {

        if (banners[i].Banner.stub) {
            stub = banners[i];
        } else {
            if ((banners[i].b < currentDate) && (currentDate < banners[i].e) && ((isComputer & banners[i].computer) || (isMobile & banners[i].phone) || (isTablet & banners[i].tablet)
            )) {
                if (AdnetDebug) {
                    console.log("[adnet]: AdnetRunBanner: Adding banner", banners[i]);
                }
                f.push(banners[i]);
            } else {
                if (AdnetDebug) {
                    console.log("[adnet]: AdnetRunBanner: Skipping banner", banners[i]);
                }
            }
        }
    }

    if (AdnetDebug) {
        console.log("[adnet]: filtering results", f);
    }

    if (f.lenght < 1) {
        if (AdnetDebug) {
            console.log("[adnet]: no banners after filtering, will use stub");
        }
        AdnetDisplayBanner(target, stub);
    } else {

        var sum = 0;
        for (let i = 0; i < f.length; i++) {
            sum += f[i].prio * 1000;
        }

        let ri = Math.floor(sum * Math.random());

        if (AdnetDebug) {
            console.log("[adnet]: sum", sum, "random integer", ri);
        }

        sum = 0;
        for (let i = 0; i < f.length; i++) {
            sum += f[i].prio * 1000;
            if (sum >= ri) {
                AdnetDisplayBanner(target, f[i]);
                return;
            }
        }
    }
}

    }
}

function AdnetStr2ab(str) {
    const buf = new ArrayBuffer(str.length);
    const bufView = new Uint8Array(buf);
    for (let i = 0, strLen = str.length; i < strLen; i++) {
        bufView[i] = str.charCodeAt(i);
    }
    return buf;
}

function AdnetDecode(input) {
    input = input.replace(/-/g, '+').replace(/_/g, '/');
    var pad = input.length % 4;
    if (pad) {
        if (pad === 1) {
            throw new Error('InvalidLengthError: Input base64url string is the wrong length to determine padding');
        }
        input += new Array(5 - pad).join('=');
    }
    return input;
}

async function AdnetVerifyToken(jwToken, pem) {

    try {
        const binaryDerString = window.atob(pem);
        const binaryDer = AdnetStr2ab(binaryDerString);
        const partialToken = jwToken.split('.').slice(0, 2).join('.')
        const signaturePart = jwToken.split('.')[2]
        let result = await window.crypto.subtle.importKey(
            "spki",
            binaryDer,
            {
                name: "ECDSA",
                namedCurve: "P-256"
            },
            true,
            ["verify"]
        ).then(publicKey =>
            window.crypto.subtle.verify(
                { name: 'ECDSA', hash: 'SHA-256' },
                publicKey,
                AdnetStr2ab(atob(AdnetDecode(signaturePart))),
                AdnetStr2ab(partialToken)
            )
        )
        if (result) {
            let x = jwToken.split('.')[1];
            let obj = atob(AdnetDecode(x));
            let cur = Math.floor(Date.now() / 1000);
            obj = JSON.parse(obj);
            //console.log("AdnetVerifyToken: validate token:", obj, cur);
            if ((cur < obj.nbf) || (cur > obj.exp)) {
                console.log("[adnet]: AdnetVerifyToken: jwt token expired")
                localStorage.removeItem("userInf")
                return false;
            }
        } else {
            console.log("[adnet]: AdnetVerifyToken: jwt token validation failed")
            localStorage.removeItem("userInf")
            return false;
        }
        return true;
    }
    catch (err) {
        console.log("[adnet]: AdnetVerifyToken: JWT validation error", err)
        localStorage.removeItem("userInf");
        return false;
    }
}
