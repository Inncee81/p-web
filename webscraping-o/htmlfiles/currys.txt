    function JSChainLoader(srclist, isRetry) {
        if (0 < srclist.length) {
            var src = srclist.shift();
            var script = document.createElement('script');
            script.onload = function(){JSChainLoader(srclist, false);};
            script.onreadystatechange = function(){JSChainLoader(srclist, false);};
            script.onerror = function(){
                if (false === isRetry) {
                    srclist.unshift(src);
                }
                JSChainLoader(srclist, true);
            };
            document.body.appendChild(script);
            script.src = src;
        }
