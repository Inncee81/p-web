     function emailIsValid (email) {
            return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)
        }

        $(document).ready(function () {
            $('#btNewsSubscribe').hide();
        });

        var correctCaptcha = function (response) {
            var recaptcha = $('#g-recaptcha-response').val(response);

            if (recaptcha === "") {
                event.preventDefault();
            }
            else {
                $('#btNewsSubscribe').show();
            }
        };

        // Change the recaptcha styling based on available width
        $(window).resize(function () {
            var recaptcha_mob = $('.g-recaptcha').parent().width();

            $('.g-recaptcha').removeClass("gcaptcha_mob1");
            $('.g-recaptcha').removeClass("gcaptcha_mob2");
            if (recaptcha_mob < 151) {
                $('.g-recaptcha').addClass("gcaptcha_mob1");
            }
            else if (recaptcha_mob < 304) {
                $('.g-recaptcha').addClass("gcaptcha_mob2");
            }
        }).resize();

        $(document).ready(function () {
            if ($(window).width() < 600 && !false) {
                $('.eNewsAlert').hide();
            }

            if (true && $(window).width() >= 600) {
                window.setTimeout(function() {
                        if (!$('.eNewsAlert').hasClass('is-open')) {
                            $('.eNewsAlert').toggleClass('is-open');
                            $('#eNewsHeaderIcon').toggleClass('is-open');
                        }
                    },
                    3000);
            }

            if (true && $(window).width() >= 600) {
                window.setTimeout(function() {
                        if (!$('.eNewsAlert').hasClass('is-open')) {
                            $('.eNewsAlert').toggleClass('is-open');
                            $('#eNewsHeaderIcon').toggleClass('is-open');
                        }
                    },
                    3000);
            }

            $('.eNewsAlert-text').click(function() {
                $('.eNewsAlert').toggleClass('is-open');
                $('#eNewsHeaderIcon').toggleClass('is-open');
            });

            $('#btNewsSubscribe').click(function(e) {
                
                e.preventDefault();

                $('.eNewsAlert-error--emailAddress').hide();
                $('.eNewsAlert-error--zipcode').hide();

                var isValidZip = /(^\d{5}$)|(^\d{5}-\d{4}$)/.test($('#eNewsZipCode').val());
                var isValidEmail = emailIsValid($('#eNewsEmailAddress').val());
                // Verify the form is valid
                    if (isValidZip && isValidEmail) {
                        var $form = $('.eNewsAlert');
                        $(this).hide();
                        $('#eNewsLoading').show();
                        callNewsletterAlertApi($form);
                    } else {
                        if (!isValidEmail) {
                            $('.eNewsAlert-error--emailAddress').show();
                        }
                        if (!isValidZip) {
                            $('.eNewsAlert-error--zipcode').show();
                        }
                    }
                return false;
            });

            $('#eNewsEmailAddress').on('input',
                function() {
                    var isValidEmail = emailIsValid($('#eNewsEmailAddress').val());
                    if (isValidEmail) {
                        $('.eNewsAlert-error--emailAddress').hide();
                    }
                });
            $('#eNewsZipCode').on('input',
                function() {
                    var isValidZip = /(^\d{5}$)|(^\d{5}-\d{4}$)/.test($('#eNewsZipCode').val());
                    if (isValidZip) {
                        $('.eNewsAlert-error--zipcode').hide();
                    }
                });

            $('#eNewsHeaderIcon').on('click', function() {
                if ($(this).hasClass('is-open')) {
                    $('.eNewsAlert').remove();
                    Cookies.set('ALZeNewsClosed', true, {expires: 14});
                }
            });

            $('.eNewsAlert-close').click(function() {
                $('.eNewsAlert').remove();
                Cookies.set('ALZeNewsClosed', true, {expires: 14});

            });
            $('.eNewsAlert-text').click(function() {
                if ($(this).parent().hasClass('is-submitted')) {
                    $('.eNewsAlert-formGroup #emailAddress').val('');
                    $('.eNewsAlert-formGroup #zipCode').val('');
                    $('.eNewsAlert').removeClass('is-open');
                    $('.eNewsAlert').removeClass('is-submitted');
                }
            });
        });

        function callNewsletterAlertApi(form) {
            // Setup for api call
            var convioUrl = '/api/convio/signupwithsurveyid';
            var email = $('#eNewsEmailAddress');
            var zipCode = $('#eNewsZipCode');
            var surveyId = 65205;

            if (email.val() === '' || zipCode.val() === '') {
                // Should not get this far without these values.  Exit if it does.
                return;
            }

            // Data for Convio api
            var postData = { cons_email: email.val(), cons_zip_code: zipCode.val(), survey_id: surveyId, cons_email_opt_in: true };
            // Use Ajax to submit form data
            $.ajax({
                url: convioUrl,
                method: 'POST',
                data: JSON.stringify(postData),
                contentType: 'application/json;charset=utf-8',
                dataType: 'json',
                success: function (data, status, jqXHR) {
                    var isSuccess = data.Data.SubmitSurveyResponse === undefined ? data.Data.submitSurveyResponse.success === 'true' : data.Data.SubmitSurveyResponse.Success === 'true';
                    if (data.Success === true && isSuccess) {
                        // gtm
                        dataLayer.push({ 'event': 'alz-event', 'eventCategory': 'newsletter', 'eventAction': 'signup', 'eventLabel': 'enews-popup-capture', 'eventValue': 0 });
                        $('.eNewsAlert').toggleClass('is-submitted');
                        $('#eNewsLoading').hide();
                    } else {
                        logError("NewsletterBar", "callNewsletterApi", status);
                        $('.eNewsAlert-heading').text("We've run into an issue");
                        $('.eNewsAlert-copy').text("We're sorry, there was an issue submitting your information. Please try again later.");
                        $('.eNewsAlert').toggleClass('is-submitted');
                        $('#eNewsLoading').hide();
                    }
                },
                error: function (jqXHR, status, error) {
                    logError("NewsletterBar", "callNewsletterApi", status);
                    $('.eNewsAlert-heading').text("We've run into an issue");
                    $('.eNewsAlert-copy').text("We're sorry, there was an issue submitting your information. Please try again later.");
                    $('.eNewsAlert').toggleClass('is-submitted');
                    $('#eNewsLoading').hide();
                }
            });
        }
