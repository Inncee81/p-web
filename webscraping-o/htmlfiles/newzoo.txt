	function subscribeFormIsReady() {

		jQuery(document).ready(function($){
			
			$(".hs-input:not(select)").each(function() {

				var input = $(this);
				var inputID = input.attr("id");
				
				input.focus(function() {
					$("label[for=\'" + inputID + "\']").addClass("has-content").addClass("has-focus");
				});
				input.focusout(function() {
					
					if( input.val() == "" ) {
						$("label[for=\'" + inputID + "\']").removeClass("has-content");
					} 

					$("label[for=\'" + inputID + "\']").removeClass("has-focus");
				});
		
			});

			$("select.hs-input").each(function() {

				var selectInput = $(this);
				var selectInputID = selectInput.attr("id");
				
				selectInput.focus(function() {
					$("label[for=\'" + selectInputID + "\']").addClass("has-content").addClass("has-focus");
				});

				selectInput.focusout(function(){

					$("label[for=\'" + selectInputID + "\']").removeClass("has-focus");

					if ($(this).val()) {
						$("label[for=\'" + selectInputID + "\']").addClass("has-content"); 
					} else {
						$("label[for=\'" + selectInputID + "\']").removeClass("has-content");
					}

				});

				selectInput.change(function() {

					if($(this).val() == '') {
						$("label[for=\'" + selectInputID + "\']").removeClass("has-content");
					} else {
						$("label[for=\'" + selectInputID + "\']").addClass("has-content");
					}

				});
		
			});			
			
		});



	} // subscribeFormIsReady();
