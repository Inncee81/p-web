  <script type="text/javascript">
   var $fontElements;
var $colorElements;
var $bkgElements;
var setFontSize;
var setLineHeight;
var setColor;
var setBackground;

jQuery(document).ready(function($){
  
    $fontElements = $("p,h1,h2,h3,h4,h5"); // Can be extended, ex. $("div, p, span.someClass")
    $colorElements = $("p,h1 span,h2,h3,h4,h5");
    $bkgElements = $("#grve-theme-wrapper,.grve-bg-primary-1,.grve-bg-primary-2,.grve-bg-primary-3,.grve-bg-primary-4,.grve-bg-primary-5");
  
    // Handle cookies
    // Font size
    setFontSize = getCookie("userFontSize");      
    if (setFontSize == "undefined" || setFontSize == ""){
        setCookie("userFontSize", 0, 30);
    }
    $fontElements.each(function(){
        var $this = $(this);
        $this.css("font-size", parseInt($this.css("font-size")) + parseInt(setFontSize) + "px");
    });
  
    // Line height
    setLineHeight = getCookie("userLineHeight");
    if (setLineHeight == "undefined" || setLineHeight == ""){
        setCookie("userLineHeight", 0, 30);
    } 
    $fontElements.each(function(){
        var $this = $(this);
        $this.css("line-height", parseInt($this.css("line-height")) + parseInt(setLineHeight) + "px");
    });
  
    // Color contrast
    setColor = getCookie("userColor");      
    if (setColor !== "undefined"){
        $colorElements.each(function(){
            var $this = $(this);
            $this.css("color", setColor);
        });
    }
  
    // Background contrast
    setBackground = getCookie("userBackground");      
    if (setBackground !== "undefined"){
        $bkgElements.each(function(){
            var $this = $(this);
            $this.css("background-color", setBackground);
        });
    }
    
    // Enable Accessibility page UI
    // Store the original styles in a data attribute so size can be reset
    $fontElements.each(function(){
        var $this = $(this);
        $this.data("normal-font", $this.css("font-size"));
        $this.data("normal-line-height", $this.css("line-height"));
        $this.data("normal-contrast", $this.css("color"));
    });

    $colorElements.each(function(){
        var $this = $(this);
        $this.data("normal-contrast", $this.css("color"));
    });

    // Apply actions to buttons
    $("#btn-larger-font").click(function(){
        changeFontSize(2);
        changeLineHeight(2);
    })

    $("#btn-smaller-font").click(function(){
        changeFontSize(-2);
        changeLineHeight(-2);
    })

    $("#btn-normal-font").click(function(){
        // Clear cookies
        clearCookie("userFontSize");
        clearCookie("userLineHeight");     
        // Reload page
        reloadPage();
    })

    $("#btn-normal-contrast").click(function(){
        // Clear cookies
        clearCookie("userColor");
        clearCookie("userBackground");
        // Reload page
        reloadPage();

        });

        // Get existing cookie value and save change to cookie

function reloadPage() {
    location.reload(true);
}

function setCookie(cname, cvalue, exdays){
    var d = new Date();
    d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
    var expires = "expires="+d.toUTCString();
    document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
}

function clearCookie(cname){
    document.cookie = cname + "=; expires=Thu, 01 Jan 1970 00:00:00 UTC;path=/";
}

function getCookie(cname){
    var name = cname + "=";
    var ca = document.cookie.split(';');
    for(var i = 0; i < ca.length; i++){
        var c = ca[i];
        while (c.charAt(0) == ' '){
            c = c.substring(1);
        }
        if (c.indexOf(name) == 0){
            return c.substring(name.length, c.length);
        }
    }
    return "";
}
