			$( this ).closest( '.newsletter-container' ).toggleClass( 'clicked' );
		});

		{
			submitForm( this );
			$( '.form-container form' ).html( '<div style="font-weight:600;" class="text-center">Thank you for signing up for a free Monitor Newsletter!</div>' );
		});

		$( document ).ready( function()
		{
			var currentNewsletterListings = $( '.newsletter-listing' ).length;
			if( currentNewsletterListings > 1 )
			{
				for( var x = 1; x < currentNewsletterListings; x++ )
				{
					$( '.newsletter-listing' )[x].remove();
				}
				triggerNewsletterModal();
			}
			else
			{
				triggerNewsletterModal();
			}
		});

		function triggerNewsletterModal()
		{
			$( '#newsletter-listing-weekender' ).foundation();
			$( '.newsletter-listing-weekender-btn' ).on( 'click', function( event )
			{
				event.preventDefault();
				$( '#newsletter-listing-weekender' ).foundation( 'open' );
			});
		}
		function submitForm( thisForm )
		{
			event.preventDefault();
			var newsletterEmail = $( thisForm ).find( 'input[type="email"]' ).val(),
				rootURL = 'https://christianscience.secure.force.com/freenewslettermonitor?email=',
				newsletterSections = getCheckedNewsletters(),
				trackingParam = getParam( thisForm );

			var url = rootURL + encodeURIComponent( newsletterEmail ) + newsletterSections + trackingParam;

			window.location = url;
		}

		function getCheckedNewsletters()
		{
			var newsletterSections = '';
			$( '#newsletter-listing-weekender input[type="checkbox"]:checked' ).each( function()
			{
				newsletterSections = newsletterSections + '&' + $( this ).attr( 'id' ) + '=True';
			});
			return newsletterSections;
		}

		function getParam( form )
		{
			var param = setDefaultParam();
			var newsletterBannerParent = $( form ).parents( '.newsletter-banner' ).parent();
			if( newsletterBannerParent.hasClass( 'param-added' ) )
			{
				param = '&icid=mkt:web:' + newsletterBannerParent.attr( 'data-param' );
			}
			return param;
		}

		function setDefaultParam()
		{
			var defaultParam = '&icid=mkt:web:wknd';
			if ( currentSection )
			{
				if ( currentSection === 'books' )
				{
					defaultParam = '&icid=mkt:web:books';
				}
				else if( currentSection === 'politics' )
				{
					defaultParam = '&icid=mkt:web:politics';
				}
				else if( currentSection === 'todays-highlights' )
				{
					defaultParam = '&icid=mkt:web:todays-highlights';
				}
			}
			if( $( 'body' ).hasClass( 'ezc-frontpage' ) )
			{
				defaultParam = '&icid=mkt:web:politicsfrontpage';
			}
			return defaultParam;
		}

			}
			else if ( section === 'books' )
			{
				$bannerIllustration.replaceWith( '<img class="float-left hide-for-small-only" src="//images.csmonitor.com/csm/2019/09/1089038_1_opened-books_standard.png?alias=original" alt="opened books">' );

				$( '#newsletter-listing-weekender #week, #newsletter-listing-weekender #poli' ).removeAttr( 'checked' ).closest( '.newsletter-container' ).removeClass( 'clicked' );
