var namesMatch = function(name, vname){
    var name1 = name.replace(/[\W_]+/g, ' ').trim().toLowerCase();
    var name2 = vname.replace(/[\W_]+/g, ' ').trim().toLowerCase();
    if(!name1 || !name2) return false;

var findMatchingVideo = function(vp, vid){
    var pl = vp.getActualPlaylist();
    vid = vid.replace(/\/+$/,'').replace(/%20/g, ' ');
    if(vid.match(/^\d+$/)){
        var number = parseInt(vid);
        return pl[number-1];
    }else if(vid.match(/^\w\w-/)){
        var s = sources[vid.substr(0, 2)];
        for(var v in pl){
            if(pl[v].source == s && pl[v].video_id == vid.substr(3)){
                return pl[v];
            }
        }
    }else{
        var bm = 0;
        var winner = -1;
        for(var i=0; i<pl.length; i++){
            var candidate = namesMatch(vid, pl[i].title);
            if(candidate > bm){
                bm = candidate;
                winner = i;
            }
        }
        if(winner == -1) return null;

        if(window.console && console.log)
            console.log('Found video: ' + winner + ' for query ' + vid);
        return pl[winner];
    }
}

var parseAction = function(vp, x){
    var video = null;
    var start = 0;
    var stop = null; // at the end of video
    if(x.match(/\//)){
        var vid_pos = x.split(/\//, 2);
        x = vid_pos[0];
        var positions = vid_pos[1].split(/\-|\.\./, 2);
        start = parsePosition(positions[0]) || 0;
        if(positions[1] && positions[1].trim()){
            stop = parsePosition(positions[1]) || null;
        }
    }
    video = findMatchingVideo(vp, x);
    return {
        'video': video,
        'start': start,
        'stop': stop
    };
};

           <h1 id="titleHeading" style="background-color: white; margin: 0;">
            Phenomenon
           </h1>
           <p id="mwBA">
            A
            <b id="mwBQ">
             phenomenon
            </b>
            (
            <a href="https://wn.com/Greek_language" id="mwBg" title="Greek language">
             Greek
            </a>
            : o,
            <i id="mwBw">
             phainomenon
            </i>
            , from the verb ,
            <i id="mwCA">
             phainein
            </i>
            , to show, shine, appear, to be manifest or manifest itself, plural
            <b id="mwCQ">
             phenomena
            </b>
            ), is any thing which manifests itself. Phenomena are often, but not always, understood as "things that appear" or "
            <a href="https://wn.com/Experience" id="mwCg" title="Experience">
             experiences
            </a>
            " for a
            <a href="https://wn.com/Sentient" id="mwCw" title="Sentient">
             sentient
            </a>
            being, or in principle may be so.
           </p>
           <p id="mwDA">
            The term came into its modern philosophical usage through
            <a href="https://wn.com/Immanuel_Kant" id="mwDQ" title="Immanuel Kant">
             Immanuel Kant
            </a>
            , who contrasted it with the
            <a href="https://wn.com/Noumenon" id="mwDg" title="Noumenon">
             noumenon
            </a>
            . In contrast to a phenomenon, a noumenon is not directly accessible to observation. Kant was heavily influenced by
            <a href="https://wn.com/Gottfried_Leibniz" id="mwDw" title="Gottfried Leibniz">
             Leibniz
            </a>
            in this part of his philosophy, in which phenomenon and noumenon serve as interrelated technical terms.
           </p>
           <h2 id="mwGA">
            Modern philosophical usage
           </h2>
           <p id="mwGQ">
            In modern philosophical use, the term 'phenomena' has come to mean 'what is experienced is the basis of
            <a href="https://wn.com/Reality" id="mwGg" title="Reality">
             reality
            </a>
            '.  In Immanuel Kant's
            <i id="mwGw">
             Inaugural Dissertation, On the Form and Principles of the Sensible and Intelligible World
            </i>
            (1770), Kant theorizes that the human mind is restricted to the logical world and thus can only interpret and understand occurrences according to their physical appearances.  He wrote that humans could infer only as much as their senses allowed, but not experience the actual object itself.  Thus, the term phenomenon refers to any incident deserving of inquiry and investigation, especially events that are particularly unusual or of distinctive importance. According to The Columbia Encyclopedia, "Modern philosophers have used 'phenomenon' to designate what is apprehended before judgment is applied." This may not be possible if observation is
            <a href="https://wn.com/Theory-ladenness" id="mwHA" title="Theory-ladenness">
             theory laden
            </a>
            .
           </p>

           <h1 id="titleHeading" style="background-color: white; margin: 0;">
            Appearance (law)
           </h1>
           <p id="mwAQ">
            In
            <a href="https://wn.com/Law" id="mwAg" title="Law">
             law
            </a>
            ,
            <b id="mwAw">
             appearance
            </b>
            (from
            <a href="https://wn.com/Latin" id="mwBA" title="Latin">
             Latin
            </a>
            <i id="mwBQ">
             apparere
            </i>
            , to appear) is the coming into court of either of the parties to a
            <a href="https://wn.com/Lawsuit" id="mwBg" title="Lawsuit">
             lawsuit
            </a>
            , and/or the formal act by which a
            <a href="https://wn.com/Defendant" id="mwBw" title="Defendant">
             defendant
            </a>
            submits himself to the
            <a href="https://wn.com/Jurisdiction" id="mwCA" title="Jurisdiction">
             jurisdiction
            </a>
            of the
            <a href="https://wn.com/Court" id="mwCQ" title="Court">
             court
            </a>
            .
           </p>
           <h2 id="mwCg">
            Legal details (outdated)
           </h2>
           <p id="mwDQ">
            The defendant in an action in the
            <a href="https://wn.com/High_Court" id="mwDg" title="High Court">
             High Court
            </a>
            of England enters his appearance to the
            <a href="https://wn.com/Summons" id="mwDw" title="Summons">
             writ of summons
            </a>
            by delivering, either at the central office of the
            <a href="https://wn.com/Supreme_Court" id="mwEA" title="Supreme Court">
             Supreme Court
            </a>
            , or a
            <a href="https://wn.com/High_Court_of_Justice" id="mwEQ" title="High Court of Justice">
             district registry
            </a>
            , a written
            <a href="https://wn.com/Memorandum" id="mwEg" title="Memorandum">
             memorandum
            </a>
            either giving his
            <a href="https://wn.com/Solicitor" id="mwEw" title="Solicitor">
             solicitor
            </a>
            's name or stating that he defends in person. He must also give notice to the
            <a href="https://wn.com/Plaintiff" id="mwFA" title="Plaintiff">
             plaintiff
            </a>
            of his appearance, which ought, according to the time limited by the writ, to be within eight days after service; a defendant may, however, appear any time before judgment.  The
            <i id="mwFQ">
             <a href="https://wn.com/Rules_of_the_Supreme_Court" id="mwFg" title="Rules of the Supreme Court">
              Rules of the Supreme Court
             </a>
            </i>
            , orders xii. and xiii., regulate the procedure with respect to the entering of an appearance, the giving of notice, the limit of
            <a href="https://wn.com/Time" id="mwFw" title="Time">
             time
            </a>
            , the setting aside and the general effect of default of appearance. In
            <a href="https://wn.com/County_court" id="mwGA" title="County court">
             county courts
            </a>
            there is no appearance other than the coming into court of the parties to the suit. In
            <a href="https://wn.com/Criminal" id="mwGQ" title="Criminal">
             criminal
            </a>
            cases the accused appears in person.  In
            <a href="https://wn.com/Civil_cases" id="mwGg" title="Civil cases">
             civil cases
            </a>
            infants appear by their guardians
            <i id="mwGw">
             ad litem
            </i>
            ;
            <a href="https://wn.com/Lunatics" id="mwHA" title="Lunatics">
             lunatics
            </a>
            by their committee; companies by a solicitor; friendly societies by the
            <a href="https://wn.com/Trustee" id="mwHQ" title="Trustee">
             trustee
            </a>
            or other officer appointed to
            <a href="https://wn.com/Lawsuit" id="mwHg" title="Lawsuit">
             sue
            </a>
            or be sued on behalf thereof.

<p id="mwAg"><b id="mwAw">Appearance</b> may refer to:</p>


<li id="mwBQ"> <a href="https://wn.com/Visual_appearance" title="Visual appearance" id="mwBg">Visual appearance</a>, the way in which objects reflect and transmit light</li>

<li id="mwBw"> <a href="https://wn.com/Human_physical_appearance" title="Human physical appearance" id="mwCA">Human physical appearance</a>, what someone looks like</li>

<li id="mwCQ"> <a href="https://wn.com/Appearances_(film)" title="Appearances (film)" id="mwCg"><i id="mwCw">Appearances</i> (film)</a>, a 1921 film directed by Donald Crisp</li>

<li id="mwDA"> <i id="mwDQ"><a href="https://wn.com/The_Appeared" title="The Appeared" id="mwDg">The Appeared</a></i>, a 2007 Spanish language horror film by <a href="https://wn.com/Paco_Cabezas" title="Paco Cabezas" id="mwDw">Paco Cabezas</a>
</li>

<li id="mwEA"> Appearance, another term for the baseball statistic <a href="https://wn.com/Games_pitched" title="Games pitched" id="mwEQ">games pitched</a>
</li>

<li id="mwEg"> <a href="https://wn.com/Appearance_(philosophy)" title="Appearance (philosophy)" id="mwEw">Appearance (philosophy)</a>, or phenomenon</li>

<li id="mwFA"> <a href="https://wn.com/Appearance_(law)" title="Appearance (law)" id="mwFQ">Appearance (law)</a>, the coming into court of either of the parties to a suit, and/or the formal act by which a defendant submits himself to the jurisdiction of the court.</li>
