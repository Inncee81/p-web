function getQueryString() {
  let result = {},
    queryString = location.search.slice(1),
    re = /([^&=]+)=([^&]*)/g,
    m;
  while (m = re.exec(queryString)) {
    result[decodeURIComponent(m[1])] = decodeURIComponent(m[2]);
  }
  return result;
}

        userAgent.match(/Opera/i) &&
        userAgent.match(/Windows.NT.5/i) &&
        userAgent.match(/HTC|Xda|Mini|Vario|SAMSUNG-GT-i8000|SAMSUNG-SGH-i9|Lumia|Generic Smartphone|Generic Feature Phone|Samsung GT-I8750/i)
      ) {
        return 'phone';
      }
      // Windows Desktop.
      if (
        userAgent.match(/Windows.(NT|XP|ME|9)/) && !userAgent.match(/Phone/i) ||
        userAgent.match(/Win(9|.9|NT)/i)
      ) {
        return 'desktop';
      }
      // Mac Desktop.
      if (userAgent.match(/Macintosh|PowerPC/i) && !userAgent.match(/Silk/i)) {
        return 'desktop';
      }
      // Linux Desktop.
      if (userAgent.match(/Linux/i) && userAgent.match(/X11/i)) {
        return 'desktop';
      }
      // Solaris, SunOS, BSD Desktop.
      if (userAgent.match(/Solaris|SunOS|BSD/i)) {
        return 'desktop';
      }
      // Desktop BOT/Crawler/Spider.
      if (
        userAgent.match(/Bot|Crawler|Spider|Yahoo|ia_archiver|Covario-IDS|findlinks|DataparkSearch|larbin|Mediapartners-Google|NG-Search|Snappy|Teoma|Jeeves|TinEye/i) &&
        !userAgent.match(/Mobile/i)
      ) {
        return 'desktop';
      }
      // Else assume it is a Mobile Device (mobile-first).
      return 'other';
    },

    getOsName: function () {
      const appVersion = window.navigator.appVersion;
      if (appVersion.indexOf('Chrome OS') !== -1) {
        return 'chromeos';
      }
      if (appVersion.indexOf('Blackberry OS') !== -1) {
        return 'blackberryos';
      }
      if (appVersion.indexOf('Windows Phone') !== -1) {
        return 'windowsphone';
      }
      if (appVersion.indexOf('Linux') !== -1) {
        return 'linux';
      }
      if (appVersion.indexOf('Ubuntu') !== -1) {
        return 'ubuntu';
      }
      if (appVersion.indexOf('Win') !== -1) {
        return 'windows';
      }
      if (appVersion.indexOf('Mac') !== -1) {
        return 'macos';
      }
      const userAgent = window.navigator.userAgent;
      if (userAgent.match(/(iPad|iPhone|iPod)/g)) {
        return 'ios';
      }
      if (userAgent.toLowerCase().indexOf('android') !== -1) {
        return 'android';
      }
      if (userAgent.toLowerCase().indexOf('wii') !== -1) {
        return 'wii';
      }
      return 'other';
    },

    getType: function () {
      const nav = window.analyticsData.nav;

      let type = 'home';
      if (window.analyticsData.type == 'index' ||
        nav == 'economie' ||
        nav == 'sport' ||
        nav == 'boulevard' ||
        nav == 'editienl') {
        type = 'navigation'
      }
      if (window.analyticsData.type == 'article') {
        if (typeof window.analyticsData.sponsor_company != 'undefined') {
          type = 'branded'
        } else {
          type = 'detail'
        }
      }

      return type;
    },

    getQueryString: function() {
      let result = {},
        queryString = location.search.slice(1),
        re = /([^&=]+)=([^&]*)/g,
        m;
      while (m = re.exec(queryString)) {
        result[decodeURIComponent(m[1])] = decodeURIComponent(m[2]);
      }
      return result;
    }
  };
})(window);
