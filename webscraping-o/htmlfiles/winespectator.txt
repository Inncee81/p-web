    function homepage_hero_random_order() {
    var cards = document.querySelectorAll(".new-card");
    var card_leader = document.querySelector(".new-card-leader");
    var a = ["1", "3", "4", "5", "6", "7"]
    var j, x, i;

    // Shuffle them
    for (i = a.length - 1; i > 0; i--) {
      j = Math.floor(Math.random() * (i + 1));
      x = a[i];
      a[i] = a[j];
      a[j] = x;
    }

    for (i = 0; i < cards.length; i++) {
      cards[i].classList.add("order-" + a[i]);
      card_leader.classList.remove("new-card-leader");

    }

    var new_card_leader = document.querySelector('.order-1');
    new_card_leader.classList.add("new-card-leader");
  }

  var sort_1 = ["1", "3", "4", "5", "6", "7"];
  var sort_2 = ["7", "1", "3", "4", "5", "6"];
  var sort_3 = ["6", "7", "1", "3", "4", "5"];
  var sort_4 = ["5", "6", "7", "1", "3", "4"];
  var sort_5 = ["4", "5", "6", "7", "1", "3"];
  var sort_6 = ["3", "4", "5", "6", "7", "1"];

  function homepage_schedule_sort(array) {
    var cards = document.querySelectorAll(".new-card");
    var card_leader = document.querySelector(".new-card-leader");

    var a = array;

    for (i = 0; i < cards.length; i++) {
      cards[i].classList.add("order-" + a[i]);
      card_leader.classList.remove("new-card-leader");
    }

    var new_card_leader = document.querySelector('.order-1');
    new_card_leader.classList.add("new-card-leader");
  }

  function homepage_hero_scheudle_order_1() {
    if (new Date().getHours() %2) {
      if (new Date().getMinutes() <= 20) {
        homepage_schedule_sort(sort_1);
      } else if (new Date().getMinutes() > 20 && new Date().getMinutes() <= 40 ) {
        homepage_schedule_sort(sort_2);
      } else {
        homepage_schedule_sort(sort_3);
      }
    } else {
      if (new Date().getMinutes() <= 20) {
        homepage_schedule_sort(sort_4);
      } else if (new Date().getMinutes() > 20 && new Date().getMinutes() <= 40 ) {
        homepage_schedule_sort(sort_5);
      } else {
        homepage_schedule_sort(sort_6);
      }
    }
  }

  function homepage_hero_scheudle_order_2() {
    switch (new Date().getHours()) {
      case 0:
      case 1:
      case 12:
      case 13:
        homepage_schedule_sort(sort_1);
        break;
      case 2:
      case 3:
      case 14:
      case 15:
        homepage_schedule_sort(sort_2);
        break;
      case 4:
      case 5:
      case 16:
      case 17:
        homepage_schedule_sort(sort_3);
        break;
      case 6:
      case 7:
      case 18:
      case 19:
        homepage_schedule_sort(sort_4);
        break;
      case 8:
      case 9:
      case 20:
      case 21:
        homepage_schedule_sort(sort_5);
        break;
      case 10:
      case 11:
      case 22:
      case 24:
        homepage_schedule_sort(sort_6);
    }
  };
