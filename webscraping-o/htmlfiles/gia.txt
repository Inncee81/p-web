    //------------------------------------------------------------------------------------------------
    // NEW: '$trapFocus()' is an alternate solution to below/standard trapFocus() in that it takes as
    // parameters a jQuery DOM element as it's first argument and an array of string value 'selectors'
    // it's second -- these selectors will determine which elements are considered "focusable"
    // and to ultimately enforce the 'trapFocus' behavior only on those elements vs. all typically
    // focusable elements as in the standard trapFocus fn
    //------------------------------------------------------------------------------------------------
    var $trapFocus = function ($el, selectorsArr) {
      var selectorStr = selectorsArr.join(" "),
        $focusableEls = $el.find(selectorStr),
        firstFocusableEl = $focusableEls[0],
        lastFocusableEl = $focusableEls[$focusableEls.length - 1],
        KEYCODE_TAB = 9;

      $el.on("keydown", function (e) {
        var isTabPressed =
          e.key === "Tab" || e.keyCode === KEYCODE_TAB || e.which === KEYCODE_TAB;

        if (!isTabPressed) {
          return;
        }

        if (e.shiftKey) {
          if (document.activeElement === firstFocusableEl) {
            lastFocusableEl.focus();
            e.preventDefault();
          }
        } else {
          if (document.activeElement === lastFocusableEl) {
            firstFocusableEl.focus();
            e.preventDefault();
          }
        }
      });
    };
