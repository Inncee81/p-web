      <h3>
       About Samba
      </h3>
      <p>
       <b>
        Samba is the standard Windows interoperability suite of programs for Linux and Unix.
       </b>
      </p>
      <p>
       Samba is
       <a href="https://www.gnu.org/philosophy/free-sw.html" target="_blank">
        Free Software
       </a>
       licensed under
	the
       <a href="/samba/docs/GPL.html">
        GNU General Public License
       </a>
       , the Samba project is a member
	of the
       <a href="https://www.sfconservancy.org/" target="_blank">
        Software Freedom Conservancy
       </a>
       .
      </p>
      <p>
       Since
       <a href="/samba/docs/10years.html">
        1992
       </a>
       , Samba has provided secure, stable and fast
	file and print services for all clients using the SMB/CIFS protocol, such as all versions of DOS
	and Windows, OS/2, Linux and many others.
      </p>
      <p>
       Samba is an important component to seamlessly integrate Linux/Unix Servers and Desktops
	into Active Directory environments. It can function both as a domain controller or as a
	regular domain member.
      </p>

   // make sure both boxes in #mainStart have the same height
			function fixMainBoxHeight() {
				var releases = document.getElementById('mainBoxReleases');
				var beyondSamba = document.getElementById('mainBoxBeyondSamba');

				if (releases.offsetHeight > beyondSamba.offsetHeight) {
					beyondSamba.style.height = '' + (releases.offsetHeight - 24)+ 'px';
				}
				else if (releases.offsetHeight < beyondSamba.offsetHeight) {
					releases.style.height = '' + (beyondSamba.offsetHeight - 24) + 'px';
				}
			}

			// move #sitemap downwards if #special ends below #sitemap
			function fixSitemapPosition() {
				var mainStart = document.getElementById('mainStart');
				var sitemap = document.getElementById('sitemap');
				var special = document.getElementById('special');

				if (special.offsetHeight > mainStart.offsetHeight + sitemap.offsetHeight) {
					sitemap.style.marginTop = '' + (special.offsetHeight - mainStart.offsetHeight - sitemap.offsetHeight - 13) + 'px';
				}
			}

			window.setTimeout(function() {
				fixMainBoxHeight();
				fixSitemapPosition();
			}, 50);
  </script>
