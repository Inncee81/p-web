  <script>
   var mobile = 576;
var pad = 992;

var nowDevice = 0;
var pastDevice = 0;

function getUrlParameter(search,name,device) {
	name = name.replace(/[\[]/, '\\[').replace(/[\]]/, '\\]');
	var regex = new RegExp('[\\?&]' + name + '=([^&#]*)');
	var results = regex.exec(search);
	var result = results === null ? '':decodeURIComponent(results[1].replace(/\+/g, ' '));   
	var regexExt = /(?:\.([^.]+))?$/;
	var ext = regexExt.exec(result)[0];   // "txt"
	
	if(result && ext){
	  var tempDevice = "";
	  if(result.lastIndexOf("_small") != -1){
		tempDevice = result.substring(0, result.lastIndexOf("_small")-1);
	  }else{
		tempDevice = result.substring(0, result.lastIndexOf("."));
	  }
	  if(device == 1){
		tempDevice += "0_small_mobile" + ext;
		search = search.replace(result, tempDevice);
	  }else if(device == 2){
		tempDevice += "0_small_pad" + ext;
		search = search.replace(result, tempDevice);
	  }else{
		tempDevice += "0_small" + ext;
		search = search.replace(result, tempDevice);
	  }
	  return search;
	}else{
	  return null;
	}
};

function findImage(device) {

	var imagemagicks = document.querySelectorAll("[data-src]");
    
    [].forEach.call(imagemagicks, function(imagemagick) {
    
        var oldImg = imagemagick.getAttribute("data-src");
        
        if(oldImg){
			var nowImge = getUrlParameter(oldImg,"file",device);
			if(nowImge){
          
				if(imagemagick.tagName == "IMG"){
					imagemagick.src = nowImge;
					imagemagick.onerror = function () {
						this.src = this.getAttribute("data-src");
					};
				}else{
					imagemagick.style.backgroundImage = "url('" + nowImge + "')";
					var imagebgi = new Image();
					imagebgi.src = nowImge;
					imagebgi.onerror = function () {
						imagemagick.style.backgroundImage = "url('" + imagemagick.getAttribute("data-src") + "')";
					}
				}
            
			}
        }
        
    });
	
}

function whichDevice(width){
	var device = 3;
	//mobile
	if(width < mobile){
		device = 1;
	//pad
	}else if(width >= mobile && width < pad){
		device = 2;
	//pc
	}else{
		device = 3;
	}
	return device;
}

function resizeFunction() {
  
  var width = document.body.clientWidth;
  
  if(pastDevice == 0){
	pastDevice = whichDevice(width);
	findImage(pastDevice);
  }else{
	nowDevice = whichDevice(width);
	if(pastDevice != nowDevice){
	  findImage(nowDevice);
	  pastDevice = nowDevice;
	}
  }
  
}

window.addEventListener("resize", resizeFunction);

window.onload = function() {
	resizeFunction();
};
