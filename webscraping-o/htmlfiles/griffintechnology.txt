  function ajaxLoad(url){
    $.ajax({
      url: url,
      type: 'get',
      data: {},
      beforeSend: function() {
        $('#wrapper_ajax').show();
      },
        success: function(data) {
          $('#wrapper_ajax').hide();
          // update content
          $('#custom-products').html($(data).find('#custom-products').html());
          $('.collection-sidebar').html($(data).find('.collection-sidebar').html());
          $('#pagination').html($(data).find('#pagination').html());
          // update url
          history.replaceState({
            page: url
          }, url, url);
          // update View All button url
          $('.item-pagination.view-all').attr('href', url+'?view=all');
          // reinitialize
          reorderSoldout('.block2');
          collectionSwatch();
          clickFilteredSwatch();
          ajaxify();
          quickview();
          initColorName('.block2');
          updateColorName('.block2');

          if($(window).width() < 700){
            resetMobileFilter();
          } 
          $('html, body').animate({
            scrollTop: $('#custom-products').offset().top - 75
          }, 500);
        },
        error: function(xhr, ajaxOptions, thrownError) {
          alert(xhr.responseText);
        }
    });
  }

  function collectionSwatch(){
    $('.block2 .swatch :radio').on('click', function() {
      var variantId = $(this).parent().attr('data-variant-id');
      var container = $(this).parents('.block2');
      var soldout = $(this).parent().hasClass('soldout') ? true : false;
      var cartBtn = $(container).find('[title="Add to Cart"]');
      // swap variant img based on variant id
      $.each($('.img-'+variantId).siblings(), function(index, value){
        if (!$(value).hasClass('no-display')){
          $(value).addClass('no-display');
        }
      });
      $('.img-'+variantId).removeClass('no-display');
      // update product link url with variant id
      $.each($(container).find('a'), function(index, value){
        var url = $(value).attr('href');
        if (url.indexOf('?variant') != -1){
          $(value).attr('href', url.substring(0, url.indexOf('?variant')));
        }
        $(value).attr('href', $(value).attr('href')+'?variant='+variantId);
      });
      // update add to cart url with variant id
      var onclick = $(container).find('[title="Add to Cart"]').attr('onclick');
      var oldId = onclick.substring(onclick.indexOf("'")+1, onclick.indexOf("'", onclick.indexOf("'")+1));
      $(container).find('[title="Add to Cart"]').attr('onclick', onclick.replace(oldId, variantId));
      // if variant is soldout disable add to cart button and update text
      if (soldout){
        $(cartBtn).prop('disabled', true);
        $(cartBtn).text('Sold Out');
      } else {
        $(cartBtn).prop('disabled', false);
        $(cartBtn).text('Add to Cart');
      }
    });
  }

  function ajaxify(){
    $('.filter-container :radio').unbind('click').on('click', function(){

    $('.filter-device, .remove-filter').unbind('click').on('click', function(e){

    });

    $('.item-pagination').unbind('click').on('click', function(e){
      e.preventDefault();
      ajaxLoad($(this).attr('href'));
    });        
  }
