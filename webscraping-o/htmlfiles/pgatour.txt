   (function ($, global) {

		global.pgatour.HomePage = global.pgatour.BaseModule.extend({

			init: function () {
				this.base();
				this.topAd = this.container.find('.ad-top');
                this.leaderboardBanner = this.container.find('.leaderboard-banner');
                this.leaderboardBanner.addClass('always-visible');
				this.t1 = this.container.find('.module-t1');
				this.disableMoving = $('body').hasClass('body-wgc-theme');
			},

			refreshTopAd: function () {
				if (this.topAd.length) {
					var api = this.topAd.find('.js-ad').data('ad-api');
					if (api) {
						api.update();
					}
				}
			},

			scroll: function () {
                pgatour.WebPerformanceTracker.trackFirstScrollEvent();
            },

            done: function () {
                setTimeout(this.proxy(this.trackDefaultNREventsForOmniture), 0);
            },

            trackDefaultNREventsForOmniture: function () {
                pgatour.WebPerformanceTracker.trackDefaultNREventsForOmniture();
            },

			layout: function (screenSize) {
				if (this.disableMoving) {
					return;
				}
				if (screenSize === 'small') {
					if (this.topAd.parent().hasClass('container')) {
						this.layoutModulesForSmall();
						this.refreshTopAd();
					}
				} else {
					if (!this.topAd.parent().hasClass('container')) {
						this.layoutModules();
						this.refreshTopAd();
					}
				}
			},

			layoutModules: function () {
				var lbParent = this.leaderboardBanner.parent();
				this.leaderboardBanner.insertBefore(this.t1);
				lbParent.append(this.t1);
				var anchor = lbParent.closest('.page');
				this.topAd.insertBefore(anchor);
				if (global.t1) {
					global.t1.stretchImages();
				}
			},

			layoutModulesForSmall: function () {
				var t1Parent = this.t1.parent();
				this.t1.insertBefore(this.leaderboardBanner);
				t1Parent.append(this.leaderboardBanner);
				this.topAd.insertAfter(this.t1);
