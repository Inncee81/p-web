        i11 = $pagesLi11.index($o11.find('.banner_pages li.current'));
        (i11==0)? i11=(len11/2-1):i11--;
        l11 = parseInt($ul11.css('left')) + w11;
        addCurrent11(i11,l11);
        return false;
