    function parseLocation(geoLoc) {
        var geoLocation = geoLoc.split("|");
        var geoLocObj = {};
        geoLocation.forEach(function (geoItem) {
            var geoKeyVal = geoItem.split(":").map(item => item.trim());
            geoLocObj[geoKeyVal[0]] = geoKeyVal[1];
        })
        geoLocObj["pollenEnabled"] = window.pollenEnabled;
        geoLocObj["airQualityEnabled"] = window.airQualityEnabled;
        return geoLocObj;
    }
