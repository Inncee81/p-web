   const KEY_POSITIONS = {

        function toolbar() {
            return {
                isOpenDesktop: false,
                isOpenMobile: false,
                timeout: undefined,
                key: "",
                lastKey: "",
                directionStyle: "",
                        mainMobile: "smore-for",
                open(key) {
                    if(this.timeout){
                        window.clearTimeout(this.timeout);
                    }
                    this.timeout = window.setTimeout(() => {
                        this.lastKey = this.key;
                        this.isOpenMobile = false;
                        this.isOpenDesktop = true;
                        this.key = key;
                        this.directionStyle = '--slide-direction: ' + this.direction(key)
                    }, 150)
                    
                },
                closeDesktop() {
                    if(this.timeout){
                    window.clearTimeout(this.timeout);
                    }
                    this.directionStyle = "";
                    this.isOpenDesktop = false;
                    if (!this.isOpenMobile) {
                        this.key = "";
                    }
                },
                close() {
                    if(this.timeout){
                    window.clearTimeout(this.timeout);
                    }
                    this.directionStyle = "";
                    this.isOpenDesktop = false;
                    this.isOpenMobile = false;
                    this.key = "";
                },
                openMobile(key) {
                    this.isOpenDesktop = false;
                    this.isOpenMobile = true;
                    this.key = key;
                },
                closeMobile() {
                    this.isOpenMobile = false;
                    this.key = "";
                },
                direction(key) {
                    if (!this.lastKey) {
                        return 0;
                    }

                    if (KEY_POSITIONS[this.lastKey] > KEY_POSITIONS[this.key]) {
                        return -1;
                    }

                    return 1;
                }
            }
        }
