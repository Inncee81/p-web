        function translateMktoForm(descriptor, xlateLangs, session) {
          var lang = xlateLangs.find(function(language) {
              return language.lang == session.lang;
            });
          lang.fields.forEach(function(xlateField) {
            descriptor.rows.forEach(function(templateRow) {
              templateRow
                .filter(function(templateField) {
                  return templateField.Name == xlateField.Name;
                })
                .forEach(function(templateField) {
                  Object.assign(templateField, xlateField);
                });
            });
          });
          descriptor["SubmitLabel"] = lang.globals["SubmitLabel"] || descriptor["SubmitLabel"];
          descriptor["ResetLabel"] = lang.globals["ResetLabel"] || descriptor["ResetLabel"];
          descriptor["ButtonText"] = lang.globals["ButtonText"] || descriptor["ButtonText"];
          descriptor["ButtonSubmissionText"] = lang.globals["ButtonSubmissionText"] || descriptor["ButtonSubmissionText"];
          descriptor["invalidInputMsg"] = lang.globals["invalidInputMsg"] || descriptor["invalidInputMsg"];
          descriptor["formSubmitFailedMsg"] = lang.globals["formSubmitFailedMsg"] || descriptor["formSubmitFailedMsg"];
          return descriptor;
        }
