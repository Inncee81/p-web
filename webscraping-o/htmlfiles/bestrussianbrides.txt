        function flexibleBlock(min, max) {
            if (window.location.pathname === '/') {
                setTimeout(function () {
                    let flexibleBlock = document.querySelector('.flexible-block.insert-clone-box');

                    if (flexibleBlock != null || flexibleBlock != undefined) {
                        min = $(flexibleBlock).offset().top;         // find the min point - start of flex
                        max = $('.video-poster').offset().top - 500; // find the max point - end of flex
                        $(window).on('scroll', function () {
                            if (document.documentElement.scrollTop > min) {
                                $(flexibleBlock).css({
                                    'position': 'fixed',
                                    'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                                    'top': '0',
                                    'opacity': '1',
                                    'transform': 'translateY(70px)',
                                    'transition': 'transform .15s linear',
                                    'z-index': '500'
                                });
                            } else if (document.documentElement.scrollTop < min) {
                                $(flexibleBlock).css({
                                    'position': 'relative',
                                    'right': '0',
                                    'transform': 'translateY(0)',
                                    'transition': 'transform .15s linear',
                                    'z-index': '500'
                                });
                            }

                            if (document.documentElement.scrollTop > max) {
                                $(flexibleBlock).css({
                                    'opacity': '0',
                                    'transition': 'transform .15s linear',
                                    'z-index': '-1'
                                });
                            }
                        })
                    }
                }, 500)
            }
        }

        if (document.documentElement.clientWidth >= 1200) {
            flexibleBlock();
        }

    });

  <script>
   jQuery(document).ready(function ($) {
        //    ------------------------------ LAZY IMAGES ------------------------------
        function lazyImagesSrc() { // For tag <img src="...">
            const lazyImages = document.querySelectorAll('.lazy-img');

            const options = {
                root: null,
                rootMargin: '200px',
                threshold: 0.1
            };

            function handleImg(myImg, observer) {
                myImg.forEach(myImgSingle => {

                    if (myImgSingle.intersectionRatio > 0) {
                        loadImage(myImgSingle.target);
                    }
                })
            }

            function loadImage(image) {
                image.src = image.getAttribute('data-l-img');
            }

            const observer = new IntersectionObserver(handleImg, options);

            lazyImages.forEach(img => {
                observer.observe(img);
            });
        }

        lazyImagesSrc();

        function lazyImagesBg() { // For blocks with style="background: url('...')"
            const lazyImages = document.querySelectorAll('.lazy-bg');

            const options = {
                root: null,
                rootMargin: '200px',
                threshold: 0.1
            };

            function handleImg(myImg, observer) {
                myImg.forEach(myImgSingle => {

                    if (myImgSingle.intersectionRatio > 0) {
                        loadImage(myImgSingle.target);
                    }
                })
            }

            function loadImage(image) {
                image.style.background = `url(${image.getAttribute('data-l-bg')})`;
            }

            const observer = new IntersectionObserver(handleImg, options);

            lazyImages.forEach(img => {
                observer.observe(img);
            });
        }

        lazyImagesBg();
        //    ------------------------------ LAZY IMAGES ------------------------------
    });
  </script>
