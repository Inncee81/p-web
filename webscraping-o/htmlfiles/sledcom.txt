   $(function () {
        var now = moment();
        var targetData = moment('2020-05-10');

        function getDiffString(diffDay) {
            return diffDay > 0 ? diffDay + '' : '';
        }

        function setDayNumber(diffString, parent) {
            parent.html('')

            for (var i = 0; i < diffString.length; i++) {
                var curNumber = diffString.charAt(i);
                parent.append($('<div class="may9-news__numbers-item">' + curNumber + '</div>'))
            }
        }

        function setDayCaption(diffString, parent) {
            var aEndings = ['день', 'дня', 'дней'];
            var caption = getNumEnding(diffString, aEndings);
            parent.text(caption)
        }

        function getNumEnding(iNumber, aEndings) {
            var sEnding, i;
            iNumber = iNumber % 100;
            if (iNumber >= 11 && iNumber <= 19) {
                sEnding = aEndings[2];
            } else {
                i = iNumber % 10;
                switch (i) {
                    case (1):
                        sEnding = aEndings[0];
                        break;
                    case (2):
                    case (3):
                    case (4):
                        sEnding = aEndings[1];
                        break;
                    default:
                        sEnding = aEndings[2];
                }
            }
            return sEnding;
        }

        var diff = getDiffString(targetData.diff(now, 'days'));

        var numbersParent = $('#may9-numbers')
        var may9day = $('#may9-day')

        if (diff === '') {
            //удаление счетчика
            numbersParent.html('')
            $('.may9-news__title').text('')
            may9day.html('')
        } else {
            setDayNumber(diff, numbersParent)
            setDayCaption(diff, may9day)
        }

    })

    $('.subscribe__form:not(.panel) form').on('submit', function (e) {

        e.preventDefault();

        var form = $(this);

        $.ajax({
            type: "GET",
            url: form.attr('action'),
            dataType: "json",
            data: form.serialize(),
            success: function(result) {

                if (result.status == 'ok') {


                    $('div.subscribe__form').after(result.data);
                    $('div.subscribe__form').hide();
                    form.find('input').val('');

                } else if (result.status == 'error') {

                    var fields = result.errors_fields;
                    var errors = result.data;

                    form.find('.has-error').removeClass('has-error');

                    for (var i = 0; i < fields.length; i++) {
                        var field = fields[i];
                        var error = errors[fields[i]];

                        var error_div = form.find('.input-info.'+field);

                        error_div.parent().addClass('has-error');

                        error_div.text(error);
                    }
                }

                form.find('input[name=captcha]').val('');

                update_captcha();
            }
        })
    })
