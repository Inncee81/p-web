       </h4>
       <!--Date -->

       </div>
       <!--Story body -->
       <p>
        Celebrate your wildlife heritage. President Theodore Roosevelt established the first national wildlife refuge on March 14, 1903, at Pelican Island, Florida, to protect brown pelicans there from slaughter by market hunters. Todayâ€™s Refuge System includes more than 560 national wildlife refuges, 38 wetland management districts and 5 marine national monuments.

  // Popover Help
$(function () {

    var overPopup = false;

    $('[rel=popover]').popover({
        trigger: 'manual',
        placement: 'top'

    // replacing hover with mouseover and mouseout
    }).mouseover(function (e) {
        // when hovering over an element which has a popover, hide
        // them all except the current one being hovered upon
        $('[rel=popover]').not('#' + $(this).attr('id')).popover('hide');
        var $popover = $(this);
        $popover.popover('show');

        // set a flag when you move from button to popover
        // dirty but only way I could think of to prevent
        // closing the popover when you are navigate across
        // the white space between the two
        $popover.data('popover').tip().mouseenter(function () {
            overPopup = true;
        }).mouseleave(function () {
            overPopup = false;
            $popover.popover('hide');
        });

    }).mouseout(function (e) {
        // on mouse out of button, close the related popover
        // in 2000 milliseconds=2sec if you're not hovering over the popover
        var $popover = $(this);
        setTimeout(function () {
            if (!overPopup) {
                $popover.popover('hide');
            }
        }, 2000);
    });
});
 </script>
