     var g_shouldShowVideo = false;
    $("#frmZipCodeVerification_PhoneNumber").validate({

            }
        },
        messages: {
            PhoneNumber: {
                required: "Please enter phone number"
            }
        },
        errorPlacement: function (error, element) {
            error.insertAfter(element);
        }
    });

    $("#frmZipCodeVerification_ZipCode").validate({
        rules: {
            txtZipCode: {
                required: true,
                minlength: {
                    param: 5,
                    depends: function () {
                        $(this).val($.trim($(this).val()));
                        return true;
                    }
                }
            }
        },
        messages: {
            txtZipCode: {
                required: "Please enter a valid zip code",
                minlength: "You need to enter a 5 characters zip code"
            }
        },
        errorPlacement: function (error, element) {
            error.insertAfter(element);
        }
    });


    $(document).ready(function () {
        $('.btnDontShowZipCodeRequest').on('click', function () {
            var params = {
                ShowZipCodeRequest: false
            };

            $.ajax({
                url: '/Home/SetDontShowAgain',
                type: "POST",
                async: false,
                cache: false,
                contentType: 'application/json; charset=utf-8',
                data: JSON.stringify({ model: params }),
                success: function (result) {
                    //console.log('Dont Show Again sent.');
                },
                error: function (xhr) {
                    console.log(xhr.responseText);
                }
            });
        });

        $("#zipCodeVerification").on({
            popupafteropen: function () {
                if (g_locationAcquired) {
                    $.ajax({
                        url: '/Home/GetZipCodeAndState',
                        type: "POST",
                        async: false,
                        cache: false,
                        contentType: 'application/json; charset=utf-8',
                        data: '',
                        success: function (result) {
                            if (result && typeof (result.ZipCode) != "undefined") {
                                $('#txtZipCode').val(result.ZipCode);
                            }
                        },
                        error: function (xhr) {
                            console.log(xhr.responseText);
                        }
                    });

                    $('.nolocationAcquired').hide();
                    $('.locationAcquired').show();
                } else {
                    $('.nolocationAcquired').show();
                    $('.locationAcquired').hide();
                }
                if (typeof (g_zipCodePartial) != "undefined") {
                    if (g_zipCodePartial.btnUpdateZipCodeClicked) {
                        $('.btnDontShowZipCodeRequestContainer').hide();
                    }
                    else {
                        $('.btnDontShowZipCodeRequestContainer').show();
                    }
                } else {
                    $('.btnDontShowZipCodeRequestContainer').show();
                }
                $('.lblTitle').html('ZIP CODE VERIFICATION');
                $('.step1').show();
                $('.step2').hide();
            },
            popupafterclose: function () {
                if (g_shouldShowVideo) {
                    $("#myYardOverviewVideo").popup("open");
                }
            }
        });

        $('#btnSubmitStep1').on('click', function () {

            var isValid = $("#frmZipCodeVerification_ZipCode").valid();
            if (isValid) {
                var zipcodevalue = $('#txtZipCode').val();
                var params = {
                    zipcode: zipcodevalue
                };

                $.ajax({
                    url: '/Home/SetLocationByZipCode',
                    type: "POST",
                    async: false,
                    cache: false,
                    contentType: 'application/json; charset=utf-8',
                    data: JSON.stringify(params),
                    success: function (result) {
                        if (result && typeof (result.Success) != "undefined") {
                            g_locationAcquired = result.Success;
                            g_menuPartial.Reload = true;
                            if (typeof (g_zipCodePartial) != "undefined") {
                                g_zipCodePartial.Reload = true;
                            }
                        }
                        else {
                            g_locationAcquired = false;
                        }
                        if (result && result.State) {
                            launchStep2(result.State);
                        }
                    },
                    error: function (xhr) {
                        console.log(xhr.responseText);
                    }
                });
            }
        });

        $("#btnSubmitStep2").click(function () {
            var isValid = $("#frmZipCodeVerification_PhoneNumber").valid();
            if (isValid) {
                $('.waitpage').show().find('.message').html('One moment please, <br /> processing your request.');

                var phoneNumberValue = $('#txtZIPCODEPhoneNumber').val();

                var params = {
                    phoneNumber: phoneNumberValue
                };

                $.ajax({
                    cache: false,
                    async: true,
                    type: "POST",
                    url: "/Plant/Signup",
                    contentType: 'application/json; charset=utf-8',
                    data: JSON.stringify(params),
                    success: function (result) {
                        $('.waitpage').hide();
                        g_shouldShowVideo = true;
                        $("#zipCodeVerification").popup('close');
                        g_menuPartial.Reload = true;
                    },
                    error: function (e, s, t) {
                        $('.waitpage').hide();
                        console.log("Error: ", e, s, t);
                        g_shouldShowVideo = true;
                        $("#zipCodeVerification").popup('close');
                    }
                });
            }
        });

        function launchStep2(stateName) {
            var hidIsSMS = $('#hidIsSMS').val();

            var isSMS = false;
            if (hidIsSMS == 's') {
                isSMS = true;
            }

            $.ajax({
                url: '/Home/getSessionValues',
                type: "POST",
                async: false,
                cache: false,
                contentType: 'application/json; charset=utf-8',
                data: '',
                success: function (result) {
                    if (result && typeof (result.IsRestrictedArea) != "undefined") {
                        if (result.IsRestrictedArea) {
                            if (isSMS) {
                                $('.txtPhoneNumberContainer').hide();
                            }
                            else {
                                $('.txtPhoneNumberContainer').show();
                            }

                            if (result.IsEndorserOptedIn) {
                                $("#zipCodeVerification").popup('close');
                            } else {
                                $('.step1').hide();
                                $('.lblTitle').html('Plant Care Updates');
                                $('.step2').show();
                            }
                        }
                        else {
                            $("#zipCodeVerification").popup('close');
                        }
                    }
                },
                error: function (xhr) {
                    console.log(xhr.responseText);
                }
            });


        }

    });
