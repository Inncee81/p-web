     $(document).ready(function(){
    $('#check_serial').on("click", function(e) {
        e.preventDefault();
        serial.checkDownload (
            $('#serial_field').val()
        );
    });
    $('#check_upgrade').on("click", function(e) {
        e.preventDefault();
        serial.checkUpgrade (
            $('#serial_field').val()
        );
    });
});

serial = {
    checkUpgrade: function(serial) {
        var app = this;
        this.check(serial, function(data) {
            $("#result").html(app.upgradeMessage(data));
        });
    },

    checkDownload: function(serial) {
        var app = this;
        this.check(serial, function(data) {
            $("#result").html(app.downloadMessage(data));
        });
    },

    check: function (serial, callback) {
        var app = this;
        var data = JSON.stringify({serial: serial});
        $.ajax({
            type: "POST",
            url: 'https://ability.com/utils/serial',
            data: data,
            dataType: "json",
            contentType: "application/json; charset=utf-8",
            headers: {
                'X-CSRF-TOKEN':'9Yat9UF4P7BXnDcc9xH1EzY01u2dTvCTpXAeaRKq'
            }
        }).done(function (data) {
            if (typeof data === 'object' && data.version > 5) {
                callback(data);
            }
            else
                app.error();

        }).fail(function (xhr, status, error) {
            // console.log(xhr, status, error);        // TODO - debug
            app.error();
        }).always(function () {
        });
    },

    downloadMessage: function(data) {
        var s = "<p>Ability Office " + data.type + " <b>version " + data.version + "</b><br />";
        var s = s + data.description;
        if (data.version < 10)
            s = s + "<p>Please go to our <a href='/en/download/legacy'>Download Page for previous versions</a>.</p>";
        return s;
    },

    upgradeMessage: function(data) {
        if (data.version == 10)
            return "You can download and install version 10 free of charge!";
        return "Sorry, you cannot upgrade for free. The latest version you can install for free is: " + data.version;
    },

    error: function() {
        $("#result").html('<p>Sorry, serial number could not be determined (could be version 5 or earlier perhaps).</p><p>Please contact support.</p>');
    }
}
