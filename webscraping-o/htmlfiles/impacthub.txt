          $ = jQuery;
	let links = $('.mod-mitabs .tabs-columns [data-columns-class]');
	let tabs = $(links).closest('.mi-column');

	function handleTabButtonClick(link, buttons, tabs) {
		$ = jQuery;
		let tab = $(link).closest('.mi-column');
		let content = $(link).attr('href');
		let button = $(link).children('button');
		let allContents = $('.' + $(link).attr('data-columns-class'));
		if($(tab).hasClass('active')) {
			//deactivate clicked tab
			hideTab(tab, button, content);
		} else {
			let oldTab = $(tabs).filter('.active');
			let oldButton = $(buttons).not('.transparent');
			let oldContent = $(allContents).filter('.active');
			console.log(oldTab);
			console.log(oldButton);
			console.log(oldContent);
			//deactivate current tab
			hideTab(oldTab, oldButton, oldContent);
			//activate the clicked tab
			displayTab(tab, button, content);
		}
	}

	function hideTab(tab, button, content) {
		$ = jQuery;
		$(tab).removeClass('active');
		$(content).removeClass('active');
		$(button).addClass('transparent');
		switchButtonText(button);
	}

	function displayTab(tab, button, content) {
		$ = jQuery;
		$(tab).addClass('active');
		$(content).addClass('active');
		$(button).removeClass('transparent');
		switchButtonText(button);
	}

	function switchButtonText(button) {
		let current = $(button).text();
		let next = $(button).attr('data-second-text');
		console.log(current);
		console.log(next);
		$(button).attr('data-second-text', current);
		$(button).text(next);
	}

	$(links).click(function(e) {
		$ = jQuery;
		let buttons = $(links).children('button');
		handleTabButtonClick(this, buttons, tabs);
	});
