      var viewTop = scrollTop + wh;
      var scrollPercent = (viewTop / height).toFixed(2) * 100;
      if (scrollPercent < 25) {
        return;

        var data = 25;

          eventCategory: "Scroll Depth",
          eventAction: pageCategory,
          eventLabel: data + "%",
        });
        firstScroll25 = false;
      }
      if (scrollPercent >= 50 && scrollPercent < 75 && firstScroll50) {
        var data = 50;
        sendGAEvent({
          nonInteraction: true,
          eventCategory: "Scroll Depth",
          eventAction: pageCategory,
          eventLabel: data + "%",
        });
        firstScroll50 = false;
      }
      if (scrollPercent >= 75 && scrollPercent < 100 && firstScroll75) {
        var data = 75;
        sendGAEvent({
          nonInteraction: true,
          eventCategory: "Scroll Depth",
          eventAction: pageCategory,
          eventLabel: data + "%",
        });
        firstScroll75 = false;
      }
      if (scrollPercent === 100 && firstScroll100) {
        var data = 100;
        sendGAEvent({
          nonInteraction: true,
          eventCategory: "Scroll Depth",
          eventAction: pageCategory,
          eventLabel: data + "%",
        });
        firstScroll100 = false;
