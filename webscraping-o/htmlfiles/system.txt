    let i = 0;
        let phrase = 0;
        let search_text = [ "customer engagement",
            "COVID-19",
            "climate change",
            "lifetime value",
            "anything" ]
        let speed = 150;

        function typeRouter() {
          if (i < search_text[phrase].length) {
            // if still typing current phrase
            type();

          } else if (phrase < search_text.length-1) {
            // if end of phrase, next phrase
            phrase++;
            i = 0;
            // document.getElementById("search").innerHTML = "";
            document.getElementById("search_span").innerHTML = "";
            type();
          } else if (phrase === search_text.length-1) {
            // if end of array, reset and loop
            phrase = 0;
            i = 0;
            // document.getElementById("search").innerHTML = "";
            document.getElementById("search_span").innerHTML = "";
            type();
          }
        }

        function type() {
          // document.getElementById("search").innerHTML += search_text[phrase].charAt(i);
          document.getElementById("search_span").innerHTML += search_text[phrase].charAt(i);
          i++;

          let speed;
          if (i === search_text[phrase].length) {
            // if it's the last i in the set, pause for a bit
            speed = 1200;
          } else if (search_text[phrase] === "anything" ) {
            speed = 300;
          } else {
            // generate a random number between 0 and 300
            speed = Math.floor(Math.random()*150);
          }
          setTimeout(typeRouter, speed);
        }

        typeRouter();
