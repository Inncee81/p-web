      navigate(id, index, !0);

      var viewportWidth = viewport.width();
      var itemWidth = $(items).find(".ntap-hp-carousel__item").width();
      var offsetLeft = (viewportWidth - itemWidth) / 2;
      var sliderPos = -(itemWidth * index - offsetLeft) - itemWidth * 2;
      if (transition === !0) {
        items.css("transition", "transform 0.5s ease-in-out");
        items.css("transform", "translate3d(" + sliderPos + "px, 0px, 0px)");

          .find(".ntap-hp-carousel__items a:nth-child(" + (index + 2) + ")")
          .addClass("active");
      } else {
        items.css("transition", "none");
        items.css("transform", "translate3d(" + sliderPos + "px, 0px, 0px)");
      }

        .find(".ntap-hp-carousel__nav-dot")
        .addClass("active");
    }

      navigate(id, index, !1);
