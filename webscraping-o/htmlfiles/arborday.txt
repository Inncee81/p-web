				var totalHeight = vh + 48; // 48px is the height of our header which we need to add - look into this further!
				$("body").animate({

		// sticky header // hide header on on scroll down
			var didScroll;
			var lastScrollTop = 0;
			var delta = 5;
			var navbarHeight = $('header[role="banner"]').outerHeight();

			$(window).scroll(function(event){
				didScroll = true;
			});

			setInterval(function() {
				if (didScroll) {
					hasScrolled();
					didScroll = false;
				}
			}, 250);

			function hasScrolled() {
				var st = $(this).scrollTop();
				
				// Make sure they scroll more than delta
				if(Math.abs(lastScrollTop - st) <= delta) return;
				
				// If they scrolled down and are past the navbar, add class .nav-up.
				// This is necessary so you never see what is "behind" the navbar.
				if (st > lastScrollTop && st > navbarHeight){
						// Scroll Down
						$('header[role="banner"]').removeClass('header-down').addClass('header-up');
				} else {
						// Scroll Up
						if(st + $(window).height() < $(document).height()) {
								$('header[role="banner"]').removeClass('header-up').addClass('header-down');
						}
				}					
				lastScrollTop = st;
			}
		//

		// note close function
			$(".note").hover(
				function(){
					$(this).append('<i class="close ss-icon">close</i>');
					$(".close").click(function(){
						$(this).parent(".note").slideUp();
					});
					$("p .note .close, label.note .close").hide();
				},
				function(){
					$(".close").remove();
				}
			);
			$(".note.no-close").hover(
				function(){ $(".close").remove(); },
				function(){}
			);
		// 

		// campaign monitor ajax subscribe form
			if ($(".list-subscribe-form").length >= 1) {
				$(".list-subscribe-form").submit(function (e) {
					e.preventDefault();
					$.getJSON(

					$(this).serialize(),
					function (data) {
						if (data.Status === 400) {
							alert("Error: " + data.Message);
						} else { // 200
							// alert("Success: " + data.Message);
							$(".list-subscribe-form").hide();
							$(".list-subscribe-form").next(".note").fadeIn();
						}
					});

						scrollTop: 0
					}, 500);
					return false;
				});
			}
		//

		// mega menu
			$(".mega-menu, .basic-menu").hover(

					$(this).prev('a').addClass("mega-on");

					$(this).prev('a').removeClass("mega-on");

					$("li.our-work a").addClass("mega-on");
				}, function() {
					$("#our-work-mega-menu").hide();
					$("li.our-work a").removeClass("mega-on");
				}
			);

		// subnav
			if ($(".subnav").length === 1) {

						if ($(".subnav-close").length === 0) {
							$(".subnav").prepend('<a class="subnav-close">×</a>');
							$(".subnav-close, subnav-close:hover").css({
								"position": "absolute",
								"top": "12px",
								"right": "24px",

						$(".subnav-close").remove();

					// console.log(thisLocationPiece);

						$(this).addClass('active');
						$(this).parents('ul').siblings('a').addClass('active');
					}
					if ( $(this).hasClass('active') ) {
						if ( $(this).siblings('ul').hasClass('hidden') ) {

						}
						if ( $(this).parents('ul').hasClass('hidden') ) {
							$(this).parents('ul.hidden').show();
							$(this).parents('ul').siblings('a').addClass("shown");
						}
					}
				});
