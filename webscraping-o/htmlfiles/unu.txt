function pad(n, width, z) {
    z = z || '0';
    n = n + '';
    return n.length >= width ? n : new Array(width - n.length + 1).join(z) + n;
}

function renderDate(date) {
    let d = new Date(date);
    return d.getFullYear() + '•' + pad(d.getMonth()+1, 2) + '•' + pad(d.getDate(), 2);
}

function renderLandingView(articles) {
    let container = jQuery( ".home-articles-external" );
    container.html('');

    jQuery.each(articles,function(){
        let article = this;
        
        let titleWrapper = jQuery('<h4 class="fs-16  fw-light">');
        let titleLink = renderLink(article.link, article.title.rendered);
        let featuredImageURL = article._links["wp:featuredmedia"][0].href;
        fetchOurWorldMedia(featuredImageURL).done(function (data) {
            let thumbnail = renderImage(data);
            let imageLink = renderLink(article.link, thumbnail);
            titleWrapper.prepend(imageLink);
        });
        titleWrapper.append(titleLink);
        
        let info = jQuery('<p class="date mar-b-0 tt-u">');
        
        let topicLink = renderLink("https://ourworld.unu.edu", "Our World article");
        let pointer = jQuery('<h6 class="pointer mar-b-5">');
        pointer.append(topicLink);
        
        let articleContainer = jQuery('<div class="cell border-bottom">');
        articleContainer.append(pointer);
        articleContainer.append(info);
        articleContainer.append(titleWrapper);
        
        container.append(articleContainer);
    });
}

function renderLink(url, text) {
    let link = jQuery('<a target="_blank">');
    link.html(text);
    link.attr('href', url);
    
    return link
}

function renderImage(data) {
    var img = jQuery('<img>');

    return img.attr({
        'src': data.media_details.sizes.medium.source_url,
        'title': data.title.rendered || "",
        'alt': data.alt_text || ""
    });
}

function fetchOurWorldMedia(url) {
    return emit(url)
}

function fetchOurWorldArticles (lang="en", noOfArticles=3) {
    let articlesEndpoint = {
		"en": "https://ourworld.unu.edu/en/wp-json/wp/v2/articles",
		"ja": "https://ourworld.unu.edu/jp/wp-json/wp/v2/articles"
	};
	
	let params = jQuery.param({
	    categories: 1479,
        per_page: noOfArticles
    });

    let url = articlesEndpoint[lang] + "?" + params;
    return emit(url);
}

function emit(url) {
    return jQuery.ajax({
        url: url,
        dataType: 'jsonp',
        jsonp: '_jsonp',
    });
}
