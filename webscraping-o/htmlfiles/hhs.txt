		function showStickyNavSearch()
		{
			$('.hhs-blue-header .inner').append('<span class="sticky-nav-search hidden" aria-hidden="false"></span>');
			$('.sticky-nav-search').append('<a href="#" role="button"></a>');

					$('.hhs-global-search').attr('aria-hidden', true);
