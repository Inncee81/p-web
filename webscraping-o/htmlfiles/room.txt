		function addHover(){
			var currentEl = null

			for(var i = 0; i < underlineableEl.length; i++){
				underlineableEl[i].addEventListener("mouseenter", function(e){
					if((window.innerWidth >= mobileView)){
						catchMeEl.style.display = "initial"
						var spanEl = e.target.querySelector("span")
						var spanWidth = spanEl.offsetWidth
						var spanOffset = spanEl.getBoundingClientRect().left
						catchMeEl.style.width = spanWidth + "px"
						catchMeEl.style.transform  = "translateX(" + spanOffset + "px)"
					}
				})

				underlineableEl[i].addEventListener("mouseleave", function(e){
					if((window.innerWidth >= mobileView)){
						catchMeEl.style.display = "none"
					}
				})
			}

			//events for inside drawers
			for (var i = 0; i < drawers.length; i++){
				//for desktop hovers
				drawers[i].addEventListener("mouseenter", function(e){
					if((window.innerWidth >= mobileView)){
						catchMeEl.style.display = "initial"
						var spanEl = e.target.querySelector("span")
						var spanWidth = spanEl.offsetWidth
						var spanOffset = spanEl.getBoundingClientRect().left
						catchMeEl.style.width = spanWidth + "px"
						catchMeEl.style.transform  = "translateX(" + spanOffset + "px)"
					}
				})

				drawers[i].addEventListener("mouseleave", function(e){
					if((window.innerWidth >= mobileView)){
						catchMeEl.style.display = "none"
					}
				})

				link = drawers[i].querySelector(".nav-link")
				link.addEventListener("click", function(e){

					if((window.innerWidth < mobileView)){
						var self = this.parentElement
						if(currentEl === self){
							currentEl.classList.remove("opened")
							currentEl = null
						}
						else if(currentEl && currentEl !== self){
							currentEl.classList.remove("opened")
							self.classList.add("opened")
							currentEl = self
						}else{
							self.classList.add("opened")
							currentEl = self
						}
					}
					e.stopPropagation();
					if(document.body.className.indexOf('mobile') > -1) {
						event.preventDefault();
						return false;
					}
				//e.preventDefault();
				})
			}

			var count = 0;

			//events for global
			for (var i = 0; i < linksMain.length; i++){
				linksMain[i].addEventListener("mouseenter", function(e){
					count ++;
					bodyEl.classList.add("navigation-drawer-open")
					// backdropEl.classList.add("visible")
				})

				linksMain[i].addEventListener("mouseleave", function(e){
					setTimeout(function(){
						count--
						if (count > 0){
							count = 1
						} else {
							count = count < 0 ? 0 : count
							bodyEl.classList.remove("navigation-drawer-open")
						}
					})
				})
			}
		}

			window.addEventListener('resize',function(){
				if(window.innerWidth >= mobileView && bodyEl.classList.contains("navigation-open")){
					bodyEl.classList.remove("navigation-open")
				}
			})
