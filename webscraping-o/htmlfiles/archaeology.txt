	{

		});

		var current = $("#current").val();
		var total = $("#total").val();
		
		if (current == total) { var next = 1; }
		else { var next = parseInt(current) + 1; }
		 
		$(".v2_slide_img").each(function(){

			current_first = parseInt(current_first) + 1;
