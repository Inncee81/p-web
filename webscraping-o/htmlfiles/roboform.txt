function getCookie(cname) {
    var name = cname + "=";
    var ca = document.cookie.split(';');
    for(var i=0; i<ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') {
           c = c.substring(1);
        }
        if (c.indexOf(name) != -1) {
           return c.substring(name.length,c.length);
        }
    }
    return "";
}

  function checkBanner () {

    if( $("*").is("#banner") ) {
      $("#shc").css({"top":$("#banner").outerHeight()});
      if ( $("*").is("#tnavproductwrapper") ) {
        $("body").css({"padding-top": $("#banner").outerHeight() + $("#shc").outerHeight() + $("#tnavproductwrapper").outerHeight()  });
        $("#tnavproductwrapper").css({"top": $("#banner").outerHeight() + $("#shc").outerHeight() });
      } else {
          $("body").css({"padding-top": $("#banner").outerHeight() + $("#shc").outerHeight() });
      }

      $( ".close-banner" ).click(function() {
        $("#banner").remove();
        $("#shc").css({"top":"0"});
        if ( $("*").is("#tnavproductwrapper") ) {
          $("body").css({"padding-top": $("#shc").outerHeight() + $("#tnavproductwrapper").outerHeight() });
          $("#tnavproductwrapper").css({"top": $("#shc").outerHeight() });
        }  else {
          $("body").css({"padding-top": $("#shc").outerHeight() });
        }
      });

    } else {
        $("#shc").css({"top":"0"});
        if ( $("*").is("#tnavproductwrapper") ) {
          $("body").css({"padding-top": $("#shc").outerHeight() + $("#tnavproductwrapper").outerHeight() });
          $("#tnavproductwrapper").css({"top": $("#shc").outerHeight() });
        }  else {
          $("body").css({"padding-top": $("#shc").outerHeight() });
        }
    }

  }

window.addEventListener("orientationchange", function() {
  checkBanner();
}, false);
$(window).resize(function() {
  checkBanner();
});
