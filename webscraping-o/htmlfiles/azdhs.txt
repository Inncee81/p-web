  function buildAgeDropdown(array) {
    var select = document.getElementById("selectAgeGroup");
    var options = array;
    for(var i = 0; i < options.length; i++) {
        var opt = options[i];
        var el = document.createElement("option");
        el.textContent = opt;
        el.value = opt;
        select.appendChild(el);
