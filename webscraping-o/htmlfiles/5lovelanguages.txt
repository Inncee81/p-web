					"email": jQuery("#freebook_email").val().trim(),
					"first_name": jQuery("#freebook_fname").val().trim(),
					"last_name": jQuery("#freebook_lname").val().trim(),
					"privacy_policy": jQuery("#freebook_privacy").is(":checked") ? "Y" : "N",

					jQuery("#freeBookModal .alert p").text("Please fill in an email address.").closest(".alert").removeClass("hide");

						dataType: "JSON",
						processData: false,
						headers: {
							"Content-Type": "application/json"

							jQuery("#freeBookModal .alert p").text("There was an error while submitting your info. Please try again later.").closest(".alert").removeClass("hide");

				    return reg.test(jQuery(this).text());

				jQuery(".resource_search_item").addClass("hide");

					jQuery(filteredDivs[i]).removeClass("hide");

	function setCookie(cname, cvalue, exdays) {
	    var d = new Date();
	    d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
	    var expires = "expires="+d.toUTCString();
	    document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
	}
	
	function getCookie(cname) {
	    var name = cname + "=";
	    var ca = document.cookie.split(';');
	    for(var i = 0; i < ca.length; i++) {
	        var c = ca[i];
	        while (c.charAt(0) == ' ') {
	            c = c.substring(1);
	        }
	        if (c.indexOf(name) == 0) {
	            return c.substring(name.length, c.length);
	        }
	    }
	    return "";
	}
