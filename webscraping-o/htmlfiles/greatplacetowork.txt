       function gptwAspectRatio() {

            element.outerHeight(Math.ceil(element[0].getBoundingClientRect()['width']));

        jQuery('.fixed-aspect').each(function() {

            var element = jQuery(this);

            if (element.hasClass('aspect-xs')) {

                setSquare(element);

            } else if (element.hasClass('aspect-sm')) {

                if (jQuery(window).width() >= globalVariables.breakpoints.sm) {
                    setSquare(element);
                } else {
                    element.css('height', 'auto');
                }

                }

            }
