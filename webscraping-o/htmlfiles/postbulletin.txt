        if((isMatchDay&&!isDay)||(!isMatchDay&&(isDay||forceDay))) {

  function setVideo(playlist, player, e) {
      var currentVideo = $(e.target).closest('span').data('mediaid');
      var videoIndex = playlist.findIndex(function(video) {
          return currentVideo === video.mediaid;
      });
      e.preventDefault();
      player.playlistItem(videoIndex);

            var currentVideoSection = $(e.target).closest('span').data('mediasection');
      var currentVideoHref = $(e.target).closest('span').data('mediahref');
      var currentVideoTime = $(e.target).closest('span').data('mediatime');
      var currentVideoTitle = $(e.target).closest('span').data('mediatitle');

            document.getElementById('video-text').innerHTML = '<div class="section-time"><a href="' + currentVideoHref + '" class="link section-label allCaps">' + currentVideoSection + '</a> <time class="timestamp">' + currentVideoTime + '</time></div> <p class="video-title">' + currentVideoTitle + '</p>';

  function setActive(e) {
      var id = e.item.mediaid;
      $('.playlist-item-link').removeClass('active')
          .filter(function(item) {
              return $(this).data('mediaid') === id;
      })

                  }
              }
