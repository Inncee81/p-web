            function glsParseAspectRatio(arStr) {
                let aspectRatio = 6 / 4; // default
                if (arStr && typeof (arStr) == "string") {
                    try {
                        let arParts;
                        if (arStr.indexOf(":") >= 0) {
                            arParts = arStr.split(":");
                        } else if (arStr.indexOf("x") >= 0) {
                            arParts = arStr.split("x");
