  function setCookie(cname, cvalue, exdays) {
    var d = new Date();
    d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
    var expires = "expires="+d.toUTCString();
    document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
  }
  function getCookie(cname) {
    var name = cname + "=";
    var ca = document.cookie.split(';');
    for(var i = 0; i < ca.length; i++) {
      var c = ca[i];
      while (c.charAt(0) == ' ') {
        c = c.substring(1);
      }
      if (c.indexOf(name) == 0) {
        return c.substring(name.length, c.length);
      }
    }
    return "";
  }
  function check_user(extid) {
    var cookie_name = "user_api_"+extid;
    var user_cookie = getCookie(cookie_name);
    if (user_cookie == "") {
      // cookie is blank
      // perform API call
      var result = {};
      //var url = "/users/api?extid="+extid // dev
      //var url = "https://rewards.thegazette.com/users/api?extid="+extid // prod
      var url = "https://rewards.thegazette.com/users/api?extid="+extid // stage
      $.ajax(url).done(function(response,status){
        result = response;
        if(result.error=="true"){
          //don't save the cookie
          //console.log("error calling the user_api");
          setCookie(cookie_name,"{}",1);
          return {};
        } else {
          setCookie(cookie_name,JSON.stringify(result),7);
          //console.log("called api -- result: ")
          //console.log(result);
          //PUSH DATA UP TO PIANO
          updateUserCookieData(result);
          return result;
        }
      });
    } else {
      // cookie is not blank
      // read cookie value
      var user = JSON.parse(user_cookie);
      updateUserCookieData(user);
      return user;
    }
  }
  function clear_cookie(extid) {
    var cookie_name = "user_api_"+extid;
    setCookie(cookie_name,"",-1);
  }

  var currentIndex = array.length, temporaryValue, randomIndex;

    randomIndex = Math.floor(Math.random() * currentIndex);
