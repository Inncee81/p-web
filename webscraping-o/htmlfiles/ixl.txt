    clickables.forEach(function(clickable) {

          eventAction: clickable.getAttribute('data-ga-click-event-name'),
          eventLabel: clickable.getAttribute('data-ga-click-event-detail') || '',
          eventValue: Date.now() - interactionStart
        });
      });
    });

    // Now some events to mark whether users scroll to reach certain elements.
    if (window.IntersectionObserver && window.Map) {
      var scrollTargets = new Map();

      var lowestScrollElement = null;

      var observer = new IntersectionObserver(function(entries) {
        var timeDiff = Date.now() - interactionStart;
        entries.forEach(function(entry) {
          var domNode = entry.target;

          if (entry.isIntersecting) {

            // Note that 'domNode' is a node which has just started emerging on
            // the screen. At the moment, we don't want to worry about 'domNode';
            // instead, we want to mark that the element above it has been fully
            // scrolled through. 'config' represents that element above
            var config = scrollTargets.get(domNode).prev;

            // Make sure that this element is the new lowest element that we
            // have seen on the page
            if (!lowestScrollElement || config.index > lowestScrollElement.index) {
              config.timeDiff = timeDiff;

              // If it is the new lowest element, save it
              lowestScrollElement = config;
            }

            // We only care about the first time when this element becomes visible, so
            // stop observing it.
            observer.unobserve(domNode);

          }
        });
      });

      // Before the page closes, fire off an event to GA noting the lowest scroll
      // point on the page.
      window.onbeforeunload = function() {
        if (lowestScrollElement) {
          gaq('send', {
            hitType: 'event',
            eventCategory: 'Home Page Scroll Tracking',
            eventAction: lowestScrollElement.name,
            eventLabel: lowestScrollElement.detail || '',
            eventValue: lowestScrollElement.scrollCoordinate || 0
          });
        }
      };

      var runningIndex = 0;
      var lastScrollTargetAdded = { name: 'above fold', index: -1 };
      var parentChildCoordinateMap = {};
      var scrollTargetNodes = Array.prototype.slice.call(document.querySelectorAll('[data-ga-scroll-event-name]'));
      scrollTargetNodes.forEach(function(scrollTargetNode) {
        if (scrollTargetNode.offsetHeight > 0) {

          // Special feature: if a node has a 'data-ga-scroll-parent-container' annotation,
          // we try to augment its config object with some information about its relative
          // vertical position within this parent container.
          var parentClassName = scrollTargetNode.getAttribute('data-ga-scroll-parent-container');
          var parent = document.querySelector('.' + parentClassName);
          if (parent) {

            // If this parent node exists and has non-zero size...
            var parentBBox = parent.getBoundingClientRect();
            if (parentBBox.height > 0) {

              // List of y-coordinates which we have already seen for child scroll targets
              // within this parent. If we have already seen a scroll target at a certain
              // percentage, no need to add another.
              var usedChildCoordinates = parentChildCoordinateMap[parentClassName] || new Set();

              // Compute this child's coordinate, and save it if it is unique
              var coordinate;
              var pageY = scrollTargetNode.getBoundingClientRect().bottom;

              }
            }
