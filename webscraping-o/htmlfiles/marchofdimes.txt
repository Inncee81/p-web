                                undefined,undefined,undefined,
                                MODGLOBALS.HEADER.searchAutoSuggestNullContentCallback    );
    }; // MODGLOBALS.HEADER.searchOnKeyUp()

        $("#search-input #autoSuggestResults").show();
    };

        $("#search-input #autoSuggestResults").hide();
    };
