	var token = getParameterByName("verifyToken");

	function VerifyAndAdd(token) {

		var url = window.location.protocol + "//" + window.location.hostname + (window.location.port ? ':' + window.location.port : '') + "/scapi/PrefCenter/IsVerifiedToken";
		$(".verifyResponse .c-status-message--warning .c-marked-text__content").html("Verifying...");
		$(".verifyResponse .c-status-message--warning").show();
				
		$.ajax({
			url: url,
			data: { token: token},
			cache: false,
			type: "POST",
			timeout: 20000,
			success: function (result) {
				VerifyEmailCallback(result);
			},
			error: function (result, textStatus) {
				if (textStatus === "timeout") { // Accept email as valid and start Verification
					$(".verifyResponse .c-status-message--warning .c-marked-text__content").html("Oh no! Something went wrong. Please try again.");
				}
				else {
					VerifyEmailCallback(result);
				}
			}
		});
	}

	function VerifyEmailCallback(result) {
		result.ResponseMessage = JSON.parse(result.ResponseMessage);
		var statusCode = result.ResponseMessage.status > 0 ? result.ResponseMessage.status : result.StatusCode;

		$(".verifyResponse .c-status-message--success").hide();
		$(".verifyResponse .c-status-message--warning").hide();
		switch (statusCode) {
			case 3200:
				$(".verifyResponse .c-status-message--success .c-marked-text__content").html(result.ResponseMessage.email + " has been verified!");
				$(".verifyResponse .c-status-message--success").show();
				break;
			case 3404:
				$(".verifyResponse .c-status-message--warning .c-marked-text__content").html("Please check your inbox for the most recent email and click on the verification link to confirm your subscription.");
				$(".verifyResponse .c-status-message--warning").show();
				break;
			case 2201:
			case 3410:
				$(".verifyResponse .c-status-message--warning .c-marked-text__content").html("Oh no! Your link has expired. We sent you another. Please check your inbox.");
				$(".verifyResponse .c-status-message--warning").show();
				break;
			default:
			case 3417:
			case 3500:
				$(".verifyResponse .c-status-message--warning .c-marked-text__content").html("Oh no! Something went wrong. Please try again.");
				$(".verifyResponse .c-status-message--warning").show();
				break;
		}
	}

	function getParameterByName(name) {
		name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
		var regexS = "[\\?&]" + name + "=([^&#]*)";
		var regex = new RegExp(regexS);
		var results = regex.exec(window.location.href);
		if (results == null)
			return "";
		else
			return decodeURIComponent(results[1].replace(/\+/g, " "));
	}
