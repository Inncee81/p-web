        if ($(e.target).closest(".user-profile").length > 0 || $(e.target).closest(".fa-user").length > 0 ||
            $(e.target).closest(".notification_count").length > 0 || $(e.target).closest(".fa-bell").length > 0) {
            if ($(e.target).closest(".user-profile").length > 0 || $(e.target).closest(".fa-user").length > 0) {
                if ($(noti).hasClass('open')) {
                    $(noti).removeClass('open');
                }
            }
            if ($(e.target).closest(".notification_count").length > 0 || $(e.target).closest(".fa-bell").length > 0) {
                if ($(profi).hasClass('open')) {
                    $(profi).removeClass('open');
                }
            }
        } else {
            if ($(profi).hasClass('open')) {
                $(profi).removeClass('open');
            }
            if ($(noti).hasClass('open')) {
                $(noti).removeClass('open');
            }
        }
    }
