	var czgFunction = function () {
		this.islogin = false;
		this.isPartner = false;
		return this;
	};
	czgFunction.prototype = {
		init: function () {
			var t = this;
			t.getIsLogin();
			t.bindEvent();
			hxpctraffic("chuangzuoguanPC", {
				"TYPE": "新闻列表页",
				"PAGENAME": "和讯网首页",
				"EVENTNAME": "浏览事件"
			});

		getIsLogin: function () {
			var t = this;
			$.ajax({
				url: "https://regtool.hexun.com/wapreg/checklogin.aspx?format=json&encode=utf-8",
				type: "GET",
				dataType: "jsonp",
				success: function (data) {
					t.isLogin = data.islogin == "True" ? true : false;
					if (t.isLogin) {
						t.getIsPartner(data.userid);
					}
				}
			});
		},
		getIsPartner: function (id) {
			var t = this;
			$.ajax({
				url: "https://partnerwriter.hexun.com/api/partner/get_partnershow_info?partnerId="+id,
				type: "GET",
				dataType: "jsonp",
				success: function (data) {
					t.isPartner = data.resultKey == "ok" ? true : false;
					if (t.isPartner) {
						$("#toczgnews a").attr("href", "https://writer.hexun.com/#/wzedit");

$("#toczgvideo a").attr("href", "https://web.hexun.com/pc/admin/videoOfflineAdmin.html");
$("#toczgvideo a").attr("target", "_blank");

						$("#toczgadmin a").attr("href", "https://writer.hexun.com/");
						$("#toczgadmin a").attr("target", "_blank");

						$("#toczgnews a").attr("target", "_blank");
						$("#toczgvideo a").attr("href", "https://jinshuju.net/f/Q9c7Np");
						$("#toczgvideo a").attr("target", "_blank");
						$("#toczgadmin a").attr("href", "https://jinshuju.net/f/Q9c7Np");
						$("#toczgadmin a").attr("target", "_blank");
					}
				}
			});
		},
