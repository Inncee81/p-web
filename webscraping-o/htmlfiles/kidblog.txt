      function loadVideos() {
        var videoList = ['moderated-publishing', 'authentic-audience', 'digital-portfolio'];
        for (var i = 0; i < videoList.length; i++) {

            '.mp4" loop="" autoplay=""></video>');

            jQuery('img.screen--image.footage').css('display', 'none');
          }
        }

      function removePlaceholders() {
        jQuery('.screen--image.footage').css('display', 'block');
        jQuery('.video-placeholder').css('display', 'none');
      }

      // checks to see if all of the element is in the browser view
      function isInViewport(element) {
        var rect = element.getBoundingClientRect();
        var html = document.documentElement;
        return (
          rect.top >= 0 &&
          rect.left >= 0 &&
          rect.bottom <= (window.innerHeight || html.clientHeight) &&
          rect.right <= (window.innerWidth || html.clientWidth)
        );
      }

      // debounce function from https://davidwalsh.name/javascript-debounce-function
      function debounce(func, wait, immediate) {
        var timeout;
        return function() {
          var context = this, args = arguments;
          var later = function() {
            timeout = null;
            if (!immediate) func.apply(context, args);
          };
          var callNow = immediate && !timeout;
          clearTimeout(timeout);
          timeout = setTimeout(later, wait);
          if (callNow) func.apply(context, args);
        };
      };

      // checks to see if each video is in the browser to determine play/pause
      var checkEachVideoInViewport = debounce(function() {
        var videos = document.querySelectorAll('.video-check');

        videos.forEach(function (video) {
          var isInView = isInViewport(video);

          // if the play/pause button was pressed the video will not auto play/pause
          if (jQuery(video).find('.screen--image').attr('id') !== 'videoControlsOn') {
            if (isInView === true) {
              jQuery(video).find('.screen--image').get(0).play();
            } else {
              jQuery(video).find('.screen--image').get(0).pause();
            }
          }
        });
      }, 200);

      // checks to see if the video is a mp4
      function videoIsMp4() {
        var videos = document.querySelectorAll('.footage');
        var videoArray = [];

        // set videos as an array
        videos.forEach(function(video) {
          videoArray.push(video);
        })

        return videoArray.some(function(video) {
          return jQuery(video).attr('src').split('.').pop() == 'mp4';
        });
      }

      jQuery(document).ready(function(){
        loadVideos();

        window.onload = function() {
          removePlaceholders();

          if (videoIsMp4()) {
            // for autoplay videos on desktop
            jQuery('body').on('mousemove', checkEachVideoInViewport);

            // for autoplay videos on ipad
            jQuery('body').on('touchmove', checkEachVideoInViewport);
          }
        }

        jQuery('.video-control').click(function () {
          var video = jQuery(this).siblings('.video-check')

          // pause video
          if (video.find('.video-placeholder').css('display') == 'none') {
            video.find('.screen--image.footage').css('display', 'none');
            video.find('.video-placeholder').css('display', 'block');
            video.find('.screen--image.footage').attr('id', 'videoControlsOn');
            jQuery(this).find('.video-status').text('Play');

            if (videoIsMp4()) {
              video.find('.screen--image.footage').get(0).pause();
            }

          // play video
          } else {
            video.find('.screen--image.footage').css('display', 'block');
            video.find('.video-placeholder').css('display', 'none');
            jQuery(this).find('.video-status').text('Pause');

            if (videoIsMp4()) {
              video.find('.screen--image.footage').get(0).play();
            }
          }
        });
