  <script>
   var $ieModal = document.getElementById('ie-warning-modal');
  var isEditionMode = $ieModal.getAttribute('data-wcm-mode') === 'EDIT'; // aem info. TODO: Pass attr to body tag
  var browser = {
    isIe: function () {
      return navigator.appVersion.indexOf('MSIE') > -1 || navigator.appVersion.indexOf('Trident') > -1;
    },
    isEdge: function () {
      return navigator.appVersion.indexOf('Edge') > -1;
    },
    getVersion: function () {
      var version = 999; // we assume a sane browser
      if (navigator.appVersion.indexOf('MSIE') > -1) {
        // bah, IE again, lets downgrade version number
        version = parseFloat(navigator.appVersion.split('MSIE')[1]);
      } else if (navigator.appVersion.indexOf('Trident') > -1) {
        version = parseFloat(navigator.appVersion.split('rv:')[1]);
      }
      return version;
    }
  };
    
  /**
   * On window load
   */
  window.onload = function () {
    var $body = document.getElementsByTagName('body')[0];

    // Show warn if browser is IE and we are not on edition mode (aem)
    if ((browser.isIe() && browser.getVersion() <= 11) && !isEditionMode) {
      setBrowserWarningText();
      document.getElementById('ie-warning-modal').removeAttribute('style');
      $body.classList.add('modal--open');
    }

    // Add body class with browser info
    if (browser.isIe() || browser.isEdge()) {
      $body.classList.add('body--ie');
    }
  }
  
  /**
   * Add warning text on ie-warning-modal depending on the language 
   */
  function setBrowserWarningText() {
    var langISO = window.navigator.language || navigator.browserLanguage;
    var defaultLangCode = 'es';
    var langCode = langISO && langISO.split('-') ? langISO.split('-')[0] : defaultLangCode;
    var title = {
      es: 'Actualiza tu navegador'
    };
    var mainText = {
      es: 'Esta versión de tu navegador no permite visualizar correctamente la página. Para que tengas una buena experiencia y mejor seguridad, te aconsejamos la descarga de cualquiera de los siguientes navegadores:'
    };
    var closingText = {
      es: 'Muchas gracias'
    };

    document.getElementById('ie-warning-title').innerHTML = title[langCode] || title[defaultLangCode];
    document.getElementById('ie-warning-main-text').innerHTML = mainText[langCode] || mainText[defaultLangCode];
    document.getElementById('ie-warning-closing-text').innerHTML = closingText[langCode] || closingText[defaultLangCode];
  }
