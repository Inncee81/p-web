        var onAffiliateCookiesParsed = function(){
			/* Just in case the other affiliate script didn't load yet: Attempt to parse directly from the URL parameters */
			if(!mbsy_campaign_uid && typeof getURLParameter !== "undefined") {
				mbsy_campaign_uid = getURLParameter("campaignid");
			}

			if (!mbsy_short_code && typeof getURLParameter !== "undefined") {
				mbsy_short_code = getURLParameter("mbsy");
			}

			if (mbsy_campaign_uid && mbsy_short_code) {
				if (document.getElementById("mbsy_campaign_uid") != null) {
					/* Already done */
					return;
				}

				/* Append params to SSO links */
				var ssoLinks = document.getElementsByClassName("ssoLoginHref");
				for (var i = 0; i < ssoLinks.length; i++) {
					var link = ssoLinks.item(i);
					link.href = link.href + "&mbsy_campaign_uid=" + mbsy_campaign_uid + "&=mbsy_short_code" + mbsy_short_code;
				}

				/* And the main signup form */
				var forms = document.getElementsByClassName("signupFormForm");
				for (var i = 0; i < forms.length; i++) {
					var form = forms.item(i);
					var input = document.createElement("input");
					input.type = "hidden";
					input.id = input.name = "mbsy_campaign_uid";
					input.value = mbsy_campaign_uid;
					form.appendChild(input);

					input = document.createElement("input");
					input.type = "hidden";
					input.id = input.name = "mbsy_short_code";
					input.value = mbsy_short_code;
					form.appendChild(input);
				}
			}
		};

		var timeout = window.location.href.indexOf("mbsy") > 0 ? 1000 : 100;
		setTimeout(function () {
			/* Pretty! We need to the delay the loading of this script, to get the other script time to load and to set the cookie, for the cases where you land directly on a page with a signup form */
			var script = document.createElement('script');
			script.onload = function () {
				onAffiliateCookiesParsed()
			};
			script.src = "https://mbsy.co/embed/v2/getcookie/wevideo";
			document.getElementsByTagName('head')[0].appendChild(script);
		}, timeout);

function changeLocalisationLanguage(elem) {
    let currentLocation = window.location.pathname;
    let newLocation = currentLocation.replace('', '');
    let selectedLang = jQuery(elem).val();
    if (selectedLang === 'en') {
        newLocation = newLocation + '?language=en';
    } else {
        selectedLang = selectedLang.replace('/', '');
        newLocation = '/' + selectedLang + newLocation + '?language=' + selectedLang;
    }
    // if a language change occurs, we're storing the preferred language in session storage
    sessionStorage.setItem('selectedLang', JSON.stringify({selectedLang: selectedLang}));
    window.location.href = newLocation;
}

function logoutUser() {
    let logoutUrl = '/login?action=logout&marketplaceId=16409042';
    let selectedLang = sessionStorage.getItem('selectedLang');
    if (typeof selectedLang !== 'undefined' && selectedLang !== null) {
        selectedLang = JSON.parse(selectedLang);
        if (selectedLang.hasOwnProperty('selectedLang')) {
            logoutUrl = logoutUrl + '&returnLogOutURL=/?language=' + selectedLang.selectedLang;
        }
    }

    window.location.href = logoutUrl;
    return false;
}
