function transXooit(othis)
{
	o = othis;
	while (o)
	{
		if (o.nodeName == 'SCRIPT' ||
				o.nodeName == 'A' ||
				o.nodeName == 'BR' ||
				o.nodeName == 'TEXTAREA' ||
				(o.nodeName == 'SPAN' && jQuery(o).hasClass('noxooitcolor')))
			return false;
		o = o.parentNode;
	}
	
	
	var data = 'xooit.com';
	var i;
	var txtO = othis.nodeValue;
	if ((i = txtO.toLowerCase().indexOf(data)) != -1)
