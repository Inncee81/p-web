var Search = {
		searchTypeCookieName: "searchType",
        init: function(){
			var parent = this;
			{
				var searchType = $.cookie(parent.searchTypeCookieName);
				if (searchType !== null) {
					$("#searchType").text(decodeURIComponent(searchType));
				} else {
					$("#searchType").text("網頁");
				}
				$('#searchBar').click(function() {
					//$('#searchType').hide();
					if ($('#search_select').css('display') == 'none') {
						$("#searchType").text("網頁");
						$.cookie(parent.searchTypeCookieName, encodeURIComponent("網頁"), {expires: 365});
						$('#search_select').show();
					} else {
						$('#search_select').hide();
					}
					return false;
				});
				$('#search_select li a').click(function() {
					$("#searchType").text($(this).text());
					$.cookie(parent.searchTypeCookieName, encodeURIComponent($(this).text()), {expires: 365});
					$('#search_select').hide();
					return false;
				});
			}
        },
		searchByType: function() {
			var k = $('#k').val();
			this.doSearch(k, $("#searchType").text());
		},
		searchService: function(id) {
			var k = $('#'+id).val();
			if ('本站搜尋' === k) {
				alert('請輸入搜尋關鍵字');
				return;
			}
			this.doSearch(k, '本站');
		},
		doSearch: function(k, searchType){
			//var k = $("#k").val();
			var ke = encodeURIComponent(k);
			var url;
			switch (searchType) {
				case "網頁":
					url = Config.hisearchUrl+"?k=" + ke + "&t=wps";
					url = Redir.getUrl('searchWeb' , url);
					break;
				case "圖片":
					url = Config.hisearchUrl+"?k=" + ke + "&t=images";
					url = Redir.getUrl('searchImg' , url);
					break;
				case "新聞":
					url = "http://times.hinet.net/search?keyword="+ ke;
					url = Redir.getUrl('searchNews' , url);
					break;
				case "影片":
					url = Config.hisearchUrl+"?k=" + ke + "&t=video";
					url = Redir.getUrl('searchVideo' , url);
					break;
				case "店家":
					url = "https://www.iyp.com.tw/search.php?a_id=0&k=" + ke;
					url = Redir.getUrl('searchStore' , url);
					break;
				case "購物":
					url = "http://mall.cheerspoint.com.tw/quidditch/cheerspoint?keywords=" + ke;
					url = Redir.getUrl('searchShopping' , url);
					break;
				case "本站":
					if (k !== '') {
						url = "http://search.cht.com.tw/Search?DataSource=Hinet_Main&searchTemplate=18&Keyword="+ ke;
						url = Redir.getUrl('searchService' , url);
					} else {
						url = Config.hisearchUrl;
						url = Redir.getUrl('searchService' , url);
					}
					break;
			}
			window.open(url);
		}
};

function getAllcookies() {
	var cookies = {};
	if (document.cookie && document.cookie != '') {
		var split = document.cookie.split(';');
		for (var i = 0; i < split.length; i++) {
			var name_value = split[i].split("=");
			name_value[0] = name_value[0].replace(/^ /, '');
			if(cookies.length == undefined){
				cookies =name_value[0]+"_"+name_value[1];
			}else{
				cookies = cookies+","+name_value[0]+"_"+name_value[1];
			}
		}
	}
	return cookies;
}

function setBlock(){
	var $block = $('.s_list_box'),
	$li = $block.find('li'),
	$img = $li.find('img'),
	_width = $img.width(),
	_height = $img.height(),
	_ratio = 1.2,
	_speed = 200;
	$li.hover(function(){
		var $this = $(this);
		$this.css('z-index', 1).find('.s_hover').stop().animate({
			top: -3
		}, _speed).find('img').attr('src',function(i,e){
				return e.replace("_o","_h");
		}).stop().animate({
			width: _width * _ratio,
			height: _height * _ratio
		}, _speed);
	}, function(){
		var $this = $(this);
		$this.css('z-index', 0).find('.s_hover').stop().animate({
			top: 0
		}, _speed).find('img').stop().animate({
			width: _width,
			height: _height
		}, _speed).attr('src',function(i,e){
				return e.replace("_h","_o");
		});
	});
}

function escapeHtml(unsafe) {
    return unsafe
         .replace(/&/g, "&amp;")
         .replace(/</g, "&lt;")
         .replace(/>/g, "&gt;")
         .replace(/"/g, "&quot;")
         .replace(/'/g, "&#039;")
		 .replace(/ /g, "&nbsp;");
}
