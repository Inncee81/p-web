 <script>
  /*seems redundant to have both css transition and jquery fadeIn() but it works?*/
    $('.menutxt').on('click', function() {
        //$('.submenu').fadeIn(100);
        $('.submenu').show();
        $('.navback').fadeToggle();
        $('#menu-toggle').toggleClass('activ');
        $('#menu-toggle2').toggleClass('activ');
        $('#menu-toggle3').toggleClass('activ');
        $('.submenu').toggleClass('invis');
        $('.cntr').toggleClass('hidden');
        $('#header').toggleClass('headerred');
        $('.header-sticky').toggleClass('headerred');
        $('.header-sticky-border').toggleClass('header-sticky-border-red');
        $('.headerlogo').toggleClass('hidden');
        $('.headerlogo-white').toggleClass('hidden');
        $('.headerlogo-mob').toggleClass('hidden');
        $('.headerlogo-mob-white').toggleClass('hidden');
        $('.menutxt').toggleClass('headerwhite');
        $('.toplinks-sticky').toggleClass('headerwhite');
        $('.movementred-apply').toggleClass('headerwhite');
        $('.languageflip').toggleClass('languageflip-red');
    });

        /*close the submenu when scrolling*/
    $(window).scroll(function(event){
        $('#menu-toggle').removeClass('activ');
        $('#menu-toggle2').removeClass('activ');
        $('#menu-toggle3').removeClass('activ');
        $('.submenu').addClass('invis');
        $('.navback').hide();
        $('.cntr').addClass('hidden');
        $('#header').removeClass('headerred');
        $('.header-sticky').removeClass('headerred');
        $('.header-sticky-border').removeClass('header-sticky-border-red');
        $('.menutxt').removeClass('headerwhite');
        $('.toplinks-sticky').removeClass('headerwhite');
        $('.movementred-apply').removeClass('headerwhite');
        $('.headerlogo').removeClass('hidden');
        $('.headerlogo-white').addClass('hidden');
        $('.headerlogo-mob').removeClass('hidden');
        $('.headerlogo-mob-white').addClass('hidden');
        $('.languageflip').removeClass('languageflip-red');
    });

    $(document).ready(function() {
    //static <-> sticky nav bar switchoff
        $(window).bind('scroll', function() {
            if ($(window).scrollTop() > 18) {
                $('.cntr-sticky').css("top","61px");
                $('#header').addClass('header-sticky-desk');
                $('#header-inner').addClass('header-inner-sticky');
                $('#top').addClass('top-sticky');
                $('.cntr').addClass('cntr-sticky');
                $(".headerlogo").attr("src","https://mvmtweb.imgix.net/movement.com/layout/mmlogo-logo.png");
                $(".headerlogo-white").attr("src","https://mvmtweb.imgix.net/movement.com/layout/mm_logowhite_icon.png");
            }
            else {
                    var datnumber = parseInt(70) - parseInt(($(window).scrollTop()));
                    //alert(datnumber);
                    $('.cntr-mobsticky').css("top",""+datnumber+"px");
                $('#header').removeClass('header-sticky-desk');
                $('#header-inner').removeClass('header-inner-sticky');
                //$('#toplinks').removeClass('toplinks-sticky');
                $('#top').removeClass('top-sticky');
                $('.cntr').removeClass('cntr-sticky');
                $(".headerlogo").attr("src","https://mvmtweb.imgix.net/movement.com/layout/mmlogo.png");
                $(".headerlogo-white").attr("src","https://mvmtweb.imgix.net/movement.com/layout/mm_logowhite.png");
            }
        });
    });
 </script>
