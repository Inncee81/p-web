	$(document).ready(function() {
			var is_m_Slider = false,
				m_width = $(window).width();

			$(window).resize(function() {
				if($(window).width() != m_width) {
					m_width = $(window).width();
					setMobileSliderWidth()
					setTimeout(function(){
						autoMobileSlider()
					},600);
				}
			});

			var setMobileSliderWidth = function() {
				m_tpl_cnt = $('.m.sp-content .m.sp-slider').children().length+1;
				$('#m_template').find('.m.sp-slider').css('width', (m_tpl_cnt+7) * $('.m.sp-slider li').width());
			}

			setTimeout(function(){
				$('#m_template').find('.m.sp-content .m.sp-slider').addClass('start-effect').addClass('sp-left-effect');
			},500);

			$(window).scroll(function(){
				var m_wscroll = $(window).scrollTop()+60,
					m_wWidth = $(window).width(),
					m_sScrollTop = $('#m_template').offset().top,
					m_sHieght = $('#m_template').height();

					if(m_wscroll > 360 && m_wscroll < m_sScrollTop + m_sHieght) {
						
						var m_scale_v = 1.1 +((m_sScrollTop-m_wscroll)/1000);
							m_scale_v = (m_scale_v < 1.1 || m_wscroll+40 > m_sScrollTop) ? 1 : m_scale_v;

						var m_scale_r = 'matrix('+m_scale_v+',0,0,'+m_scale_v+',0,0)';

							if(m_scale_v) {
								is_m_Slider = true;
								var is_m_Reset = ($('#m_template').find('.m.sp-content .m.sp-slider').css('opacity') == 0) ? true : false;
								if(is_m_Reset) $('#m_template').find('.m.sp-content .m.sp-slider li').eq(8).find('img').click(); 
								$('#m_template').find('.m.sp-content .m.sp-slider').css({'opacity':'1'});
								$('#m_template').find('.m.sp-content .m.sp-slider').removeClass('sp-left-effect2').addClass('sp-left-effect');
								$('#m_template').find('.m.sp-content .m.sp-slider').removeClass('start-effect');
							} else {
								is_m_Slider = false;
								$('#m_template').find('.m.sp-content .m.sp-slider').removeClass('sp-left-effect').addClass('sp-left-effect2');
								$('#m_template').find('.m.sp-content .m.sp-slider').css({'opacity':'0'});
								$('#m_template').find('.m.sp-content .m.sp-slider').css('left',m_wWidth);
							}
					}
			});

			$('#m_template .m.sp-slider li img').click( function (e) {
				getMobileMoveSlider($(this).closest('li').attr('data-name'));
				clearInterval(autoMobilePlay);
				setTimeout(function() {
					autoMobilePlay = setInterval(function() {
						autoMobileSlider()
					},4000);
				});			
			});

			$('#m_template .m.sp-slider li').live({
				mouseenter: function(){
					if($('#m_template .m.sp-slider').find('.m-over-box').length> 0 ) $('#m_template .m.sp-slider').find('.m-over-box').remove();
					if(!$(this).hasClass('active')) return false;
					var m_src = (is_m_Slider) ? "http://" + $(this).data('sid') + ".creatorlink.net/render" : "javascript:;",
						m_target = (is_m_Slider) ? "_blank" : "";
					$(this).css('cursor','pointer');
					$(this).append('<div class="m-over-box"><a href="'+ m_src + '" target="'+m_target+'">VIEW MORE</a></div>');
				},
				mouseleave: function(){
					$('#m_template .m.sp-slider').find('.m-over-box').remove();
				}
			});

			$('#m_template .m.slider-control').click(function (e) {
				var m_show_tpl = $('#m_template').find('.m.sp-content .m.sp-slider li.active');
				if($(this).attr('data-which') == 'right') {	
					getMobileMoveSlider(m_show_tpl.next().attr('data-name'));
				} else if($(this).attr('data-which') == 'left') {
					getMobileMoveSlider(m_show_tpl.prev().attr('data-name'));
				}
				clearInterval(autoMobilePlay);
				setTimeout(function() {
					autoMobilePlay = setInterval(function() {
						autoMobileSlider()
					},4000);
				});
			});

			var autoMobilePlay = setInterval(function() {
				autoMobileSlider()
			},4000);

			var autoMobileSlider = function() {
				if(is_m_Slider) {
					if($('#m_template').find('.m.sp-content .m.sp-slider li.active').next().length == 0) {
						$('#m_template').find('.m.sp-content .m.sp-slider li').eq(0).find('img').click();
					} else {
						$('#m_template').find('.m.sp-content .m.sp-slider li.active').next().find('img').click();
					}
				}
			} 

			var getMobileMoveSlider = function(tplname) {
				var m_wCenter = $(window).width()/2,
					m_slider = $('#m_template .m.sp-content .m.sp-slider'),
					m_slider_left = m_slider.offset().left,
					m_iCenter = m_slider.find('li[data-name="'+tplname+'"] img').offset().left + (m_slider.find('li[data-name="'+tplname+'"] img').width()/2);

				if($('#m_template .m.sp-slider').find('.m-over-box').length>0) $('#m_template .m.sp-slider').find('.m-over-box').fadeOut().remove();

				m_slider.find('li').removeClass('active');
				m_slider.find('li[data-name="'+tplname+'"]').addClass('active');

				if(m_slider.find('li').eq(0).hasClass('active')) {
					$('#m_template').find('.m.slider-control.left').addClass('hide');
					$('#m_template').find('.m.slider-control.right').removeClass('hide');

				} else if (m_slider.find('li').last().hasClass('active')) {
					$('#m_template').find('.m.slider-control.right').addClass('hide');
					$('#m_template').find('.m.slider-control.left').removeClass('hide');

				} else {
					$('#m_template').find('.m.slider-control').removeClass('hide');
				}

				$('#m_template').find('.m.sp-content .m.sp-slider').css({'left':m_slider_left + (m_wCenter-m_iCenter)});
			}
	});

			if($(window).width() > 992 && $(window).width() < 1200) {
				$('#responsive .sp-content .sp-slider').css('width', (tpl_cnt+16) * 430);

			if(wscroll > 360 && wscroll < sScrollTop+sHeight) {

				var scale_r = 'matrix('+scale_v+',0,0,'+scale_v+',0,0)',
					m_width = (wWidth<1200) ? '-22' : '-22';

			$('#goto-top-m').fadeIn(200);
		} else {
			$('#goto-top-m').fadeOut(200);
		}
    });
    
	// Animate the scroll to top
	$('#goto-top-m').click(function(event) {
		event.preventDefault();
		$('body,html').animate({scrollTop: 0}, 400);
	});
