      $(function () {

		var check_phone_number = function (value, inputSelector, countryDropdown, countrySelectedSelector) {
			if (countryDropdown.length) {
				return intlTelInputUtils.isValidNumber(value, $(countrySelectedSelector).data('iso2'));
			} else if (inputSelector) {
				return iti[inputSelector].isValidNumber();
			}

			return false;
		};

		var iti = []

		// init plugin
		window.initTelPlugin = function (inputSelector, countrySelector, countrySelectedSelector, isAjax = false) {
			if (iti[inputSelector] !== undefined) {
				return
			}

			let countryDropdown = $(countrySelector);
			let input = $(inputSelector);

			let hiddenInputName = null;
			if (!isAjax) {
				// Hidden input used instead
				hiddenInputName = input.attr('name');
				input.attr('name', '');
			}

			$(inputSelector).on('blur', function (e) {
				if ($(this).val() &&
					 !check_phone_number($(this).val(), inputSelector, countryDropdown, countrySelectedSelector)) {
					$(this).addClass('error');
				} else {
					$(this).removeClass('error');
				}
			});

			required_js.add_callback('check-phone-number', function (value) {
				let response = {valid: true, text: "is not valid"};

				if (value) {
					response.valid = check_phone_number(value, inputSelector, countryDropdown, countrySelectedSelector);
				}

				return response;
			});

			iti[inputSelector] = window.intlTelInput(document.querySelector(inputSelector), {
				initialCountry: countryDropdown.find('option:selected').data('iso2') ?
					countryDropdown.find('option:selected').data('iso2') : 'us',
				utilsScript: "https://cdn.newswire.com/assets/lib/intl-tel-input/build/js/utils.js", // just for formatting/placeholders etc
				separateDialCode: true,
				hiddenInput: hiddenInputName,
			});

			// set it's initial value
			setSelectedCountry(countryDropdown, inputSelector);

			// listen to the telephone input for changes
			input[0].addEventListener('countrychange', function (e) {
				setSelectedCountry(countryDropdown, inputSelector);
			});

			// listen to the address dropdown for changes
			countryDropdown.on('change', function () {
				let iso2;

				if ($(this).find('option:selected').length) {
					iso2 = $(this).find('option:selected').data('iso2')
				} else {
					iso2 = $(this).data('iso2')
				}

				if (iso2) {
					iti[inputSelector].setCountry(iso2);
				}
			});

		}

		
		function setSelectedCountry(countryDropdown, inputSelector) {
			let options = countryDropdown.find('option[data-iso2]');

			if (options.length) {
				options.removeAttr('selected');
				options.prop('selected', false);
				countryDropdown.find('option[data-iso2="' + iti[inputSelector].getSelectedCountryData().iso2 + '"]').prop('selected', true);
			} else {
				countryDropdown.data('iso2', iti[inputSelector].getSelectedCountryData().iso2);
			}

			countryDropdown.trigger('change');
			if (countryDropdown.parent().hasClass('bootstrap-select')) {
				countryDropdown.selectpicker('refresh');
			}

			let mask = $(inputSelector).attr('placeholder').replace(/[0-9]/g, 9);
			let im = new Inputmask({mask: mask, greedy: false, showMaskOnHover: false, jitMasking: true});
			im.mask($(inputSelector));
		}

	});

     <script>
      $(function () {

		var $location_input = $("#capture_location_pr_guide"),
			$location_id_input = $("#capture_location_pr_guide_id"),
			allow_custom_value = false,
			value_field = "name",
			location_query = '',
			location_id = '',
			location_name = ""		;

		if ($location_input.length) {

			$location_input.attr('autocomplete', 'off');

			$location_input.autoComplete({
				minLength: 2,
				autoSelect: false,
				resolverSettings: {
					url: '/common/location/find'
				},
				formatResult: function (item) {
					if (typeof item === 'undefined' || item === null) {
						return {
							id: null,
							text: null
						};
					}

					let html_value = '',
						html_extras = [];

					if (item.hasOwnProperty('html')){
						html_value = item.html;
					} else {
						html_value = item.name;

						if (item.hasOwnProperty('subdivision_name') && item.subdivision_name) {
							html_extras.push(item.subdivision_name);
						}

						if (item.hasOwnProperty('country_name') && item.country_name) {
							html_extras.push(item.country_name);
						}

						if (html_extras.length) {
							html_value += ' <small class="text-muted">('
								+ html_extras.join(', ')
								+ ')</small>';
						}

						if (item.hasOwnProperty('is_common') && item.is_common)
							html_value = '<strong>' + html_value + '</strong>';
					}

					return {
						id: item.id,
						text: item[value_field],
						html: html_value,
					};
				},
				events: {
					searchPre: function (typed_value) {
						return location_query = typed_value;
					},
					searchPost: function (result, elem) {
						if (allow_custom_value) {
							let obj = {
								id: null,
								html: '<em>Use custom location "<strong>' + location_query + '</strong>"</em>',
							};
							obj[value_field] = location_query;
							result.push(obj);
						}

						// Fix the library issue with links in the list
						setTimeout(function () {
							let dropdown = elem.next('.dropdown-menu');
							dropdown.find('a').on('click', function (event) {
								event.preventDefault();
							});
						}, 100);

						return result;
					},
				},
			});

			$location_input.on('autocomplete.select', function (event, item) {
				// check if item exists because library does not filter.
				if (typeof item === 'undefined' || item === null) {
					return;
				}

				$location_id_input.data('iso2', item.iso2);
				$location_input.val(item[value_field]);
				$location_id_input.val(item.id).trigger('change');
				location_id = item.id;
				location_name = item[value_field];
			});

			$location_input.on('change', function (event) {
				// Reverse the last change if the user didn't select an option
				setTimeout(function () {
					if (location_name.length) {
						$location_input.val(location_name);
					}
				}, 100);
			});

			$location_input.on('keydown', function (event) {
				switch (event.keyCode) {
					case 13:
					case 16:
					case 18:
						return false;
					default:
						break;
				}
			});

		}

	});
     </script>

      $(function() {

						}
					},
				});
