            <script>
             let timer = 0; // Registered globally to be able to clear it later
    let currentIndex = 0;
    const scrollInterval = 3000;

    // FUNCTIONS CALLED VIA HTML

    function goToLeft() {
        let nextIndex = currentIndex - 1;
        if (nextIndex < -1) { // We use -1 because we need to jump to the very right before smoothly animating left
            let elements = document.getElementsByClassName("dn-block-slider__paging-indicator");
            nextIndex = elements.length - 1;
        }
        scrollToSlide(nextIndex);
    }

    goToRight = function(automated) { // Different style to call this from setTimer() and setIntervall()
        let nextIndex = currentIndex + 1;
        let elements = document.getElementsByClassName("dn-block-slider__paging-indicator");
        if (nextIndex >= elements.length + 1) {
            nextIndex = 0;
        }
        scrollToSlide(nextIndex, automated);
    };

    // CORE FUNCTIONS

    function scrollToSlide(index, automated) {
        setPagingIndicators(index);
        scrollToIndex(index);
        pauseVideos();
        playVideo(index);
        if (!automated) {
            stopAutoScroll();
        }

        return false; // https://stackoverflow.com/questions/128923/whats-the-effect-of-adding-return-false-to-a-click-event-listener
    }

    // SCROLL EVERY 3 SECONDS

    startAutoScroll = function () {
        timer = setInterval(function () {
            goToRight(true);

        clearInterval(timer);

    // HELPER FUNCTIONS

    function playVideo(index) {

        let slide, element;
        slide = document.getElementsByClassName("dn-block-slider__slide")[index];
        if (!slide) {
            return;
        }
        element = slide.firstChild;
        if (element.tagName && element.tagName === "VIDEO") {
            element.play();
        }
    }

    function pauseVideos() {

        let videos = document.getElementsByTagName("video");
        for (let video of videos) {
            video.pause()
        }
    }

    function transform(stage, index) {
        const widthOfSlide = 0.67; //67%
        const offsetOfSlide = (1 - widthOfSlide) / 2;
        const marginRightOfSlide = 10;
        return "translateX(-" + Math.floor(((stage.offsetWidth * widthOfSlide + marginRightOfSlide) * (index + 1)) - stage.offsetWidth * offsetOfSlide) + "px)";
    }

    function scrollToIndex(index) {

        let stages = document.getElementsByClassName("dn-block-slider__stage");
        for (let stage of stages) {
            stage.style.transform = transform(stage, index);

            // Make animation smooth
            if (index != 0) {
                stage.style.transition = "transform 1000ms ease-in-out";
            }

            //After scrolling from the last slide to the "first" and vice versa, we need to jump to the first slide again to create our infinite loop
            let slides = document.getElementsByClassName("dn-block-slider__thumbnail");

            // Scroll from the last to the "first" (which is a clone) slide.
            if (index == slides.length) {

                setPagingIndicators(0);
                setTimeout(function () {
                    let stages = document.getElementsByClassName("dn-block-slider__stage");
                    for (let stage of stages) {
                        stage.style.transform = transform(stage, 0);
                        stage.style.transition = "";
                    }
                }, 1000);

            } else if (index == -1) { // Scroll from the last to the "first" (which is a clone) slide.

                // Jump
                currentIndex = slides.length;
                stage.style.transform = transform(stage, currentIndex);
                stage.style.transition = "";

                // Animate smooth
                setPagingIndicators(slides.length - 1);
                stage.style.transform = transform(stage, currentIndex);
                stage.style.transition = "transform 1000ms ease-in-out";
            }
        }
    }
    // Scroll to first slide
    scrollToIndex(0);

    function setPagingIndicators(index) {

        console.log("setPagingIndicators index: " + index);
        currentIndex = index;
        // Remove old .active on paging indicator
        let pagingIndicators = document.getElementsByClassName("dn-block-slider__paging-indicator--active");
        for (let pagingIndicator of pagingIndicators) {
            pagingIndicator.classList.remove("dn-block-slider__paging-indicator--active")
        }

        // Set .active on paging indicator
        let pagingIndicator = document.getElementsByClassName("dn-block-slider__paging-indicator")[index];
        if (pagingIndicator) {
            pagingIndicator.classList.add("dn-block-slider__paging-indicator--active");
        }
    }
