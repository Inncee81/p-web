    /*                          Constants                             */

    // Global days until the cookie consent decision expires
    window.EWSCookieConsentDurationInDays = 365;

    // Global name of the consent cookie
    window.EWSCookieConsentName = "EWSCookieConsent";

    // Global website ID to retrieve the cookie list
    window.EWSCookieWebsiteID = "7ef9413a-7b58-46f6-bca5-4d7cddf6a5fe";

    // Global cookie consent value to deny all cookies
    window.EWSCookieConsentDenyAllValue = "denyAll";

    // Global cookie consent value to allow all cookies
    window.EWSCookieConsentAllowAllValue = "allowAll";

    // Global cookie consent value to allow all cookies
    window.EWSCookieConsentCustomValue = "custom";

    /* -------------------------------------------------------------- */
    /*                      Global Variables                          */
    /* -------------------------------------------------------------- */

    // Global queue to store cookies until consent decision was made
    window.EWSCookieConsentQueue = [];

    // Global queue to store cookies until cookie list is loaded
    window.EWSCookieLoadingQueue = [];

    // Global dictionary of cookie categories and the cookies in them
    window.EWSCookieCategories = null;

    /*                     Convenience Functions                      */

    // Get the name of a cookie from the document.cookie "cookie set" call
    function EWSGetCookieName(cookie) {
        return cookie.split("=")[0];
    }

    // Get the value of a cookie
    function EWSGetCookie(cname) {
        var name = cname + "=";
        var decodedCookie = decodeURIComponent(document.cookie);
        var ca = decodedCookie.split(';');
        for(var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') {
                c = c.substring(1);
            }
            if (c.indexOf(name) == 0) {
                return c.substring(name.length, c.length);
            }
        }
        return null;
    }

    // Set a cookie
    function EWSSetCookie(cname, cvalue, exdays) {
        var d = new Date();
        d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
        var expires = "expires=" + d.toUTCString();
        var setCookieString = cname + "=" + cvalue + ";" + expires + ";path=/;SameSite=Lax";
        if(location.protocol === 'https:') {
            setCookieString += ";secure";
        }
        document.cookie = setCookieString;
    }

    /* -------------------------------------------------------------- */
    /*                        Main Functions                          */
    /* -------------------------------------------------------------- */

    // Replace document.cookie
    function EWSProxyDocumentCookie() {
        // Proxy document.cookie to delay/prevent settings of cookies
        var cookieDesc = Object.getOwnPropertyDescriptor(Document.prototype, 'cookie');
        if (cookieDesc && cookieDesc.configurable) {
            Object.defineProperty(document, 'cookie', {
                get: function () {
                    return cookieDesc.get.call(document);
                },
                set: function (val) {
                    // Queue cookie request until cookie list is fetched
                    if(window.EWSCookieCategories === null) {
                        window.EWSCookieLoadingQueue.push(val);
                        return;
                    }

                    // Check if we are allowed to set the cookie
                    if(EWSCheckCookieForPermission(val)) {
                        // If so set the cookie
                        cookieDesc.set.call(document, val);
                    }
                }
            });
        }
    }

    // Check if a cookie is allowed to be set
    function EWSCheckCookieForPermission(cookieToSet) {
        // Extract the cookie name
        let cookieNameToSet = EWSGetCookieName(cookieToSet);

        // Iterate over all categories
        for(var categoryID in window.EWSCookieCategories) {
            // Iterate over all cookies
            for(var i = 0; i < window.EWSCookieCategories[categoryID]["cookies"].length; i++) {
                // Get the cookie name
                let cookieName = window.EWSCookieCategories[categoryID]["cookies"][i];

                // Check if the cookie to be set matches the cookie in this category
                if(EWSCompareCookieNames(cookieNameToSet, cookieName)) {
                    // Check if this cookie is part of a blocked category
                    if(window.EWSCookieCategories[categoryID].blocked === true) {
                        return false;
                    }

                    // Check if the user made already a cookie consent decision
                    if(EWSGetCookie(window.EWSCookieConsentName) !== null) {
                        // The user made aleady a cookie consent decision
                        if(EWSGetCookie(window.EWSCookieConsentName) === window.EWSCookieConsentAllowAllValue) {
                            // The user agreed to all cookies
                            return true;
                        }
                        else if(EWSGetCookie(window.EWSCookieConsentName) === window.EWSCookieConsentDenyAllValue) {
                            // The user denied all cookies
                            if(window.EWSCookieCategories[categoryID].required === true) {
                                // Set it anyways if it is required
                                return true;
                            }
                            else {
                                // Do not set cookie since the user denied all cookies
                                return false;
                            }
                        }
                        else {
                            // Return custom cookie policy
                            let cookiePolicy = JSON.parse(EWSGetCookie(window.EWSCookieConsentName));
                            return cookiePolicy[categoryID];
                        }
                    }
                    else {
                        // If the user did not make a cookie consent decision yet
                        // Check if this cookie is required
                        if(window.EWSCookieCategories[categoryID].required === true) {
                            return true;
                        }
                        else {
                            // If this cookie is not required
                            // Queue cookie request until user made a cookie decision
                            window.EWSCookieConsentQueue.push(cookieToSet);
                            return false;
                        }
                    }
                }
            }
        }

        // The cookie was not part of the cookie list
        // Submit new cookie to the backend
        EWSSubmitNewCookie(cookieNameToSet);

        // If an unexpected cookie was found, allow it for now until it was categorized
        return true;
    }

    // Compare a cookie name to a regex describing a cookie name
    function EWSCompareCookieNames(cookieName, regexStr) {
        let matchedName = cookieName.match(RegExp(regexStr));
        return matchedName == cookieName;
    }


    /* -------------------------------------------------------------- */
    /*                        Fetch Cookie List                       */
    /* -------------------------------------------------------------- */

    // Fetch the cookie list
    function EWSFetchCookieList() {
        var xhr = new XMLHttpRequest();
        xhr.onreadystatechange = function() {
            if(xhr.readyState === XMLHttpRequest.DONE && xhr.status === 200) {
                let cookieDictionary = JSON.parse(xhr.responseText);

                // Save fetched cookies list
                window.EWSCookieCategories = cookieDictionary.categories;

                // Set all cookies that were queued until the cookie list was fetched
                while(window.EWSCookieLoadingQueue.length > 0) {
                    let cookie = window.EWSCookieLoadingQueue.pop();
                    document.cookie = cookie;
                }
            }
        };
        xhr.open("GET", "https://cookie-edge.elgato.com/websites/catalogs/" + window.EWSCookieWebsiteID + ".json", true);
        xhr.send();
    }

    // Submit a new cookie to the backend
    function EWSSubmitNewCookie(name) {
        var xhr = new XMLHttpRequest();
        xhr.open("POST", "https://cookie.elgato.com/api/website/" + window.EWSCookieWebsiteID + "/create-cookie/", true);
        xhr.setRequestHeader("Content-Type", "application/json");
        let data = {"name": name};
        xhr.send(JSON.stringify(data));
    }


    /* -------------------------------------------------------------- */
    /*                           Executions                           */
    /* -------------------------------------------------------------- */

    // User custom document.cookie method
    EWSProxyDocumentCookie();

    // Fetch cookie list
    EWSFetchCookieList();
