            dateEnd: termin.dateStart,
            time: '',
            type: 'empty',
              column: termin.column,
            empty: true
          })
      }
      terminsWithEmpty.push(termin)
      if(nextTermin && calculateMinutesBetweenTermins(termin.dateEnd, nextTermin.dateStart) > 0){

            empty: true
          })
        }
      }

  Object.keys(column).forEach(function (day) {
    var groupsByDayKey = parseInt(day) - 1
      column[day].forEach(function(element){
        var heightTermin = calculateHourBetweenTermins(element.dateStart, element.dateEnd)
