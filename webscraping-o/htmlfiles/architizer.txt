     <script id="header-script" type="text/javascript">
      var GlobalHeader = function() {};
GlobalHeader.prototype.isChildOf = function(child, parent) {
if (!child.parentNode) {
return false;
} else if (child.parentNode === parent) {
return true;
} else {
return this.isChildOf(child.parentNode, parent);
}
};
////////////////////////////////////////////////////////////////////////////////
// User Dropdown
////////////////////////////////////////////////////////////////////////////////
var UserDropdown = function() {
this.link = document.getElementById('dropdown-link');
this.dropdown = document.getElementById('user-dropdown');
this.dropdownContent = document.getElementById('dropdown-content');
};
UserDropdown.prototype = Object.create(GlobalHeader.prototype);
UserDropdown.prototype.bindEventListeners = function () {
if (this.dropdown) {
this.dropdown.addEventListener('click', this.toggleUserDropdown.bind(this));
}
};
UserDropdown.prototype.toggleUserDropdown = function() {
if (this.dropdownContent.style.display === 'block') {
this.closeUserDropdown();
} else {
this.openUserDropdown();
}
};
UserDropdown.prototype.openUserDropdown = function() {
// Display dropdown content
this.dropdownContent.style.display = 'block';
// Add event listeners to close on resize and outside clicks
document.body.addEventListener('click', this.closeUserDropdown.bind(this));
window.addEventListener('resize', this.closeUserDropdown.bind(this));
};
UserDropdown.prototype.closeUserDropdown = function(e) {
// Don't close if click is on child of dropdown
if (e && e.target && this.isChildOf(e.target, this.link)) {
return;
}
// Hide dropdown content
this.dropdownContent.style.display = 'none';
// Remove event listeners
document.body.removeEventListener('click', this.closeUserDropdown.bind(this));
window.removeEventListener('resize', this.closeUserDropdown.bind(this));
};
////////////////////////////////////////////////////////////////////////////////
// Sign In Link
////////////////////////////////////////////////////////////////////////////////
var SignInLink = function () {
this.signInLink = document.getElementById('sign-in-link');
this.sidebarLink = document.getElementById('sidebar-sign-in-link');
this.joinNowLink = document.getElementById('join-now-link');
};
SignInLink.prototype = Object.create(GlobalHeader.prototype);
SignInLink.prototype.bindEventListeners = function () {
if (this.signInLink) {
this.signInLink.addEventListener('click', setNextParameter.bind(this, this.signInLink.pathname, this.joinNowLink.pathname));
}
if (this.sidebarLink) {
this.sidebarLink.addEventListener('click', setNextParameter.bind(this, this.signInLink.pathname, this.joinNowLink.pathname));
}
};
var setNextParameter = function (clickedLinkHref, secondLinkHref, event) {
event.preventDefault();
var currentPath = window.location.pathname;
var currentQueryParams = window.location.search;
// Append URI-encoded current path and query parameters to the href of the Sign In link
if (currentPath && currentPath !== '/') {
var LinkWithNext = '';
// if the user in the same link keep him at the same page
if (currentPath === clickedLinkHref) {
LinkWithNext = window.location;
}
// if the user in the is switching between join now and sign in pages, pass the params to the other page
else if (currentPath === secondLinkHref) {
LinkWithNext = clickedLinkHref + currentQueryParams;
}
// if the user in any other page and he clicks sign in or join now links, pass the page path as next param
else {
LinkWithNext = clickedLinkHref + '?next=' + encodeURIComponent(currentPath + currentQueryParams);
}
// Redirect to the Join Now or Sign In urls with the ?next= parameter appended
window.location.assign(LinkWithNext);
} else {
// Otherwise just use the Join Now or Sign In links without a ?next= parameter appended
window.location.assign(clickedLinkHref);
}
};
////////////////////////////////////////////////////////////////////////////////
// Sign Out Link
////////////////////////////////////////////////////////////////////////////////
var SignOutLink = function () {
this.signOutLinks = document.getElementsByClassName('sign-out-link');
this.resetIntercomSession = function (e) {
// Reset Intercom session
if (typeof window.Intercom === 'function') {
window.Intercom('shutdown');
}
}
};
SignOutLink.prototype = Object.create(GlobalHeader.prototype);
SignOutLink.prototype.bindEventListeners = function () {
if (this.signOutLinks) {
for (var i = 0; i < this.signOutLinks.length; i++) {
this.signOutLinks[i].addEventListener('click', this.resetIntercomSession);
}
}
};
////////////////////////////////////////////////////////////////////////////////
// Join Now Link
////////////////////////////////////////////////////////////////////////////////

this.addActiveClass(this.firmSearchLinks);
break;

this.addActiveClass(this.productsLinks);
break;

case (/(\/blog\/)/.test(pathname)):
this.addActiveClass(this.journalLinks);
break;
}
}
