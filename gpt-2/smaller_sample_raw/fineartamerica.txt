        // Return back the calculated height of all of the images next to one another.
        return width / aspectRatio;
    }

    function setheight(images, height)
    {
        rowheights.push(height);

        var widthContainer = parseInt(document.getElementById('imageflowcontainerdiv').offsetWidth) - 0;
        var widthRow = 0;
        var display = 'inline-block';

        for(var i = 0; i < images.length; ++i)
        {
            widthRow += $(images[i]).data('width')/$(images[i]).data('height')*height;
        }

        if(widthRow < 0.85 * widthContainer || numRows >= maxRows)
        {
            display = 'none';
        }

        if(images.length == 1)
        {
            var targetHeight = widthContainer * $(images[0]).data('height')/$(images[0]).data('width');

            if(targetHeight <= height * 1.2)
            {
                $(images[0]).css({'display':display, 'width': widthContainer, 'height': targetHeight});
                numRows++;
            }
            else
            {
                $(images[0]).css({'display': 'none'});
            }

            return;
        }

        for(var i = 0; i < images.length; ++i)
        {
            $(images[i]).css({'display': display, 'width': parseInt(height * $(images[i]).data('width') / $(images[i]).data('height')), 'height': height});
        }

        numRows++;

    }

    function resize(images, width)
    {
        setheight(images, getheight(images, width));
    }

    function run(max_height)
    {
        var widthcontainer = parseInt(document.getElementById('imageflowcontainerdiv').offsetWidth) - 0;
        numRows = 0;

        // Get all of the images inside of #imagecontainerdiv.
        var images = $('#imageflowcontainerdiv > .flowdiv');

        w: while(images.length > 0)
        {
            for(var i = 1; i < images.length + 1; i++)
            {
                // Continue adding images to the "slice" array, and after you add each image, check the new minimum height of that slice of images
                // by calling getheight.

                var slice = images.slice(0, i);

                var h = getheight(slice, widthcontainer);

                if(h < max_height)
                {
                    setheight(slice, h);

                    // Continue with the remaining images from index "i" on up.
                    images = images.slice(i);

                    continue w;
                }
            }

            setheight(slice, Math.min(max_height, h));

            break;
        }

    }
