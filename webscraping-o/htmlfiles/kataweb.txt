function tabber() {
    $(".radio-content").hide();
    $("ul.tabs li:first").addClass("active").show(); 
    $(".radio-content:first").show();
    $("ul.tabs li").click(function() {
        $("ul.tabs li").removeClass("active"); 
        $(this).addClass("active");
        $(".radio-content").hide();
        var activeTab = $(this).find("a").attr("href"); 
        $(activeTab).fadeIn(); 
        return false;
    });
}
