	var $jq = jQuery.noConflict();
	
	$jq(function($jq) {

 	function showTimeoutMsg() {
		
		var msg = 'Your session will expire in 3 minutes. ' +
			'To continue working, click on the "OK" button below. <br /><br />' +
    		'Note:<br />' + 
   			'This is being done to protect your privacy. Unsaved changes will be lost.'
   			
		msg += '<br /><br />' +
			'<div align="center">' +
			'<input type="button" value="  OK  " onclick="sessionTimeoutContinue();" />' +
			'</div>';
		
		document.getElementById( "sessionTimeoutDiv" ).innerHTML = msg;
		$jq( "#sessionTimeoutDiv" ).dialog( "open" );
		return false;
	}
	
	
	function updateSessionTimeout( check ) {
		var url = top.location.href;
		var skipUpdate = ( ( url.match( /search-grants\.html/ ) != null ) || ( url.match( /view-opportunity\.html/ ) != null ) );
		/* console.log( "grants theme - skip session timeout update: " + skipUpdate ); */
		
		var id = getGGUserProfileUID();
		if ( id == "" || skipUpdate ) { return; }
		/* var queryStr = 'jp={"id":"' + id + '"'; */
		var queryStr = 'jp={"id":"' + encodeURIComponent( id ) + '"';
		if ( check == false ) { queryStr += ',"check":false'; }
		queryStr += '}';
		
		$jq.ajax({
			"url": "https://www.grants.gov/grantsws/SessionTimeout",
			"data": queryStr,
			"dataType": "jsonp",
			"error": function( error ) {},
			"success": function( data ) {
				var id = data.id;
				var timeRemainingInMillis = data.timeRemainingInMillis;
				
				if ( timeRemainingInMillis == 0 ) {
					sessionTimeoutLogin();
				} else {
					updateSessionTimer( timeRemainingInMillis );
				}
			}
		});	
	}
	
	
	var sessionTimeoutTimerObj;
	
	function updateSessionTimer( timeInMillis ) {
		clearTimeout( sessionTimeoutTimerObj );
		var sessionTimeoutWarningInMillis = 3 * 60 * 1000;
		if ( timeInMillis > sessionTimeoutWarningInMillis ) {
			var timeToWarningInMillis = timeInMillis - sessionTimeoutWarningInMillis;
			/* set difference to display warning */
			sessionTimeoutTimerObj = setTimeout( "updateSessionTimeout(true)", timeToWarningInMillis );
			$jq( "#sessionTimeoutDiv" ).dialog( "close" );
		} else {
			sessionTimeoutTimerObj = setTimeout( "updateSessionTimeout(true)", timeInMillis );
			showTimeoutMsg();
		}/* if-else */
		
	}
	
	
	function sessionTimeoutContinue() {
		updateSessionTimeout( false );
		$jq( "#sessionTimeoutDiv" ).dialog( "close" );
	}
	
	
	function sessionTimeoutLogin() {
		top.location.href = "/custom/spoeeLogout.jsp";
	}
