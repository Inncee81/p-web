		function TriggerResize() {
			var event;
			if (typeof(Event) === 'function') {
				event = new Event('resize');

				event = document.createEvent('Event');

		var HkexAjaxUtils = {
			getKANHANString: function () {
				dest = '' + window.parent.location;
				dest = dest.toLowerCase();
				if (dest.toLowerCase().indexOf('/sc.hkex.com.hk/tunis/') != -1) {
					return 'sc.hkex.com.hk/TuniS/';
				} else if (dest.toLowerCase().indexOf('/sc.hkex.com.hk/gb/') != -1) {
					return 'sc.hkex.com.hk/gb/';
				} else {
					return '';
				}
			},
			stripKANHAN: function (url_in) {
				var khstr = HkexAjaxUtils.getKANHANString();
				var re = new RegExp(khstr, "i");
				return url_in.replace(re, "");
			},
			getDomain: function (url) {
				var lead_slashes = url.indexOf("://");
				var domain_start = lead_slashes + 3;
				var without_resource = url.substring(domain_start, url.length);
				var next_slash = without_resource.indexOf("/");
				var domain = without_resource.substring(0, next_slash);
				return domain;
			},
			getAjaxLink: function (path) {
				var dest = window.parent.location.href;
				dest = HkexAjaxUtils.stripKANHAN(dest);
				var domain = HkexAjaxUtils.getDomain(dest);
				var khstr = HkexAjaxUtils.getKANHANString();
				var http = window.parent.location.protocol == "https:" ? "https" : "http";
				return http + "://" + khstr + domain + path;
			},
			convertTag: function (html) {
				return html.split("___LT___").join("<").split("___GT___").join(">");
			}
		};

		function GetSpCalendar() {
			$(".sp-market-calendar").each(function () {

				if (spJson) {
					var path = spJson.path;
					var lang = spJson.lang;
					if (path.length > 0) {
						GetCalendar(sid, path, lang);
					}
				}
			});

		function GetCalendar(sid, path, lang) {
			var currentTime = new Date();
			var now = Date.now();
			for (var i = 0; i < (minMonth + 1 + maxMonth); i++) {
				var calDate = new Date(currentTime.getFullYear(), currentTime.getMonth() + (i - minMonth + 1), 0);
				var year = calDate.getFullYear().toString();
				var mth = (calDate.getMonth() + 1).toString();
				mth = mth.length > 1 ? mth : "0" + mth;
                // prevent no cn version json
                if (lang.toLowerCase() == "zh_cn") {
                    lang = "zh_hk";
                }
				var url = path + "/" + year + mth + "_" + lang + ".json?" + now;
				$.ajax({
					dataType: "json",
					async: false,
					contentType: "application/json; charset=utf-8",
					type: "GET",
					url: HkexAjaxUtils.getAjaxLink(url),
					success: function (data) {
						$.merge(calendarRecord, data);
					},
					error: function (xhr, option, thrownError) {
						//console.log(xhr.status);
					}
				});
			}

        function cardCalendarInit() {
			$('.hkex_card__datepicker').each(function () {
				var datePickerMonth = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"]
				var datePickerDay = ["S", "M", "T", "W", "T", "F", "S"]
				var yearSuffix = ""
				if ($("html").attr("lang") != "en") {
					datePickerMonth = ["1月", "2月", "3月", "4月", "5月", "6月", "7月", "8月", "9月", "10月", "11月", "12月"];
                    datePickerDay = ["日", " 一", "二", "三", "四", "五", "六"];
                    yearSuffix = "年";
				}
				var currentTime = new Date();
				var maxDate = new Date(currentTime.getFullYear(), currentTime.getMonth() + (maxMonth + 1), 0);
				var minDate = new Date(currentTime.getFullYear(), currentTime.getMonth() - (minMonth - 1), 0);

				$(this).datepicker({
					dateFormat: 'dd/mm/yy',
					inline: true,
					minDate: minDate,
					maxDate: maxDate,
					changeMonth: false,
					selectOtherMonths: true,
					monthNames: datePickerMonth,
					dayNamesMin: datePickerDay,
                    yearSuffix: yearSuffix,
					beforeShowDay: function beforeShowDay(date) {
						if (date > currentTime) {
							return [false, "calendar-day further-day day-" + date.getFullYear() + '-' + (date.getMonth() + 1) + '-' + date.getDate()];
						} else {
							return [false, "calendar-day day-" + date.getFullYear() + '-' + (date.getMonth() + 1) + '-' + date.getDate()];
						}
					},
					onChangeMonthYear: function onChangeMonthYear(date) {
						setTimeout(function () {
							renderDateEvent()
						}, 50);
						clickRecentlyDay(false)
					}
				});
				renderDateEvent()
				initCardCalendarClick()
				clickRecentlyDay(true)
				initCardCalendarSwiper()
            });

        function generateSliderCardHtml(sid, data) {
            if (data != "") {

                            dataHtml += '<div class="hkex_card_img ">';

		function dynamicCardInit() {
			$(".card-dynamic").each(function () {
				var order = parseInt($(this).attr("data-order"));

					if ($(this).attr("data-normal").toLowerCase() == "true") {
