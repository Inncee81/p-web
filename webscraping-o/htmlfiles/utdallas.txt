    function setUrgentCookie(cname,cvalue) {
  var d = new Date();
  d.setTime(d.getTime() + (5*60*1000));
  var expires = "expires=" + d.toGMTString();
  document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
}

function getUrgentCookie(cname) {
  var name = cname + "=";
  var decodedCookie = decodeURIComponent(document.cookie);
  var ca = decodedCookie.split(';');
  for(var i = 0; i < ca.length; i++) {
    var c = ca[i];
    while (c.charAt(0) == ' ') {
      c = c.substring(1);
    }
    if (c.indexOf(name) == 0) {
      return c.substring(name.length, c.length);
    }
  }
  return "";
}

function checkUrgentCookie() {
  var urgent=getUrgentCookie("UTDURGENT");
  if (urgent != "" && urgent == "2021-03-16T19:08:00+00:00") {

     urgent = "2021-03-16T19:08:00+00:00";
     if (urgent != "" && urgent != null) {
       jQuery('#urgent').modal('show');
       setUrgentCookie("UTDURGENT", urgent);

     }
  }
}
