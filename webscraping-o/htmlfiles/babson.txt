      // Change opacity if another overlay detected

        if ($('.cookie-handler').length > 1 && $('.cookie-handler:hidden').length == 0) {
          $('#cookie-handler').css({ opacity: 1 });

      }, 0);

      $('#overlay').fadeIn('fast', function () {
        if (detectmob()) {
          $('#cookie-handler').animate({ 'top': '100px' }, 500);

          $('#cookie-handler').animate({ 'top': '220px' }, 500);

        }
      });
      focusRestrict();
    });
  }
  $(function () {
    $('.boxclose').click(function () {  //Close message box
      closePopUp();
    });
  });

  function detectmob() {
    if (window.innerWidth <= 400 && window.innerHeight <= 800) {
      return true;
    } else {
      return false;
    }
  }

  $(document).keydown(function (event) {
    if (event.keyCode == 27) {
      closePopUp();
    }
  });

  function closePopUp() {
    document.cookie = "babsonedu-babovisited2=true;path=/;max-age=" + 1000 * 60 * 60 * 24; // Creates cookie and sets cookie for a year
    if (detectmob()) {
      fade_top = '-600px';
    } else {
      fade_top = '-400px';
    }
    $('#cookie-handler').animate({ 'top': fade_top }, 500, function () {
      $('#overlay').fadeOut('fast', function () {
        $('#overlay').hide();
        $('#cookie-handler').hide();
      });
    });
    modalOpen = false;
  }

  function focusRestrict(event) {
    document.addEventListener('focus', function (event) {
      if (modalOpen && (
        (typeof modal.has === "function" && !modal.has(event.target).length) ||
        (typeof modal.contains === "function" && !modal.contains(event.target)) ||
        $(event.target).hasClass("last-tab"))
      ) {
        event.preventDefault();
        event.stopPropagation();
        firstFocusField.focus();
      }
    }, true);
  }
