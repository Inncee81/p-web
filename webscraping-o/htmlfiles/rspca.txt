    function submitDonateForm() {
        var target = event.target;
        var form = $(target).parents('form:first');
        var link = $(form).find("#donateButton");
        var url = $(link).data('url');
        var amount = $(form).find("#_donationAmount").val();
        var minAmount = 1;

        if (url == "/getinvolved/donate/online/single") {

        } else {
                minAmount = 2

                var maxAmount = 1000
        }
        
        var regEx = /^\d+(\.\d{1,2})?$/;
       
        if (amount == '' || isNaN(amount)) {
            $(form).find("#errDiv").html( "Please enter amount");
            event.preventDefault();
            return;   
        } else if (amount < minAmount) {
            $(form).find("#errDiv").html("Due to administration costs we are unable to accept donations under £" + minAmount);
            event.preventDefault();
            return;
        } else if (typeof(maxAmount) != "undefined" && amount > maxAmount) {
            $(form).find("#errDiv").html("Due to administration costs we are unable to accept donations over £" + maxAmount);
            event.preventDefault();
            return;
        } else if (!regEx.test(amount)) {
            $(form).find("#errDiv").html("Please enter a valid amount (no commas and only 2 digits after the decimal)");
            event.preventDefault();
            return;   
        } else {
            var finalUrl = '';

            if (url.indexOf('?') > 0) {
                finalUrl = url + '&amount=' + amount;    
            } else {
                finalUrl = url + '?amount=' + amount; 
            }
            form.attr('action', finalUrl);       
        }
    }
