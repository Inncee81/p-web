            function destyleMktoForm(mktoForm, moreStyles){

        var formEl = mktoForm.getFormElem()[0],
            arrayify = getSelection.call.bind([].slice);

        // remove element styles from <form> and children
        var styledEls = arrayify(formEl.querySelectorAll("[style]")).concat(formEl);	
        styledEls.forEach(function(el) {
            el.removeAttribute("style");
            el.removeAttribute('placeholder');
        });

        // disable remote stylesheets and local <style>s
        var styleSheets = arrayify(document.styleSheets);	
        styleSheets.forEach(function(ss) {
            if ( [mktoForms2BaseStyle,mktoForms2ThemeStyle].indexOf(ss.ownerNode) != -1 || formEl.contains(ss.ownerNode) ) {
                ss.disabled = true;
            }
        });
            
        if(!moreStyles) {
            formEl.setAttribute("data-styles-ready", "true");
        }

        /**
         * Add select to manipulate hidden inputs
         */
        // Get button row
        let button = formEl.querySelector('.mktoButtonRow'),
            inputs = formEl.querySelectorAll('.mktoFormRow > input[type=hidden]'),
            dropdown = document.createElement('select'),
            names = {
                'BrandsNewsletter': 'Brands',
                'SportsNewsletter': 'Sports',
                'EnergyNewsletter': 'Energy',
                'EntertainmentNewsletter': 'Entertainment',
                'FinanceNewsletter': 'Finance',
                'HealthNewsletter': 'Health',
                'TechnologyNewsletter': 'Tech',
                'WashingtonNewsletter': 'Washington',
                'EconomicIntelligenceSignups': 'Economic Intel',
                'PoliticalIntelligenceSignups': 'Political Intel',
                'OurBestIntel': 'Our Best Intel'
            };


        for(i = 0; i < inputs.length; i++) {
            let option = document.createElement('option');
            option.value = inputs[i].name;
            option.text = names[option.value];
            dropdown.appendChild(option);
        }

        formEl.insertBefore(dropdown, button);

        let selectr = new Selectr(dropdown, {
            multiple: true,
            placeholder: 'Select an option'
        });

        // Change text box value to null when option removed.
        selectr.on('selectr.deselect', function(option) {
            let input = formEl.querySelector('input[name=' + option.value + ']');
            input.value = null;
        });

        // Change text box value to 1 when option added.
        selectr.on('selectr.select', function(option) {
            let input = formEl.querySelector('input[name=' + option.value + ']');
            input.value = 1;
        });
    };


    function validateForm(form) {

        // Get all form inputs
        let fields = [...document.querySelectorAll('.mktoForm input.mktoRequired, .mktoForm select.mktoRequired, .mktoForm textarea.mktoRequired')],
            error = 'This field is required',
            error_email = 'Must be valid email. example@yourdomain.com',
            selects = [...document.querySelectorAll('.mktoForm select.mktoRequired')];

        fields.forEach((field) => {

            // Generate each field error message and hide it on load.
            let error_container = document.createElement('div'),
                error_inner = document.createElement('div'),
                error_text = '';

            error_container.classList.add('mktoErrored');
            error_container.style.display = 'none';
            error_inner.classList.add('mktoErrorMsg');
            error_inner.setAttribute('role', 'alert');

            if (field.type == 'email') {
                error_text = document.createTextNode(error_email);
            } else {
                error_text = document.createTextNode(error);
            }
            
            // Append text to inner container
            error_inner.appendChild(error_text);

            // Append inner container to error container
            error_container.appendChild(error_inner);

            // Add error container to DOM
            field.parentNode.appendChild(error_container);

            // Toggles mktoInvalid class and error.
            field.addEventListener('blur', function() {
                if (! field.value) {
                    field.classList.add('mktoNotValid');
                    error_container.style.display = 'block';
                } else {
                    // If this is an email field, regex
                    if (field.type == 'email') {
                        if (/^[a-zA-Z0-9.!#$%&'*+/=?^_`{|}~-]+@[a-zA-Z0-9-]+(?:\.[a-zA-Z0-9-]+)*$/.test(field.value)) {
                            field.classList.remove('mktoNotValid');
                            error_container.style.display = 'none';
                        } else {
                            field.classList.add('mktoNotValid');
                            error_container.style.display = 'block';
                        }
                    } else {
                        field.classList.remove('mktoNotValid');
                        error_container.style.display = 'none';
                    }
                }
            })
        })

        // Listen for the validate event
        form.onValidate(function() {

            // Get the values
            let vals = form.vals(),
                submittable = true;

            fields.forEach((field) => {

                let error_container = field.parentNode.querySelector('.mktoErrored');
                
                if (! field.value) {
                    field.classList.add('mktoNotValid');
                    error_container.style.display = 'block';
                    submittable = false;
                } else {
                    // If this is an email field, regex
                    if (field.type == 'email') {
                        if (/^[a-zA-Z0-9.!#$%&'*+/=?^_`{|}~-]+@[a-zA-Z0-9-]+(?:\.[a-zA-Z0-9-]+)*$/.test(field.value)) {
                            field.classList.remove('mktoNotValid');
                            error_container.style.display = 'none';
                        } else {
                            field.classList.add('mktoNotValid');
                            error_container.style.display = 'block';
                            submittable = false;
                        }
                    } else {
                        field.classList.remove('mktoNotValid');
                        error_container.style.display = 'none';
                    }
                }
            });

            form.submittable(submittable);
        });

        form.getFormElem().find(".mktoFieldDescriptor").each(function(idx,el) {
            var originalFocusInListeners;

            if (MktoForms2.$.hasData(el)) {
                if (originalFocusInListeners = MktoForms2.$._data(el).events.focusin){
                    MktoForms2.$(el).data("reservedFocusIn", originalFocusInListeners[0]);
                    MktoForms2.$(el).off("focusin");     
                }
            } else {
                MktoForms2.$(el).data("reservedFocusIn", originalFocusInListeners[0]);
                MktoForms2.$(el).off("focusin");     
            }   
        });
    }
