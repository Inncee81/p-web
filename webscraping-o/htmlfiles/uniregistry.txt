	                customer_email: `${Unireg.appData.session.user_email}`,
	                customer_id: `${Unireg.appData.session.user_id}`,
	                language_code: `${Unireg.appData.session.locale}`,
                };
            }

            if (typeof utag !== "undefined" && utag.view) {

            }
        } catch (e) {
            console.error("Tealium failure");
            // Log this to sentry
        }

    }

    function onUrlChange(event_function) {
        /*
         * calls an event function when the url changes.
         * had to patch the pushState and replace state since
         * there isn't any existing functionality that lets us
         * listen to the url changes, either that or we manually
         * use the event from the react router for each app we
         * have.
         */

        // override pushState so we can call our event function with it
        var pushState = history.pushState;
        history.pushState = function(state, title, url) {
            var return_value = pushState.bind(history)(state, title, url);
            event_function(state, title, url);
            return return_value;
        };

        // override replaceState so we can call our event function with it
        var replaceState = history.replaceState;
        history.replaceState = function(state, title, url) {
            var return_value = replaceState.bind(history)(state, title, url);
            event_function(state, title, url);
            return return_value;
        };
    }

    // Tealium is loaded asynchronously so track page views after the DOM is ready
    window.addEventListener('load', (event) => {
        tealiumUrlUpdate();
        onUrlChange(tealiumUrlUpdate);
    });
