   jQuery(document).ready(function ($) {
        const mainContent = document.querySelector('#custom-wrap .custom-wrap__content');
        const pars = mainContent.querySelectorAll('p');
        const docW = document.documentElement.clientWidth;
        const lists = mainContent.querySelectorAll('ol');
        const unLists = mainContent.querySelectorAll('ul');

        function showMoreContent() {
            if (docW >= 768) {
                // paragraphs
                pars.forEach((par, index) => {
                    let links = par.querySelectorAll('a');
                    let lastLink = links[links.length - 1];

                    if (par.textContent.length > 430 && par.lastElementChild !== lastLink) {
                        const substr = par.textContent.slice(0, 430);
                        const str = par.textContent;
                        par.innerHTML = `${substr}...`;

                        const showMore = document.createElement('span');
                        showMore.classList.add('show-more-par-text');
                        showMore.innerText = 'Show more';
                        par.appendChild(showMore);

                        showMore.addEventListener('click', function () {
                            par.innerText = str;
                        })
                    }
                });

                // lists
                lists.forEach(list => {
                    if (list.textContent.length > 470) {
                        let listItems = list.querySelectorAll('li');
                        listItems.forEach(listItem => {
                            if (listItem !== listItems[0]) {
                                listItem.style.display = 'none';
                            }
                        });
                        const showMore = document.createElement('span');
                        showMore.classList.add('show-more-par-text');
                        showMore.innerText = 'Show more...';
                        listItems[0].appendChild(showMore);

                        showMore.addEventListener('click', function () {
                            listItems.forEach(listItem => {
                                if (listItem !== listItems[0]) {
                                    listItem.style.display = 'flex';
                                }
                                showMore.style.display = 'none';
                            });
                        })
                    }

                });

                // Unordered lists
                unLists.forEach(list => {

                        par.innerHTML = `${substr}...`;

                        showMore.innerText = 'Show more';
                        par.appendChild(showMore);

                        showMore.addEventListener('click', function () {
                            par.innerText = str;
                        })

                });

                // lists

                    if (list.textContent.length > 140) {
                        let listItems = list.querySelectorAll('li');
                        listItems.forEach(listItem => {
                            if (listItem !== listItems[0]) {
                                listItem.style.display = 'none';
                            }
                        });
                        const showMore = document.createElement('span');
                        showMore.classList.add('show-more-par-text');
                        showMore.innerText = 'Show more...';

  <script>
   jQuery(document).ready(function ($) {
        function addSidebarBlocks() {

            /*Находим начальную точку, в которую будет добавляться первый блок*/
            const firstPoint = document.getElementById('first-entry-point');

            /*Находим все инсёрт-боксы*/
            const insertBoxes = document.querySelectorAll('.insert-box');

            /*Находим первый инсёрт-бокс*/
            const firstInsertBox = insertBoxes[0];

            /*Делаем клон первого инсёрт-бокса*/
            const firstInsertClone = $(firstInsertBox)
                .clone()
                .removeClass('insert-box')
                .addClass('insert-clone-box');

            /*Помещаем первый клон в первый бокс*/
            $(firstPoint).append($(firstInsertClone));

            /*Находим все заголовки на одном уровне*/
            $('#heading-finder').siblings('h1').addClass('head-lock');
            $('#heading-finder').siblings('h2').addClass('head-lock');
            $('#heading-finder').siblings('h3').addClass('head-lock');
            $('#heading-finder').siblings('h4').addClass('head-lock');
            $('#heading-finder').siblings('h5').addClass('head-lock');
            $('#heading-finder').siblings('h6').addClass('head-lock');


            /*Находим все тайтлы с классом .head-lock*/
            const titles = document.querySelectorAll('.head-lock');

            /*Условие если количество инсёрт-блоков меньше или равно количеству заголовков*/
            if (insertBoxes.length <= $('.head-lock').length) {

                const firstEntryPoint = document.querySelector('#first-entry-point');

                /*Перебираем каждый блок*/
                for (let i = 1; i < insertBoxes.length; i++) {

                    /*Перебираем каждый блок начиная со второго*/
                    let insertBox = insertBoxes[i];
                    let cloneBox = $(insertBox).clone().addClass('insert-clone-box');

                    /*Перебираем каждый заголовок начиная со второго*/
                    let title = titles[i];

                    /*Вычисляем высоту каждого элемента относительно верхушки документа*/
                    let titleOffset = $(title).offset().top + $(title).outerHeight(true);

                    /*Добавляем пустые блоки с классом .entry-point*/
                    let entryPoint = document.createElement('div');

                    $(entryPoint)
                        .addClass('entry-point')
                        .css({
                            'display': 'block',
                            'width': '256px',
                            'position': 'absolute',
                            'top': titleOffset + 'px',
                            'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                        });

                    $('#cust-side').append($(entryPoint).append($(cloneBox)));


                }

                $(window).on('resize', function () {
                    $('.entry-point').css({
                        'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                    });
                })
            }
        }

        addSidebarBlocks();

        function fixEntrypoints() {
            if (document.documentElement.clientWidth > 1200) {
                const firstEntry = document.querySelector('#first-entry-point');
                const firstEntryTop = firstEntry.getBoundingClientRect().bottom;

                const entry = document.querySelectorAll('.entry-point');

                const secEntry = entry[0];
                const secEntryTop = secEntry.getBoundingClientRect().top;
                const secEntryBottom = secEntry.getBoundingClientRect().bottom;

                const thirdEntry = entry[1];
                const thirdEntryTop = secEntry.getBoundingClientRect().top;
                const thirdEntryBottom = secEntry.getBoundingClientRect().bottom;

                const headLock = document.querySelectorAll('.head-lock');


                function secondAdding() {
                    const insertBoxes = document.querySelectorAll('.insert-box'),
                        titles = document.querySelectorAll('.head-lock');

                    if (insertBoxes.length <= $('.head-lock').length) {
                        // const firstEntryPoint = document.querySelector('#first-entry-point');

                        /*Перебираем каждый блок*/
                        for (let i = 1; i < insertBoxes.length; i++) {

                            /*Перебираем каждый блок начиная со второго*/
                            let insertBox = insertBoxes[i];
                            let cloneBox = $(insertBox).clone().addClass('insert-clone-box');

                            /*Перебираем каждый заголовок начиная со второго*/
                            let title = titles[i];

                            /*Вычисляем высоту каждого элемента относительно верхушки документа*/
                            let titleOffset = $(title).offset().top + $(title).outerHeight(true);

                            /*Добавляем пустые блоки с классом .entry-point*/
                            let entryPoint = document.createElement('div');

                            $(entryPoint)
                                .addClass('entry-point')
                                .css({
                                    'display': 'block',
                                    'width': '256px',
                                    'position': 'absolute',
                                    'top': titleOffset + 'px',
                                    'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                                });

                            $('#cust-side').append($(entryPoint).append($(cloneBox)));


                        }

                        $(window).on('resize', function () {
                            $('.entry-point').css({
                                'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                            });
                        })
                    }
                }

                if (secEntryTop < firstEntryTop && firstEntryTop !== null) {
                    $('.entry-point').empty();
                    headLock[1].classList.remove('head-lock');
                    secondAdding();
                } else if (thirdEntryTop < secEntryBottom) {
                    $('.entry-point').empty();
                    headLock[2].classList.remove('head-lock');
                    secondAdding();
                }

            }
        }

        fixEntrypoints();
    });

  <script>
   jQuery(document).ready(function ($) {


        function flexibleBlock(min, max) {
            if (window.location.pathname === '/') {
                setTimeout(function () {
                    let flexibleBlock = document.querySelector('.flexible-block.insert-clone-box');

                    if (flexibleBlock != null || flexibleBlock != undefined) {
                        min = $(flexibleBlock).offset().top;         // find the min point - start of flex
                        max = $('.video-poster').offset().top - 500; // find the max point - end of flex
                        $(window).on('scroll', function () {
                            if (document.documentElement.scrollTop > min) {
                                $(flexibleBlock).css({
                                    'position': 'fixed',
                                    'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                                    'top': '0',
                                    'opacity': '1',
                                    'transform': 'translateY(70px)',
                                    'transition': 'transform .15s linear',
                                    'z-index': '500'
                                });
                            } else if (document.documentElement.scrollTop < min) {
                                $(flexibleBlock).css({
                                    'position': 'relative',
                                    'right': '0',
                                    'transform': 'translateY(0)',
                                    'transition': 'transform .15s linear',
                                    'z-index': '500'
                                });
                            }

                            if (document.documentElement.scrollTop > max) {
                                $(flexibleBlock).css({
                                    'opacity': '0',
                                    'transition': 'transform .15s linear',
                                    'z-index': '-1'
                                });
                            }
                        })
                    }
                }, 500)
            }
        }

        if (document.documentElement.clientWidth >= 1200) {
            flexibleBlock();
        }

    });
  </script>

  <script>
   jQuery(document).ready(function ($) {
        //    ------------------------------ LAZY IMAGES ------------------------------
        function lazyImagesSrc() { // For tag <img src="...">
            const lazyImages = document.querySelectorAll('.lazy-img');

            const options = {
                root: null,
                rootMargin: '200px',
                threshold: 0.1
            };

            function handleImg(myImg, observer) {
                myImg.forEach(myImgSingle => {

                    if (myImgSingle.intersectionRatio > 0) {
                        loadImage(myImgSingle.target);
                    }
                })
            }

            function loadImage(image) {
                image.src = image.getAttribute('data-l-img');
            }

            const observer = new IntersectionObserver(handleImg, options);

            lazyImages.forEach(img => {
                observer.observe(img);
            });
        }

        lazyImagesSrc();

        function lazyImagesBg() { // For blocks with style="background: url('...')"
            const lazyImages = document.querySelectorAll('.lazy-bg');

            const options = {
                root: null,
                rootMargin: '200px',
                threshold: 0.1
            };

            function handleImg(myImg, observer) {
                myImg.forEach(myImgSingle => {

                    if (myImgSingle.intersectionRatio > 0) {
                        loadImage(myImgSingle.target);
                    }
                })
            }

            function loadImage(image) {
                image.style.background = `url(${image.getAttribute('data-l-bg')})`;
            }

            const observer = new IntersectionObserver(handleImg, options);

            lazyImages.forEach(img => {
                observer.observe(img);
            });
        }

        lazyImagesBg();
        //    ------------------------------ LAZY IMAGES ------------------------------
    });
  </script>
