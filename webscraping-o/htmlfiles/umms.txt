      function GetEntity(url) {
        if (!url) url = window.location.href;
        var entity = getParameterByName('si', url); // 
        if (entity != null) return entity;
        if (url.indexOf("/shore/") > 0) return "shore";
        else if (url.indexOf("/bwmc/") > 0) return "bwmc";
        else if (url.indexOf("/charles/") > 0) return "charles";
        else if (url.indexOf("/sjmc/") > 0) return "sjmc";
        else if (url.indexOf("/rehab/") > 0) return "rehab";
        else if (url.indexOf("/uch/") > 0) return "uch";
        else if (url.indexOf("/midtown/") > 0) return "midtown";
        else if (url.indexOf("/ummc/") > 0) return "ummc";
        else if (url.indexOf("/childrens/") > 0) return "childrens";
        else if (url.indexOf("/umgccc/") > 0) return "cancer";
        else if (url.indexOf("/capital/") > 0) return "capital";
        else if (url.indexOf("/fpi/") > 0) return "fpi";
        return "";
    }

    function getParameterByName(name, url) {
        if (!url) url = window.location.href;
        name = name.replace(/[\[\]]/g, '\\$&');
        var regex = new RegExp('[?&]' + name + '(=([^&#]*)|&|#|$)'),
            results = regex.exec(url);
        if (!results) return null;
        if (!results[2]) return '';
        return decodeURIComponent(results[2].replace(/\+/g, ' '));
    }

      (function ($) {
        var $typeaheadMenu = $('.js-typeahead-sss-menu-header');

        GenerateRowHeader = function (i, highlightedText, fields) {
            if (!fields.item_type_string) {
                return;
            }
            var $ul = $('ul[data-typeaheadtype=' + fields.item_type_string + ']');
            if ($ul.length <= 0) {
                return;
            }
            //if ($ul.children('li').length >= 7)
            //    return;

            var limitReached = false;
            $ul.each(function (i) {
                if ($(this).children('li').length >= 3) {
                    limitReached = true;
                }
            }
            );
            if (limitReached) return;

            var url = fields.url.replace(/^.*\/\/[^\/]+/, '')
            if (fields.item_type_string == "Doctors") {
                var entity = GetEntity();
                $ul.append('<li class="typeahead-sss__result"><a href="' + url + '?si=' + entity + '">' + fields.content_typeahead_search + '</a></li>');
            }
            else {
                if (fields.item_type_string == "Locations") {
                    if (fields.site_string=="MWPH") {
                        url = url.replace("/mwph", "");
                        var newurl = window.location.protocol + "//www.mwph.org";
                        url = newurl.concat(url);

                    }
                    if (fields.site_string=="FPI") {
                        url = url.replace("/fpi", "");
                        var newurl = window.location.protocol + "//www.umfpi.org";
                        url = newurl.concat(url);

                    }
                }
                $ul.append('<li class="typeahead-sss__result"><a href="' + url + '">' + fields.content_typeahead_search + '</a></li>');

            }
            $ul.parent().show();
            $typeaheadMenu.show();
            return null;
        }
