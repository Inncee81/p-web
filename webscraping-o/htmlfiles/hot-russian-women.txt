                }

                $(window).on('resize', function () {
                    $('.entry-point').css({
                        'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                    });
                })
            }
        }

        addSidebarBlocks();

        function fixEntrypoints() {
            if (document.documentElement.clientWidth > 1200) {
                const firstEntry = document.querySelector('#first-entry-point');
                const firstEntryTop = firstEntry.getBoundingClientRect().bottom;

                const entry = document.querySelectorAll('.entry-point');

                const secEntry = entry[0];
                const secEntryTop = secEntry.getBoundingClientRect().top;
                const secEntryBottom = secEntry.getBoundingClientRect().bottom;

                const thirdEntry = entry[1];
                const thirdEntryTop = secEntry.getBoundingClientRect().top;
                const thirdEntryBottom = secEntry.getBoundingClientRect().bottom;

                const headLock = document.querySelectorAll('.head-lock');


                function secondAdding() {
                    const insertBoxes = document.querySelectorAll('.insert-box'),
                        titles = document.querySelectorAll('.head-lock');

                    if (insertBoxes.length <= $('.head-lock').length) {
                        // const firstEntryPoint = document.querySelector('#first-entry-point');

                        /*Перебираем каждый блок*/
                        for (let i = 1; i < insertBoxes.length; i++) {

                            /*Перебираем каждый блок начиная со второго*/
                            let insertBox = insertBoxes[i];
                            let cloneBox = $(insertBox).clone().addClass('insert-clone-box');

                            /*Перебираем каждый заголовок начиная со второго*/
                            let title = titles[i];

                            /*Вычисляем высоту каждого элемента относительно верхушки документа*/
                            let titleOffset = $(title).offset().top + $(title).outerHeight(true);

                            /*Добавляем пустые блоки с классом .entry-point*/
                            let entryPoint = document.createElement('div');

                            $(entryPoint)
                                .addClass('entry-point')
                                .css({
                                    'display': 'block',
                                    'width': '256px',
                                    'position': 'absolute',
                                    'top': titleOffset + 'px',
                                    'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                                });

                            $('#cust-side').append($(entryPoint).append($(cloneBox)));


                        }

                        $(window).on('resize', function () {
                            $('.entry-point').css({
                                'right': ($(document).outerWidth() - $('.custom-wrap').outerWidth()) / 2,
                            });
                        })
                    }
                }

                if (secEntryTop < firstEntryTop && firstEntryTop !== null) {
                    $('.entry-point').empty();
                    headLock[1].classList.remove('head-lock');
                    secondAdding();
                } else if (thirdEntryTop < secEntryBottom) {
                    $('.entry-point').empty();
                    headLock[2].classList.remove('head-lock');
                    secondAdding();
                }

            }
        }

        fixEntrypoints();
    });

  <script>
   jQuery(document).ready(function ($) {
        //    ------------------------------ LAZY IMAGES ------------------------------
        function lazyImagesSrc() { // For tag <img src="...">
            const lazyImages = document.querySelectorAll('.lazy-img');

            const options = {
                root: null,
                rootMargin: '200px',
                threshold: 0.1
            };

            function handleImg(myImg, observer) {
                myImg.forEach(myImgSingle => {

                    if (myImgSingle.intersectionRatio > 0) {
                        loadImage(myImgSingle.target);
                    }
                })
            }

            function loadImage(image) {
                image.src = image.getAttribute('data-l-img');
            }

            const observer = new IntersectionObserver(handleImg, options);

            lazyImages.forEach(img => {
                observer.observe(img);
            });
        }

        lazyImagesSrc();

        function lazyImagesBg() { // For blocks with style="background: url('...')"
            const lazyImages = document.querySelectorAll('.lazy-bg');

            const options = {
                root: null,
                rootMargin: '200px',
                threshold: 0.1
            };

            function handleImg(myImg, observer) {
                myImg.forEach(myImgSingle => {

                    if (myImgSingle.intersectionRatio > 0) {
                        loadImage(myImgSingle.target);
                    }
                })
            }

            function loadImage(image) {
                image.style.background = `url(${image.getAttribute('data-l-bg')})`;
            }

            const observer = new IntersectionObserver(handleImg, options);

            lazyImages.forEach(img => {
                observer.observe(img);
            });
        }

        lazyImagesBg();
        //    ------------------------------ LAZY IMAGES ------------------------------
    });
  </script>
