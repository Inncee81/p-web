    /*
    스크롤이 광고div 위치에 있는지 체크
    */
    function isAdDivVisible( obj )
    {
        if( obj.length == 0 ) return false;
        var target           = obj.offset().top;      //타겟 위치값
        var window_height    = $(window).height();    //윈도우크기
        var window_scrollTop = $(window).scrollTop(); //브라우저의 스크롤 위치값
        var gap              = 500;                    //간격조정 px
        var isVisible        = false;

        isVisible = ( target - gap <= (window_height) )                    ? true : isVisible;
        isVisible = ( target - gap <= (window_height + window_scrollTop) ) ? true : isVisible;

        return isVisible;
    }

function jfnGifLayerPopupView(imgurl, w, h)
{

    var img_w = w;
    var img_h = h;
    if( $(document).width() < w - 20 )
    {
        img_w = $(document).width() - 20;
        img_h = Math.round((h) * (img_w/w));
    }
    w = img_w; h = img_h;


    $('#gif_down_layer_box').css({width:w, height:h});
    $('#gif_download_img').css({width:w, height:h});

    $("#gif_download_img").attr("src", imgurl);
    jfnGifLayerPopupShow(w,h);
}

function jfnGifLayerPopupShow(w,h){

    var $el = $('#gif_down_layer');        //레이어의 id를 $el 변수에 저장
    var isgif_dim = $el.prev().hasClass('gif_dimBg');   //gif_dimmed 레이어를 감지하기 위한 boolean 변수

    // if( $(document).width() < w - 20) w = $(document).width() - 70;
    $('#gif_down_layer_box').css('width', w+'px');
    $('#gif_down_layer_box').css('height', h);

    isgif_dim ? $('.gif_down-layer').fadeIn() : $el.fadeIn();

    var $elWidth = ~~($el.outerWidth()),
        $elHeight = ~~($el.outerHeight()),
        docWidth = $(document).width(),
        docHeight = $(document).height();

    // 화면의 중앙에 레이어를 띄운다.
    if ($elHeight < docHeight || $elWidth < docWidth) {
        $el.css({
            marginTop: -$elHeight/2,
            marginLeft: -$elWidth/2
        })
    } else {
        $el.css({top: 0, left: 0});
    }
}

   // get SNS oembed
$('.cmt-media-link').on('click', function(e){
    var self = this;
    var $next = $(self).next();
    var url = $(self).data('url');
    var media = $(self).data('media');
    var api_url = '';
    var data = {};

    if( $(self).hasClass('active') ){
        $next.hide();
        $(self).removeClass('active');
        return false;
    }

    if( $next.html() ){
        $next.show();
        $(self).addClass('active');
        return false;
    }

    switch( media )
    {
        case 'instagram' :
            if( url.match(/instagram.com\/p\/|instagram.com\/tv\//) ){
                api_url = "https://graph.facebook.com/v9.0/instagram_oembed?";
                data = {
                    "url" : url,
                    "access_token" : "427974708216137|02c9c2734c062288a55445eb068469d4"
                };
            }else{
                return false;
            }
            break;

        case 'twitter' :
            if( url.match(/\/status\//) ){
                api_url = "https://publish.twitter.com/oembed?callback=?";
                data = {"url" : url};
            }else{
                return false;
            }
            break;

        case 'facebook' :
            // video
            if( url.match(/\/videos\/|\/video\.php\?/) ){
                api_url = "https://graph.facebook.com/v9.0/oembed_video?";
            // post
            }else if( url.match(/\/posts\/|\/activity\/|\/photos\/|\/media\/|\/questions\/|\/notes\/|\/photo.php\?|\/permalink.php/) ){
                api_url = "https://graph.facebook.com/v9.0/oembed_post?";
            // page
            }else{
                api_url = "https://graph.facebook.com/v9.0/oembed_page?";
            }
            data = {
                "url" : url,
                "access_token" : "427974708216137|02c9c2734c062288a55445eb068469d4"
            };
            break;

        case 'youtube' :
            var res = '<iframe width="560px" height="315px" src="https://www.youtube.com/embed/' +url+ '" frameborder="0" allow="accelerometer; encrypted-media; gyroscope; picture-in-picture" allowfullscreen=""></iframe>'
            $next.html(res);
            $(self).addClass('active');
            return false;

        case 'kakao' :
            var res = '<iframe title="" width="640" height="360" src="https://play-tv.kakao.com/embed/player/cliplink/' +url+ '" allowfullscreen frameborder="0" scrolling="no" allow="autoplay"></iframe>'
            $next.html(res);
            $(self).addClass('active');
            return false;

        case 'naver' :
            var res = '<iframe src="https://tv.naver.com/embed/' +url+ '" frameborder="no" scrolling="no" marginwidth="0" marginheight="0" WIDTH="544" HEIGHT="306" allow="autoplay" allowfullscreen></iframe>'
            $next.html(res);
            $(self).addClass('active');
            return false;

        default :
            return false;
    }

    $.getJSON(api_url, data).done(function(res){
        if( typeof res.html != 'undefined' ){
            $next.html(res.html);
            $(self).addClass('active');
        }
    });

    return false;
});
