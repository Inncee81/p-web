         if($(this)[0].min_price == $(this)[0].max_price && $(this)[0].max_price == 0){
         		$(thHtml).find('.event_data').text('Free');

         		$(thHtml).find('.event_data').text($(this)[0].currency_symbol+$(this)[0].max_price+'.00');

         		$(thHtml).find('.event_data').text($(this)[0].currency_symbol+$(this)[0].min_price+'.00 ~ '+$(this)[0].currency_symbol+$(this)[0].max_price+'.00');

      })

            const date = new Date(utcDateTimeString);
            const localDateTimeString = new Intl.DateTimeFormat([], {
              month: 'short',
              day: 'numeric',
              hour: 'numeric',
              minute: 'numeric',
              timeZoneName: 'short'
            }).format(date);
            $(this).html(localDateTimeString);
      });
