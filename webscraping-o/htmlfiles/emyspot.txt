offers.refreshBasket = function(){
	var selectedOffers = new Array();
	var toPay = 0;
	$('#offers-basket-content').html('');
	$('#offers-options input[type="checkbox"]:checked').each(function(){
		var offer = $(this).data('offer');
		if(selectedOffers.indexOf(offer) == -1)
		{
			selectedOffers.push(offer);
		}
	});
	selectedOffers.sort();
	localStorage.setItem('offersBasket', selectedOffers.join('|'));
	for(i = 0; i < selectedOffers.length; i++){
		toPay += offers.prices[offers.currency][selectedOffers[i]];
	    
		if(selectedOffers[i] != 'FREE'){
			$('#offers-basket-content').append('<li class="clearfix"><a href="' + $('[data-offer-title="' + selectedOffers[i] + '"]').attr('href') + '">' + $('[data-offer-title="' + selectedOffers[i] + '"]').attr('title') + '</a> <strong>' + offers.format[offers.country][0].replace('%',offers.prices[offers.currency][selectedOffers[i]]).replace('*', offers.symbol[offers.currency]) + '</strong></li>');
		}
	}
	var displayedPrice = offers.format[offers.country][0].replace('%',toPay);
	displayedPrice = displayedPrice.replace('*', offers.symbol[offers.currency]);
	$('#offers-final-price').html(displayedPrice);
	
	if(toPay != 0){
		$('#free-offer-infos, #offers-to-pay .btn').hide();
		$('#offers-basket').show();
	}
	else
	{
		$('#free-offer-infos, #offers-to-pay .btn').show();
		$('#offers-basket').hide();
	}
	$('#offers-prices').fix('refresh');
	$(window).bind('resize');
};

offers.fixBasket = function(){
	var obj = $('#offers-prices');
	
	//LAUNCH
	init(obj);
	position(obj);
	
	//LISTENERS
	$(window).on({
		'resize'	: function(){
			init(obj);
			position(obj);
		},
		'scroll'	: function(){
			position(obj);
		}
	});
    function init(obj){
        obj.data('_posY', 50);
        obj.data('_start', $('#offers-options').find('.row').offset().top);
        obj.data('_end', ($('#form-offers').offset().top + $('#form-offers').outerHeight()) - $('#form-offers').find('.media-thumbnails').css('margin-bottom').replace('px', '') - $('#offers-prices').outerHeight());
		obj.data('_start', obj.data('_start') - $('.navbar-fixed-top').outerHeight());
		obj.data('_end', obj.data('_end') - $('.navbar-fixed-top').outerHeight());
	};
	function position(obj){
	    if(window.matchMedia("(max-width: 991px)").matches)
		{   
    	    if($(window).scrollTop() >= obj.data('_start'))
        	{	
	            obj.closest('.column-content').height(obj.outerHeight());
        	    obj.addClass('fixed');
        		if($(window).scrollTop() >= obj.data('_end'))
        		{
        			obj.css({
        				'top'		: ((obj.data('_end') + obj.data('_posY')) - $(window).scrollTop()) + 'px',
        			});
        		}
        	}
        	else
        	{
	            obj.closest('.column-content').css('height', '');
        	    obj.removeClass('fixed');
        	}
		}
	};
};

offer.fixHeader = function(){
	var fromTop		= $('#offer-detail .row').offset().top - $('.navbar').height();

	var stopOn		= $('.offer-table-options:last').offset().top - ($('#offer-table-header').height() + $('.navbar').height());
	var affixTop	= $('.offer-table-options:last').position().top - ($('#offer-table-header').height() + $('.navbar').height()) + 'px';
	var fixWidth    = $('#offer-table-header').closest('.column').width() + 'px';
	
    $('#offer-table-header').find('.arrow').each(function(){
        $(this).css({
            'border-top-color'  : $(this).closest('td').css('background-color'),
            'border-width'  : '20px ' + ($(this).closest('td').width() / 2)+ 'px 0'
        })
    });
    
	if($(document).scrollTop() >= fromTop && $(document).scrollTop() <= stopOn){
		$('#offer-table-header').removeClass('affix-top').addClass('affix').css({
		    'top'   : fixTop,
		    'width' : fixWidth
		});
		$('.offer-table-options:first').css('margin-top',$('#offer-table-header').height() + 'px');
	}
	else if($(document).scrollTop() > stopOn){
		$('#offer-table-header').removeClass('affix').addClass('affix-top').css('top', affixTop);
	}
	else
	{
		$('#offer-table-header').removeClass('affix').removeClass('affix-top').css({
		    'top'   : '',
		    'width' : ''
		});
		$('.offer-table-options:first').css('margin-top','');
	}
};

var onSubmit = function(token) {
    return new Promise(function(resolve, reject) {

        if (grecaptcha === undefined) {
            reject('reCAPTCHA n\'a pas pu être chargé correctement');
        }

        var response = grecaptcha.getResponse();

        if (!response) {
            reject('Une erreur est arrivée');
        }
        is_captcha_ok = true;
        var $statutConnexion = $("#connexionStatus");
        $.ajax({
            type: "POST",
            url: "//www.e-monsite.com/sessions/checklogin",
            data: $('form#connexionManager').serialize(),
            dataType: "json",
            beforeSend: function() {
                $statutConnexion.removeClass('ems_error_label');
                var emsLoader = "<div class=\"lds-ring\"><div></div><div></div><div></div><div></div></div>";
                $("#ems_connect_button").html(emsLoader);
            },
            success: function(response) {
                if (response.valid === true) {
                    $("#connexionManager").submit();
                    resolve();
                } else {
                    is_captcha_ok = false;
                    grecaptcha.reset();
                    $statutConnexion.addClass('ems_error_label');
                    $("#ems_connect_button").text("Se connecter").removeClass('auth disabled');
                    $("#connexionManager input").addClass('error');
                    reject("Nom d'utilisateur ou mot de passe incorrect");
                }
            },
            error: function() {
                is_captcha_ok = false;
                grecaptcha.reset();
                $statutConnexion.addClass('ems_error_label');
                $("#ems_connect_button").text("Se connecter").removeClass('auth disabled');
                $("#connexionManager input").addClass('error');
                reject("La communication avec le système d'authentification n'a pas pu être établie. Veuillez réessayer.");
            }
        });
    }).catch(function(e) {
        $("#connexionStatus").html(e);
    });
};
