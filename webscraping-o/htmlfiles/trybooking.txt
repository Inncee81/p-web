        function GetSessionDates(event) {
            var dateFormat = 'dddd D MMMM YYYY';
            var dateOnlyFormat = 'D MMMM YYYY';
            var firstSessionTime = dayjs(event.EventStartDate).format(dateFormat);
            var lastSessionTime = dayjs(event.EventEndDate).format(dateFormat);
            var firstEventDate = dayjs(event.EventStartDate).format(dateOnlyFormat);
            var lastEventDate = dayjs(event.EventEndDate).format(dateOnlyFormat);

            if (typeof event.AlternateLabel !== "undefined" && event.AlternateLabel !== null) {
                return event.AlternateLabel;
            }

            if (firstEventDate === lastEventDate) {
                return lastSessionTime;
            } else {
                return "Between " + firstSessionTime + " and " + lastSessionTime;
            }
        }

        function ConfigureFeaturedEvent(tabName, event, isMobile, eventCounter, isFeatured, isNearby, isEo) {
            var containerId = event.Id + "-container";

            var featuredEventId = event.Id + "-item";
            var featuredEventImgId = event.Id + "-image";
            var featuredEventDetailsId = event.Id + "-deatils";

            if (isFeatured) {
                containerId = containerId + "-featured";
                featuredEventLinkId = featuredEventLinkId + "-featured";
                featuredEventId = featuredEventId + "-featured";
                featuredEventImgId = featuredEventImgId + "-featured";
                featuredEventDetailsId = featuredEventDetailsId + "-featured";
            }

            if (isNearby) {
                containerId = containerId + "-nearby";
                featuredEventLinkId = featuredEventLinkId + "-nearby";
                featuredEventId = featuredEventId + "-nearby";
                featuredEventImgId = featuredEventImgId + "-nearby";
                featuredEventDetailsId = featuredEventDetailsId + "-nearby";
            }

            if (isEo) {
                containerId = containerId + "-eo";
                featuredEventLinkId = featuredEventLinkId + "-eo";
                featuredEventId = featuredEventId + "-eo";
                featuredEventImgId = featuredEventImgId + "-eo";
                featuredEventDetailsId = featuredEventDetailsId + "-eo";
            }

            // Create the containers of a featured event under `featured-events-section`
            if (isMobile) {
                containerId = containerId + "-mobile";
                featuredEventLinkId = featuredEventLinkId + "-mobile";
                featuredEventId = featuredEventId + "-mobile";
                featuredEventImgId = featuredEventImgId + "-mobile";
                featuredEventDetailsId = featuredEventDetailsId + "-mobile";

                if (eventCounter === 1) {
                    $("#" + tabName).append("<div id=\"" + containerId + "\" class=\"item active\"></div>");
                } else {
                    $("#" + tabName).append("<div id=\"" + containerId + "\" class=\"item\"></div>");

            } else {
                $("#" + tabName).append("<div id=\"" + containerId + "\" class=\"col-md-4 col-sm-6\"></div>");
            }

            // Create a link to the featured event
            $("#" + containerId).append("<div id=\"" + featuredEventLinkId + "\"></div>");
            $("#" + featuredEventLinkId).append("<div id=\"" + featuredEventId + "\" class=\"featured-event-slider-item\"></div>");

            $("#" + featuredEventId).append("<div id=\"" + featuredEventImgId + "\" class=\"featured-event-img-container\"></div>");

            if (event.ThumbnailImage || !event.HeroImage) {
                if (event.ThumbnailImage) {
                    $("#" + featuredEventImgId).append("<a href=\"book/event?eid=" + event.Id + "\" target=\"_blank\" class=\"featured-event-link\"><img src=\"" + event.ThumbnailImage + "\" class=\"featured-event-img\" alt=\"Thumbnail Image\" loading=\"lazy\" /></a>");

        function CreateFeaturedEventsCarousel(eventCounter, carouselContainerName, carouselContainerNameMobile, events, isFeatured, isNearby, isEo) {
            // Carousel for iPad
            if (width === 991 || width === 810 || width === 768) {
                if (eventCounter <= 2) {
                    ConfigureFeaturedEvent(carouselContainerName, events, false, eventCounter, isFeatured, isNearby, isEo);
                } else if (eventCounter <= 4) {
                    ConfigureFeaturedEvent(carouselContainerName + "2", events, false, eventCounter, isFeatured, isNearby, isEo);
                } else {
                    ConfigureFeaturedEvent(carouselContainerName + "3", events, false, eventCounter, isFeatured, isNearby, isEo);
                }

                // Create carousel items for mobile
                ConfigureFeaturedEvent(carouselContainerNameMobile, events, true, eventCounter, isFeatured, isNearby, isEo);

                return;
            }

            // Create carousel items for desktop
            if (eventCounter <= 3) {
                ConfigureFeaturedEvent(carouselContainerName, events, false, isFeatured, isNearby, isEo);
            } else {
                ConfigureFeaturedEvent(carouselContainerName + "2", events, false, isFeatured, isNearby, isEo);
            }

            // Create carousel items for mobile
            ConfigureFeaturedEvent(carouselContainerNameMobile, events, true, eventCounter, isFeatured, isNearby, isEo);
        }

        function CreateFeaturedEventsCarouselIndicators(carouselContainerNameMobile, carouselIndicatorsContainerName, eventCount) {
            for (var i = 0; i < eventCount; i++) {
                if (i === 0) {
                    $("#" + carouselIndicatorsContainerName).append("<li data-target=\"#" + carouselContainerNameMobile + "\"" + " data-slide-to=\"" + i + "\"" + " class=\"\active\"></li>");
                } else {
                    $("#" + carouselIndicatorsContainerName).append("<li data-target=\"#" + carouselContainerNameMobile + "\"" + " data-slide-to=\"" + i + "\"" + " class=\"\"></li>");
                }
            }
        }

        function GetSuburbSuggestions(postCode) {
            var suburbSuggestions = [];
            $.get("/api/v2/search/getSuburbsByFilterCriteria?filterCriteria=" + postCode + "&maxCount=5", function (data) {
                $.each(data, function (e, obj) {
                    var suggestion = {
                        label: obj.Locality + ", " + obj.Postcode,
                        value: obj.Locality
                    };

                    suburbSuggestions.push(suggestion);
                });
            });

            return suburbSuggestions;
        }

        function DisplayFeaturedEvents() {
            $("#featured-events-container").show();
            $("#nearby-events-container").hide();
            $("#eo-events-container").hide();

            $("#featured-events-mobile #featured-events-container").show();
            $("#featured-events-mobile #nearby-events-container").hide();
            $("#featured-events-mobile #eo-events-container").hide();

            $("#featured-events-prev-button").attr("href", "#featured-events-carousel-container");
            $("#featured-events-next-button").attr("href", "#featured-events-carousel-container");
        }

        function DisplayNearbyEvents() {
            $("#nearby-events-container").show();
            $("#featured-events-container").hide();
            $("#eo-events-container").hide();

            $("#featured-events-mobile #nearby-events-container").show();
            $("#featured-events-mobile #featured-events-container").hide();
            $("#featured-events-mobile #eo-events-container").hide();

            $("#featured-events-prev-button").attr("href", "#nearby-events-carousel-container");
            $("#featured-events-next-button").attr("href", "#nearby-events-carousel-container");
        }

        function DisplayEoEvents() {
            $("#eo-events-container").show();
            $("#featured-events-container").hide();
            $("#nearby-events-container").hide();

            $("#featured-events-mobile #eo-events-container").show();
            $("#featured-events-mobile #featured-events-container").hide();
            $("#featured-events-mobile #nearby-events-container").hide();

            $("#featured-events-prev-button").attr("href", "#eo-events-carousel-container");
            $("#featured-events-next-button").attr("href", "#eo-events-carousel-container");
        }

        function GetFeaturedEvents() {
            $.get("/api/v2/search/featuredEventsByIPAddress?maxEventCount=6", function (data) {
                var eventCount = data.length;
                if (eventCount === 0) {
                    $("#nav-featured-events").hide();
                    $("#nav-nearby-events").addClass("active");
                    DisplayNearbyEvents();
                    return;
                }

                var counter = 1;
                $.each(data, function (e, obj) {
                    CreateFeaturedEventsCarousel(counter, "featured-events-carousel", "featured-events-carousel-inner", obj, true, false, false);
                    counter = counter + 1;
                });

                // Create carousel indicators for mobile
                CreateFeaturedEventsCarouselIndicators("featured-events-carousel-container-mobile", "featured-events-carousel-indicators", eventCount);

                if (width === 991 || width === 810 || width === 768) {
                    if (eventCount <= 2) {
                        $("#featured-events-carousel-item2").removeClass("item");
                        $("#featured-events-carousel-item3").removeClass("item");
                    } else if (eventCount > 4){
                        $("#featured-events-carousel-item3").addClass("item");
                    }
                } else {
                    if (eventCount <= 3) {
                        $("#featured-events-carousel-item2").removeClass("item");
                        $("#featured-events-carousel-item3").removeClass("item");
                    }
                }

                $("#nav-featured-events").show();
                $(".featured-events").show();
            })
                .fail(function () {
                    $("#nav-nearby-events").addClass("active");
                    $("#featured-events-carousel-item3").removeClass("item");
                    $("#nav-featured-events").hide();
                    $(".featured-events").hide();
                    DisplayNearbyEvents();
                });
        }

        function GetNearbyEvents() {
            $.get("/api/v2/search/eventsNearbyByIPAddress?maxEventCount=6", function (data) {
                var eventCount = data.Events.length;
                if (eventCount === 0) {
                    $("#nav-nearby-events").hide();
                    return;
                }

                var events = data.Events;
                var counter = 1;
                $.each(events, function (e, obj) {
                    CreateFeaturedEventsCarousel(counter, "nearby-events-carousel", "nearby-events-carousel-inner", obj, false, true, false);
                    counter = counter + 1;
                });

                // Create carousel indicators for mobile
                CreateFeaturedEventsCarouselIndicators("nearby-events-carousel-container-mobile", "nearby-events-carousel-indicators", eventCount);

                if (width === 991 || width === 810 || width === 768) {
                    if (eventCount <= 2) {
                        $("#nearby-events-carousel-item2").removeClass("item");
                        $("#nearby-events-carousel-item3").removeClass("item");
                    } else if (eventCount > 4) {
                        $("#nearby-events-carousel-item3").addClass("item");
                    }
                } else {
                    if (eventCount <= 3) {
                        $("#nearby-events-carousel-item2").removeClass("item");
                        $("#nearby-events-carousel-item3").removeClass("item");
                    }
                }

                $("#nav-nearby-events").show();
                $(".featured-events").show();
            })
                .fail(function () {
                    $("#nearby-events-carousel-item3").removeClass("item");
                    $("#nav-nearby-events").hide();
                });
        }

        function GetEoEvents() {
            var encodedlocalStorageValue = window.localStorage.getItem("tbeos");
            if (encodedlocalStorageValue) {
                var localStorageValue = atob(encodedlocalStorageValue);
                $.get("/api/v2/search/eventOrganisers?accountIds=" + localStorageValue + "&maxEventCount=6", function (data) {
                    var eventCount = data.Events.length;
                    if (eventCount > 0) {
                        var counter = 1;
                        $.each(data.Events, function (e, obj) {
                            CreateFeaturedEventsCarousel(counter, "eo-events-carousel", "eo-events-carousel-inner", obj, false, false, true);
                            counter = counter + 1;
                        });

                        // Create carousel indicators for mobile
                        CreateFeaturedEventsCarouselIndicators("eo-events-carousel-container-mobile", "eo-events-carousel-indicators", eventCount);

                        if (width === 991 || width === 810 || width === 768) {
                            if (eventCount <= 2) {
                                $("#eo-events-carousel-item2").removeClass("item");
                                $("#eo-events-carousel-item3").removeClass("item");
                            } else if (eventCount > 4) {
                                $("#eo-events-carousel-item3").addClass("item");
                            }
                        } else {
                            if (eventCount <= 3) {
                                $("#eo-events-carousel-item2").removeClass("item");
                                $("#eo-events-carousel-item3").removeClass("item");
                            }
                        }

                        $("#nav-eo-events").show();
                        $(".featured-events").show();
                    } else {
                        $("#nav-eo-events").hide();
                        $(".featured-events").hide();
                    }
                })
                    .fail(function () {
                        $("#eo-events-carousel-item3").removeClass("item");
                        $("#nav-eo-events").hide();
                    });
            } else {
                $("#nav-eo-events").hide();
            }
        }
        
        function GetLocationFilter() {
            var locationFilter = window.localStorage.getItem("tblp");
            if (locationFilter) {
                $("#home-search-suburb").val(locationFilter);
            }
        }

        GetFeaturedEvents();
        GetNearbyEvents();
        GetEoEvents();
        GetLocationFilter();

        $("#lnkFeatured").click(function () {
            DisplayFeaturedEvents();
        });

        $("#lnkNear").click(function () {
            DisplayNearbyEvents();
        });

        $("#lnkEo").click(function () {
            DisplayEoEvents();
        });

        $("#home-search-suburb").keyup(function () {
            var suburbs = [];
            var postCode = $(this).val();
            suburbs = GetSuburbSuggestions(postCode);

            $("#home-search-suburb").autocomplete({
                source: suburbs,
                select: function (event, ui) {
                    $("#home-search-suburb").val(ui.item.value);
                    return false;
                }
            });
        });
    });
