   var rotateId = 0;

$(document).ready(function(){



  $("#next").click(function()
    {
      var numArticles = $("#feature-select article").length;
      var index = $("#feature-select article.featured").index()+1;
      var newIndex = 0;

      ( index == numArticles ) ? newIndex = 1 : newIndex = index + 1;

      $("#feature-select article.featured").fadeOut(350).removeClass("featured").attr('aria-hidden', 'true');
      $("#feature-select article:nth-child(" + (newIndex) + ")").fadeIn(350).addClass("featured").attr('aria-hidden', 'false');

      $("#dotIndicators button:nth-child(" + index + ") span").removeClass("active-dot").addClass("inactive-empty-dot");
      $("#dotIndicators button:nth-child(" + (newIndex) + ") span").addClass("active-dot").removeClass("inactive-empty-dot");

      window.clearInterval(rotateId);

      if ( $(window).width() >= 992 )
      {
        $("#pause").hide();
        $("#play").show();
      }

    });

  $("#prev").click(function()
    {
      var numArticles = $("#feature-select article").length;
      var index = $("#feature-select article.featured").index()+1;
      var newIndex = 0;

      ( index == 1 ) ? newIndex = numArticles : newIndex = index - 1;

      $("#feature-select article.featured").fadeOut(350).removeClass("featured").attr('aria-hidden', 'true');
      $("#feature-select article:nth-child(" + (newIndex) + ")").fadeIn(350).addClass("featured").attr('aria-hidden', 'false');

      $("#dotIndicators button:nth-child(" + index + ") span").removeClass("active-dot").addClass("inactive-empty-dot");
      $("#dotIndicators button:nth-child(" + (newIndex) + ") span").addClass("active-dot").removeClass("inactive-empty-dot");

      window.clearInterval(rotateId);
      if ( $(window).width() >= 992 )
      {
        $("#pause").hide();
        $("#play").show();
      }
    });


  $("#dotIndicators button").click(function()
    {
      var index = $(this).index()+1;

      $("#feature-select article.featured").fadeOut(350).removeClass("featured").attr('aria-hidden', 'true');
      $("#feature-select article:nth-child(" + (index) + ")").fadeIn(350).addClass("featured").attr('aria-hidden', 'false');

      $("#dotIndicators button span").removeClass("active-dot").addClass("inactive-empty-dot");
      $("#dotIndicators button:nth-child(" + (index) + ") span").addClass("active-dot").removeClass("inactive-empty-dot");

      window.clearInterval(rotateId);
      if ( $(window).width() >= 992 )
      {
        $("#pause").hide();
        $("#play").show();
      }
    });

  $("#pause").click(function()
    {
        $("#pause").hide();
        $("#play").show().focus();

        window.clearInterval(rotateId);
    });

  $("#play").click(function()
    {
        $("#play").hide();
        $("#pause").show().focus();

        rotateId = window.setInterval(featureNext, 10000);
    });

  $(".eventsDown").click(function()
  {

    $(".eventsPanel1").slideUp().attr('aria-hidden', 'true');;
    $(".eventsPanel2").slideDown().attr('aria-hidden', 'false');;

  });

  $(".eventsUp").click(function()
  {

    $(".eventsPanel1").slideDown().attr('aria-hidden', 'false');
    $(".eventsPanel2").slideUp().attr('aria-hidden', 'true');;

  });

  if ( $(window).width() >= 992 && $("div.features article").length > 1 )
  {
    rotateId = window.setInterval(featureNext, 10000);
  }
