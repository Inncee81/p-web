     var currentLocal = "en";
	function setCookie(cname, cvalue, exdays) {
		var d = new Date();
		d.setTime(d.getTime() + (exdays*24*60*60*1000));
		var expires = "expires="+d.toUTCString();
		document.cookie = cname + "=" + cvalue + "; " + expires + "; path=/; SameSite=Lax; Secure;";
	}

	function changeLanguage(locale){
		if (currentLocal !== locale) {
			if (locale === "en") {
				locale = "www";
			}
			if (currentLocal === "en") {
				currentLocal = "www";
			}
			var url = window.location.href;
			if (url.indexOf("?") > 0) {
				url = url + "&overrideLanguage=true";
			}
			else {
				url = url + "?overrideLanguage=true";
			}
			var hash = "#translation-message";
			document.cookie = "overrideLanguage=" + locale + "; path=/; domain=." +
					location.hostname.split('.').reverse()[1] + "." +
					location.hostname.split('.').reverse()[0] + ";";
			if (!window.location.hash) {
				url = url.replace(currentLocal, locale) + hash;
			}
			else {
				url = url.replace(currentLocal, locale).replace(window.location.hash, "") + hash;
			}
			window.location.href = url;

		}
	}

    function getCookie(cname) {
        var name = cname + "=";
        var ca = document.cookie.split(';');
        for(var i = 0; i <ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0)==' ') {
                c = c.substring(1);
            }
            if (c.indexOf(name) == 0) {
                return c.substring(name.length,c.length);
            }
        }
        return "";
    }
