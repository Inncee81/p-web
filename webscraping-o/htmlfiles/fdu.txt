   <script>
    (function($) {
						if ('object' !== typeof FWP) {
							return;
						}
						
            FWP.reset();
            var throttleTimer = null;
            var throttleDelay = 100;
            var offset = 800;
            if ( $('.loader').length ){
                $('.loader-end').hide();
            }

            $(function() {
                /** scroll */
                function scrollhandler() {
                  clearTimeout( throttleTimer );
                  throttleTimer = setTimeout(function() {
										var bottom_offset = $( window ).height() +
																				$('.mainFooter').height() +
																				parseInt($('.loader').css('margin-top')) +
																				parseInt($('.loader').css('margin-bottom'));
                    if ( $( window ).scrollTop() >= $( document ).height() - bottom_offset ) {
                        $(window).off( 'scroll', scrollhandler );
                        $('.fwp-load-more').trigger('click');
                        if (FWP.settings.pager.page >= FWP.settings.pager.total_pages - 1) {
                          $('.search-results .loader, .directory__results .loader').hide();
													$('.search-results .loader-end, .directory__results .loader-end').show();
                        }
                      }
                    }, throttleDelay );
                }
                function addhandler() {
									if ( FWP.settings.pager.page < FWP.settings.pager.total_pages ) {
                      $(window).on( 'scroll',  scrollhandler );
                      $('.search-results .loader').show();
                  } else {
										$('.search-results .loader').hide();
										$('.search-results .loader-end').show();
									}
                }

                addhandler();

                /** re-add handler after facetwp reloads */
                FWP.hooks.addAction( 'facetwp/loaded', function() { addhandler() } );
            });
        })(jQuery)
   </script>
