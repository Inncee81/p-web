   (function($) {
        $(function() {
            // Mobile Navigation Settings
            let mobileNavigationWidth = 320     // Width of mobile navigation - compare with css

            // Animate Navigation-Toggle-Button
            let mobileNavigationToggleButton = $('#mobile-navigation-toggle-button');
            let mobileNavigationOverlay = $('#mobile-navigation-overlay');
            let mobileNavigation = $('#mobile-navigation');
            let body = $('body');
            let html = $('html');

            function toggleNavigation() {
                // Close search if is open
                closeSearch();

                // Animate Navigation-Toggle-Button
                mobileNavigationToggleButton.toggleClass('active');

                // Overflow Hidden on body when Navigation is open
                body.toggleClass('mobile-navigation-open');
                html.toggleClass('mobile-navigation-open');

                // Animate Navigation
                mobileNavigation.toggleClass('active');

                // root class is active
                $('.root').addClass('active');
            }

            function closeNavigation() {
                // Animate Navigation-Toggle-Button
                mobileNavigationToggleButton.removeClass('active');

                // Overflow Hidden on body when Navigation is open
                body.removeClass('mobile-navigation-open');
                html.removeClass('mobile-navigation-open');

                // Animate Navigation
                mobileNavigation.removeClass('active');
            }

            mobileNavigationToggleButton.on('click', toggleNavigation);
            mobileNavigationOverlay.on('click', toggleNavigation);

            // Navigate between levels
            let nextLevel = $('.next-level');
            let prevLevel = $('.prev-level');

            function toNextLevel(e) {
                let thisTarget = $(this);

                thisTarget.parent().parent().parent().animate({
                    scrollTop: thisTarget.parent().parent().parent().offset().top - 60
                }, 200, function() {
                    thisTarget.parent().parent().children('ul').css('left', '0');
                    thisTarget.parent().parent().children('ul').addClass('active');
                    thisTarget.parent().parent().parent().removeClass('active');
                });

                e.preventDefault();
            }

            function toPrevLevel() {
                $(this).parent().parent().parent().parent().addClass('active');
                $(this).parent().parent().css('left', mobileNavigationWidth + 'px');
            }

            nextLevel.on('click', toNextLevel);
            prevLevel.on('click', toPrevLevel);

            // Open and close mobile search
            let mobileSearchToggleButton = $('#mobile-search-toggle-button');
            let mobileSearch = $('#mobile-search');
            let mobileSearchInput = $('#mobile-search-input');

            function toggleSearch() {
                if (mobileNavigation.is(':visible')) {
                    closeNavigation();
                }

                // Autofocus on Input
                if (mobileSearch.is(':visible')) {
                    closeSearch();
                } else {
                    openSearch();
                }
            }

            function openSearch() {
                // Open mobile search
                mobileSearch.fadeIn();

                mobileSearchToggleButton.children('i').addClass('fa-close');
                mobileSearchToggleButton.children('i').removeClass('fa-search');

                mobileSearchInput.focus();
            }

            function closeSearch() {
                // Close mobile search
                mobileSearch.fadeOut();

                mobileSearchToggleButton.children('i').removeClass('fa-close');
                mobileSearchToggleButton.children('i').addClass('fa-search');
            }

            mobileSearchToggleButton.on('click', toggleSearch)
        })
    }) (jQuery)
