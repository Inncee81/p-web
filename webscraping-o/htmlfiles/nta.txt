function kszNumChk(e) {
  if ((e.which < 48 || 57 < e.which) && e.which != 0 && e.which != 8 && e.which != 9 && e.which != 13) {
    return false;
  } else if(e.which == 13) {
    return SubmitKszForm();
  } else {
    return true;
  }
}

function SubmitKszForm() {
  var zipcode1 = document.knskForm.kszc1;
  var zipcode2 = document.knskForm.kszc2;
  var m = document.getElementById('kszMsg');

  if ((zipcode1.value == "") || (zipcode2.value == "")) {
    m.innerHTML = '<strong>郵便番号は必ず入力してください。</strong>';
    m.style.display = 'block';
    m.style.color = '#b00';
  } else if ((zipcode1.value.match(/[^0-9]+/)) || (zipcode2.value.match(/[^0-9]+/))) {
    m.innerHTML = '<strong>入力桁数誤りもしくは全角で入力しています。</strong>';
    m.style.display = 'block';
    m.style.color = '#b00';
  } else if ((zipcode1.value.length != 3) || (zipcode2.value.length != 4)) {
    m.innerHTML = '<strong>入力桁数誤りもしくは全角で入力しています。</strong>';
    m.style.display = 'block';
    m.style.color = '#b00';
  } else {
    document.knskForm.KSTYPE.value = "ksz";
    document.knskForm.submit();
    emulAcceptCharset(document.knskForm);
    return true;
  }
  return false;
}

function ksaChk(e) {
  if(e.which == 13) {
    return SubmitKsaForm();
  }
}

function SubmitKsaForm() {
  var todofuken = document.knskForm.ksaTodofuken;
  var addr = document.knskForm.ksaddr;
  var addrStr = "";

  if(addr.value != ""){
    addrStr = addr.value.replace("市区町村・町名を入力", "");
  }

  var m = document.getElementById('ksaMsg');

  if ((todofuken.value == "")) {
    if ((addrStr == "")) {
      m.innerHTML = '<strong>再度住所を入力してください。</strong>';
      m.style.display = 'block';
      m.style.color = '#b00';
    } else {
      m.innerHTML = '<strong>都道府県を選択してください。</strong>';
      m.style.display = 'block';
      m.style.color = '#b00';
    }
  } else if((addrStr == "")) {
    if ((todofuken.value == "")) {
      m.innerHTML = '<strong>再度住所を入力してください。</strong>';
      m.style.display = 'block';
      m.style.color = '#b00';
    } else {
      m.innerHTML = '<strong>市区町村名等を入力してください。</strong>';
      m.style.display = 'block';
      m.style.color = '#b00';
    }
  } else if(!(addrStr.match(/^[^ -~｡-ﾟ]*$/))) {
    m.innerHTML = '<strong>全角文字で入力してください。</strong>';
    m.style.display = 'block';
    m.style.color = '#b00';
  } else {
    document.knskForm.KSTYPE.value = "ksa";
    var str_todofuken = convertToAscii(todofuken.value);
    document.knskForm.TODOFUKEN_TO_ASCII.value = str_todofuken;
    var str_addr = convertToAscii(addrStr);
    document.knskForm.ADDR_TO_ASCII.value = str_addr;
    document.knskForm.submit();
    emulAcceptCharset(document.knskForm);
    return true;
  }
  return false;
}

function SubmitKnskForm(prm) {
  if(prm == "ksa"){
    return SubmitKsaForm();
  } else if(prm == "ksz") {
    return SubmitKszForm();
  } else {
    return false;
  }
}

function emulAcceptCharset(form) {
  if (form.canHaveHTML) {
    document.characterSet = form.acceptCharset;
  }
  return true;
}

function convertToAscii(str){
  var strToAscii = "";
  for(var i = 0; i < str.length; i++){
    var c = str.charCodeAt(i);
    if(i != 0){ strToAscii += " "; }
    strToAscii += c;
  }
  return strToAscii;
}
