    return str.length > 0 ? str[0].toUpperCase() + str.slice(1) : str;

  function extractTrackingProps(el) {
      let obj = {};

          const attrName = el.attributes[i].nodeName

            const prettyName = attrName.replace('data-tracking-', '').split('-').map(capitalize).join(' ');
