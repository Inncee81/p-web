    Reely.Carousel['TopStoriesVideo'].VideoRequestManager = {
      // Link back to parent object
      carousel: Reely.Carousel['TopStoriesVideo'],
      // First video properties
      initialDelay: null, // INIT OPTION - Initial delay in milliseconds before requesting the first video
                          // `null` value means just go with the itemInterval consideration
      timer: 0,           // Keep track of time between videos
      tick: 200,          // Time between videoRequestManager work operations
      // Subsequent video properties
      itemInterval: 2,    // INIT OPTION - Number of Items to display between Video requests
      itemCounter: 0,  // Number of Items passed between Video requests
      itemLastIndex: null,
      pauseTrackingOutOfView: false, // INIT OPTION - Don't do videoRequestManager state tracking when out of view
      // Video requested stuff
      firstVideoRequested: false, // One off track if the first Video was requested
      videoRequested: false,      // Track if Video was requested
      videoRequestCompleted: false, // Track if Video was retrieved
      videoRequestFailed: false,  // Track if Video errored or failed
      tracker: null,
      get waitForFirstVideoDelay () {
        return !this.firstVideoRequested && this.initialDelay !== null
      },
      get canUpdateRequestState () {
        if (this.pauseTrackingOutOfView) {
          return this.carousel.inView
        }
        return Reely.Carousel['TopStoriesVideo'].waitTillInViewToBegin === false
        // return true
      },
      get canRequestVideo () {
        if (!this.videoRequested) {
          if (this.waitForFirstVideoDelay) {
            // Enough time has passed to request the first one
            if(this.timer >= this.initialDelay) {
              return true 
            }
          }
          // Carousel has played enough items to request another Video
          else if (this.itemCounter > this.itemInterval) {
            return true 
          }
        }
        return false // Video is currently playing
      },
      attemptToShowVideo () {
        // logToConsole('requesting ad.')
        Reely.Carousel['TopStoriesVideo'].VideoLoader.requestAds(Reely.Carousel['TopStoriesVideo'].VideoRequest)
        this.firstVideoRequested = true
        this.videoRequested = true
        this.videoRequestCompleted = false
        this.videoRequestFailed = false
      },
      updateRequestState () {
        // logToConsole(
        //   'this.timer:',this.timer,
        //   'this.tick:',this.tick,
        //   'this.initialDelay:',this.initialDelay,
        //   'this.firstVideoRequested:',this.firstVideoRequested,
        //   'this.videoRequested:',this.videoRequested,
        //   'this.videoRequestCompleted:',this.videoRequestCompleted,
        //   'this.videoRequestFailed:',this.videoRequestFailed,
        //   'this.itemCounter:',this.itemCounter,
        //   'this.itemLastIndex:',this.itemLastIndex)

        if (this.canUpdateRequestState) {
          // Request video before updating request state
          if (this.canRequestVideo) {
            this.attemptToShowVideo()
          }

          // Update video request state
          if (this.waitForFirstVideoDelay) {
            this.timer += this.tick
          }
          else {
            if (this.videoRequested) {
              // Reset some tracking at end of request
              if (this.videoRequestCompleted) {
                this.videoRequested = false
                this.videoRequestCompleted = false
                this.videoRequestFailed = false
                this.itemCounter = 0 // Reset counter to beginning
              }
              else if (this.videoRequestFailed) {
                this.videoRequested = false
                this.videoRequestCompleted = false
                this.videoRequestFailed = false
                this.itemCounter = this.itemInterval - 1 // Reset counter to retry on the next item
              }
            }
            // Track Carousel item change
            else if (this.carousel.itemIndex !== this.itemLastIndex) {
              this.itemCounter += 1
            }
            this.itemLastIndex = this.carousel.itemIndex
          }
        }
      },
      init (options) {
        if (options) {
          this.initialDelay = options.initialDelay || this.initialDelay
          this.itemInterval = options.itemInterval || this.itemInterval
          this.pauseTrackingOutOfView = options.pauseTrackingOutOfView || this.pauseTrackingOutOfView
          this.tick = options.tick || this.tick
        }

        this.tracker = setInterval(function() {
          this.updateRequestState()
        }.bind(this), this.tick)
      },
    }

      validVolume (value) { return Math.min(Math.max(value, this.minVolume), this.maxVolume) },

        // this.logStats()
      },
      get muted () {
        return this.volume === 0
      },
      setVolume (value) {
        var value = Number.parseFloat(value)
        // logToConsole(value, typeof(value))
        if (value !== 0) {
          this.volume = this.validVolume(value)
          this.setUnMuteVolume(this.volume)
        }
        else if (value === 0 && !this.muted){
          this.setUnMuteVolume(this.volume)
          this.volume = 0
        }
        // this.logStats()
      },
      mute () {
        if (!this.muted) {
          this.setVolume(0)
          return true
        }
        // warnToConsole('-- Already muted. --')
        // this.logStats()
        return false
      },
      unMute () {
        if (this.muted) {
          this.setVolume(this.validVolume(this.unMuteVolume))
          return true
        }
        // warnToConsole('-- Already UN-muted. --')
        this.logStats()
        return false
      },
