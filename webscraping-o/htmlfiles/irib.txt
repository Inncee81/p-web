   $(document).ready(function () {
        $("input[name*='reg_']").attr('autocomplete','off');
        ///////////////////////////////////////////////////////////
        //captcha refresh
        $('.refresh').on('click',function(e){
            e.preventDefault();
            var d = new Date();
            $('.captcha  img').attr('src','/include/captcha.php?' + d.getTime());
        });
        ///////////////////////////////////////////////
        //login
        $('button#login').on('click', function (e) {
            e.preventDefault();
            e.stopPropagation();
            $('#alert-not-found').addClass('hidden');

            $('div.help-block').text('');

            var email = $('input#email').val();
            var password = $('input#password').val();
            var captcha = $('input#login_captcha').val();
            if( $('input#remember_me').is(':checked') ){
                var remember_me = '1';
            }else{
                var remember_me = '0';
            }

            $.post('/include/pages/loginProcess.php', {
                email: email,
                password: password,
                captcha: captcha,
                remember_me: remember_me
            }, function (res) {
                var JsonResult = $.parseJSON(res); /*console.log(JsonResult);*/ //alert(JsonResult);
                if( JsonResult['captcha'] != undefined )
                    $('input#login_captcha').siblings('div.help-block').text(JsonResult['captcha']);
                if( JsonResult['email'] != undefined )
                    $('input#email').siblings('div.help-block').text(JsonResult['email']);
                if( JsonResult['password'] != undefined )
                    $('input#password').siblings('div.help-block').text(JsonResult['password']);
                if( JsonResult['notFound'] != undefined )
                    if( JsonResult['notFound'] != 'found' )
                        $('#alert-not-found').text(JsonResult['notFound']).removeClass('hidden');
                    else
                        location.reload(true);
            });
        });
        $('button#reg').on('click', function (e) {
            $('#reg_alert').addClass('hidden');
            e.preventDefault();
            e.stopPropagation();
            var first_name = $('input#reg_first_name').val();
            var last_name = $('input#reg_last_name').val();
            var email = $('input#reg_email').val();
            var mobile = $('input#reg_mobile').val();
            var password = $('input#reg_password').val();
            var captcha = $('input#reg_captcha').val();
            if( $('input#get_sms').is(':checked') ){
                get_sms = '1';
            }else{
                get_sms = '0';
            }
            var post_object = {
                first_name: first_name,
                last_name: last_name,
                mobile: mobile,
                email: email,
                password: password,
                captcha: captcha,
                get_sms : get_sms
            };
            $.post('/include/pages/registerProcess.php', post_object, function (res) {
                //alert('res: ' + res);

                var JsonResult = $.parseJSON(res);
                if( JsonResult['first_name'] != undefined ){
                    $('input#reg_first_name').siblings('div.help-block').text(JsonResult['first_name']);
                }

                if( JsonResult['last_name'] != undefined ){
                    $('input#reg_last_name').siblings('div.help-block').text(JsonResult['last_name']);
                }

                if( JsonResult['captcha'] != undefined ){
                    $('input#reg_captcha').siblings('div.help-block').text(JsonResult['captcha']);
                }

                if( JsonResult['email'] != undefined ){
                    $('input#reg_email').siblings('div.help-block').text(JsonResult['email']);
                }

                if( JsonResult['password'] != undefined ){
                    $('input#reg_password').siblings('div.help-block').text(JsonResult['password']);
                }

                if( JsonResult['mobile'] != undefined ){
                    $('input#reg_mobile').siblings('div.help-block').text(JsonResult['mobile']);
                }

                if( JsonResult['other'] != undefined ){
                    $('#reg_alert').removeClass('hidden').text(JsonResult['other']);
                }

                if( JsonResult['success'] != undefined ){
                    $('#reg_alert').removeClass('hidden').removeClass('alert-danger').addClass('alert-success').text('ثبت نام با موفقیت انجام شد.');
                    $("input[name*='reg_']").val('');
                }
            });
        });

        var trigger = $('.hamburger'),
            overlay = $('.overlay'),
            isClosed = false;
        trigger.click(function () {
            hamburger_cross();
        });

        function hamburger_cross() {

            if (isClosed == true) {
                overlay.hide();
                trigger.removeClass('is-open');
                trigger.addClass('is-closed');
                isClosed = false;
            } else {
                overlay.show();
                trigger.removeClass('is-closed');
                trigger.addClass('is-open');
                isClosed = true;
            }
        }
