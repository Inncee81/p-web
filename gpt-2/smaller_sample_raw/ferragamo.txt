					}

					//desktop

						category = $.trim($(this).closest('ul').siblings().text());

					subCategory = '';
					}
			// 		debug.debug("category:"+category+" subCategory:"+subCategory);
					ga('send', 'event', 'Menu', category, subCategory);
				}
			});
		});
	}

			var gender = $(".plp__breadcrumb-list--r19").attr("data-parent-top-cat");
			var clickedLabel = "";
			if (gender !== undefined && gender !== "") {
				clickedLabel = gender + " " + $.trim($( this ).text());
			} else {
				clickedLabel = $.trim($( this ).text());
			}
	//		debug.debug(clickedLabel);
			
			var elSeparator = " | ";

	// 		var allBreadcrumb = breadcrumbStep1 + elSeparator + breadcrumbStep2 + elSeparator + breadcrumbStep3;
			var allBreadcrumb = "";
			$('.plp__breadcrumb-item--r19').each(function(index,item) {
				//ciclo tutti gli elementi con class "ga-breadcrumb-step<N>" per costruire il breadcrumb intero
				if (allBreadcrumb === "" || (allBreadcrumb.lastIndexOf(elSeparator) === allBreadcrumb.length - elSeparator.length) > 0) {
					allBreadcrumb = allBreadcrumb + $.trim($(item).find('a').text());
				} else {
					if(index === ($('.plp__breadcrumb-item--r19').size() - 1)){
						allBreadcrumb = allBreadcrumb + elSeparator + $.trim($(item).text());
					}else{
						allBreadcrumb = allBreadcrumb + elSeparator + $.trim($(item).find('a').text());
					}
				}
			});
	//		debug.debug(allBreadcrumb);

			ga('send', 'event', clickedLabel, allBreadcrumb, B_GOOGLE_ANALYTICS.frgCanonicalPath);
		});
	}
