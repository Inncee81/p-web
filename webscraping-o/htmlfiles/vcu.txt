function shouldBeVisible({ visibility }) {
    // If there's no visibility set, return false...
    if (!visibility) return false;
    // ... otherwise, return whether the visibility includes "vcu" or not
    return visibility.includes('vcu');
}

function withinDateRange({ start, end }) {
    // If there's no date range set, return false
    if (!start && !end) return false;
    // Generate a JS date object from the alert's start/end datestamps...
    const startDate = new Date(start),
        endDate     = new Date(end),
    // ... as well as the current time
    currentDate = new Date();
    // Return the condition that the start date is before/at the current date and the end date after/at the current date
    return currentDate >= startDate && currentDate <= endDate;
}

function loadAlerts() {
    
    var alertUrl = 'https://alert.vcu.edu/alerts.json';

    var $alert = $('.alert-section');
    jQuery.ajax(
      {
        url: alertUrl,
        dataType: 'json',
        crossDomain: true ,

      }).done(function(data) {

	      //console.log('Alert data', data);

        data = data.filter(shouldBeVisible).filter(withinDateRange);

        //console.log('Filtered alert data', data);

	
        var alertTitle = "";
        var alertCutline = "";
        var itemCount = (data) ?  data.length : 0;

        if (itemCount == 0) {
            return;
        }
        $alert.empty();
 	
        data.forEach(function (elem, i, array) {
        
          //console.log('Alert item', elem);
          alertTitle = elem.title;
          alertCutline = elem.cutline;
          
          if (!alertTitle) {
            return true;
          }
          
          var $item = $('<dl></dl>');

          if (i === 0) {
            $item.append($('<dt class="important-msg-header">').text('VCU Alerts'));
          } else {
            $item.append($('<dt class="important-msg-header">').text('VCU Alerts'));
          }
         
          var $info = $('<dd class="important-msg-message">');
          $info.append(alertTitle + ' ');

          if (alertCutline) {
            $info.append(' &ndash; '+ alertCutline + " ");
          }

          $info.wrapInner('<p>');
          $item.append($info);

          var $more = $('<dd></dd>', { class: 'important-msg-more', appendTo: $item});
          $more.append('<a href="https://alert.vcu.edu" class="btn btn-pill btn-large important-msg-box-link" aria-label="Read more on VCU Alert website"><strong>Read more</strong></a>');

          $item.append($more);
          	
          
          $alert.append($item);
        });

    });
}
