      window.onload = function() {

  var slide= document.getElementById("slideRotate");
  
  if (slide) {
  
    var container = document.querySelector('[data-block-name="img-rotator"]');
    var arr = Array.prototype.slice.call(slide.children);
    var parentNode = slide.parentNode.parentNode;
    var imgElem = document.createElement("img");
  
    imgElem.src = arr[0].children[0].firstElementChild.src;
    parentNode.insertBefore(imgElem, parentNode.firstChild);
  
    var current = 0

    arr[current].classList.add('show');
  
    container.classList.add('fadein');
  
    var slideTimer = setInterval(rotateSlides, 6000);

    slide.addEventListener("mouseover", function(){ clearInterval(slideTimer)});
    slide.addEventListener("mouseout", function(){ slideTimer = setInterval(rotateSlides, 6000);});

    function rotateSlides() {
      
      for (var i = 0; i < arr.length; i++) {
        arr[i].classList.remove('show');
      }

      current = (current != arr.length -1) ? current + 1 : 0;
      arr[current].classList.add('show');
      
    }
  
  }

}
