function link_topimage () {
	var tic_dest = '//www.uzh.ch/cmsssl/de/outreach/events/rv/2021fs/zazh-naturkatastrophen.html'; 
	$('.mod-top-image-rotator').css('cursor','pointer');

		window.location.href = tic_dest;
	});
}

function link_news_teaser () {
	// find all items of UZH News RSS-Feed
	var el = $('h2:contains("News")').next().children();
        //console.log (el);
       $('h2:contains("News")').find('a').attr('href','https://www.news.uzh.ch');
$('h3:contains("Alle News")').find('a').attr('href','https://www.news.uzh.ch');
	
	// add pointer to images
	$(el).find('img').css('cursor','pointer');
	
	// add click-action to images
	$(el).find('img').on('click', function () {
		$(this).css('cursor','pointer');
		var p = $(this).parents('li');
		var url = $(p).find('a').attr('href');
		window.location.href=url;
	});
}
