var fsidSignupLinkModifier = function() {
  if (typeof(URL) === 'undefined'
    || typeof(URLSearchParams) === 'undefined'
    || typeof(localStorage) === 'undefined'
  ) {
    // Don't bother supporting really old browsers
    return;
  }

  document.addEventListener('DOMContentLoaded', function () {
    const localStorageToFsidMappings = {
      gspk: 'gspk',
      utm_content: 'internal_utm_content',
      utm_campaign: 'internal_utm_campaign',
      utm_source: 'internal_utm_source',
      utm_medium: 'internal_utm_medium',
      utm_terms: 'internal_utm_terms',
    };

    const attributesToInclude = {};
    Object.keys(localStorageToFsidMappings).forEach(function(localStorageKey) {
      var fsidKey = localStorageToFsidMappings[localStorageKey];
      var value = localStorage.getItem(localStorageKey);
      if (value) {
        attributesToInclude[fsidKey] = value;
      }
    });

    if (Object.keys(attributesToInclude).length === 0) {
      // Nothing to send along so no need to modify links;
      return;
    }

    document.querySelectorAll('a[fsid-params-passthrough~="true"]').forEach(function(link) {
      var href = link.href;
      var url = new URL(href);
      var searchParams = new URLSearchParams(url.search);

      Object.keys(attributesToInclude).forEach(function(key) {
        var value = attributesToInclude[key];
        searchParams.append(key, value);
      });

      url.search = searchParams.toString();

      link.href = url.toString();
    })
  });
}

fsidSignupLinkModifier();
