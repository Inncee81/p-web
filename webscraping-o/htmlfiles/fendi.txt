function updateFendiLabels(){
	var lang = $("html").attr("lang");
	if(lang!=null){
		var labels = fendi_labels[lang];
		if(labels!=null){
			$("[fendi-label]").each(function(){
				var k = $(this).attr("fendi-label");
				var v = labels[k];
				if(v!=null){
					$(this).html(v);
					$(this).removeAttr("fendi-label");
				}
			});
		}
	}
}

$(function(){
	updateFendiLabels();
});
