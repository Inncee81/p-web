      function getCookie(key) {

	var result = "";

	var cookies = document.cookie ? document.cookie.split('; ') : [];

	for (var i = 0, l = cookies.length; i < l; i++) {
		var parts = cookies[i].split('=');
		var name = decodeURIComponent(parts.shift());
		var cookie = parts.join('=');

		// Prevent storing or parsing of a cookie that we couldn't decode.
		if ( cookie && key === name ) {

			// removes surrounding quotes
			if ( cookie.indexOf('"') === 0) {
				// This is a quoted cookie as according to RFC2068, unescape...
				cookie = cookie.slice(1, -1).replace(/\\"/g, '"').replace(/\\\\/g, '\\');
			}

			// parse utf8 characters
			// cannot use decode_utf8 because it will escape the value
			try {
				cookie = decodeURIComponent(cookie);
			} catch (e){ }

			result = cookie;
			// we found it, don't bother looking at other cookies
			break;
		}
	}

	return result;
};

    };

            for (index = 0; index < hashes.length; index++) {

            var currentDate = new Date;
            var currentCookieValue;

            // thus 86400000 = 24 hours = 1 day
            var interval = timeUnit == 'm' ? 60000 : 86400000;
            cookieName = cookieName ? cookieName : 'digitaldata_gvo';
            value = value ? value : '';
            expiry = expiry ? expiry : 0;
            currentCookieValue = this.getCookie(cookieName);
            if (value) {

        getCookie: function (key) {

            var result = "";
            var cookies = document.cookie ? document.cookie.split('; ') : [];
            for (var i = 0; i < cookies.length; i++) {
                var parts = cookies[i].split('=');
                var name = decodeURIComponent(parts.shift());
                var cookie = parts ? parts.join('=') : [];

                // Prevent storing or parsing of a cookie that we couldn't decode.
                if (cookie && key === name) {

                    // removes surrounding quotes
                    if (cookie.indexOf('"') === 0) {
                        // This is a quoted cookie as according to RFC2068, unescape...
                        cookie = cookie.slice(1, -1).replace(/\\"/g, '"').replace(/\\\\/g, '\\');
                    }

                    // parse utf8 characters
                    try {
                        cookie = decodeURIComponent(cookie);
                    } catch (e) {}

                    result = cookie;
                    // we found it, don't bother looking at other cookies
                    break;
                }

            return result;

         */
        initted: false,
        /**
         * @return {Array} method returns an array of a given object's key properties
         */
        init: function () {
            
            this.emailCampaignRecord();

        },
        
        emailCampaignRecordFired: false,
        
        emailCampaignRecord: function () {
        	// Don't fire the record action if the merchandising data is not initialized
        	if (digitalData.merchandisingDataInitted == false) {
        		console.log("digitalDataAnalyticsListener: Not ready to fire email campaing record")
        		return;
        	}
        	
        	// Don't fire the record action if there was already a record event
        	if (this.emailCampaignRecordFired == true) {
        		console.log("digitalDataAnalyticsListener: Email campaign record was successfully fired once before")
        		return;
        	}
        	
            var mobject = digitalData.merchandising;
            var emid = mobject.emid;

            if (typeof emid == "undefined" || emid == null || emid == "") {
            	console.log("digitalDataAnalyticsListener: Email campaign tracking NOT triggered as emid is not present");
            	return;
            }

        	console.log("digitalDataAnalyticsListener: Email campaign tracking recored about to fire");

            $.ajax({
                headers: {
                    "Content-Type": "application/json; charset=utf-8"
                },
                url:environmentURL.DATAHUB+'/datahub/api/v1/datalayer/record',
                dataType: 'json',
                type: "POST",
                contentType: 'application/json',
                data: JSON.stringify( {
                	"hit_domain":window.location.hostname,
                    "referrer":mobject.referrer,
                    "analyticsVisitorid":mobject.analyticsVisitorID,
                    "marketingCloudid":mobject.marketingCloudVisitorID,
                    "dmOfferSessionid":mobject.sessionId,
                    "s_vi_low":mobject.s_vi_low,
                    "s_vi_high":mobject.s_vi_high,
                    "icid":mobject.icid,
                    "cid":mobject.cid,
                    "emid":mobject.emid,
                    "amoid":mobject.amoid,
                    "formid":digitalData.form,
                    "gigya_uid":digitalData.user.gigyaId
                } ),
                success : function(response) {
                    console.log( 'digitalDataAnalyticsListener: Success', response );
                    digitalDataAnalyticsListener.emailCampaignRecordFired = true;
                },
                error : function(err) {
                    console.log( 'digitalDataAnalyticsListener: Data not persisted', err );
                    digitalDataAnalyticsListener.emailCampaignRecordFired = false;
                }
            });
        }
    }
})();
window.digitalDataAnalyticsListener.init();
