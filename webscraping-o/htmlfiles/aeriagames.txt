// check for HTML5 support and disable for mobile devices
function isCanvasSupported() {
    var iPadAgent = navigator.userAgent.match(/iPad/i) != null;
    var iPodAgent = navigator.userAgent.match(/iPhone/i) != null;
    var AndroidAgent = navigator.userAgent.match(/Android/i) != null;
    var webOSAgent = navigator.userAgent.match(/webOS/i) != null;

    if(iPadAgent || iPodAgent || AndroidAgent || webOSAgent) return false;

    var elem = document.createElement('canvas');
    return !!(elem.getContext && elem.getContext('2d'));
}

// inject HTML5 video into target
function injectHTML5Content(slideNum){
    var html5VideoName = $('slide'+ slideNum).getProperty('rel');

    /*
    When using the new resource reference generate the s3 links
     */
    var s3_type = '';
    if (html5VideoName.search(/^\/\/files\//) > -1){ //check if rel starts wil //files/
        s3_type = AG_S3.S3_METHOD_CMS_CONTENT;
        html5VideoName = html5VideoName.replace(/^\/\/files\//, 'files/'); // Remove //at the beginning of string
    }

    if(html5VideoName.indexOf(".mp4") != -1) html5VideoName = html5VideoName.split(".mp4")[0];
    else if(html5VideoName.indexOf(".ogv") != -1) html5VideoName = html5VideoName.split(".ogv")[0];

    var contentInject = '<video width="' + slideWidth + '" height="510" loop="loop" autoplay="autoplay" preload="auto" autobuffer>' +
        '<source src="' + AG_S3.get_url(s3_type, html5VideoName + '.mp4') + '" type="video/mp4" />' +
        '<source src="' + AG_S3.get_url(s3_type, html5VideoName + '.ogv') + '" type="video/ogg;codecs=theora,vorbis" />' +
        '</video>';

    $('slide' + slideNum + '-html5').set('html',contentInject).fade('out');
}

// play or pause HTML5 video of slide
function controlHTML5Content(type,slideNum){
    var slideID = $('slide' + slideNum + '-html5');
    if(type == 'pause'){
        slideID.fade('out');
        slideID.getElement('video').pause();
    }
    else if(type == 'play'){
        slideID.fade('in');
        slideID.getElement('video').play();
    }
}
