        function submitEvent() {

            if (!eventSubmitted) {
        
                eventSubmitted = true;
			    stopTimer();
                var link = document.createElement('a');
                link.href = href;
        		if (a_target !== "") {
					link.setAttribute('target', a_target);
        		}
                document.body.appendChild(link);
                link.click();
        
            }
        }
		
		function stopTimer() {
			clearTimeout(submitTimeout);
		}
}
