         function changeTab(e) {
		var tabs = $('searchNav').childElements();
		var toTab = Event.element(e).up('li');

		tabs.each(function(tab){
			if(toTab == tab) tab.addClassName("current");
			else tab.removeClassName("current");
		});

		var myButton = $("search-button");
		var myButtonContent = myButton.down().down();
		if (toTab.id == "tab-transfer") {
			myButtonContent.update("Next");
			myButton.removeClassName("find-large");
			myButton.addClassName("login-large");
			$("bulk-transfer").show();
			$("search").action = "https://cjb.shopco.com/transfer/search";
		} else if (toTab.id == "tab-names") {
			myButtonContent.update("Find");
			myButton.removeClassName("login-large");
			myButton.addClassName("find-large");
			$("bulk-transfer").hide();
			$("search").action = "https://cjb.shopco.com/search";
		} else {
			myButtonContent.update("Find");
			myButton.removeClassName("login-large");
			myButton.addClassName("find-large");
			$("bulk-transfer").hide();
			$("search").action = "https://cjb.shopco.com/search";
		}

		var myInput = $("query");
		if (toTab.id == "tab-domain") myInput.defaultValue = "add some keywords, your location, your name, etc.";
		else if (toTab.id == "tab-transfer") myInput.defaultValue = "enter the domain to transfer here";
		else myInput.defaultValue = "enter your first and last name";

	}
	document.observe("dom:loaded", function() {
		$('searchNav').childElements().each(function(tab) {
			tab.down('a').observe('click', changeTab);
		});
	});
        </script>
