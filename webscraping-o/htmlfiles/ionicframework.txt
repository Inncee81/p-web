window.testimonials = function() {
  // Testimonials rotation
  const $testimonials = document.getElementById('testimonials');
  let testimonialIntervalID = null;
  let queued = false;
  const $highlight = document.getElementById('testimonials__highlight');

  function activateNextQuote() {
    const $oldSpeaker = deactivateSpeaker();
    let $active = null;
    // go back to the beginning if at the end
    if (!$oldSpeaker.nextElementSibling) {
      $active = $testimonials.querySelector('.quotees li:first-child');
    } else {
      $active = $oldSpeaker.nextElementSibling;
    }

    setActive($active);
  }
  
  // what happens if a speaker is clicked
  function handleClick(event) {
    deactivateSpeaker();

    // activate the clicked speaker
    setActive(event.currentTarget);

    // stop the rotation for a bit
    // the queued check prevents the rotation from starting multiple times
    // if the user clicks another speaker before the rotation restarts
    clearInterval(testimonialIntervalID);
    if (! queued) {
      queued = true;
      setTimeout(start, 13000);
    }
  }

  function start() {
    queued = false;
    testimonialIntervalID = setInterval(activateNextQuote, 8000);
  }

  // finds and deacivates just the current speaker, not the quote
  function deactivateSpeaker() {
    const $speaker = $testimonials.querySelector('.quotees .is-active')
    $speaker.classList.remove('is-active');
    return $speaker;
  }

  // add is-active classes to the passed quotee and their related quote
  function setActive($speaker) {
    $speaker.classList.add('is-active');
    // deactivate the old quote first
    $testimonials.querySelector('blockquote.is-active').classList.remove('is-active');
    $testimonials.querySelector(`blockquote.${$speaker.dataset.speaker}`).classList.add('is-active');
    updateHighlight($speaker);
  }

  // update the speaker highlight position
  function updateHighlight($active) {
    // get $active element if it is not passed
    if (!$active) {
      $active = $testimonials.querySelector('.quotees .is-active');
    }
    // console.log('updating');

    const $ul = $testimonials.querySelector('.quotees');
    $highlight.style.top = `${$active.offsetTop + $ul.offsetTop}px`;
    $highlight.style.left = `${$active.offsetLeft + $ul.offsetLeft}px`;
    $highlight.style.height = `${$active.offsetHeight}px`;
    $highlight.style.width = `${$active.offsetWidth}px`;
  }

  function addListeners() {
    for (let $quotee of $testimonials.querySelectorAll('.quotees li')) {
      $quotee.addEventListener('click', handleClick);
    }
  }

  // check the highlight position when the page resizes
  let resizeTimer;
  window.addEventListener('resize', function(e) {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(updateHighlight, 150);
  });

  // kick things off
  updateHighlight();
  addListeners();
  start();
  }();
