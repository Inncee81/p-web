    jQuery(document).ready(function () {
        jQuery("#mob-main-nav").hide();

        var anchors = getAnchorsWithId();

        anchors.map((anchor) => {
            anchor.classList.add('anchor-target');
        });

        function getAnchorsWithId() {
            var anchorsWithId = [];

            jQuery('a[href*="#"]').toArray().map((item) => {
                var aWithId = jQuery('a[id]').toArray().filter((aTag) => { return ('#' + aTag.id) == item.hash });
                if (aWithId.length > 0) {
                    anchorsWithId.push(aWithId[0]);
                }
            });

            return anchorsWithId
        }

        function handleTab() {
            var esc = jQuery.Event("keydown", { keyCode: 27 });
            function hideMegaMenu() {
                hideMask("rwdOverlayMask");
                jQuery('.mm-dropdown').removeClass('open');
                jQuery('.mm-menu').removeClass('open');
                jQuery('.mm-li').removeClass('fadee');
                jQuery('#fullLogo').show();
                jQuery('#leafLogo').hide();
                jQuery('body').removeClass('mm-open');
                hideMenu();
            }

            jQuery(document).on("keydown", "body.mm-open .mm-container", function (e) {
                var focused = jQuery(':focus');
                if (e.keyCode == 27) {
                    hideMegaMenu();
                }
            });

            jQuery(document).on("keydown", ".mm-dropdown > a", function (e) {
                if (e.keyCode == 13) {

                } else if (e.keyCode == 9) {
                    if (!jQuery('body').hasClass('mm-open')) {
                        hideMegaMenu();
                    }
                }
            });

            jQuery(".mm-container .mm-menu").each(function () {
                var lastLink = jQuery(this).find("a").last();
                jQuery(lastLink).on("keydown", function (e) {
                    console.log(lastLink, ' keydown');
                    if (e.keyCode == 9) {
                        if (!e.shiftKey) {
                            jQuery(".mm-container").trigger(esc);
                        }
                    }
                });
            });
        }

        jQuery(window).on('load', function () {
            console.log('loaded');
            handleTab();
        });

        jQuery('.mm-header').on('click', '.mm-dropdown > a', function (e) {
            e.preventDefault();
            var mmLink = jQuery(this);
            var parentLi = mmLink.closest('li');
            var isOpen = parentLi.hasClass('open');
            var mmId = mmLink.data('mm-id');
            var mmMenu = jQuery('#mm-' + mmId);

            jQuery('.mm-dropdown').not(parentLi).removeClass('open');
            jQuery('.mm-li').not(parentLi).addClass('fadee');
            jQuery('.mm-menu').not(mmMenu).removeClass('open');

            if (mmMenu) {
                if (isOpen) {
                    hideMask("rwdOverlayMask");
                    parentLi.removeClass('open');
                    mmMenu.removeClass('open');
                    jQuery('.mm-li').removeClass('fadee');
                    jQuery('#fullLogo').show();
                    jQuery('#leafLogo').hide();
                    jQuery('body').removeClass('mm-open');
                } else {
                    showMask("rwdOverlayMask");
                    parentLi.addClass('open');
                    mmMenu.addClass('open');
                    jQuery('#leafLogo').show();
                    jQuery('#fullLogo').hide();
                    jQuery('body').addClass('mm-open');
                }
            }

            hideSearch();
        });

        jQuery('.srch-toggle').click(function () {
            toggleSearchForm();
        });

        jQuery('.mm-toggle-menu').click(function () {
            jQuery('.mm-container').toggleClass('mm-menu-expanded');
            if (jQuery('.mm-container').hasClass('mm-menu-expanded')) {
                showMask("rwdOverlayMask");
                hideSearch()
                jQuery('.topnav').show();
            } else {
                hideMask("rwdOverlayMask");
                jQuery('.topnav').hide();
            }
        });

        function toggleSearchForm() {
            jQuery('#srchForm').toggle();
            hideMenu();

            resetMegaMenu();

            if (!$j('#srchForm').is(':hidden')) {
                showMask("rwdOverlayMask");
            } else {
                hideMask("rwdOverlayMask");
            }
        }

        function resetMegaMenu(li, menu) {
            li = li || jQuery('.mm-li');
            menu = menu || jQuery('.mm-menu');
            li.removeClass('open');
            menu.removeClass('open');
            jQuery('.mm-li').removeClass('fadee');
        }

        function hideMenu() {
            jQuery('.mm-container').removeClass('mm-menu-expanded');
        }

        function hideSearch() {
            jQuery('#srchForm').hide();
        }

        function resetMegaMenuElements() {
            hideMenu();
        }

        jQuery(window).scroll(function () {

            if (!jQuery('.mm-header').hasClass('sticky')) {
                jQuery('.anchor-target-sticky').attr('class', 'anchor-target');
            }
            else {
                jQuery('.anchor-target').attr('class', 'anchor-target-sticky');
            }

            if ((jQuery(document).height() - 235) > jQuery(window).height()) {
                if (jQuery(this).scrollTop() > 70) {
                        jQuery('.mm-header').addClass('sticky');
                    if (Modernizr.mq('(min-width: 996px)')) {
                        jQuery('.mm-header').addClass('s-xl');
                    }
                } else {
                        jQuery('.mm-header').removeClass('sticky');
                        jQuery('.mm-header').removeClass('s-xl');
                }
            }
        });

    });

    jQuery(window).resize(function () {
        jQuery('.topnav').hide();
        if (Modernizr.mq('(min-width: 996px)')) {
            if (jQuery('.mm-header').hasClass('sticky')) {
                jQuery('.mm-header').removeClass('sticky');
            }
        }
    });
