        function SetAddPax(objElement, prefix) {
            var value = jQuery('input[name="' + objElement + '"]').val();

            value = (value == "" ? 0 : parseInt(value));

            if (prefix == '+') {
                value = value + 1;
            } else if (prefix == '-') {
                value = value - 1;
            }

            jQuery('input[name="' + objElement + '"]').val(value);
            jQuery('#' + objElement).text(value);
            var selectedClass = jQuery("input[name='class']:checked").attr('id');
            jQuery("#passangers").val(getTotalPax() + " Travelers, " + jQuery("label[for='" + selectedClass + "']").text());

        }

        function getTotalPax() {
            var adults = jQuery('input[name="adult"]').val();
            adults = (adults == "" ? 0 : parseInt(adults));
            var childs = jQuery('input[name="child"]').val();
            childs = (childs == "" ? 0 : parseInt(childs));
            var infants = jQuery('input[name="infant"]').val();
            infants = (infants == "" ? 0 : parseInt(infants));

            var total_pax = adults + childs + infants;
            return total_pax;
