            $(".de-navbar__menu").toggleClass("de-navbar__menu--open"); //show menu

            if ($(".de-navbar__search").hasClass("de-navbar__search--open")) {
                $(".de-navbar__search").removeClass("de-navbar__search--open");

                $(".de-navbar__search-trigger").removeClass("de-navbar__search-trigger--open");
            }

        });

        // -------------------------------------------------------------------------------- 02. Search bar trigger
        $(".de-navbar__search-trigger").on("click", function(e) {
            e.preventDefault();

            $(".de-navbar__search").toggleClass("de-navbar__search--open");
            $(".de-navbar__search-trigger").toggleClass("de-navbar__search-trigger--open");
            $(".de-navbar").toggleClass("de-navbar--search-open");

            if ($(".de-navbar__menu").hasClass("de-navbar__menu--open")) {
                $(".de-navbar__menu-trigger").removeClass("de-navbar__menu-trigger--open");
                $(".de-navbar").removeClass("de-navbar--open");
                $(".de-navbar__menu").removeClass("de-navbar__menu--open");
