      function attachLightbox(selector) {
	function updateButtonPos(instance) {
		if (instance.current==undefined)
			return;

		var pos=instance.current.$content.css([ 'transform', 'width', 'height' ]);
		var xform=pos.transform.match(/(-?[\d\.]+), (-?[\d\.]+)\)$/);
		if (xform==null)
			return;

		instance.$refs.inner.find('.fancybox-tpu-nav').css({
			'left': xform[1]+'px',
			'top': xform[2]+'px',
			'width': pos['width'],
			'height': pos['height']
		});

		if (instance.$refs.stage.children().length==1) {
			$('.fancybox-tpu-btn-prev, .fancybox-tpu-btn-next').hide();
		}
	}

	loadjs.ready(['jquery', 'fancybox'], function() {
		var items;
		if (selector instanceof Function)
			items=selector();
		else
			items=$(selector);

		items.fancybox({
			loop: false,
			wheel: 'auto',
			buttons: [],
			smallBtn: false,
			onInit: function(instance, slide) {
				var maxWidth=$('body').css('maxWidth');
				$('body').addClass('tpu-fancybox-body-wrap').wrapInner('<div class="tpu-fancybox-wrap"></div>');
				$('.tpu-fancybox-wrap').css('maxWidth', maxWidth);
				instance.$refs.stage.on('transitionend', function() {
					updateButtonPos(instance);
				});
			},
			onUpdate: updateButtonPos,
			afterShow: function(instance, slide) {
				updateButtonPos(instance);
				instance.$refs.inner.find('.fancybox-tpu-nav').show();
			},
			beforeClose: function(instance, slide) {
				instance.$refs.inner.find('.fancybox-tpu-nav').hide();
			},
			afterClose: function(instance, slide) {
				$('.tpu-fancybox-wrap').contents().unwrap();
				$('body').removeClass('tpu-fancybox-body-wrap')
			},
			baseTpl: '<div class="fancybox-container" role="dialog" tabindex="-1">' +
				'<div class="fancybox-bg"></div>' +
				'<div class="fancybox-inner">' +
					'<div class="fancybox-tpu-nav">' +
						'<button data-fancybox-close class="fancybox-button fancybox-tpu-btn fancybox-tpu-btn-close" title="Close"></button>' +
						'<button data-fancybox-prev class="fancybox-button fancybox-tpu-btn fancybox-tpu-btn-prev" title="Previous"></button>' +
						'<button data-fancybox-next class="fancybox-button fancybox-tpu-btn fancybox-tpu-btn-next" title="Next"></button>' +
					'</div>' +
					'<div class="fancybox-stage"></div>' +
					'<div class="fancybox-caption"></div>' +
				'</div>' +
			'</div>',
		});
	});
}
	loadjs.ready(['jquery', 'swiper'], function() {
		attachLightbox('[data-fancybox]');

		if ($(window).width()<600) {
			$('.imgcontainer').each(function() {
				var $this=$(this);
				if (($this.find('a').length==1) || ($this.find('a').length>7))
					return;

				$this.addClass('swiper-container');
				$this.find('a').addClass('swiper-slide').css('width', 'auto').wrapAll('<div class="swiper-wrapper" />');

				new Swiper ($this.eq(0), { slidesPerView: 'auto', slidesPerGroup: 1, spaceBetween: 15, pagination: { el: '.swiper-pagination', clickable: true } });

		$('<div class="news-search" title="Search TechPowerUp News"></div>').appendTo('.datesep:first').on('click', function() {
			$('.news-search').slideUp(100);
			$('.news-search-box').slideDown(200);
			$('#news-search').focus();
		});

		var req=null;
		$('#news-search').on('focusout', function() {
			if ($(this).val()=='') {
				$('.news-search-box').slideUp(200, function() { $('.news-search').slideDown(200); });
				$('.hiddenbysearch').show().removeClass('hiddenbysearch');
			}
		}).on('keyup', function() {
			if (req) { req.abort(); req=null; }
			$('#ajaxresults').remove();
			$('.content article, .datesep, .loadmore').not('.hiddenbysearch').addClass('hiddenbysearch').hide();
			val=$(this).val();
			if (val=='')
				return;
			req=$.ajax({
				url: '/news-search?q='+val,
				cache: false,
				beforeSend: function() { $('#news-search-indicator').show(); $('#news-search').addClass('nomsclear'); },
				complete: function() { $('#news-search-indicator').hide(); $('#news-search').removeClass('nomsclear'); },
				success: function(data) {
					$('.news-search-box').after($('<div id="ajaxresults">').html(data));
					$('#ajaxresults a').highlight(val.split(' '));
					$('#ajaxresults a').on('click', function() {
						location.hash='#q-'+val;
					});
					attachLightbox('[data-fancybox]');
				}
			});
		});

		var hash=location.hash.slice(1);
		if ((hash!='') && (hash.indexOf('q-')==0)) {
			$('.news-search').trigger('click');
			$('#news-search').val(decodeURIComponent(hash.replace(/^q-/, ''))).trigger('keyup');
		}
	});

	var reviewCarousel = {
		swiper: undefined,
		init: function() {
			var initialSlide=0;
			var hash = location.hash.substr(1);
			if (hash.startsWith('review-')) {
				hash=hash.split('-');
				hash=hash[1];
				$.each(slides, function(key, val) {
					if (val['ts']!=hash)
						return;
					initialSlide=key;
				});
			}

			this.swiper = new Swiper('.review-carousel__items', {
				slidesPerView: 3,
				spaceBetween: 0,
				initialSlide: initialSlide,
				threshold: 1,
				breakpoints: {
					320: {
						slidesPerView: 3,
					},
					640: {
						slidesPerView: 4,
					},
					768: {
						slidesPerView: 5,
					},
					900: {
						slidesPerView: 3,
					},
					950: {
						slidesPerView: 4,
					},
					1100: {
						slidesPerView: 5,
					},
					1250: {
						slidesPerView: 6,
					},
					1400: {
						slidesPerView: 7,
					},
				},
				virtual: {
					slides: slides,
					renderSlide: this.renderSlide,
				},
			});
			this.swiper.on('progress', function() { reviewCarousel.checkAjax.call(reviewCarousel); });

			$('.review-carousel__button--prev').on('click', function(e) { e.preventDefault(); reviewCarousel.clickPrev(); });
			$('.review-carousel__button--next').on('click', function(e) { e.preventDefault(); reviewCarousel.clickNext(); });

			$('.review-carousel__items').on('click', 'a', function() {
				var current=slides[reviewCarousel.swiper.activeIndex]['ts'];
				window.history.replaceState(null, '', '/#review-'+current);
			});

			this.markNewReviews();
			this.workaroundForSmallClickMovement();
		},
		checkAjax: function() {
			var swiper=this.swiper;
			var delta = swiper.virtual.slides.length - swiper.activeIndex;
			if (delta>10)
				return;

			if ($('.review-carousel').hasClass('review-carousel--is-loading'))
				return;

			this.loadMore();
		},
		loadMore: function() {
			var swiper=this.swiper;
			var $container=$('.review-carousel');
			$container.addClass('review-carousel--is-loading');

			var lastId=swiper.virtual.slides[swiper.virtual.slides.length - 1].ts;

			$.getJSON('/lastreviews/'+lastId+'?cnt=20', function(data) {
				var items=[];

				$.each(data, function(key, val) {
					swiper.virtual.appendSlide(val);
				});

				$container.removeClass('review-carousel--is-loading');
			});
		},
		markNewReviews: function() {
			try {
				var key='reviewcarousel_last';
				var time=Math.round((new Date()).getTime() / 1000);

				var lastActivity=parseInt(localStorage.getItem(key));
				localStorage.setItem(key, time);
				if (isNaN(lastActivity))
					lastActivity=0;

				if (lastActivity<time-24*3600)
					lastActivity=time-24*3600;

				$('.review-carousel-item').each(function(i ,e) {
					if ($(this).data('ts')>lastActivity)
						$(this).addClass('review-carousel-item--new');
					if ($(this).data('ts')>time-3600)
						$(this).addClass('review-carousel-item--new');
				});
			} catch(e) {}
		},
		renderSlide: function(slide, index) {
			result='<div class="swiper-slide review-carousel-item" data-ts="'+slide['ts']+'">';
			result+='<a class="'+slide['class']+'" href="'+slide['url']+'">';
			result+='<img class="review-carousel-item__image" alt="" src="'+slide['small1x']+'" width="100" height="100" sizes="100px" srcset="'+slide['small1x']+' 100w, '+slide['small']+' 256w" />';
			result+='<span class="review-carousel-item__title">'+slide['title']+'</span></a></div>';

			return result;
		},
		clickPrev: function() {
			var swiper=this.swiper;
			var limit = swiper.params.slidesPerView != undefined ? swiper.params.slidesPerView : swiper.slidesPerView;
			var target = swiper.activeIndex - limit;
			if (target<=0)
				target=0;
			if (target==swiper.activeIndex)
				return;

			function keepScrolling() {
				if ((swiper.activeIndex>target) && (!swiper.isBeginning))
					swiper.slidePrev(33);
				else {
					$('.review-carousel .swiper-wrapper').css('transition-timing-function', '');
					swiper.off('transitionEnd', keepScrolling);
				}
			}

			$('.review-carousel .swiper-wrapper').css('transition-timing-function', 'linear');
			swiper.on('transitionEnd', keepScrolling);
			swiper.slidePrev(33);
		},
		clickNext: function() {
			var swiper=this.swiper;
			var limit = swiper.params.slidesPerView != undefined ? swiper.params.slidesPerView : swiper.slidesPerView;
			var target= swiper.activeIndex + limit;
			if (target>=swiper.virtual.slides.length)
				return;

			function keepScrolling() {
				if ((swiper.activeIndex<target) && (!swiper.isEnd))
					swiper.slideNext(33);
				else {
					$('.review-carousel .swiper-wrapper').css('transition-timing-function', '');
					swiper.off('transitionEnd', keepScrolling);
				}
			}

			$('.review-carousel .swiper-wrapper').css('transition-timing-function', 'linear');
			swiper.on('transitionEnd', keepScrolling);
			swiper.slideNext(33);
		},
		workaroundForSmallClickMovement: function() {
			var mouseDownData = undefined;

			$('.review-carousel .swiper-wrapper').on('mousedown', 'a', function(e) {
				if (e.which!=1)
					return;
				mouseDownData = e.originalEvent;
			});

			$('.review-carousel .swiper-wrapper').on('mouseup', 'a', function(e) {
				var debug = false;

				if (debug) console.log(e);

				if (e.which!=1)
					return;

				e = e.originalEvent;
				var deltaT = e.timeStamp - mouseDownData.timeStamp;
				if (deltaT>300) {
					if (debug) console.log('deltaT');
					return;
				}

				var deltaX = e.pageX - mouseDownData.pageX;
				if (deltaX > 10) {
					if (debug) console.log('deltaX');
					return;
				}

				var velocity = Math.abs(deltaX / deltaT);
				if (velocity > 0.05) {
					if (debug) console.log('velocity');
					return;
				}

				var $link = $(e.target).closest('a');

				$link.trigger('click');
				window.location = $link.attr('href');
			});
		}
	};

	loadjs.ready(['jquery', 'swiper'], function() {
		reviewCarousel.init();
	});

       <span>
        <span class="flags">

       <span class="byline-item">
        <a class="discusslink hotnews" href="/279938/trendforce-consumer-dram-pricing-to-increase-20-in-2q2021-due-to-increased-demand#comments">
         Discuss (25 Comments)
        </a>
       </span>
      </div>
      <div class="text p">
       According to TrendForce, we technology enthusiasts will have other rising prices to contend with throughout 2021, adding to the already ballooning discrete GPU and latest-gen CPUs from the leading manufacturer. The increased demand due to the COVID pandemic stretched the usual stocks to their limits, and due to the tremendous, multiple-month lead times between semiconductor orders and their fulfillment from manufacturers, the entire supply infrastructure was spread too thin for the increased worldwide needs. This leads to increased component pricing, which in turn leads to higher ASP pricing for DRAM. Adding to that equation, of course, is the fact that companies are now more careful, and are placing bigger orders so as to be able to weather these sudden demand changes.
       <br/>
       <br/>
       TrendForce says that DRAM pricing has already increased 3-8% in 1Q2021, and that market adjustments will lead to an additional increase somewhere between 13-18% for contract pricing. Server pricing is projected to increase by 20%; graphics DRAM is expected to increase 10-15% in the same time-span, thus giving us that strange stomach churn that comes from having to expect even further increases in graphics card end-user pricing; and overall DRAM pricing for customers is expected to increase by 20% due to the intensifying shortages. What a time to be a system builder.

When NVIDIA launched GeForce NOW for Windows, and Macintosh in 2017 they switched to a "bring your own games" model which required the user to own a compatible game on their Steam or other storefront account. Current Founders members will receive a Founders for Life membership which renews at 4.99 USD/month for the lifetime of the subscription. This latest price increases comes as NVIDIA continues to expand the service launching in new countries and nearing 10 million members.

        Confronting NVIDIA's DLSS: AMD Confirms FidelityFX Super Resolution (FSR) to Launch in 2021
       </a>
      </h1>
      <div class="byline">

        <a class="discusslink hotnews" href="/279883/confronting-nvidias-dlss-amd-confirms-fidelityfx-super-resolution-fsr-to-launch-in-2021#comments">
         Discuss (59 Comments)
        </a>
       </span>
      </div>
      <div class="text p">
       AMD, via its CVP &amp; GM at AMD Radeon Scott Herkelman, confirmed in video with PCWorld that the company's counterpart to NVIDIA's DLSS technology - which he defines as the most important piece of software currently in development from a graphics perspective - is coming along nicely. Launch of the technology is currently planned for later this year. Scott Herkelman further confirmed that there is still a lot of work to do on the technology before it's ready for prime time, but in the meantime, it has an official acronym: FSR (FidelityFX Super Resolution). If you're unfamiliar with DLSS, it's essentially an NVIDIA-locked, proprietary upscaling algorithm that has been implemented in a number of games now, which leverages Machine Learning hardware capabilities (tensor cores) to upscale a game with minimal impact to visual quality. It's important because it allows for much higher performance in even the latest, most demanding titles - especially when they implement raytracing.
       <br/>
       <br/>
       As has been the case with AMD, its standing on upscaling technologies defends a multiplatform, compatible approach that only demands implementation of open standards to run in users' systems. The idea is to achieve the broadest possible spectrum of game developers and gamers, with tight, seamless integration with the usual game development workflow. This is done mostly via taking advantage of Microsoft's DirectML implementation that's baked straight into DX 12.

       </a>
      </h1>
      <div class="byline">
       <span>
        <span class="flags">
        </span>
       </span>
       <span class="byline-item">
        by
        <address>
         <a href="/forums/members/187454/">

       </span>
      </div>
      <div class="text p">
       AMD is slowly preparing to launch its next-generation client-oriented accelerated processing unit (APU), which is AMD's way of denoting a CPU+GPU combination. The future design is codenamed after Van Gogh, showing AMD's continuous use of historic names for their products. The APU is believed to be a design similar to the one found in the SoC of the latest PlayStation 5 and Xbox Series X/S consoles. That means that there are Zen 2 cores present along with the latest RDNA 2 graphics, side by side in the same processor. Today, one of AMD's engineers posted a boot log of the quad-core Van Gogh APU engineering sample, showing some very interesting information.
       <br/>
       <br/>
       The boot log contains information about the memory type used in the APU. In the logs, we see a part that says "[drm] RAM width 256bits DDR5", which means that the APU has an interface for the DDR5 memory and it is 256-bit wide, which represents a quad-channel memory configuration. Such a wide memory bus is typically used for applications that need lots of bandwidth. Given that Van Gogh uses RDNA 2 graphics, the company needs a sufficient memory bandwidth to keep the GPU from starving for data. While we don't have much more information about it, we can expect to hear greater details soon.
