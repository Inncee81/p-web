          <h2 class="h-highlight">
           <a class="text-dark" href="https://www.tshaonline.org/handbook">
            Handbook of Texas
           </a>
          </h2>
          <p>
           The
           <em>
            Handbook of Texas
           </em>
           was the brainchild of past TSHA President Walter Prescott Webb, who boldly declared it would be “a reference to practically any topic on Texas history.” The initial
           <em>
            Handbook
           </em>
           was published as a two-volume set in 1952, with a supplemental volume published in 1976.
          </p>
          <p>
           Today, the
           <em>
            Handbook
           </em>
           is entirely online and includes 27,284 encyclopedic entries on the influential people, eras, and events of Texas history. Its staff of editors continuously works to add and update entries to ensure its accuracy and comprehensiveness.
          </p>

   <!-- font resize script -->
   <script>
    $(function() {
    $('.increase-font').on('click', function(event) {
        event.preventDefault();
        $('.resize').each(function() {
            changeTextSize(this, change);
        });
    });
    $('.decrease-font').on('click', function(event) {
        event.preventDefault();
        $('.resize').each(function() {
            changeTextSize(this, -change);
        });
    });
    $('.reset-font').on('click', function(event) {
        event.preventDefault();
        $('.resize').each(function() {
            resetTextSize(this);
        });
    });
});
var min = 8, max = 30, change = 2;
function changeTextSize(element, value, relative = true) {
    var currentSize;
    var newSize;
    var lineHeight;
    const phi = 1.618;
    if (relative) {
        currentSize = parseFloat($(element).css('font-size'));
        newSize = currentSize + value;
    } else {
        newSize = value;
    }
    if (newSize <= max && newSize >= min) {
        $(element).css('font-size', newSize + 'px');
        /* change line-height as well */
        $(element).css('line-height', phi * newSize + 'px');
    }
    if (typeof(Storage) !== "undefined") {
        // Store in local storage
        localStorage.setItem("font-size", newSize);
    }
}
function resetTextSize(element) {
    const phi = 1.618;
    $(element).css('font-size', '1rem');
    $(element).css('line-height', phi + 'rem');
    if (typeof(Storage) !== "undefined") {
        // delete from local storage
        localStorage.removeItem("font-size");
    }
}
// load from local storage and change font size on initial load if found */
if (typeof(Storage) !== "undefined") {
    var size = localStorage.getItem("font-size");
    $('.resize').each(function() {
        changeTextSize(this, size, false);
    });
}
