   jQuery(document).ready(function($) {
		/**
		 * Make all links to other domains open in new tabs/windows
		 */
		$('a').live('click', function() {
		    // If we're dealing with an Amber cache link from the popup, keep propagating normally.
            if ($(this).hasClass('amber-cache-link')) {
                return true;
            }

			// Validate that we have a link before proceeding. If we don't, propagate the event normally.
			var href = $(this).attr('href');
			if (typeof href === typeof undefined || href === false) {
				return true;
			}

			// Get the "data-amber-behavior" attribute in our link. If we have one, we want to check
            // if the link leads to an offline page or not. If the page is offline, we keep propagating
            // the event normally since a popup will appear instead of following the link.
            var amber_behaviour = $(this).attr('data-amber-behavior');
            if ('string' == typeof amber_behaviour && amber_behaviour.indexOf('down') !== -1) {
			    return true;
            }

			// Regex to extract the top level domain
			var pattern = /(\w*\.\w*)$/i;

			// Extract top level domains
			var link_domain = pattern.exec(this.host)[1];
			var host_domain = pattern.exec(location.hostname)[1];

			// Standardize Global Voices domain names
			if ('globalvoicesonline.org' == link_domain) {
				link_domain = 'globalvoices.org';
			}
			if ('globalvoicesonline.org' == host_domain) {
				host_domain = 'globalvoices.org';
			}

			if (link_domain != host_domain) {
				window.open(href);
			    return false;
			}
		});
	}); // end document.ready

   jQuery(document).ready(function($) {

		/**
		 * Clone a .toggle-menu from an existing div with a button and content inside it.
		 * 
		 * Creates a new menu for use with gv_toggle_menu_button(). Used to repurpose blocks of HTML
		 * hidden by media queries by cloning them to new locations where they can be collapsed.
		 * 
		 * Result is a div.target_css_class with a div.toggle-menu.menu-closed inside, .menu-toggle 
		 * and a .icon added to toggle_selector, and .toggle-menu-content-container added 
		 * to .menu_content_selector, with a new inner div.toggle-menu-content wrapping contents of 
		 * .menu_content_selector
		 * 
		 * @param string source_container jQuery selector for container of source content
		 * @param string target_container jQuery selector for existing container that will hold new menu
		 * @param string target_css_class CSS class (no .) to add to new menu's NEW wrapper div
		 * @param string toggle_selector jQuery selector for the toggle button already inside menu content
		 * @param string menu_container_selector jQuery selector for new menu content container (below heading)
		 * @param string menu_content_selector jQuery selector for new menu content inside it's container
		 * @returns false
		 */
		function gv_clone_menu(source_container, target_container, target_css_class, toggle_selector, menu_container_selector, menu_content_selector) {
			source_container_html = $(source_container).html();
			
			if (source_container_html) {
				// Add the content to the new container wrapped in div.target_css_class
				$(target_container).append("<div class='" + target_css_class + " toggle-menu menu-closed'>" + source_container_html + "</div>");
				// Add .menu-toggle and an html arrow to the toggle
				$(target_container).find(toggle_selector).addClass('menu-toggle');
				$(target_container).find(toggle_selector).addClass('toggle-menu-heading');
				$(target_container).find(toggle_selector).append(" <span class='icon icon-caret-down'></span>");
				
				// Add .menu-content-container to the menu_container_selector
				$(menu_container_selector).addClass('toggle-menu-content-container');
				// Wrap the entire contents of the container in div.toggle-menu-content
				$(menu_container_selector).contents().wrapAll("<div class='toggle-menu-content'></div>");
			}
			return false;
		}

		 * 
		 * Displays the contents of menu_content_selector as an absolute-positioned popup
		 * below toggle_selector and adds callback to hide it whenver anything else is clicked.

		 * Toggles the .menu-opened/.menu-closed selectors on menu_selector and toggles up/down icon inside

		 * If aldready_open_container is set then if it's visible the toggle will do nothing.

		 * @returns false Always return false to avoid click recursion

		function gv_toggle_menu_button(toggle_selector, menu_container_selector, menu_content_selector, within_selector) {
		
			/**
			 * Use 'body' as default value of within_selector, should always be a containing div
			 */
		   if(typeof(within_selector)==='undefined') within_selector = 'body';

//console.log('gv_toggle_menu_button! toggle_selector: ' + toggle_selector);

			/**
			 * Define cllback function to hide the menu when anything else is clicked
			 * Removes itself from body.click after running
			 */
			var hide_popup = function(e) {
				// Manually stop propagation
				// Fixes multiple-firing for touch events etc.
				e.stopPropagation();
				//e.preventDefault(); // Breaks links in menus! Use carefully!
	
				// PreventDefault/Return false when toggle is 'touch'ed to avoid touch+click double action
				// First make sure the toggle itself was clicked
				if ($(e.target).is(toggle_selector)) {					
					// AND this was a touch rather than click event
					if (e.type == 'touchend') {
						// PreventDefault, i.e. don't accept the click 
						e.preventDefault(); // Breaks links in menus!
						
						// PreventDefault should be enough but Android 4.x needs false or it opens+closes in a row
						return false;
					} 
				}

				// If this click/touch was not on the menu content itself
				$popup_menu = $(menu_content_selector);
				if (!$(e.target).parents().is($popup_menu)) {
					// IF the menu is opened swap it's classes to hide and toggle arrow
					if ($(menu_container_selector).hasClass('menu-opened')) {
						$(menu_container_selector).toggleClass('menu-opened');
						$(menu_container_selector).toggleClass('menu-closed');
						gv_toggle_font_icon_arrow(toggle_selector);
					}
					
					// Remove this callback no matter what
					$("body").off('click', hide_popup);
					$("body").off('touchend', hide_popup);

					return true;
				}
			};

			// Close all open menus on the page before we do anything
			$(".menu-opened").not(menu_container_selector).each(function() {
				$(this).toggleClass('menu-opened');
				$(this).toggleClass('menu-closed');
				gv_toggle_font_icon_arrow(this);
			});

			// Swap the open/closed classes
			$(menu_container_selector).toggleClass('menu-opened');
			$(menu_container_selector).toggleClass('menu-closed');
			
			gv_toggle_font_icon_arrow(toggle_selector);

			// If the menu is now open position the menu and add the body.click callback
			if ($(menu_container_selector).hasClass('menu-opened') ) {
			   				// Determine my/at parameters for .position(), flipped for toggle-menu-left
				if ($(menu_container_selector).hasClass('toggle-menu-left') ) {
					position_my = 'left top';
					position_at = 'left bottom';
				} else {
					position_my = 'right top';
					position_at = 'right bottom';
				}
				
				// Position the menu content
				$(menu_content_selector).position({					
					my: position_my
					,at: position_at
					,of: toggle_selector
					,collision: 'fit none'
					,within: within_selector
				});

				// Trigger custom toggle_open event
				$(toggle_selector).trigger('gv.toggle_menu.toggle_open');

				$("body").on('click', hide_popup);
				$("body").on('touchend', hide_popup);
			} 

			//console.log('end of gv_toggle_menu_button');

			// Return false to avoid the body click being called immediately. 
			// http://stackoverflow.com/questions/10021984/using-liveclick-inside-a-click-function-triggers-itself
			return false;			
		}

		/**
		 * Toggle the up/down font icon arrow inside parent_selector
		 * 
		 * @param string parent_selector jQuery selector for element that contains the span.icon 
		 */
		function gv_toggle_font_icon_arrow(parent_selector) {
			// Define the font icon classes to add to span
			icon_up_class = 'icon-caret-up';
			icon_down_class = 'icon-caret-down';
			
			// Get the .icon and make sure it exists
			arrow_icon = $(parent_selector).find('.icon');
			if (!arrow_icon.length) {
				return;
			}
				
			// Swap the icon classes
			if (arrow_icon.hasClass(icon_up_class)) {
				arrow_icon.removeClass(icon_up_class);
				arrow_icon.addClass(icon_down_class);
			} else if (arrow_icon.hasClass(icon_down_class)) {
				arrow_icon.removeClass(icon_down_class);
				arrow_icon.addClass(icon_up_class);
			}
		}

	function gv_data_close() {
		var cookies = $.cookie();

		// Scan all data-close elements
		$('[data-close]').each(function(){
			var id = $(this).attr('data-close');

			// Hide the element if gv_hide_$id cookie is present
			if (typeof $.cookie('gv_hide_' + id) !== 'undefined') {
				$('#' + id).hide();
			// Explicitly show the element if the cookie doesn't exist
			} else if (typeof $.cookie('gv_hide_' + id) === 'undefined') {
				$('#' + id).show();
			}
		});

		// Add on click events to all data-close elements
		$('[data-close]').on('click', function() {
			var id = $(this).attr('data-close');

			$.cookie('gv_hide_' + id, true, { path: '/', expires: 30 });
			$('#' + id).hide();
		});
	}
	gv_data_close();

	function gv_wpp_popup_js_fixes() {
		var cookies = $.cookie();
		var wpp_popup = $('[id^=wpp_html_theme-auto_popup-]');
		var wpp_popup_active = false;
		var wpp_popup_cookie_name = null;

		// Checking the length of an element is used to determined if it exists
		if (wpp_popup.length) {
			var match = /^wpp_html_theme-auto_popup-(.+)/im.exec(wpp_popup.attr('id'));

			// Extra validation to make sure we found a match.
			if (null !== match) {
				wpp_popup_cookie_name = 'wpp_popup_' + match[1] + '_closed'
				wpp_popup_active = true;
			}
		}

		// Scan all cookies for WP Popup cookie to see if it is closed. Update flag if found.
		for (var name in cookies) {
			if (wpp_popup_cookie_name === name) {
				wpp_popup_active = false;
				break;
			}
		}

		// Don't hide anything if popup isn't active.
		if (wpp_popup_active === false) {
			return;
		}

		// Scan all data-close elements and close the ID they refer to
		$('[data-close]').each(function(){
			var id = $(this).attr('data-close');
			$('#' + id).hide();
		});
	}

	// Checks if the given element is in view inside the browser
	function isScrolledIntoView(el) {
		var elemTop = el.getBoundingClientRect().top;
		var elemBottom = el.getBoundingClientRect().bottom;

		return elemTop < window.innerHeight && elemBottom >= 0;
	}

	// Register the scroll event handler
	$(document).scroll(scrollEventHandler);
