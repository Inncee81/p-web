	var browser_chrome = false;
	var browser_ff = false;
	var show_multisuche = false;
	if (navigator.userAgent.match(/Chrome\/\d+/) !== null && navigator.userAgent.indexOf("MSIE") < 1 && navigator.userAgent.indexOf("Edge") < 1) {
		browser_chrome = true;
		show_multisuche = true;
	}
	if (navigator.userAgent.search(" Firefox/") >= 0) {
		browser_ff = true;
		show_multisuche = true;
	}

	if (show_multisuche) {
		$("#multi-suche-install").show();
	}

	// Install button click event handler
	if (multisucheInstall) {
		$("#slidebox").css("cursor", "pointer").click(function() {
			location.href = "/multi_suche";
		});
		multisucheInstall.addEventListener("click", function (event) {
			event.preventDefault();
			event.stopPropagation();
			if (browser_ff) {
				addon_overlay();
				// AMO XPI file URL
				var url = "https://addons.mozilla.org/firefox/downloads/latest/dejure-org-multi-search/latest.xpi";

				// Check if InstallTrigger is available
				if (typeof (InstallTrigger) !== "undefined" && InstallTrigger.enabled) {
					// Trigger the install
					InstallTrigger.install ({ "dejure.org Multi-Suche": url });

					// And prevent the default event behavior
					event.preventDefault ();
					return false;
				}
			} else if (browser_chrome) {
				addon_overlay("chrome");
				// Check if Chrome inline installation is available
				if (typeof (chrome) !== "undefined") {
					// If so, trigger install
					chrome.webstore.install ();

					// And prevent the default event behavior
					event.preventDefault ();
					return false;
				}
			}
			return false;
		}, false);
	}
