        //Variables to control search related DOM elements
                  const searchContainer = document.querySelector('#search-container');
                  const searchExpandButton = document.querySelector('#search-expand-button');
                  const searchFormContainer = document.querySelector('#search-form-container');
                  const searchInput = document.querySelector('#grlp-search-input');
                  const headerNavContainer = document.querySelector('#header-nav');
                  const timesCloseButton = document.querySelector('#times-close-button');
                  const timesContainer = document.querySelector('#times-container');
                  const tabletMobileActivated = window.matchMedia("(max-width: 992px)");
                  
                  //Search Expander to open Search Bar
                  searchExpandButton.addEventListener('click', function() {
                    
                    
                    headerNavContainer.classList.remove('navbar-headers-visible');
                    headerNavContainer.classList.add('navbar-headers-invisible');
                    
                    setTimeout(function() {
                      if (searchContainer.style.display = 'block') {
                      searchContainer.style.display = 'none';
                      }
                      if (timesContainer.style.display === 'none') {
                        timesContainer.style.display = 'block';
                      }
                      headerNavContainer.classList.add('navbar-display-none');
                      if (searchFormContainer.style.display === 'none') {
                        searchFormContainer.style.display = 'flex';
                      }
                      searchInput.focus();
                    }, 100);
                    
                  })
        
                  //Times Btn to close Search Bar with animation
                  timesCloseButton.addEventListener('click', function() {
                    closeSearch();
                  })
        
                  //If viewport goes below boostrap md, actiavte close immediately without animation
                  function checkViewportWidth(tabletMobileActivated) {
                    if (tabletMobileActivated.matches) {
                        
                        searchInput.blur();
        
                      if (timesContainer.style.display === 'block') {
                        timesContainer.style.display = 'none';
                      }
                      if (searchContainer.style.display = 'none') {
                        searchContainer.style.display = 'block';
                      }
                      if (searchFormContainer.style.display === 'flex') {
                        searchFormContainer.style.display = 'none';
                      }
        
                      headerNavContainer.classList.remove('navbar-display-none');
                      headerNavContainer.classList.remove('navbar-headers-invisible');
                    }
                  }
        
                  checkViewportWidth(tabletMobileActivated);
                  tabletMobileActivated.addListener(checkViewportWidth);
                  
                  //If Search Bar does not have active focus on it, then close the search bar
                  searchInput.addEventListener('blur', closeSearch);
        
                  //Function to close search bar with animation
                  function closeSearch() {
                    searchInput.blur();
        
                    setTimeout(function() {
                      if (timesContainer.style.display === 'block') {
                      timesContainer.style.display = 'none';
                      }
                      if (searchContainer.style.display = 'none') {
                        searchContainer.style.display = 'block';
                      }
                      headerNavContainer.classList.add('navbar-headers-visible');
                      headerNavContainer.classList.remove('navbar-headers-invisible');
                    }, 200);
        
                    setTimeout(function() {
                      if (searchFormContainer.style.display === 'flex') {
                        searchFormContainer.style.display = 'none';
                      }
                      headerNavContainer.classList.remove('navbar-display-none');
                    }, 300);
                  }
