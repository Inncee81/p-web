            e.preventDefault();
            var form = e.target;

            form.submit();

   function AddFavorite(sURL, sTitle)
{	
	var flag=true;
	try{
	if (window.sidebar) {
		flag=false;
		window.sidebar.addPanel(sTitle, sURL,"");
		} else if( document.all ) {
		window.external.AddFavorite( sURL, sTitle);

		}
	}catch(e){
		if(flag){
		alert("加入收藏失败，请使用Ctrl+D进行添加");
		}
	}
}
