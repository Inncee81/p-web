    (function($) {
        //ensures the default selection string doesn't
        //dangle any periods at the end of the redirect URL
        function cleanUrl(urlString) {
            var newUrl = urlString.trim();
            var trailingSlash = /\/$/;
            if (newUrl.match('/content/news')) {
                newUrl = newUrl.replace('/content/news', "");
            }
            if (newUrl.match(trailingSlash)) {
                newUrl = newUrl.replace(trailingSlash, "");
            }
            return newUrl;
        }
        $(function() {
            var regionCookie = News.region.getRegionCookieVal();
            var redirectNode = $('.redirect');
            var defaultPage = redirectNode.data('defaultpage');
            var returnFromDefault = Boolean(redirectNode.data('returnfromdefault'));
            if (returnFromDefault) {
                var selector = window.location.pathname.replace('/', '.')
                defaultPage = defaultPage + selector
            }
            var fromPages = redirectNode.data('frompages');
            var toPages = redirectNode.data('topages');
            if(regionCookie && fromPages && toPages){
                var from = fromPages.split(',');
                var to = toPages.split(',');
                var matchedRegion = from.indexOf(regionCookie);
                if (matchedRegion > -1) {
                    window.location.replace(cleanUrl(to[matchedRegion]));
                } else {
                    window.location.replace(cleanUrl(defaultPage));
                }
            } else {
                //If no region cookie is set go to the splash page, let user select region
                // splash page will detect the redirect and handle accordingly.
                var newLocation = defaultPage;
                //If a user goes to ny1 with no cookie set they are redirected
                //to nyc/all-boroughs per the splash page behavior discussions
                if (window.location.hostname.indexOf("ny1.com") > -1) {
                    newLocation = "/nyc/all-boroughs"
                }
                window.location.replace(newLocation);
            }
        });
