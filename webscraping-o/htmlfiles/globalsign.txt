function setCookieC(cookieName, cokkieValue)
{
    var expDays = 0;
    var d = new Date(cokkieValue, expDays);
    d.setTime( d.getTime() + ( expDays * 24 * 60 * 60 * 1000 ) );
    var expires = "expires=" + d.toGMTString(); 
    document.cookie = cookieName + "=" + cokkieValue + ";" + expires +"; Path=/;" + "Secure;" + "SameSite=Strict";      
}

function deleteCookieC(cookieName){
    document.cookie = cookieName + '=; expires=Thu, 01 Jan 1970 00:00:01 GMT;';
}


function getCookieC( cname ) {
    var name = cname + "=";
    var ca = document.cookie.split( ';' );
//console.log("sss:"+ca);
    for ( var i = 0; i < ca.length; i++ ) {
        var c = ca[ i ];
        while ( c.charAt( 0 ) == ' ' ) c = c.substring( 1 );
        if ( c.indexOf( name ) != -1 ) return c.substring( name.length, c.length );
    }
    return "";
}
