const initDesktopAnimation = () => {
  $(window).on('scroll', function(){

    });
  }

const showScrollNotification = (element) => {
  element.classList.add('visible');
};

const hideScrollNotification = (element) => {
  element.classList.remove('visible');
};

const scrollPercentageIntoView = (ele, eleIndex) => {
  let windowBottom = $(this).scrollTop() + $(this).height();
  let elementTop = $(ele).offset().top;
  let percentage = (windowBottom - elementTop) / $(ele).height() * 85;

  if(eleIndex == 1) {
    // Because we nudge the first section, we need to alter the scroll percentage
    percentage = (windowBottom - elementTop - 85) / ($(ele).height() - 85) * 100;
  }

  if(eleIndex != 99) {
    handleCrossOutAndImageAnimation(windowBottom, elementTop, percentage, eleIndex, ele);
  } else {
    // Donate form animation
    if (percentage >= 1 && getParameterByName('wps') != 'true') {
      $('body').addClass('donate-form-scrolled-info-view');

      $('body').removeClass('donate-form-scrolled-info-view');

  }
};

const scrollStyleIntoView = ($el, bgSize) => {
  return $el.css('background-size', `${bgSize} 67%`);
};

const handleCrossOutAndImageAnimation = (windowBottom, elementTop, percentage, eleIndex, ele) => {
  let $elementToSelect = $('[data-sentence='+ eleIndex +']');

  if (percentage >= 100) {
    scrollStyleIntoView($elementToSelect, '100%');
    $(ele).css('transform', 'scale(1.05) translateZ(21px)');
  } else if (windowBottom >= elementTop) {
    if($elementToSelect.length > 1) {
      if(percentage < 50) {
        scrollStyleIntoView($elementToSelect.eq(0), `${Math.round(percentage * 2)}%`);
      } else {
        scrollStyleIntoView($elementToSelect.eq(1), `${Math.round(percentage * 2) - 100}%`);
      }
    } else {
      scrollStyleIntoView($elementToSelect, `${Math.round(percentage)}%`);

    scrollStyleIntoView($elementToSelect, '0%');
    $(ele).css('transform', 'scale(1) translateZ(0px)');
  }
};

const getParameterByName = (name, url = window.location.href) => {
  name = name.replace(/[\[\]]/g, '\\$&');
  var regex = new RegExp('[?&]' + name + '(=([^&#]*)|&|#|$)'),
      results = regex.exec(url);
  if (!results) return null;
  if (!results[2]) return '';
  return decodeURIComponent(results[2].replace(/\+/g, ' '));
};

         <h2 class="h40 modal-title">
          Other Ways to Give
         </h2>
         <div class="row">
          <div class="large-4 columns">
           <h5 class="info-label">
            Donate by Check
           </h5>
           <p class="info-body">
            Make checks payable to charity: water. If you're donating in honor of someone, sponsoring a water project or want to direct your donation to our operating costs, please write that in the memo of your check.
           </p>
           <p class="info-note">
            Unfortunately, we cannot accept donations to The Spring via check.
           </p>
           <p class="info-note">
            Please note, donations made by check will not be matched.
           </p>
           <p class="info-note">
            Tax receipts will be mailed within four to six weeks after the donation has been processed.
           </p>
          </div>
          <div class="large-4 columns">
           <h5 class="info-label">
            International Giving
           </h5>
           <p class="info-body">
            charity: water is a registered charity in the UK. To make a tax-efficient donation in GBP, visit
            <a href="https://charitywateruk.org">
             charitywateruk.org
            </a>
            . For other international donations, we accept bank issued money orders in USD. Money orders should be made payable to charity: water and can be mailed to the address below. For donations over $6,000, we also accept wire transfers; email
            <a href="mailto:info@charitywater.org">
             info@charitywater.org
            </a>
            for more information.
           </p>
           <p class="info-note">
            Please note, only donations made by eligible US taxpayers may be tax-deductible.
           </p>
          </div>
          <div class="large-4 columns">
           <h5 class="info-label">
            Donate Stock
           </h5>
           <p class="info-body">
            We're able to accept donations in the form of stock, government (including municipal) debt and corporate debt through the Depository Trust Company (DTC). For security, settlement and reduced paperwork, electronic transfer through the DTC is preferred. This can be arranged from most brokerage accounts.
           </p>
           <p class="info-note">
            Unfortunately, we cannot accept donations to The Spring via stock.
           </p>
           <p class="info-note">
            Please contact
            <a href="mailto:info@charitywater.org">
             info@charitywater.org
            </a>
            for more information on making a stock donation.
           </p>
          </div>
