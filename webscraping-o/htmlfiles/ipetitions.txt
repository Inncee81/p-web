    function sendToPaypalFromDonateModal() {
        amount = $("#donate-form input[name=_amount]:checked").val();
        let monthlyPaymentVal = $("#donate-form input[name=monthly_payment]:checked").val();
        let monthlyPayment = (monthlyPaymentVal != undefined && monthlyPaymentVal == 'true');

        if (isNaN(monthlyPayment)) {
            monthlyPayment = false;
        }

        if (isNaN(amount) || amount <= 0) {
            amount = $("#other-amount").val();
        }

        if (isNaN(amount) || amount <= 0) {
            amount = 50;
        }

        $.post("/track/payment-click", {
            'ps': 'paypal',
            'amount': amount,
            'loc': window.payment_location
        });

        $.cookie('homepage_popup_pause', '1', {path: '/', expires: 15});

        payWithPaypal(
            amount,
            'contribute',
            window.payment_location,
            null,
            null,
            null,
            monthlyPayment
        );
    }

    $("#other-amount").on("keypress change click tap touch", function (e) {
