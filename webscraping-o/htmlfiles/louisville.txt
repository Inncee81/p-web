  <script>
   var alert_template = $("#alert-template").html();
    Mustache.parse(alert_template);

    function checkAlerts() {
        $.get("https://api.louisville.edu/v1/alert", function (data) {
            var rendered_template = Mustache.render(alert_template, data);
            $("#alerts").html(rendered_template);
        }).always(function () {
            setTimeout(checkAlerts, 60000);
        })
    }

    checkAlerts();
  </script>
