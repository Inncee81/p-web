    function getCurrentColorMode() {
        var body = document.getElementsByTagName("BODY")[0];

        color_mode = window.getCookieLocal("color_mode");

        // COOKIE
        if (color_mode) {
            if (color_mode && color_mode == 'dark-mode') {
                body.className += " dark-mode";
            }
        } else {
            // OS THEME

            if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
                // dark mode
                color_mode = 'dark-mode';
                body.className += " dark-mode";

            } else {
                //TIME
                currentTime = new Date();
                time = currentTime.getTime();
                hours = currentTime.getHours();

                if (isBetweenHours('22', '7')) {
                    color_mode == 'dark-mode';
                    body.className += " dark-mode";
                }
            }
        }
    }

    function isBetweenHours(startHour, endHour) {
        var now = new Date().getHours();

        return (now >= startHour || now <= endHour);

    function searchController() {
        var offset = 15;
        var type_filter = "";
        var max_results = (document.getElementById('query') !== null ? document.getElementById('query').dataset.searchResultsNum : 1000);
        var inBottom = false;

        document.querySelector('.search-input input').addEventListener('blur', function () {
            if (this.value !== "")
                document.querySelector('.search-input label').classList.add('not-empty');
            else
                document.querySelector('.search-input label').classList.remove('not-empty');
        });

        document.querySelectorAll('ul#filter_types li button').forEach(function (button) {

            button.addEventListener('click', function () {

                if (functionLock)
                    return;

                document.querySelectorAll('.article').forEach(function (article) {
                    article.remove();
                });

                if (button.dataset.filter === 'all')
                    type_filter = "";
                else
                    type_filter = encodeURIComponent(button.dataset.filter);
                offset = 0;
                max_results = 1000;

                document.querySelector('ul#filter_types li button.selected').classList.remove('selected');
                button.classList.add('selected');

                searchLoadMore();
                document.querySelector('body').scrollIntoView({behavior: 'smooth', block: 'start'});
            });
        });


        function isOnViewport() {
            var elementTop = $('#footer').offset().top;
            var elementBottom = elementTop + $('#footer').outerHeight();
            var viewportTop = $(window).scrollTop();
            var viewportBottom = viewportTop + $(window).height();

            return elementBottom > viewportTop && elementTop < viewportBottom;
        };

        var functionLock = false;

        function searchLoadMore() {
            var _locale = websiteLocaleValue + "/";
            if (_locale == "en-gb/") {
                _locale = "";
            }
            if (functionLock)
                return;

            var xhttp = new XMLHttpRequest();

            xhttp.onreadystatechange = function () {
                if (this.readyState == 4 && this.status == 200) {
                    var data = JSON.parse(this.responseText);
                    offset = data.offset;
                    max_results = data.search_results_number;
                    addSearchResults(data.search_results);
                    functionLock = false;
                }

                if (this.readyState == 4 && this.status == 500) {
                    functionLock = false;
                }
            };
            xhttp.open('GET', window.location.origin + '/' + _locale + 'search?q=' + document.getElementById('query').dataset.query + '&offset=' + offset + '&type=' + type_filter, true);

            xhttp.send();

            functionLock = true;
        }

        $(window).on("scroll", function () {
            if (offset < max_results)
                if (isOnViewport() && !inBottom) {
                    searchLoadMore();
                    inBottom = true;
                } else if (inBottom && !isOnViewport())
                    inBottom = false;
        });
    }
