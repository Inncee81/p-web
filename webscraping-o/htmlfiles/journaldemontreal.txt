	function activateLeger(){
		
    //---------------------------------------------------------------------
// HOME PAGE PERCENTAGE ANIMATION
//---------------------------------------------------------------------

    var jsonUrl = 'https://indicedebonheur.com/surveys/mood_en.json';
    var prcTwoEnd;
    var prcLanguage;
    console.log('leger: '+jsonUrl);

    $.ajax({
        url: jsonUrl,
        dataType: 'json',
        success: function( data ) {
            console.log(data);
            prcTwoEnd = parseFloat( data.average);
            prcLanguage = data.average;
            emojiCodeWrap();
        },
        error: function( data ) {
            prcTwoEnd = 75.5;
            prcLanguage = 'fr';
            emojiCodeWrap();
        }
    });


    function emojiCodeWrap(){
        // TWO ----------------------------------------------
        var prcTwo = $('[data-prc-two]');

        var prcTwoStart = prcTwoEnd - 15;
        if(prcTwoStart < 0) prcTwoStart = 0;

        var prcSTwo = {val:prcTwoStart};


				var myInt;
				var myIntCount = 0;
				onUpdatePrcTwo();
				var numberOfInt = 45
				myInt = setInterval(function(){ 
					onUpdatePrcTwo();
					prcSTwo.val = prcSTwo.val + (prcTwoEnd - prcTwoStart)/numberOfInt;
					myIntCount++
					if (myIntCount === numberOfInt){
						clearInterval(myInt);
						onCompleteTwo();
					}
				}, 50);				
				
				
        var diffPrevTwo = prcSTwo.val;
        var incTwo = 0;

        function onUpdatePrcTwo(){
            if(true){
                prcTwo.html(prcSTwo.val.toFixed(1).toString().replace("." , ","));
            }
            else{
                prcTwo.html(prcSTwo.val.toFixed(1).toString());
            }

            var diff = prcSTwo.val - diffPrevTwo;
            diffPrevTwo = prcSTwo.val;
            incTwo += diff;

            if(incTwo > 1.25){
                incTwo = 0;

                var r = Math.round(Math.random()*7);

                $('[data-emoji-two]').css('visibility', 'hidden');
								$('[data-emoji-two]').css('opacity', '0');
                $('[data-emoji-two]').eq(r).css('visibility', 'visible');
								$('[data-emoji-two]').eq(r).css('opacity', '1');
            }
        }

        var emojiTwo;
        switch (true){
            case (prcTwoEnd <= 55):
                emojiTwo = 1;
                break;
            case (prcTwoEnd > 55 && prcTwoEnd <= 61):
                emojiTwo = 2;
                break;
            case (prcTwoEnd > 61 && prcTwoEnd <= 66):
                emojiTwo = 3;
                break;
            case (prcTwoEnd > 66 && prcTwoEnd <= 71):
                emojiTwo = 4;
                break;
            case (prcTwoEnd > 71 && prcTwoEnd <= 76):
                emojiTwo = 5;
                break;
            case (prcTwoEnd > 76 && prcTwoEnd <= 81):
                emojiTwo = 6;
                break;
            case (prcTwoEnd > 81 && prcTwoEnd <= 86):
                emojiTwo = 7;
                break;
            case (prcTwoEnd > 86 && prcTwoEnd <= 100):
                emojiTwo = 8;
                break;
        }
			

        function onCompleteTwo(){
            $('[data-emoji-two]').css('visibility', 'hidden');
						$('[data-emoji-two]').css('opacity', '0');
						$('[data-emoji-two]').eq(emojiTwo).css('visibility', 'visible');
						$('[data-emoji-two]').eq(emojiTwo).css('opacity', '1');
        }
    }
