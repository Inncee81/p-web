                    element.value = "";
                    element.focus();

                    if (element.selectionStart === 0 || element.selectionStart >= element.value.length) {
                        var len = element.value.length;
                        element.setSelectionRange(len, len);
                    }
                }

                    oSel.moveStart("character", -element.value.length);
