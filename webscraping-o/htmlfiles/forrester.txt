//function to set state picklist. if Json retrival failure, back up state picklist in stateDataLocal variable(above) will be used
function setStatepicklist() {
  var dataSource = stateDataLocal;
  if (jsonSuccess == 1) {
    dataSource = countryStateData;
  }

  if ($('#countryField').val() == 'USA' || $('#countryField').val() == 'US') {
    $("#stateField").empty();
    dataSource.statesUSA.forEach(function(item, i) {
      $("#stateField").append('<option value="' + item.Value + '">' + item.Display + '</option>');
    });
    // if there was a stae value returned from WDL put it back into the field
    if ((stateValue) && (countryValue == 'US' || countryValue == 'USA')) {
      $("#stateField").val(stateValue);
    } else {
      $("#stateField").val('');
    }
  } else if ($('#countryField').val() == 'CA') {
    $("#stateField").empty();
    dataSource.statesCA.forEach(function(item, i) {
      $("#stateField").append('<option value="' + item.Value + '">' + item.Display + '</option>');
    });
    // if there was a stae value returned from WDL put it back into the field
    if ((stateValue) && countryValue == 'CA') {
      $("#stateField").val(stateValue);
    } else {
      $("#stateField").val('');
    }

  } else if ($('#countryField').val() == 'AU') {
    $("#stateField").empty();
    dataSource.statesAUS.forEach(function(item, i) {
      $("#stateField").append('<option value="' + item.Value + '">' + item.Display + '</option>');
    });
    // if there was a state value returned from WDL put it back into the field
    if ((stateValue) && countryValue == 'AU') {
      $("#stateField").val(stateValue);
    } else {
      $("#stateField").val('');
    }
  }
}

//function to set country picklist. if Json retrival failure, back up country picklist in html field will remain
function setCountryOptions() {
  if (jsonSuccess == 1) {
    $("#countryField").empty();
    countryStateData.countries.forEach(function(item, i) {
      $("#countryField").append('<option value="' + item.Value + '">' + item.Display + '</option>');
    });
    $("#countryField").val('');
    $("#countryFieldHidden").val('');
    setStatepicklist();
  } else {
    setStatepicklist();
  }

}

//function to retrieve json for country and state data, if failure jsonSuccess variable will = 0 and local data will be used - (stateDataLocal variable for states and the actual picklist in the form for countries. ALERT! countriesConverterValues needs to be updated if any changes are made to Country json)
function countryDataJSON() {
  $.ajax({
    type: 'GET',
    url: url,
    jsonpCallback: cbFunction,
    dataType: 'jsonp',
    contentType: "text/jsonp; charset=utf-8",
    crossDomain: true,
    timeout: 500,
    success: function(json) {
      countryStateData = json;
      jsonSuccess = 1;
      setCountryOptions();

    },
    error: function(e, xhr, status, error) {
      // AJAX Error Messaging
      // alert(e.message);
      console.log('not OK ' + xhr);
      console.log('not OK ' + status);
      console.log('not OK ' + error);
    }
  });
}




//check form values in hidden fields. calls formStatusState function to do visibility setting based on result
function formStatusHiddens() {
  //if any of the hidden fields don't have a value, show them.
  if (($("#companyAddressField").val() == '' || $("#companyAddressField").val() == null) || ($("#cityField").val() == '' || $("#cityField").val() == null) || ($("#zipCodeField").val() == '' || $("#zipCodeField").val() == null)) {
    formStatusState(1);
  } else {
    formStatusState(2);
  }
}

        stateDBupdate = true;
      }

    }

    else if (dbCallmade == false) {

        $('#companyAddressField').val("");
        companyAddressDBupdate = false;

      if (zipDBupdate == true) {
        $('#zipCodeField').val("");
        zipDBupdate = false;

        $('#cityField').val("");
        cityDBupdate = false;
      }
      if (stateDBupdate == true) {
        $('#stateField').val("");
        stateDBupdate = false;
      }

    }
    formStatusHiddens();
  }, 1000);

  companyNameFieldBlur = true;


});

//function to set visibility of hidden fields based on country field (called by formStatusHiddens function)
function formStatusState(x) {
  //if country = USA/Canada/Australia and state field is empty, show all hidden fields
  if ($('#countryField').val() == 'USA' || $('#countryField').val() == 'US' || $('#countryField').val() == 'CA' || $('#countryField').val() == 'AU') {
    if ($('#stateField').val() === null || $('#stateField').val() == '') {
      $('#stateField').prop('required', true);
      $('.state_row').slideDown(500);
      $('.js-eloqua-company-address-2').slideDown(500);
      fieldsExposed = true;


    }
    //if country is US/AU/CA but formStatusHiddens indicated other hidden fields are missing data - expose all
    else if (x == 1) {

      $('#stateField').prop('required', true);
      $('.state_row').slideDown(500);
      $('.js-eloqua-company-address-2').slideDown(500);
      fieldsExposed = true;
    }
    //if country is US/AU/CA and formStatusHiddens did not indicate any other hidden fields are missing values
    else {
      if (fieldsExposed != true) {
        $('#stateField').prop('required', false);
        $('.state_row').slideUp(100);
        $('.js-eloqua-company-address-2').slideUp(100);
      }
    }

  }
  //if country != USA/Canada/Australia and state field is empty, but formStatusHiddens indicated other hidden fields are missing values, show other hiddens but hide statefiled
  else if (x == 1) {

    $('#stateField').prop('required', false);
    $('.state_row').slideUp(100);
    $('.js-eloqua-company-address-2').slideDown(500);
    fieldsExposed = true;
  }
  //if fields haven't already been exposed, country != USA/Canada/Australia, and formStatusHiddens doesnt indicate any other hiddens are missing values
  else {
    if (fieldsExposed != true) {
      $('#stateField').prop('required', false);
      $('.state_row').slideUp(100);
      $('.js-eloqua-company-address-2').slideUp(100);
    }
    //if fields haven already been exposed, but country != USA/Canada/Australia when it previously was, and formStatusHiddens doesnt indicate any other hiddens are missing values, hide the statefield
    else {
      $('#stateField').prop('required', false);
      $('.state_row').slideUp(100);
    }
  }
}

function countryConverter(countryInput) {
  var foundValue = countriesConverterValues.find(function(obj) {
    return obj.Display === countryInput;
  });

  if (foundValue) {
    newCode = foundValue.Value;
  } else {
    foundValue = countriesConverterValues.find(function(obj) {
      return obj.Value === countryInput;
    });
    newCode = foundValue.Value;
  }

  countryValue = newCode;
}
