			// Probably most of the DOM manipulation that happens on page load can be moved into the module instead. I checked for other uses of the JS-only file and only found one currently still in use with an HTML snippet
			document.addEventListener("DOMContentLoaded", function () {
				var div;
				var n;
				var v = document.getElementsByClassName("youtube-player");
				for (n = 0; n < v.length; n++) {
					// Get the button
					var button = v[n].querySelector('button');
					// As far as I know there's no particular reason we can't do this in the module code
					// I think the reason it was built that way in the original  Labnol version is to simplify the HTML that editors have to generate, but that's not a problem here
					button.setAttribute("data-id", document.getElementsByClassName("youtube-player")[n].getAttribute("data-id"));
					button.innerHTML = labnolThumb(document.getElementsByClassName("youtube-player")[n].getAttribute("data-thumbnail"), document.getElementsByClassName("youtube-player")[n].getAttribute("data-id"));

					// Don't need a separate keyboard event since it's an actual button (so the click event captures mouse and keyboard by itself already)
					button.addEventListener('click', labnolIframe, false);
					button.idParam = document.getElementsByClassName("youtube-player")[n].getAttribute("data-id");
					button.titleParam = document.getElementsByClassName("youtube-player")[n].getAttribute("data-title");
				}

			function labnolThumb(thumbnail, id) {
				var thumbProvided = '<img src="THUMB_URL" alt="">';
				var thumbSD = '<img src="https://img.youtube.com/vi/ID/sddefault.jpg" alt="">';
				var play = '<span class="play-btn"></span>';

				if (!thumbnail) {
					return thumbSD.replace("ID", id) + play;
				} else {
					return thumbProvided.replace("THUMB_URL", thumbnail) + play;
				}
			}

			function labnolIframe(event) {
				var videoDiv = thisOrClosest(event.target, 'youtube-player');
				var iframe = document.createElement("iframe");
				var embed = "https://www.youtube.com/embed/ID?rel=0&autoplay=1";
				iframe.setAttribute("src", embed.replace("ID", this.idParam));
				iframe.setAttribute("frameborder", "0");
				iframe.setAttribute("title", this.titleParam);
				iframe.setAttribute("allowfullscreen", "1");
				iframe.setAttribute("autoplay", "1");
				iframe.setAttribute("allow", "autoplay");
				var button = videoDiv.getElementsByClassName("youtube-thumb")[0];
				videoDiv.appendChild(iframe);
				videoDiv.focus();
				button.remove();
			}

			var thisOrClosest = function (elem, className) {
				if(elem.classList.contains(className)){
					return elem;
				}else{
					for ( ; elem && elem !== document; elem = elem.parentNode ) {
						if (elem.classList.contains(className)) return elem;
					}
				}
				return null;
			};

     "use strict";

 function tabToggle(elem) {
                var y = elem.attr('tabindex');

              if (y == "0") {
                     y = "-1";
           } else {
                  y = "0";
            }

           elem.attr('tabindex', y);
 }

   function setCookie(name,value) {
          document.cookie = name + "=" + (value || "");
 }
 function getCookie(name) {
                var nameEQ = name + "=";
            var ca = document.cookie.split(';');
              for(var i=0;i < ca.length;i++) {
                     var c = ca[i];
                    while (c.charAt(0)==' ') c = c.substring(1,c.length);
                     if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
           }
         return null;
      }

   document.addEventListener("DOMContentLoaded", function() {
          var chatbotButton = $('#chatbot-tab');
            var chatbotContent = $('#chatbot-drawer');
                var chatbotClose = $('#chatbot-close');
           var chatbotIframe = $('#chatbot-iframe');

           var attrToggle = function (elem, attr) {
                  var x = elem.attr(attr);

                    if (x === "true") {
                         x = "false";
                        } else {
                          x = "true";
                 }

                   if (attr === 'aria-hidden' && x === "false") {
                          elem.removeAttr(attr);
                    } else {
                          elem.attr(attr, x);
                       }
         }

           chatbotButton.click(function() {
                  attrToggle(chatbotButton, 'aria-expanded');
                       attrToggle(chatbotButton, 'aria-pressed');
                        attrToggle(chatbotContent, 'aria-hidden');
                        chatbotButton.toggleClass('expanded');
                    chatbotContent.toggleClass('expanded');

                     tabToggle(chatbotClose);

                    if (!chatbotIframe.attr('src')) {
                         chatbotIframe.attr('src','https://jhmcoronavirusselfchecker.azurewebsites.net/');
                 }

                   if (!chatbotButton.hasClass('chatbot-viewed')) {
                          chatbotButton.addClass('chatbot-viewed');
                 }
         });

         chatbotClose.click(function() {
                   attrToggle(chatbotButton, 'aria-expanded');
                       attrToggle(chatbotButton, 'aria-pressed');
                        attrToggle(chatbotContent, 'aria-hidden');
                        chatbotButton.toggleClass('expanded');
                    chatbotContent.toggleClass('expanded');

                     tabToggle(chatbotClose);

                    chatbotButton.focus();
            });

         var cookiePresent = getCookie('JHMchatbotView');
          if (cookiePresent && !chatbotButton.hasClass('chatbot-viewed')) {
                     chatbotButton.addClass('chatbot-viewed');
         } else if (!cookiePresent) {
                      setCookie('JHMchatbotView', true);
                }
 });
