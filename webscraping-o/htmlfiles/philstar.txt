         (function($){
	fmt = {
		sep : ' - ', 
		del : ['(In exact order)'],
		cur : [', ', ' ', ','],
	}
	function split_cur(str){
		spl = "";
		$.each(fmt.cur, function(key, val){
			if (str.search(val) > 0 ){
				spl = val;
				return false;
			}
		});
		return str.split(spl);
	}
	function sanitize_name(str){
		$.each(fmt.del, function(key, val){
			str = str.replace(val, "");
		});
		return str;
	}
	function sanitize_number(str){
		return split_cur(str);
	}
	function sanitize_price(str){
		return str.replace("P", "").trim();		
	}
	function isStr(str){
		r = "int";
		arStr = split_cur(str);
		$.each(arStr, function(key, val){
			if ( isNaN( parseInt(val) ) ){
				r = "str";
				return false;
			}
		});
		if (str.search(".00") > 0){
			r = "amt";
		}
		return r;
	}
	function display_lotto(lotDat){
		str = '<div class="widget_lotto_wrap">';
		lotDat.reverse();
		console.log(lotDat);
		$.each(lotDat, function(key, attr){
			lotNum = '';
			$.each(attr.number, function(k, v){
				lotNum += '<span class="num">'+v+'</span>';
			});
			if (attr.price == "NaN"){
				lotPri = '';
			}else{
				lotPri = '<div class="lotto-price"><span class="jp">JACKPOT PRIZE: </span>'+attr.price+' PHP</div>';
			}
			str += '<div class="item item-'+key+'">\
						<div class="lotto-name">'+attr.name+'</div>\
						<div class="lotto-number">'+lotNum+'</div>\
						'+lotPri+'\
				   </div>';
		});
		str += '</div>';
		return str;
	}
	function init_adjust_format(lrh){		
		b = "";
		lrh.find("p").each(function(key){
			a = $(this).html().replace(/<br>\n\t<br>/gi,'</p><p>');
			b += "<p>"+a+"</p>";
		});
		lrh.html(b);
	}
	function lotto_detect_format(lrh){
		lotNam = "";
		lotDat = [];
		lrh.find("p").each(function(key){
			thiTex = $(this).html();
			arrLot = thiTex.split( fmt.sep );
			lnam = "";
			lnum = "";
			lpri = "NaN";
			if (arrLot.length == 1){
				if (isStr(arrLot[0]) == "amt"){
					l = lotDat.length - 1;
					lotDat[l].price = sanitize_price(arrLot[0]);
				}else{
					lotNam = sanitize_name(arrLot[0]);
					lotNam = lotNam.trim() + " ";
				}
			}else{
				if ( isStr(arrLot[0]) == "str" ){
					lotNam = "";
				}
				$.each(arrLot, function(key, val){
					arrLot[key] = val.trim();
					if ( isStr(arrLot[key]) == "str" ){
						lnam = arrLot[key];
					}else{
						lnum = arrLot[key];
						if (arrLot[key].search("<br>") > 0){
							spli_lnum = arrLot[key].split("<br>");
							if (spli_lnum.length > 1){
								lnum = spli_lnum[0];
								if (isStr(spli_lnum[1]) == "amt"){
									lpri = sanitize_price(spli_lnum[1]);
								}
							}
						}
					}
				});				
				lotDat.push({
					name : lotNam + lnam,
					number : sanitize_number(lnum),
					price : lpri,
				});
			}
		});
		lrh.show().empty().html(display_lotto(lotDat));

		  dots: false,

		  speed: 200,

		  slidesToShow: 1,
		  adaptiveHeight: false,
		})

   function carousel_actual_config(){
      return {
            items: 1,
            loop: true,
            autoplay: true,
            autoplayTimeout: 4000,
            autoplayHoverPause: true,
            responsiveClass: true,
            responsive: {

               },
               768: {
                  nav: true
               },
            }
      };

   function owl_carousel_bites_config() {
      return {
            items: 1,
            nav: true,
            responsiveClass: true,
            responsive: {
               0: {
                  autoHeight: true,
               },
               600: {
                  slideBy: 2,
                  items: 2,
               },
            }
      };
   }

            responsive: {
               0: {
                  items: 1,
               },

                  slideBy: 2,

      };
   }
