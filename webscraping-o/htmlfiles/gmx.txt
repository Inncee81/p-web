   (function() {
  if (checkIfConsentCookieIsPresent()) {
    return;
  }

  if (checkBypassCookiePresent()) {
    return;
  }

  if (checkIfCookiesAreWritable()) {
    saveHashAndQueryStringInSessionStorage();
    saveReferrers();
    redirectToInterceptionPage();
  }

  // Helper functions

  /**
   * Checks if the consent cookie is present
   * @returns {boolean} Whether the consent cookie is present or not
   */
  function checkIfConsentCookieIsPresent() {
    var localTCString = getCookie('euconsent-v2');
    return (localTCString && localTCString.length > 0);
  }

  /**
   * Checks if the bypass cookie is present
   * @returns {boolean} Whether the bypass cookie is present or not
   */
  function checkBypassCookiePresent() {
    var bypassCookieName = 'euconsent-bypass';
    var bypassCookieValue = getCookie(bypassCookieName);
    return (bypassCookieValue && bypassCookieValue === '1');
  }

  /**
   * Gets a cookie value
   * @param {string} cname The name of the cookie to get the value of
   * @returns {string} The cookie value or empty string if not found
   */
  function getCookie(cname) {
    var name = cname + '=';
    var decodedCookie = decodeURIComponent(document.cookie);
    var ca = decodedCookie.split(';');
    for (var i = 0; i < ca.length; i++) {
      var c = ca[i];
      while (c.charAt(0) === ' ') {
        c = c.substring(1);
      }
      if (c.indexOf(name) === 0) {
        return c.substring(name.length, c.length);
      }
    }
    return '';
  }

  /**
   * Checks if cookies are enabled or not
   * @returns {boolean}
   */
  function checkIfCookiesAreWritable() {
    try {
      // Create cookie
      document.cookie = 'cookietest=1';
      var ret = document.cookie.indexOf('cookietest=') !== -1;
      // Delete cookie
      document.cookie = 'cookietest=1; expires=Thu, 01-Jan-1970 00:00:01 GMT';
      return ret;
    } catch (e) {
      return false;
    }
  }

  /**
   * Save URL info in session storage
   */
  function saveHashAndQueryStringInSessionStorage() {
    var qs = window.location.search;
    var hash = window.location.hash;

    window.sessionStorage.setItem('redirectHash', hash);
    window.sessionStorage.setItem('redirectQueryString', qs);
  }

  /**
   * Save referrers in session storage
   */
  function saveReferrers() {
    // store the page to redirect to once the user has given its consent preference,
    // i.e. the current page
    var redirectReferrer = window.location.origin + window.location.pathname + window.location.search;
    window.sessionStorage.setItem('redirectReferrer', encodeURI(redirectReferrer));

    // the page where the user came from
    window.sessionStorage.setItem('originalReferrer', document.referrer);
  }

  /**
   * Perform a redirection to the consent page
   */
  function redirectToInterceptionPage() {
    window.location.href = window.ui.consentPageUrl;
  }
})();
