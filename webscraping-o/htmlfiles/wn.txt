	    function wp_parse(data) {
	        var r = [];
	        if(!data.query || !data.query.search) return r;
	        $.each(data.query.search, function(key, row) {
	            r.push({
	                data: row.title.toLowerCase(),
	                value: row.title
	            });
	        });
	        return r;
	    }

	    function bing_parse(data) {
	        var r = [];
	        var t = data.SearchResponse.RelatedSearch;
	        if(!t) return r;
	        $.each(t.Results, function(key, row) {
	            r.push({
	                data: row.Title.toLowerCase(),
	                value: row.Title
	            });
	        });
	        return r;
	    }

	    function yt_parse(data) {
	        var r = [];
	        if(!data[1]) return r;
	        $.each(data[1], function(key, row) {
	            r.push({
	                data: row[0],
	                value: row[0]
	            });
	        });
	        return r;
	    }

    function activateTab(tabName, fromClick){
        clog("Activate tab:", tabName, 'fromClick=', fromClick);
        if(tabName.match('^(local|edition)/')){ // decompose
            setPageEdition(tabName);
            tabName = 'local';
        }	
        var tabUrl = (tabName == 'local') ? window.pageEdition : tabName;

        if(tabUrl == window.currentTab) return;
        window.currentTab = tabUrl;

        var $activeTab = $('.nav a[data-url="' + tabName+'"]');	
        //$activeTab.tab('show');
				$('.nav a[data-url="' + tabName+'"]').tab('show');
        if(!fromClick){
            $activeTab.click(); // call onclick="" handler.						
        }
        updateUrl(tabUrl);
        loadTab(tabName);
        $("html, body").animate({ scrollTop: 0 }, "slow");
				$('.nav a[data-url="' + tabName+'"]').next('.submenu').slideDown();	
				$('.submenu a[data-url="' + tabName+'"]').parent().parent().slideDown();	
    };

    function get_forecast(city_name, days_count, global_geo_obj, country_code) {
        forecast_status('<p style="margin: 40px 0; text-align:center;"><img src="//ecdn4.wn.com/or/assets/assets/wn/img/712.GIF" alt="" width="64" height="64"></p>');
        $.ajax({
            data: {
                city: city_name,
                report: 'daily'
            },
            dataType: 'jsonp',
            cache: true,
            jsonpCallback: 'cbWeatherForecast',
            url: '//wn.com/api/upge/cheetah-photo-search/weather_forecast_4days',
            success: function(data) {
                if(!data) { text = ('weater data temporarily not available'); }
                forecast_status(data.city.name);
                // loop through the list of weather info
                var weather_info = '';
                var weather_day_loop = 0;
                var weather_list = data.list;
                for(var i=0; i < weather_list.length; i++){
                    if (i == 0 || i >= days_count+1) {
                        continue;
                    }
                    var weather = weather_list[i].weather.length && weather_list[i].weather[0];
                    var value = weather_list[i];
                    var clouds = value.clouds;
                    d = new Date(value.dt*1000);
                    t = d.getMonth()+1 + '-' + d.getDate() + '-' + d.getFullYear()
                    tempC = parseInt(parseFloat(value.temp.day)-273.15);
                    tempF = parseInt(tempC*1.8+32);
                    moment.lang('en', {
                        calendar : {
                            lastDay : '[Yesterday]',
                            sameDay : '[Today]',
                            nextDay : '[Tomorrow]',
                            lastWeek : '[last] dddd',
                            nextWeek : 'dddd',
                            sameElse : 'L'
                        }
                    });
                    mobj = moment(value.dt*1000)
                    // skip today
