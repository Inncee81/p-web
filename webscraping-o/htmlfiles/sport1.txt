        function topTeaserScroller(data) {
            var elements = data.elements,
                type = data.type,
                direction = 'next',
                currentIndex = 0,
                buttons,
                scrollBar,
                topTeaserElement = $('.s1-module.s1-top-teaser-chart.' + data.id),
                width = 320;

            function init() {
                buttons = topTeaserElement.find('.control > .box');
                scrollBar = topTeaserElement.find('.line');
                scrollBar.css('margin-left', 0);
                window.dataLayer = window.dataLayer || [];

                buttons.on('click', function() {
                    if ($(this).hasClass('active')) {
                        if($(this).hasClass('prev')) {
                            currentIndex--;
                            direction = 'prev';
                        } else {
                            currentIndex++;
                            direction = 'next';
                        }
                        window.dataLayer.push({
                            'ad_type': type,
                            'ad_placement': direction,
                            'ad_partner': 'int_widget',
                            'event': 'ad.click'
                        });
                    }
                    setScroll();
                });
            }

            function setScroll() {
                scrollBar.animate({
                    'margin-left': -1 * (currentIndex * width)
                }, 600);

                buttons.each(function() {
                    var currentButton = $(this);

                    if (currentButton.hasClass('prev')) {
                        currentIndex > 0 ? currentButton.addClass('active') : currentButton.removeClass('active');
                    } else {
                        currentIndex < elements - 2 ? currentButton.addClass('active') : currentButton.removeClass('active');
                    }
                });
            }

            if (topTeaserElement.length) {
                init();
            }
        }
    })();
