        // Extract time and replace with current timestamp (syntax is odd tri-part".../_training_start_date_timestamp/value/1523493622/...")
        var currentTimestamp = Math.floor(Date.now() / 1000);
        feedUrl = feedUrl.toLowerCase();
        // set to https
        feedUrl = feedUrl.replace("http://","https://");
        if(feedUrl.indexOf('json') > 10) {
            feedUrl = feedUrl.replace(/_training_start_date_timestamp\/value\/\d+\//, '_training_start_date_timestamp/value/'+currentTimestamp+'/');

            feedUrl = feedUrl.replace(/_training_start_date\/value\/\d+\//, '_training_start_date_timestamp/value/'+currentTimestamp+'/');

			type: "GET",

			dataType: 'json',

				if(typeof data !== 'undefined'){					
					buildItemRows(data);
				}
			}

            var rowCount = 0; // row counter
            var currentRowTarget = containerTarget +" "+ rowTargetPrefix + rowCount;

                if (data.hasOwnProperty(key)) {
                    var objResource = data[key];
                    //get the date of the publication
                    var d = new Date(objResource._training_start_date_timestamp*1000);           
                    objResource.fromLongDateTimeString = dateTimeFormatString(d,'longdatetime');
                    objResource.fromLongDateString = dateTimeFormatString(d,'longdate');
                    objResource.fromMonthString = dateTimeFormatString(d,'month');                        
                    objResource.fromDayString = dateTimeFormatString(d,'day');

                    // use moment.js + moment-timezone.js to get TimeZone from custom date field "1550156400|America/New_York"
                    if(objResource._training_start_date) {
                        var dateTimeWithRegion = objResource._training_start_date.split("|");                        
                        var timeZoneString = dateTimeWithRegion.pop();
                        if(timeZoneString && timeZoneString.length > 0 && typeof moment !== 'undefined' && typeof moment.tz !== 'undefined') {
                            objResource.timeZone = moment().tz(timeZoneString).format('z');
                        }

                    // Location string
                    if(objResource._training_location) {
                        objResource.location = objResource._training_location;

                    // Skip online, else add 'in' to location
                    if(objResource.location.toLocaleLowerCase() === "online") {
                        objResource.location = "";
                    } else {
                        objResource.location = " in " + objResource.location;
                    }
                    
                    // truncate to 3 char
                    objResource.fromMonString = "";
                    if(objResource.fromMonthString.length > 0) {    
                        objResource.fromMonString = dateTimeFormatString(d,'month').substring(0, 3);
