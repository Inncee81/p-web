						Percents.push("Businese");

				return json;

	var DealTypeJsonbanner = function(json,type) {

                break;

		if (content && content !== "") {
			that.find(".banner").find("a").find(".wording-background").find(".wording-content").html(content);
			hasWord = true;
		}
		if (hasWord) {

			that.find("a").attr("target", "_blank");

                    startTouchPosition.posY = null;
