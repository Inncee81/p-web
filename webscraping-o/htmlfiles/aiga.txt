    function removeUserCookie(event){

    function checkAndSetMenuActive() {
        var rootItems = $('ul#menu-main > li');
        for (var i = 0; i < rootItems.length; i++) {
            var item = $(rootItems[i]);
            var isActive = isMenuItemActive(item);
            if (isActive) {
                $(item).addClass('active');
                $(item).attr('data-active', true);
            }
        }
    }

    function isMenuItemActive(menuItem) {
        var activeMenuUrl = $('#hdActiveMenuUrl').val();
        var menuItems = $(menuItem).find('div > ul > li');
        if (menuItems.length == 0) menuItems = $(menuItem).find('ul > li');

        var isActive = false;
        var currentHref = window.location.href.toLowerCase();
        for (var i = 0; i < menuItems.length; i++) {
            var item = $(menuItems[i]);
            var $aTag = $(item).find('> a');
            if ($aTag.length > 0 && $aTag.attr('href') &&
                currentHref.indexOf($aTag.attr('href').toLowerCase()) >= 0) {
                isActive = true;
            }
            else {
                isActive = isMenuItemActive(item);
            }

            if (!isActive) {
                 isActive = isMenuItemActiveWithActiveMenuUrl(activeMenuUrl, $aTag);
            }

            if (isActive) {
                $(item).addClass('active');
                $(item).attr('data-active', true);
                break;
            }
        }

        if (!isActive) {
            isActive = isMenuItemActiveWithActiveMenuUrl(activeMenuUrl, $(menuItem).find('> a'));
        }

        return isActive;
    }

    function isMenuItemActiveWithActiveMenuUrl(activeMenuUrl, atag) {
        var isActive = false;
        if (activeMenuUrl) {
            var href = $(atag).attr('href');
            if (activeMenuUrl && href &&
                href.toLowerCase().indexOf(activeMenuUrl.toLowerCase()) >= 0) {
                isActive = true;
            }
        }
        return isActive;
    }
