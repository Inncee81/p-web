    function isURL(str_url){
        str_url = str_url.toLowerCase();
        var regular = /^((https|http|ftp|rtsp|mms)?:\/\/)?(([0-9a-z_!~*'().&amp;=+$%-]+: )?[0-9a-z_!~*'().&amp;=+$%-]+@)?((\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.(\d{1,2}|1\d\d|2[0-4]\d|25[0-5]$)|([0-9a-z_!~*'()-]+\.)*([0-9a-z][0-9a-z-]{0,61})?[0-9a-z]\.[a-z]{2,6})(:[0-9]{1,5})?(.*)$/
        if (regular.test(str_url)) {
            return true;
        }
        else {
            return false;
        }
    }

	function checkWallHost() {
        var host= getHost();
        let host_tmp = host.split('\n');
        let host_arr = [];
        let res = {};
        host_tmp.forEach((item,index)=>{
            var item = item.replace(/\s*/g,"");
            if(item){
                let is_url = isURL(item);
                if (!is_url){
                    res.code = 202
                    return res;
                }
                host_arr.push(item);
            }
        });

        if (host_arr.length < 1){
            res.code = 202
            return res;
        }
        res.code = 200;
        res.data = host_arr.toString();
        return res;

    }

    function checkip(ip)
    {
        var pattern=/^(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9])\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[1-9]|0)\.(25[0-5]|2[0-4][0-9]|[0-1]{1}[0-9]{2}|[1-9]{1}[0-9]{1}|[0-9])$/;
        flag_ip=pattern.test(ip);
        if(!flag_ip) {
            return false;
        }else{
            return true;
        }
    }
