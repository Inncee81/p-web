   function bookmark(title,url){
	if (window.sidebar) // firefox
		window.sidebar.addPanel(title, url, "");
	else if(window.opera && window.print){ // opera
		var elem = document.createElement('a');
		elem.setAttribute('href',url);
		elem.setAttribute('title',title);
		elem.setAttribute('rel','sidebar');
		elem.click();
	}
	else if(document.all)// ie
		window.external.AddFavorite(url, title);
}
function createbookmarke(title, url){
	if ((window.opera && window.print) || (document.all)) {
		var a = document.createElement('a');
		a.innerHTML = "update your bookmarks";
		a.setAttribute("title", "update your bookmarks");
		a.setAttribute("href", "javascript:bookmark('" + title + "', '" + url + "');");
		document.getElementById('bookmarke').innerHTML = "&nbsp;and ";
		document.getElementById('bookmarke').appendChild(a);
	}
	else {
		document.getElementById('bookmarke').innerHTML = " and update your bookmarks";
	}
}
function createbookmarkc(title, url){
	if ((window.opera && window.print) || (document.all)) {
		var a = document.createElement('a');
		a.innerHTML = "更新你的書籤 (bookmarks)";
		a.setAttribute("title", "更新你的書籤 (bookmarks)");
		a.setAttribute("href", "javascript:bookmark('" + title + "', '" + url + "');");
		document.getElementById('bookmarkc').innerHTML = "及";
		document.getElementById('bookmarkc').appendChild(a);
	}
	else {
		document.getElementById('bookmarkc').innerHTML = "及更新你的書籤 (bookmarks)";
	}
}
