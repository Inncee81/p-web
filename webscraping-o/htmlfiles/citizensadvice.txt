  <script>
   (function(){
                
        if ( /\bCABExtent=[^;]*(;|$)/.test(document.cookie) || /\bno-cookies\b/.test(document.documentElement.className)) {
            return;
        }
                
        document.getElementById('home-extent-popup').style.display = "";

        $(function ($) {

            // Set keyboard focus to first language (i.e. England).
            $('a.home-extent-popup__link').first().focus();

            var lang = document.documentElement.lang;

            function chooseCurrent() {
                $(document).off('.home-extent-popup');
                $('#home-extent-popup').remove();
                document.cookie = 'CABExtent=' + lang + '; path=/; expires=' + new Date(Date.now() + 172800000).toGMTString();
            }

            // Don't reload the page for the current language
            $(document)
                .on('keydown.home-extent-popup', function (e) {
                    // Close the dialog using the escape key
                    if (!e.isDefaultPrevented() && e.keyCode === 27) {
                        chooseCurrent();
                    }
                });

            // if shift-tab pressed on first language then focus on last language (England -> Wales).
            $('a.home-extent-popup__link').first().on('keydown', function (e) {
                if (e.key === 'Tab' && e.shiftKey) {
                    e.preventDefault();
                    $('a.home-extent-popup__link').last().focus();
                }
            });

            // if tab pressed on last language then focus on first language (i.e. Wales -> England).
            $('a.home-extent-popup__link').last().on('keydown', function (e) {
                if (e.key === 'Tab' && !e.shiftKey) {
                    e.preventDefault();
                    $('a.home-extent-popup__link').first().focus();
                }
            });

            $('#home-extent-popup').on('click', '[lang]', function (e) {
                if (this.lang === lang) {
                    e.preventDefault();
                    chooseCurrent();
                }
            });
        });

    }());
  </script>
