  <script type="text/javascript">
   function validateEmail(email) {
            var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
            return re.test(email);
        }
        function displayError (description) {
            $('#success').hide();
            $('#errors').show();
            $('#errors').text(description);
        }
        function postContactToGoogle(){
            var email = $('#email').val();
            var note = $('#note').val();
            if (!email || !note) {
              return displayError('Please fill out all fields')
            }
            if (!validateEmail(email)) {
              return displayError('Please enter a valid email address')
            }
            $.ajax({
                url: "https://docs.google.com/forms/d/1TZqrTcmogfkt4d_w1ogaW3mvMgJhpQs2uPmY1a80Vb8/formResponse",
                data: {"entry.1609812892" : email, "entry.1120558534" : note},
                type: "POST",
                dataType: "xml",
                statusCode: {
                    0: function (){

                        $('#email').val("");
                        $('#note').val("");
                        //Success message
                        $('#errors').hide();
                        $('#success').show();
                    },
                    200: function (){
                        $('#email').val("");
                        $('#note').val("");
                        //Success Message
                    }
                }
            });
        }
  </script>
