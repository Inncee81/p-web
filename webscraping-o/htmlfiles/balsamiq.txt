     document.addEventListener("DOMContentLoaded", function() {
        var menuButton = document.getElementById("menu-toggle");
        var menu = document.getElementById("main-menu");

        menuButton.addEventListener("click", function() {
          if (menu.getAttribute("aria-expanded") === "true") {
            BLQ.collapsible.collapse(menu);
            menu.setAttribute("aria-expanded", "false")
          } else {
            BLQ.collapsible.expand(menu);
            menu.setAttribute("aria-expanded", "true")
          }
        });

        var loginButton = document.getElementById("loginButton");
        var loginDropdown = loginButton.parentNode;

        function isLoginVisible() {
          return loginDropdown.classList.contains("open");
        }

        function onClickWhenLoginIsVisible(e) {
          if (loginButton.contains(e.target) || !isLoginVisible()) {
            return;
          }

          hideLogin();
        }

        function showLogin() {
          loginButton.setAttribute("aria-expanded", "true");
          loginDropdown.classList.add("open");
          document.addEventListener('click', onClickWhenLoginIsVisible);
        }

        function hideLogin() {
          loginButton.setAttribute("aria-expanded", "false");
          loginDropdown.classList.remove("open");
          document.removeEventListener('click', onClickWhenLoginIsVisible);
        }

        loginButton.addEventListener("click", function() {
          if (isLoginVisible()) {
            hideLogin();
          } else {
            showLogin();
          }
        });

        BalsamiqSearchAutoComplete.initAutoCompleteInputDiv("searchInNav");
      });
    </script>

   function setupTOC() {
        var tocContainer = document.getElementById('toc-container');

        if (!tocContainer) {
            return;
        }

        var content = document.getElementById('content');
        tocContainer.style.height = content.offsetHeight + 'px';

        if (Stickyfill.stickies.length) {
            Stickyfill.refreshAll();
        }
        else {
            var toc = document.getElementById('toc') || document.getElementById('lefttoc');
            Stickyfill.addOne(toc);
        }

        var tocList = document.getElementById('lefttoclist');
        if (tocList) {
            tocList.style.height = window.innerHeight-tocList.offsetTop + 'px';
        }
    }

    function lazyLoad() {
        // Lazily load images and iframes
        lozad(
            document.querySelectorAll([
                'img[data-src]',
                'iframe[data-src]',
                '*[data-background-image]',
            ].join(', ')),
        )
        .observe();
    }

    document.addEventListener('DOMContentLoaded', function () {
        // Lazily load stuff
        lazyLoad();

        // Highlight the current parent in the navs

            var currentNav = [].slice.call(document.querySelectorAll('ul.'+navName+'__menu a')).sort((a, b) => a.length > b.length ? 1: -1).filter(nav => window.location.href.startsWith(nav.href))[0];

        });
        var localNav = [].slice.call(document.querySelectorAll('ul.localNav a')).sort((a, b) => a.length > b.length ? 1: -1).filter(nav => window.location.href.startsWith(nav.href))[0];

            //Breadcrumb Menus
            var currentBreadcrumbMenuDiv = null;
            var outTimer = null;

            function isbreadcrumbMenuVisible() {
                return currentBreadcrumbMenuDiv && currentBreadcrumbMenuDiv.style.display === 'block';
            }

            function startClosingMenu() {
                if (isbreadcrumbMenuVisible()) {
                    if (!outTimer) {
                        outTimer = setTimeout(function() {
                            if (currentBreadcrumbMenuDiv) {
                                clearTimeout();
                                currentBreadcrumbMenuDiv.setAttribute("aria-expanded", "false");
                                currentBreadcrumbMenuDiv.style.display = "none";
                                currentBreadcrumbMenuDiv = null;
                            }
                        }, 200);
                    } // else already closing
                }
            }

            function clearOutTimer() {
                clearTimeout(outTimer);
                outTimer = null;
            }

            function breadcrumbAmouseOver() {
                clearOutTimer();

                if (isbreadcrumbMenuVisible() && currentBreadcrumbMenuDiv !== this.nextElementSibling) {
                    //it's not my menu, close it right away
                    currentBreadcrumbMenuDiv.setAttribute("aria-expanded", "false");
                    currentBreadcrumbMenuDiv.style.display = "none";
                    currentBreadcrumbMenuDiv = null;
                }

                if (!isbreadcrumbMenuVisible()) {
                    //show the menu
                    currentBreadcrumbMenuDiv = this.nextElementSibling;
                    currentBreadcrumbMenuDiv.setAttribute("aria-expanded", "true");
                    currentBreadcrumbMenuDiv.style.display = "block";
                    currentBreadcrumbMenuDiv.addEventListener('mouseover', function() {
                        clearOutTimer();
                    });
                    currentBreadcrumbMenuDiv.addEventListener('mouseout', function(event) {
                        var e = event.toElement || event.relatedTarget;
                        if (e.parentNode.parentNode == currentBreadcrumbMenuDiv || e.parentNode.parentNode.parentNode == currentBreadcrumbMenuDiv || e == this) {
                            return; //ignore children
                        }
                        startClosingMenu();
                    });
                }
            }

            var breadcrumbs = document.getElementById('breadcrumbsUL');
            if (breadcrumbs) {
                var listenersOn = false;
                function onResizeForBreadcrumbs() {
                    var breadcrumbAs = document.getElementsByClassName('breadcrumbA');
                    if (breadcrumbs.offsetHeight <= 52) {
                        //they fit in one line, add the menus
                        if (!listenersOn) {
                            for (var i = 0; i < breadcrumbAs.length; i++) {
                                var menuBtn = document.createElement('span');
                                menuBtn.innerHTML = " ▾";
                                breadcrumbAs[i].append(menuBtn);
                                breadcrumbAs[i].addEventListener("mouseover", breadcrumbAmouseOver);
                                breadcrumbAs[i].addEventListener("mouseout", startClosingMenu);
                                listenersOn = true;
                            }
                        }
                    } else {
                        //they don't fit in one line, just have the breadcrumbs
                        if (listenersOn) {
                            for (var i = 0; i < breadcrumbAs.length; i++) {
                                breadcrumbAs[i].removeChild(breadcrumbAs[i].childNodes[1]);
                                breadcrumbAs[i].removeEventListener("mouseover", breadcrumbAmouseOver);
                                breadcrumbAs[i].removeEventListener("mouseout", startClosingMenu);
                            }
                            listenersOn = false;
                        }
                    }
                }
                window.addEventListener('resize', onResizeForBreadcrumbs);
                onResizeForBreadcrumbs();
            }
        }        
    });
