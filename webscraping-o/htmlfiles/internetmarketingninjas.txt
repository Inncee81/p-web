      function twitterfy(text) {
      	text = text.replace(/(\b(https?|ftp|file):\/\/[-A-Z0-9+&@#\/%?=~_|!:,.;]*[-A-Z0-9+&@#\/%=~_|])/i, "<a href='$1' target='_blank'>$1</a>")
                 	   .replace(/(^|\s)#(\w+)/, "$1<a href='http://search.twitter.com/search?q=%23$2' target='_blank'>#$2</a>")
                 	   .replace(/(^|\s)@(\w+)/, "$1<a href='http://www.twitter.com/$2' target='_blank'>@$2</a>")

          return text;
      }

      function parseTwitterDate(date) {
      	var months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
      	var d = new Date(date);
      	var month = months[d.getMonth()];
      	var day = d.getDate();
      	var hours = d.getHours();

      	// 12 hour instead of 24
      	var suff = "am";
      	if (hours==0) { hours=12; }
      	else if (hours==12) { suff="pm"; }
      	else if (hours>12) { hours=(hours-12); suff="pm"; }

      	// 0 in front of single digits
      	var mins = d.getMinutes();
      	if (mins<10) { mins = "0"+mins; }

        	return month+' '+day+' @ '+hours+":"+mins+suff;
      }
