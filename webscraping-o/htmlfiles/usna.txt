            // convert to integers from strings
            a = parseInt($(a).attr("data-timestamp"));
            b = parseInt($(b).attr("data-timestamp"));

            // compare
            if (a > b) {
               return 1;
            } else if (a < b) {
               return -1;
            } else {
               return 0;
            }
         });

         if (listLimit>0 && events.length > listLimit) {
            eventsList.children(":gt("+(listLimit-1)+")").hide();
         }
         parentContainer.html(eventsList);
               });
