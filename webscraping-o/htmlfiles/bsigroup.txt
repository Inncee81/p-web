        function isValidEmail(s) {
            if (isEmpty(s)) return false;
            if (isWhitespace(s)) return false;
            var i = 1;
            var sLength = s.length;
            while ((i < sLength) && (s.charAt(i) != "\@")) {
                i++;
            }
            if ((i >= sLength) || (s.charAt(i) != "\@")) return false;
            else i += 2;
            while ((i < sLength) && (s.charAt(i) != ".")) {
                i++;
            }
            if ((i >= sLength - 1) || (s.charAt(i) != ".")) return false;
            else return true;
        }

            }
            return true;
