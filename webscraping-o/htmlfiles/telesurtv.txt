    // Get the modal
const $modal = document.getElementById("myModal");
$modal.style.background = "rgba(0,0,0,0.5)";

const $modalContent = document.querySelector('.modal-content');

// Get the button that opens the modal
const $liveButton = document.querySelector(".button-red.large");

const $minimizeStreaming = document.querySelector('.minimizeStreaming');
const $maximizeStreaming = document.querySelector('.maximizeStreaming');

//Get the <input type="range"> element that control the modal's opacity
const $opacityStreaming = document.getElementById('inputOpacityStreaming');
let opacityStreamingValue = 0.5;

function rangeValue(){
  opacityStreamingValue = $opacityStreaming.value;
  $modal.style.background = "rgba(0,0,0,"+opacityStreamingValue+")";
}

$opacityStreaming.addEventListener("input", rangeValue);

function showHiddenWindowOptions() {
  if(isStreamingMin){
    $maximizeStreaming.classList.remove('d-none');
    $minimizeStreaming.classList.add('d-none');
    
  }else{
    $maximizeStreaming.classList.add('d-none');
    $minimizeStreaming.classList.remove('d-none');
  }
}

showHiddenWindowOptions();

function overflowController(){
  if(window.screen.width <= 900 && isActiveStreaming == true){
    document.getElementsByTagName("html")[0].classList.add('overflow-hidden');
    //document.getElementsByTagName("html")[0].style.position = "relative";
    document.getElementsByTagName("body")[0].classList.add('overflow-hidden');
    document.getElementsByTagName("body")[0].classList.add('p-relative');
    
  }else{
    document.getElementsByTagName("html")[0].classList.remove('overflow-hidden');
    //document.getElementsByTagName("html")[0].style.position = "unset";
    document.getElementsByTagName("body")[0].classList.remove('overflow-hidden');
    document.getElementsByTagName("body")[0].classList.remove('p-relative');
    
  }
}

function showHiddenStreaming(state) {
  if(state === 'show'){
    isActiveStreaming = true;
    $modal.style.display = "block";
    mostrarStreaming();
    overflowController();
  }

  if(state === 'hidden'){
    isActiveStreaming = false;
    $modal.style.display = "none";
    document.getElementsByTagName("html")[0].style.overflow = "auto";
    ocultarStreaming();
    overflowController();
  }
}

function isMinimize(modalIsMinimize) {

  const $iframeEnVivo = document.querySelector('.box-streaming #envivoid');
  const $liveRedes = document.querySelector('.liveRedes');
  const $opacityOptions = document.querySelector('.opacityOptions');

  if(modalIsMinimize){
  
    isStreamingMin = true;

    document.getElementsByTagName("html")[0].style.overflow = "auto";
  
    $modal.classList.remove('modal');
    $modal.classList.add('min-modal');
    $modal.style.background = "transparent";
  
    $modalContent.classList.remove('modal-content');
    $modalContent.classList.add('min-modal-content');
  
    $iframeEnVivo.classList.add('min-envivoid');
  
    $liveRedes.classList.add('d-none');
  
    $opacityOptions.classList.add('d-none');
    
    showHiddenWindowOptions();

  }else{
  
    isStreamingMin = false;

    //document.getElementsByTagName("html")[0].style.overflow = "hidden";
  
    $modal.classList.add('modal');
    $modal.classList.remove('min-modal');
    $modal.style.background = "rgba(0,0,0,"+opacityStreamingValue+")";
  
    $modalContent.classList.add('modal-content');
    $modalContent.classList.remove('min-modal-content');
  
    $iframeEnVivo.classList.remove('min-envivoid');
  
    $liveRedes.classList.remove('d-none');
  
    $opacityOptions.classList.remove('d-none');
    
    showHiddenWindowOptions();
  }
}

const requestVideo = ( namespace, route ) => {

    const URL = namespace + route;
    let destacada = [];
    let secundarias = [];

    const headers = new Headers({
        'Content-Type': 'application/json',
    });

    fetch( URL, {
        method: 'GET',
        headers: headers,
        //credentials: 'same-origin',
    } )
    .then( response => {
        return response.ok ? response.json() : '¡Ops! Al parecer ha ocurrido un error...';
    })
    .then( json_response => {

        //const dataPackage = json_response[0] !== null ? json_response[0] : json_response;
        const dataPackage = json_response;
        const videoSource = json_response[1] !== null ? json_response[0] : null;
        const thumbnailSource = json_response[2] !== null ? json_response[0] : null;
        const cloudfrontLink = 'https://dddfnn6w75cj1.cloudfront.net/';

        for( i = 0; i <= dataPackage.length-1; i++ ) {
            const title = dataPackage[i].title;
            const abstract = dataPackage[i].abstract;
            const duration = dataPackage[i].media_duration;
            //const thumbnail = thumbnailSource[i] + dataPackage[i].thumbnail_url;
            const thumbnail = cloudfrontLink + dataPackage[i].thumbnail_url;
            const embed = 'https://videos.telesurtv.net/embed/0/' + dataPackage[i].id;
            const data = {
                title,
                abstract,
                duration,
                thumbnail,
                embed,
            }

            if( i == 0 ) {
                destacada.push( data );
            } else {
                secundarias.push( data );
            }
        }

	const destacadaElement = generateDestacada( destacada[0] );
	const destacadaContent = document.querySelector( '#VideoDestacado .important' );
	destacadaContent.insertAdjacentElement('afterbegin', destacadaElement);
	
	const secundariasElement = generateSecundarias( secundarias );
	const secundariasContent = document.querySelector( '.secundarias.aside' );
	secundariasContent.insertAdjacentElement( 'beforeend', secundariasElement );
        
    } )
    .catch( error => {
        console.error( 'Error:', error );
    } );

    return { destacada, secundarias };
}

     <script>
      class SliderOpinion {
  constructor(numElementsToWatch, content){
    this.content = content
    this.elements = [...this.content.querySelectorAll('.boxOpi')]
    this.elementsNoWatch = []
    this.elementsToWatch = []

    this.model(numElementsToWatch)
    this.view()
    this.controller()
    this.painter(this.elementsToWatch)
  }

  model(numElementsToWatch){
    for(let i = 0; i < numElementsToWatch; i++){
      this.elementsToWatch.push(this.elements[i])
    }
    this.elements.splice(0, numElementsToWatch)
    this.elements.forEach((element => {
        this.elementsNoWatch.push(element)
    }))
  }

  view(){
    this.elementsNoWatch.forEach((element) => {
      element.style.width = '7.5em'
      element.style.float = 'left'
      element.style.display = 'block'
    })
  }

  controller(){
    document.getElementById('slideopinionmore')
      .addEventListener('click', () => {
        this.elementsNoWatch.push(this.elementsToWatch.shift())
        this.elementsToWatch.push(this.elementsNoWatch.shift())

        this.painter(this.elementsToWatch)
       })
       
    document.getElementById('buttonReturnOpi')
      .addEventListener('click', () => {
        this.elementsToWatch.unshift(this.elementsNoWatch.pop())
        this.elementsNoWatch.unshift(this.elementsToWatch.pop())

        this.painter(this.elementsToWatch)
    })
  }

  painter(elementsToWatch){
    this.content.innerHTML = ''

    elementsToWatch.forEach((element) => {
        this.content.insertAdjacentElement('beforeend', element)
    })
  }

}
