  $(function(){
    var $el = $('.parallax-bg');
    if ($el.length == 0) {
        return;
    }
    $(window).scroll(function () {
        para($el);
    });
    para($el);
});
var speed = .8;
function para($el) {
    var diff = $(window).scrollTop() - $el.offset().top;
    var yPos = -(diff * speed);
    var coords = '50%' + yPos + 'px';
    $el.css({
        backgroundPosition: coords
    });
}
