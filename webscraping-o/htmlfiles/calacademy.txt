      if (page.indexOf(val) !== -1) {
        boo = true;
        return false;
      }
    });

    return boo;
  }

  /*
  one popup per session
  AND
  max one donate + one RO popup per day
  EOY lightbox on every visit

  var cookieDayNumber = 1;

    var visitation = [
    '',
    'eticket-program',
    'buy-tickets',
    'become-a-member',
    'press',
    'plan-your-visit',
    'hours-admission',
    'getting-here',

    'exhibits-events',
    'exhibits',

    'penguinspajamas-sleepovers',
    'events'
    ];

    var page = window.location.pathname.toLowerCase();
    var part = page.split('/')[1];
    var boo = false;

    $.each(visitation, function (i, val) {
      if (val == part) {
        boo = true;
        return false;
      }
    });
