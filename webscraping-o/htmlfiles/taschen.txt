    function search_2_submit_search_modal()
	{
	//console.log("function search_2_submit_search_modal()");

	// delimiter f. Phrase (immer als Phrase behandeln)
	//var tmp_delimiter = '-';
	var tmp_delimiter = '+';

	var tmp_search_string = $('#search_string').val();
	var tmp_search_subnav = $('#search_subnav').val();

	//console.log("tmp_search_string: "+tmp_search_string);
	//console.log("tmp_search_subnav: "+tmp_search_subnav);

		// ggf. von Hochkommas umgeben? Suchphrase!
		if((tmp_search_string.charAt(0) == '"')
		&& (tmp_search_string.charAt(tmp_search_string.length-1) == '"'))
		{
		//console.log("von Hochkommas umgeben!");

		// Hochkommas weg:
		tmp_search_string = tmp_search_string.substring(1, (tmp_search_string.length-1));

		//tmp_delimiter = '-';
		}

	// Slash durch Underscore ersetzen
	tmp_search_string = tmp_search_string.replace(/\//g, '_'); // global replacement

	// klein
	tmp_search_string = tmp_search_string.toLowerCase();

	// Leerzeichen durch + oder - ersetzen
	tmp_search_string = tmp_search_string.replace(/ /g, tmp_delimiter); // global replacement

	//console.log("tmp_search_string: "+tmp_search_string);

	// encode uri
	tmp_search_string = encodeURI(tmp_search_string);

	// ( und ) encoden - macht encodeURI irgendwie nicht
	tmp_search_string = tmp_search_string.replace(/\(/g, '%28'); // global replacement
	tmp_search_string = tmp_search_string.replace(/\)/g, '%29'); // global replacement

	var search_location = 'https://www.taschen.com/pages/en/search/'+tmp_search_string;

	//console.log("search_location: "+search_location);

	window.location.href = search_location;

	return false;
	}
