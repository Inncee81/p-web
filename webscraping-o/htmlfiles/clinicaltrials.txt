	if ((userAgent.indexOf("msie") > -1) || (userAgent.indexOf("rv:11") > -1)) 
	{
		$( ".ct-clearFieldIE" ).css('padding', '0 4px');
	} else {
    	$( ".ct-clearField" ).css('display', 'inline');
	};

				$( "#Country" ).on('change', function() {
		var theCountry = $(this).val();
		$( "#State" ).val(""); 
		$( "#City" ).val("");
		$( "#Distance" ).val(""); 
		if (theCountry == "") {
			$( "#StateRow" ).hide();
		} else {
			if (theCountry == "US") {
				$( "#StateField" ).show();
			} else {
				$( "#StateField" ).hide();
			};
			$( "#StateRow" ).show();
		}
	});

	$( "#State" ).on('change', function() {
		var theState = $(this).val();
		if(theState != "") $("#StateWarning").text("").hide();
	});
	
	$( "#City" ).on('change', function() {
		var theCity = $(this).val();
		if(theCity == "") $("#StateWarning").text("").hide();
	});
	
				$( "#MainForm" ).on('submit', function( event ) {
		var theCountry = $( "#Country" ).val();
				if (theCountry == "") return true;
		var theState = $( "#State" ).val();
		var theCity = $( "#City" ).val();
		if (theCountry == "US" &&
			theState == "" && theCity != "")
		{
			$("#StateWarning").text("Please select a State when specifying a US City").show();
			$("#State").trigger('focus');
			event.preventDefault();
			return false;
		}
				if($('input:invalid').length==0)
		{
			showWorking();
		} 
		return true;
	});
	
	$( "input[name='recrs']" ).checkboxradio();
	
		$("input[name='recrs']").prop("checked", false);
	$("input#home-search-condition-query").val("");
	$("input#home-search-other-query").val("");
	$("#Country").val("");
	$("#State").val("");
	$("#City").val("");
	$("#Distance").val("");
});

	// Dynamically builds a new Advanced Search URL and loads that page
	var gotoAdvSearchURL = function() {
		var advSearchURL = "/ct2/search/advanced";
		var params	= new Object();
		params.cond		= $("input#home-search-condition-query").val();
		params.term		= $("input#home-search-other-query").val();
		params.cntry			= $("#Country").val();
		params.state			= $("#State").val();
		params.city				= $("#City").val();
		params.dist			= $("#Distance").val();

		advSearchURL = advSearchURL + '?' + $.param(params); // URL encodes too
		window.location.href = advSearchURL;
	};
