      //Adding the locale preference to the cookie for 120 days
function setLocaleCookie(obj) {
var frontDoorCookie = "EMR_FRONTDOOR_LOCALE";
var localeTagValue = obj.getAttribute('data-locale-tag');
var totalDays = 120;
var localeSwitchUrl = window.location.protocol + '//' + window.location.host + '/' + localeTagValue;
var date = new Date();
date.setTime(date.getTime() + (totalDays * 24 * 60 * 60 * 1000));
var expires = "; expires=" + date.toGMTString();
document.cookie = frontDoorCookie + "=" + localeTagValue + expires + "; path=/";
if (typeof localeSwitchUrl !== 'undefined' && localeSwitchUrl.length > 0) {
window.location.href=localeSwitchUrl;
}
}
