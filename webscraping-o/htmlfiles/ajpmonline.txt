    function gpt_InitAds() {
        googletag.cmd.push(function () {

            //***********************************************
            //Start Define Size mapping based on client Viewport
            //***********************************************
            var gptLeaderboardSizeMapping = googletag.sizeMapping().
                addSize([992, 0], [[728, 90]]). 						// Desktop
                addSize([768, 0], [[728, 90]]). 								// Tablet Landscape
                addSize([0, 0], [[320, 50], [300, 50]]). 						// Phone & Tablet Portrait
                build();

            var gptSideBarSizeMapping = googletag.sizeMapping().
                addSize([1140, 795], [[300, 600], [160, 600], [120, 600], [300, 250]]). 	// SkyScraper - Desktop (Require min 630 vertical)
                addSize([1140, 445], [[300, 250]]).			 				// BoomBox Fall Back - Desktop (Require min 280 vertical)
                addSize([0, 0], []). 										// Phone
                build();

            var gptBoomBoxSizeMapping = googletag.sizeMapping().
                addSize([768, 0], [[300, 250], [336, 280]]). 					// Desktop & Tablet Landscape
                addSize([0, 0], []). 											// Phone
                build();

            var gptFluidMapping = googletag.sizeMapping().
                addSize([0, 0], ['fluid']).
                build();

            var gptPrestitialMapping = googletag.sizeMapping().
                addSize([768, 0], [[1, 1]]). 					// Desktop & Tablet Landscape
                addSize([0, 0], []). 							// Phone
                build();

            //***********************************************
            //End Define Size mapping based on client Viewport
            //***********************************************

            //***********************************************
            //Start detection for ad slots on page
            //***********************************************
            var gptMasterPage = GPT_CheckForNULL($('#gptPage').val());
            var gptMasterSite = GPT_CheckForNULL($('#gptSite').val());

            //If there is no master site, check for a defaultsite
            if (gptMasterSite == null) {
                gptMasterSite = GPT_CheckForNULL($('#gptDefaultSite').val());
            }

            //If there is no master Page, check for a Page using the URL
            if (gptMasterPage == null) {
                gptMasterPage = GPT_GetPageByURL();
            }

            $(".gpt-ad").each(function (index) {
                var gptPage = '';
                var adSizeMapping = '';
                var adId;
                var gptPOS;
                var gptSite = null;
                var tempSlot = null;

                //Check if the ad slot has it's own site defined
                if (!typeof $(this).data('site') == 'undefined') {
                    gptSite = GPT_CheckForNULL($(this).data('site'));
                }
                else {
                    //Use the Master Site
                    gptSite = gptMasterSite;
                }

                if (gptSite != null) {
                    adId = 'gpt-ad-' + index + 1;

                    gptPOS = parseInt($(this).data('pos')) || -1;
                    if (gptPOS == -1) {
                        gptPOS = index + 1;
                    }

                    //Check if the ad slot has it's own page defined
                    if (!typeof $(this).data('page') == 'undefined') {
                        gptPage = GPT_CheckForNULL($(this).data('page'));
                    }
                    else {
                        //Use the Master Page
                        gptPage = gptMasterPage;
                    }

                    //If the page is not null, preceed it with a slash
                    if (gptPage != null) {
                        gptPage = '/' + gptPage;
                    }
                    else {
                        gptPage = '';
                    }

                    //Determine sizemapping based on data-adsize
                    switch ($(this).data('adsize')) {
                        case 'cell-leaderboard':
                        case 'leaderboard':
                            adSizeMapping = gptLeaderboardSizeMapping;
                            tempSlot = googletag.defineSlot('/6053/els.' + gptSite + gptPage, [[728, 90], [320, 50], [300, 50]], adId);
                            break;
                        case 'bottom':
                        case 'boombox':
                            adSizeMapping = gptBoomBoxSizeMapping;
                            tempSlot = googletag.defineSlot('/6053/els.' + gptSite + gptPage, [[300, 250]], adId);
                            break;
                        case 'sidebar':
                        case 'skyscraper':
                            adSizeMapping = gptSideBarSizeMapping;
                            tempSlot = googletag.defineSlot('/6053/els.' + gptSite + gptPage, [[120, 600], [160, 600], [300, 600], [300, 250]], adId);
                            break;
                        case 'native':
                        case 'fluid':
                            adSizeMapping = gptFluidMapping;
                            tempSlot = googletag.defineSlot('/6053/els.' + gptSite + gptPage, 'fluid', adId);
                            break;
                        case 'prestitial':
                            adSizeMapping = gptPrestitialMapping;
                            tempSlot = googletag.defineSlot('/6053/els.' + gptSite + gptPage, 'fluid', adId);
                            break;
                    }
                }

                //Ignore any invalid adsizes that could be on the page
                if (adSizeMapping != '') {
                    $(this).addClass('gpt-ad-defined');
                    $(this).attr('id', adId);
                    tempSlot.defineSizeMapping(adSizeMapping).addService(googletag.pubads()).setTargeting("pos", gptPOS);
                }
                else {
                    $(this).remove();
                }
            });
            //***********************************************
            //End detection for ad slots on page
            //***********************************************

            //***********************************************
            //Start page level targeting
            //***********************************************
            //Logged in or IP Access
            gpt_TargetAccessType();

            //Check for DFP Test
            gpt_GetDFPTestId();

            //Check for Staging
            gpt_CheckForStaging();

            //Check for Additional Targeting
            gpt_GetAdditionalTargeting()

            //***********************************************
            //End page level targeting
            //***********************************************

            //Enable google dfp services
            googletag.pubads().enableSingleRequest();
            googletag.pubads().collapseEmptyDivs();
            googletag.enableServices();

            //Add Listener for Leaderboard
            googletag.pubads().addEventListener('slotRenderEnded', function (event) {
                if (!event.isEmpty) {
                    $('#' + event.slot.getSlotElementId()).closest(".GPT-ad-placeholder").css("display", "block");
                }
            });

            //Populate Defined Ad Slots
            $(".gpt-ad-defined").each(function (index) {
                var adId = $(this).attr('id');
                googletag.cmd.push(function () { googletag.display(adId); });
            });

        });
    }

                      Featured This Month
                     </h2>
                     <div class="widget-body body body-none body-compact-all">
                      <div data-pb-dropzone="contents">
                       <div class="widget layout-two-columns none widget-none widget-compact-all" id="ba9a1002-df4c-47a6-92c3-5d07e69607e2">
                        <div class="wrapped">
                         <div class="widget-body body body-none body-compact-all">

                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <a class="contentLink floatLeft" href="/article/S0749-3797(21)00012-X/fulltext" shape="rect">
                                 <img alt="girl looking into empty refrigerator" class="imageWrapStyle border background" src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/Health%20Advance/journals/amepre/Nagata_COVID-19-mental-health-food-insufficiency.jpg" width="150"/>
                                </a>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                           <div class="width_11_16">
                            <div class="pb-autoheight" data-pb-dropzone="right">
                             <div class="widget general-html none widget-none widget-compact-all" id="7a0d778c-22ad-4e93-8565-5287045b4c43">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <p>
                                 <a href="/article/S0749-3797(21)00012-X/fulltext">
                                  <b>
                                   Food Insufficiency and Mental Health in the U.S. During the COVID-19 Pandemic
                                  </b>
                                  <!--<img src="https://www.ajpmonline.org/templates/jsp/_style2/_marlin/images/markers/cmeamp_s.png" border="0" alt="" width="18" height="12" valign="bottom"><img src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/webfiles/images/icons/icon_free.gif" border="0" alt="" width="14" height="14" valign="bottom"> -->
                                 </a>
                                 <br/>
                                 Data from the U.S. Census Household Pulse Survey showed a near 25% increase in food insufficiency, or extreme food insecurity, during the early months of the COVID-19 pandemic. People who were younger, Black or Latinx, in larger households, and with several other social factors were more likely to experience food insufficiency. Those with food insufficiency were more likely to have symptoms of poor mental health. However, receiving free meals or groceries attenuated this relationship.
                                </p>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                          </div>
                         </div>
                        </div>
                       </div>
                       <div class="widget layout-two-columns none widget-none widget-compact-all" id="3c38d2f2-80cc-4ee1-b30b-f75441b94bcc">
                        <div class="wrapped">
                         <div class="widget-body body body-none body-compact-all">
                          <div class="pb-columns row-fluid">
                           <div class="width_5_16">
                            <div class="pb-autoheight" data-pb-dropzone="left">
                             <div class="widget general-html none widget-none widget-compact-all" id="492138b5-877b-48e9-987c-13a03dd5c395">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <a class="contentLink floatLeft" href="/article/S0749-3797(21)00019-2/fulltext" shape="rect">
                                 <img alt="man smoking an ecigarette" class="imageWrapStyle border background" src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/Health%20Advance/journals/amepre/Torner_ecigarettes-and-attention-deficit-hyperactivity-disorder2.jpg" width="150"/>
                                </a>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                           <div class="width_11_16">
                            <div class="pb-autoheight" data-pb-dropzone="right">
                             <div class="widget general-html none widget-none widget-compact-all" id="63160562-a869-414d-aee8-cf41d6c422e5">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <p>
                                 <a href="/article/S0749-3797(21)00019-2/fulltext">
                                  <b>
                                   Association of Attention-Deficit/Hyperactivity Disorder With E-Cigarette Use
                                  </b>
                                  <!-- <img src="https://www.ajpmonline.org/templates/jsp/_style2/_marlin/images/markers/cmeamp_s.png" border="0" alt="" width="18" height="12" valign="bottom"><img src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/webfiles/images/icons/icon_free.gif" border="0" alt="" width="14" height="14" valign="bottom">-->
                                 </a>
                                 <br/>
                                 A survey of almost 200,000 U.S. undergraduate and graduate students found that 7.89% used e-cigarettes and 8.10% had been diagnosed with attention-deficit/hyperactivity disorder (ADHD). Adjusted analysis found that those with a history of ADHD diagnosis were significantly more likely to use e-cigarettes than those with no such history, putting them at risk for health consequences of e-cigarette use.
                                </p>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                          </div>
                         </div>
                        </div>
                       </div>

                        <div class="wrapped">
                         <div class="widget-body body body-none body-compact-all">
                          <div class="pb-columns row-fluid">
                           <div class="width_5_16">
                            <div class="pb-autoheight" data-pb-dropzone="left">
                             <div class="widget general-html none widget-none widget-compact-all" id="011bceb1-51ba-4216-b5d2-7b7d1848c454">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <a class="contentLink floatLeft" href="/article/S0749-3797(21)00017-9/fulltext" shape="rect">
                                 <img alt="aspirin" class="imageWrapStyle border background" src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/Health%20Advance/journals/amepre/Leupker_aspirin-use-misuse-for-cardiovascular-disease.jpg" width="150"/>
                                </a>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                           <div class="width_11_16">
                            <div class="pb-autoheight" data-pb-dropzone="right">
                             <div class="widget general-html none widget-none widget-compact-all" id="67def78a-d691-4b21-9888-c4266b3cc86b">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <p>
                                 <a href="/article/S0749-3797(21)00017-9/fulltext">
                                  <b>
                                   Aspirin Use and Misuse for the Primary Prevention of Cardiovascular Diseases
                                  </b>
                                 </a>
                                 <!--<img src="https://www.ajpmonline.org/templates/jsp/_style2/_marlin/images/markers/cmeamp_s.png" border="0" alt="" width="18" height="12" valign="bottom"><img src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/webfiles/images/icons/icon_free.gif" border="0" alt="" width="14" height="14" valign="bottom"> -->
                                 <br/>
                                 Many adults use aspirin daily for primary prevention of cardiovascular disease. Authors found that aspirin was indicated for about 1 in 3 adults surveyed who were not in a secondary prevention category. However, more than half of these did not use aspirin despite indications. Further, more than 1 in 5 surveyed used aspirin when not indicated, suggesting suboptimal adherence to guidelines.
                                </p>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                          </div>
                         </div>
                        </div>

                        <div class="wrapped">
                         <div class="widget-body body body-none body-compact-all">
                          <div class="pb-columns row-fluid">
                           <div class="width_5_16">
                            <div class="pb-autoheight" data-pb-dropzone="left">
                             <div class="widget general-html none widget-none widget-compact-all" id="575bdd40-e936-45de-8b13-882262661b22">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <a class="contentLink floatLeft" href="/article/S0749-3797(21)00010-6/fulltext" shape="rect">
                                 <img alt="stethoscope and money" class="imageWrapStyle border background" src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/Health%20Advance/journals/amepre/Chua_influenza-hospitalizations-Medicare.jpg" width="150"/>
                                </a>
                               </div>
                              </div>
                             </div>
                            </div>
                           </div>
                           <div class="width_11_16">
                            <div class="pb-autoheight" data-pb-dropzone="right">
                             <div class="widget general-html none widget-none widget-compact-all" id="dfb22810-619e-47ed-83e2-375b69861656">
                              <div class="wrapped">
                               <div class="widget-body body body-none body-compact-all">
                                <p>
                                 <a href="/article/S0749-3797(21)00010-6/fulltext">
                                  <b>
                                   Out-of-Pocket Spending for Influenza Hospitalizations in Medicare Advantage
                                  </b>
                                  <!-- <img src="//els-jbs-prod-cdn.jbs.elsevierhealth.com/pb/assets/raw/webfiles/images/icons/icon_free.gif" border="0" alt="" width="14" height="14" valign="bottom"> -->
                                 </a>
                                 <br/>
                                 Many Medicare Advantage plans have waived cost sharing for COVID-19 hospital stays, but these waivers are voluntary and could end. To understand potential cost sharing for Medicare Advantage patients hospitalized for COVID-19, authors analyzed out-of-pocket spending for 14,278 influenza hospitalizations, which require similar types of care, among this group. Mean out-of-pocket spending was almost $1,000, with 3% exceeding $2,500. Legislation mandating and extending cost sharing waivers for COVID-19 hospitalization may encourage the ill to seek earlier care.
                                </p>
