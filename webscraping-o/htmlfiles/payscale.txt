      var carouselApp1 = new Vue({
            el: '#brandedcolumns-1',
            data: {
                activeSlide: 1,
                slidesLength: 3,
                to: null,
            },
            mounted() {
                this.setTimer()
            },
            methods: {
                setTimer() {
                    if ( this.to ) {
                        clearTimeout(this.to);
                    }
                    this.to = setTimeout(this.autoSlide, 10000);
                },
                chooseSlide(slide) {

                    this.setTimer();
                    this.activeSlide = slide;
                    this.slideBG(slide);
                },
                autoSlide() {
                        if (this.activeSlide == this.slidesLength) {
                            var nextslide = 1;
                        } else {
                            var nextslide = this.activeSlide + 1;
                        }
                        this.chooseSlide(nextslide);

                        this.setTimer();
                },
                slideBG(location) {
                    gsap.to(".two-columns-branded .gradient-bg", {
                        backgroundPositionX: `${((location - 1) / this.slidesLength) * 100}%`,
                        ease: 'ease-in-out',
                        duration: .4
                    })
                }
            }
        })
