        var slideWrapper = $(".marquee-slider"),

hv = window.hv || {};
(function(window, document, $, hv) {
	hv.homeMarquee = {
		init: function(){
			hv.homeMarquee.bindEventsSelectors();
		},
		postMessageToPlayer: function(player, command){
			if (player == null || command == null) return;
			player.contentWindow.postMessage(JSON.stringify(command), "*");
		},
		playPauseVideo: function(slick, control){
			var currentSlide, slideType, startTime, player, video;

			currentSlide = slick.find(".slick-current");
			slideType = currentSlide.attr("class").split(" ")[1];
			player = currentSlide.find("iframe").get(0);
			startTime = currentSlide.data("video-start");
			if (slideType === "video") {
				$('.hv-hero-marquee .vjs-social-overlay').addClass('vjs-hidden');
				video = currentSlide.find("video").get(0);
				if (video != null) {
					if (control === "play") {
						video.play();
					} else {
						video.pause();
					}
				}
			}
		},
		bindEventsSelectors: function() {
			var ctime = Number($('.hv-hero-marquee .carousel-time').text());
			var cstatus = $('.hv-hero-marquee .carousel-auto').text();
			// Initialize
			slideWrapper.on("init", function (slick) {
				slick = $(slick.currentTarget);
				setTimeout(function () {
					hv.homeMarquee.playPauseVideo(slick, "play");
				}, 1000);
				//resizePlayer(iframes, 16 / 9);
			});
			slideWrapper.on("beforeChange", function (event, slick) {
				slick = $(slick.$slider);
				hv.homeMarquee.playPauseVideo(slick, "pause");
			});
			slideWrapper.on("afterChange", function (event, slick) {             
			// hv.buildShowMobNav.homeTransparent();
				slick = $(slick.$slider);
				hv.homeMarquee.playPauseVideo(slick, "play");
			});
			slideWrapper.on("lazyLoaded", function (event, slick, image, imageSource) {
				lazyCounter++;
				if (lazyCounter === lazyImages.length) {
					lazyImages.addClass('show');
					// slideWrapper.slick("slickPlay");
				}
			});

			//start the slider
			slideWrapper.slick({
				fade:true,
				autoplaySpeed: ctime,
				lazyLoad: "progressive",
				speed: 600,
				arrows: false,
