         <script>
          $(document).ready(function () {

  var config = {
    duration: 300
  }

  var imageElement = $('.kb-fullview-image');
  var images = $('.kb-fullviewable');
  var overlay = $('.kb-fullview');
  var wrapper = $('.kb-fullview-wrapper');
  var logo = $('.kb-logo');

  function isOpen() {
    return overlay.attr('data-open') === 'true';
  }

  function closeOverlay() {
    overlay.removeClass('kb-fullview--open').delay(config.duration).hide(1);
    imageElement.attr('src','#');
    overlay.attr('data-open', 'false');
    enableBodyScroll();
    //logo.show();   
    overlay.removeClass('kb-fullview--zoomed');
  }

  function enableBodyScroll() {
    $('.kb-logo').css('margin-right', '');
    setTimeout(function () {
      $('body').removeClass('kb-overflow-hidden').css('margin-right', '0');
      SolUtils.restoreTabIndex();
      logo.css('z-index', '');
    }, config.duration);
  }

  function disableBodyScroll() {
    $('body').addClass('kb-overflow-hidden');
    SolUtils.removeTabIndex('.kb-fullview a, .kb-fullview button');
    if (SolUtils.hasScroll()) {
      $('body').css('margin-right', '' + (SolUtils.getScrollbarWidth()) + 'px');
      $('.kb-logo').css('margin-right', '' + (SolUtils.getScrollbarWidth() / 2) + 'px');
    }
  }

  function showOverlay() {
    overlay.show(1).addClass('kb-fullview--open');
    overlay.attr('data-open', 'true');
    logo.css('z-index', '1');
  }

  function loadImage(src) {
    imageElement.attr('src', src);
  }


  function clickImage(event) {
    event.stopPropagation();

    var clickPosition = {
      x: event.pageX - imageElement.offset().left,
      y: event.pageY - imageElement.offset().top
    }

    var scrollPercentage = {
      x: clickPosition.x / imageElement.width(),
      y: clickPosition.y / imageElement.height()
    }

    overlay.toggleClass('kb-fullview--zoomed');
    logo.toggle();

    // scroll and center on click
    $(wrapper).scrollLeft((event.target.width * scrollPercentage.x) - ($(window).width() / 2));
    $(wrapper).scrollTop((event.target.height * scrollPercentage.y) - ($(window).height() / 2));
  }


  images.on('click', function(e) {
    var imagesrc = this.dataset.fullviewsrc;
    if (!imagesrc)
      return;

    loadImage(imagesrc);
    showOverlay();
    disableBodyScroll();

  });


  overlay.on('click', closeOverlay);
  imageElement.on('click', clickImage);


  $(document).keyup(function (e) {
    if (e.keyCode === 27 && isOpen()) closeOverlay();
  });


});
