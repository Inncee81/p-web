        getCookie: function (cname, cdefault = '') {
            let name = cname + "=";
            let ca = document.cookie.split(';');
            for (let i = 0; i < ca.length; i++) {
                let c = ca[i];
                while (c.charAt(0) === ' ') {
                    c = c.substring(1);
                }
                if (c.indexOf(name) === 0) {
                    return c.substring(name.length, c.length);
                }
            }

            return cdefault;
        },

        // Set Cookie.
        setCookie: function (cname, cvalue, exdays) {
            let d = new Date();
            d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));

            let expires = "expires=" + d.toUTCString();
            document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
        },

        // Get Session Var.
        getSessionVar: function (key, value = '') {

            // Start New Session.
            let regex = '^https?:\/\/' + this.getDomain();
            if (!this.getCookie(key) || !document.referrer.match(regex)) {
                this.setCookie(key, value, 1);
            }

            // Return Session Value.
            return this.getCookie(key);
        },

        // Get GA.
        getGA: function () {
            return this.getCookie('_ga').replace(/GA\d+?\.\d+?\./gi, '');
        },

            this.setCookie('cpm-page', window.location.pathname, 1);

            return window.location.hostname;
